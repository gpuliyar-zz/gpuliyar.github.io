SELECT org.org_entity_name AS organization, f.risk_score AS inherent_risk,
       a.inherent_rollup_rating AS inherent_risk_score,
       e.risk_score AS residual_risk,
       a.residual_rollup_rating AS residual_risk_score,
       a.organization_id,a.scenario_id,a.assessment_type,b.level_num
  FROM ms_rsk_organization_rollup a,
       ms_rsk_org_levels b,
       ms_rsk_scenarios_scn c,
       ms_rsk_assessed_score e,
       ms_rsk_assessed_score f,
       si_org_entities_t org
 WHERE a.organization_id = b.org_id
   AND a.scenario_id = c.scenario_id
   AND a.organization_id = org.org_entity_id
  -- and c.default_scenario = '1'     
   AND a.version_no = (SELECT MAX (d.version_no)
                         FROM ms_rsk_organization_rollup d
                        WHERE d.organization_id = a.organization_id AND d.scenario_id = a.scenario_id 
AND d.assessment_type = a.assessment_type)
   AND round(a.residual_rollup_rating) BETWEEN round(e.minimum_value) AND round(e.maximum_value)
   AND round(a.inherent_rollup_rating) BETWEEN round(f.minimum_value) AND round(f.maximum_value) 
   and b.parent_org_id = :1
