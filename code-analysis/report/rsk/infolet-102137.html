SELECT COUNT(u.user_name)   as cnt
FROM
               SI_ORG_ROLES G,
               SI_USER_ORG_ROLES E ,
               SI_ROLE_ACTIVITIES A,
               SI_ACTIVITIES_T C,
               si_users_t u
           WHERE
                G.ORG_ROLE_ID = E.ORG_ROLE_ID
            AND  A.ROLE_ID = G.ROLE_ID
           AND  A.ACTIVITY_ID = C.ACTIVITY_ID
             AND  e.USER_ID =  u.user_id
             AND u.user_name=:1
               AND  ACTIVITY_NAME ='RSK - Manage Risk Factors'
