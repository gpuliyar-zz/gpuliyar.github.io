SQL 
SELECT risk_id, risk_name, residual_rollup_rating, inherent_rollup_rating,
       scenario_id, assessment_type, level_num + 1 AS next_level, level_num AS current_level, 1 as default_level, scenario_name
  FROM (SELECT a.risk_id, rsk.object_name AS risk_name,
               a.residual_rollup_rating, a.inherent_rollup_rating,
               a.scenario_id, a.assessment_type, a.version_no, b.level_num,
               c.object_name AS scenario_name,
               RANK () OVER (ORDER BY a.residual_rollup_rating DESC) res_rank
          FROM ms_rsk_risk_rollup a,
               ms_rsk_risk_levels b,
               ms_rsk_scenarios_scn c,
               ms_grc_risk rsk
         WHERE a.risk_id = b.risk_id
           AND a.risk_id = rsk.object_id
           AND a.scenario_id = c.scenario_id
           AND b.level_num = 1
           AND a.scenario_id = :1
           AND a.assessment_type = :2
           AND a.version_no = (SELECT MAX (d.version_no)
                                 FROM ms_rsk_risk_rollup d
                                WHERE d.risk_id = a.risk_id AND d.scenario_id = a.scenario_id 
AND d.assessment_type = a.assessment_type))
 WHERE res_rank <= 10
