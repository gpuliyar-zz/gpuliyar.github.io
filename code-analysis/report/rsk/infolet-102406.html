select distinct CORE_OBJECT_NAME,CORE_OBJECT_ID from  Ms_Rsk_Hm_Master where core_object_id in (SELECT DISTINCT (

      (SELECT parent_objective_name
      FROM ms_grc_objectives
      WHERE object_id=CORE_OBJECT_ID
      )

      CASE
        WHEN CORE_OBJECT_ID LIKE '%RQMT%'
        THEN
          (SELECT object_parent FROM ms_grc_requirement WHERE object_id= CORE_OBJECT_ID
          )
        ELSE
          CASE
            WHEN CORE_OBJECT_ID LIKE '%RQMT%'
            THEN
              (SELECT object_parent FROM ms_grc_product WHERE object_id= CORE_OBJECT_ID
              )
            ELSE
              (SELECT object_parent FROM MS_GRC_CORE_OBJECT WHERE object_id= CORE_OBJECT_ID
              )
          END
      END
  END) AS CO_Owner
