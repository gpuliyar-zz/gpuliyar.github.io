SELECT DISTINCT  c.core_object_id,e.OBJECT_NAME CORE_OBJECT_NAME, b.object_name SCENARIO_NAME, c.assessment_type,
         ms_rsk_assessments_task.fn_get_previous_range (residual_rollup_rating) as residual_rollup_rating,e.object_level--d.level_num
    FROM ms_rsk_core_object_rollup c,
            --MS_GRC_CORE_OBJECT e,
            ms_rsk_all_assmnt_type e,
         ms_rsk_scenarios_scn b
       --ms_rsk_core_object_levels d
   WHERE version_no =
            (SELECT MAX (version_no)
               FROM ms_rsk_core_object_rollup d
              WHERE c.core_object_id = d.core_object_id
                AND c.assessment_type = d.assessment_type
                AND c.scenario_id = d.scenario_id)
      AND e.OBJECT_ID = c.CORE_OBJECT_ID
     AND c.assessment_type = '1'
     AND c.scenario_id = b.scenario_id
     --AND d.core_object_id = c.core_object_id
     AND e.object_level = 1 --d.LEVEL_NUM = 1
GROUP BY b.object_name,
         c.assessment_type,
        -- d.level_num,
        e.object_level,
         c.core_object_id,
         residual_rollup_rating,
         e.OBJECT_NAME
