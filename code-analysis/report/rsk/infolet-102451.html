SELECT MS_JOIN_FOR_DISPLAY(ob_id) AS ob_id ,
  MS_JOIN_FOR_DISPLAY(ob_name)    AS ob_name
FROM
  (SELECT table_id AS ob_id,
    table_name     AS ob_name
  FROM si_data_tables
  WHERE table_name='MS_RSK_HEATMAP_RANGE'
  UNION
  SELECT metric_id AS ob_id,
    metric_name    AS ob_name
  FROM si_metrics_t
  WHERE metric_name='MS_RSK_SF_CONFIGURATION'
  AND metric_type  =10
  UNION
  SELECT infocenter_id AS ob_id,
    infocenter_name    AS ob_name
  FROM si_user_infocenter_t
  WHERE infocenter_name ='Risk Management Setup'
  AND user_id           =
    (SELECT user_id FROM si_users_t WHERE user_name=:1
    )
  )
