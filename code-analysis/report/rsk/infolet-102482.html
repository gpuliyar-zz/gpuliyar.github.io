SELECT c.creation_date startdate, c.object_name title,
       c.object_id || ' [ Completed ]' description, 0 ismonthevent,
          'Pushinfolet?id='
       || ms_apps_utilities.get_infolet_id (c.dd_object_type)
       || CHR (38)
       || 'proc='
       || c.process_instance_id
       || CHR (38)
       || 'instid='
       || c.instance_id
       || CHR (38)
       || 'edit_flag=Y'
       || CHR (38)
       || 'flag=1'
       || CHR (38)
       || 'emd=1' url,
       (SELECT ms_rsk_utils.get_scenario_name (c.scenario_name)
          FROM DUAL) scenario, c.risk_assessment_type, c.assessor, c.reviewer,
       c.organization_name, c.process_name, c.rsk_question_title,
       NULL end_date, NULL objid, NULL objtype
  FROM ms_rsk_risk_assessment_v c, ms_rsk_schedule_risk_a x
 WHERE EXISTS (
          SELECT object_id
            FROM ms_rsk_risk_assessment b
           WHERE c.schedule_risk_id = b.schedule_risk_id
             AND b.progress_status = 'Completed')
   AND instance_rec_num = 1
   AND instance_id =
          (SELECT MAX (instance_id)
             FROM ms_rsk_risk_assessment_v d
            WHERE d.process_instance_id = c.process_instance_id
              AND c.progress_status = 'Completed')
   AND 1 = 1
   AND x.object_id = c.schedule_risk_id
   AND ms_apps_security_pkg.check_object_access (x.dd_object_type,
                                                 x.restrict_access_to,
                                                 x.owner_organizations,
                                                 :1,
                                                 'VIEW'
                                                ) = 'ACCESS'
