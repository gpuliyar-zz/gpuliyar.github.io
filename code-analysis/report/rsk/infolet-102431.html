SELECT   SUBSTR (aa.object_name, 1, 30) display_name, aa.object_id object_id,
         (SELECT ms_join (risk_category)
             FROM MS_RSK_QUALT_ASSMNT_FA_rst hh
            WHERE aa.object_id = hh.object_id) AS rsk_categories, 
         (SELECT MS_JOIN_FOR_DISPLAY
                     (ms_apps_utilities.get_display_value
                                          ((SELECT enterprise_id
                      FROM si_ent_active_v),
                                           'MS GRC Risk Category',
                                           risk_category
                                          )
                     )
             FROM MS_RSK_QUALT_ASSMNT_FA_rst ff
            WHERE aa.object_id = ff.object_id) AS risk_category,
         (SELECT ms_join (owner_organizations)
             FROM MS_RSK_QUALT_ASSMNT_FA_OWs ee
            WHERE ee.object_id = aa.object_id) AS owner_organization,
         valid_from valid_from, valid_until valid_until,
         dd_object_type object_type, 'Y' AS over_ride_action,
            'proc='
         || TO_CHAR (dd_process_instance_id)
         || '|instid='
         || TO_CHAR (dd_instance_id)
         || '|edit_flag='
         || NVL (dd_edit_flag, 'N') AS Edit_action_parameters,
         ms_rsk_utils.include_expired (valid_from,
                                       valid_until
                                      ) AS expired_flag
    FROM ms_rsk_qualt_assmnt_fa aa
   WHERE ms_apps_security_pkg.check_object_access (dd_object_type,
                                                   restrict_access_to,
                                                   (SELECT ms_join (owner_organizations)
             FROM MS_RSK_QUALT_ASSMNT_FA_OWs z
            WHERE z.object_id = aa.object_id),
                                                   :1,
                                                   'VIEW'
                                                  ) = 'ACCESS'
ORDER BY aa.object_name
