Ms_rsk_assessments_task.Fn_Get_Previous_range(residual_rollup_rating) as residual_rollup_rating,
c.assessment_type ,E.OBJECT_LEVEL --d.level_num
from   Ms_Rsk_core_Object_Rollup c,
       Ms_Rsk_Scenarios_Scn b,
       --ms_rsk_core_object_levels d,
       --MS_GRC_CORE_OBJECT e
       MS_RSK_ALL_ASSMNT_TYPE E
Where  version_no      = (Select max(version_no) 
                          from Ms_Rsk_core_Object_Rollup f
                          Where c.core_object_id = f.core_object_id
                          and c.assessment_type  = f.assessment_type
                          and   c.scenario_id    = f.scenario_id )
AND e.OBJECT_ID = c.CORE_OBJECT_ID                          
And  c.scenario_id = b.scenario_id  
--and   d.core_object_id = c.core_object_id 
and   c.assessment_type  = :1
and   b.scenario_id     in (:2)  
and   c.core_object_id  in (:3)
and d.LEVEL_NUM = '1'
group by   b.object_name,c.assessment_type,E.OBJECT_LEVEL,
