SELECT DISTINCT USRS.user_id
, USRS.user_name
, ms_apps_utilities.get_user_full_name(USRS.user_id) USER_FULL_NAME
FROM SI_FLOW_DOWN_ORG_V ENT,
  SI_ORG_ROLES ORGR,
  SI_USER_ORG_ROLES UOR,
  SI_ACTIVITIES ACT,
  SI_ROLE_ACTIVITIES RACT,
  SI_USERS_T USRS
WHERE ORGR.ORG_ENTITY_ID = ENT.ORG_ENTITY_ID
AND RACT.ACTIVITY_ID     = ACT.ACTIVITY_ID
AND RACT.ROLE_ID         = ORGR.ROLE_ID
AND ORGR.ORG_ROLE_ID     = UOR.ORG_ROLE_ID
AND UOR.USER_ID          = USRS.USER_ID
AND (TRIM (ACTIVITY_NAME) = 'RSK - Assess Risks')
AND ((:2         IN ('3','2')
  AND (MS_RSK_CheckInList(ENT.org_entity_id,:1) = 1))
  OR (:2              ='1'))
order by USER_FULL_NAME
