Name MS_RSK_Process_For_Dashboard (102417)
SQL 
SELECT core_object_id,
       process_name,
       residual_rollup_rating,
       inherent_rollup_rating,
       scenario_id,
       assessment_type,
       level_num + 1 AS next_level,
       level_num AS current_level,
       1 as default_level,
       scenario_name
  FROM (SELECT DISTINCT a.core_object_id,
               e.object_name AS process_name,
               a.residual_rollup_rating,
               a.inherent_rollup_rating,
               a.scenario_id,
               a.assessment_type,
               a.version_no,
               --b.level_num,
               E.OBJECT_LEVEL AS level_num,
               c.object_name AS scenario_name,
               RANK() OVER(ORDER BY a.residual_rollup_rating DESC) res_rank
          FROM ms_rsk_core_object_rollup a,
               --ms_rsk_core_object_levels b,
               ms_rsk_scenarios_scn      c,
               --ms_grc_core_object        e
               MS_RSK_ALL_ASSMNT_TYPE E
         WHERE --a.core_object_id = b.core_object_id and
               a.core_object_id = e.object_id
           AND a.scenario_id = c.scenario_id
           AND E.OBJECT_LEVEL = 1 )--b.level_num = 1)
           AND a.scenario_id = :1
           AND a.assessment_type = :2
           AND a.version_no =
               (SELECT MAX(d.version_no)
                  FROM ms_rsk_core_object_rollup d
                 WHERE d.core_object_id = a.core_object_id
                   AND d.scenario_id = a.scenario_id
                   AND d.assessment_type = a.assessment_type))
 WHERE res_rank <= 10
