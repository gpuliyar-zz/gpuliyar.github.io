SQL 
SELECT g.object_name AS risks, f.risk_score AS inherent_risk,
       a.inherent_rollup_rating AS inherent_risk_score,
       e.risk_score AS residual_risk,
       a.residual_rollup_rating AS residual_risk_score, a.risk_id,
       a.scenario_id, a.assessment_type
  FROM ms_rsk_risk_rollup a,
       ms_rsk_risk_levels b,
       ms_rsk_scenarios_scn c,
       ms_rsk_assessed_score e,
       ms_rsk_assessed_score f,
       ms_grc_risk g
 WHERE a.risk_id = b.risk_id
   AND a.scenario_id = c.scenario_id
   AND a.risk_id = g.object_id
   AND a.scenario_id = :2
   AND a.assessment_type = :3
   AND a.version_no = (SELECT MAX (d.version_no)
                         FROM ms_rsk_risk_rollup d
                        WHERE d.risk_id = a.risk_id AND d.scenario_id = a.scenario_id 
AND d.assessment_type = a.assessment_type)
   AND round(a.residual_rollup_rating) BETWEEN round(e.minimum_value) AND round(e.maximum_value)
   AND round(a.inherent_rollup_rating) BETWEEN round(f.minimum_value) AND round(f.maximum_value)
   AND b.parent_risk_id = :1
