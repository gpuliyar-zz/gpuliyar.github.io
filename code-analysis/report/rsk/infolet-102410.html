    (SELECT object_name

    (SELECT ms_rsk_utils.Get_Edit_Url(dd_object_type,dd_process_instance_id,dd_instance_id,'Y',object_name)

  FROM Ms_Rsk_Scenarios_detail_Vw a,

    (SELECT SCENARIO_ID,
