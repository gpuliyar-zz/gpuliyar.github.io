SELECT  g.object_name AS core_objects, f.risk_score AS inherent_risk,
       a.inherent_rollup_rating AS inherent_risk_score,
       e.risk_score AS residual_risk,
       a.residual_rollup_rating AS residual_risk_score,
a.core_object_id,a.scenario_id,a.assessment_type,g.object_level--b.level_num
  FROM ms_rsk_core_object_rollup a,
       --ms_rsk_core_object_levels b,
       ms_rsk_scenarios_scn c,
       ms_rsk_assessed_score e,
       ms_rsk_assessed_score f,
       --ms_grc_core_object g
       ms_rsk_all_assmnt_type g
 WHERE --a.core_object_id = b.core_object_id AND
       a.scenario_id = c.scenario_id
   AND a.core_object_id = g.object_id
   --and c.default_scenario = '1' -- for default scenario
   AND g.object_level = :1--b.level_num = :1
   AND a.scenario_id = :2
   AND a.assessment_type = :3
   AND a.version_no = (SELECT MAX (d.version_no)
                         FROM ms_rsk_core_object_rollup d
                        WHERE d.core_object_id = a.core_object_id AND d.scenario_id = a.scenario_id 
AND d.assessment_type = a.assessment_type)
   AND round(a.residual_rollup_rating) BETWEEN round(e.minimum_value) AND round(e.maximum_value)
   AND round(a.inherent_rollup_rating) BETWEEN round(f.minimum_value) AND round(f.maximum_value)
