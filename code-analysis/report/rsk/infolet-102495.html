  where a.instance_id =    (Select max(instance_id)

                            where a.OBJECT_ID = mi.OBJECT_ID)
  and a.process_instance_id = (Select max(process_instance_id) 
                               from ms_rsk_risk_assessment_v mp
                               where a.OBJECT_ID = mp.OBJECT_ID)

                            where a.dd_current_stage   <> 'PUBLISH' 
                            and a.instance_id =    (Select max(instance_id) 
                                               from ms_rsk_risk_assessment_v mi
                                               where a.OBJECT_ID = mi.OBJECT_ID)
                             and a.process_instance_id = (Select max(process_instance_id) 
                                                         from ms_rsk_risk_assessment_v mp
                                                         where a.OBJECT_ID = mp.OBJECT_ID))
