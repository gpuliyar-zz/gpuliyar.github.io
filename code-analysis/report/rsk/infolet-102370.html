SELECT distinct 1 as ORGANIZATION_NAME,
  1 as ORGANIZATION_ID,
  SCENARIO_ID,
  SCENARIO_NAME,
  ASSESSMENT_TYPE,
 OBJECT_LEVEL_NUM+1
FROM TABLE(ms_rsk_lnd_reporting_utils.fn_lnd_rk_rt_org_level1()) 
WHERE ORGANIZATION_ID IN
  (SELECT DISTINCT owner_organization
  FROM ms_rsk_scenarios_orn
  WHERE scenario_id IN (scenario_id)
  )
AND assessment_type <> '1'
AND ORGANIZATION_ID IN
  (SELECT DISTINCT TO_CHAR(ORG_ENTITY_ID) AS object_id
  FROM SI_ORG_ROLES
  WHERE ORG_ROLE_ID IN
    (SELECT ORG_ROLE_ID
    FROM SI_USER_ORG_ROLES
    WHERE user_id=
      (SELECT USER_ID FROM SI_USERS_T WHERE USER_NAME =:1
      )
    )
  )
