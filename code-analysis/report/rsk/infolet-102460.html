Name MS_RSK_REPORT_ORGS_INFOL (102460)
SQL 
SELECT org.org_entity_id   AS org_entity_id
, org.parent_org_entity_id    AS parent_org_entity_id
, org.org_entity_name          AS org_entity_name
, porg.org_entity_name       AS parent_org_entity_name
, org.description                   AS org_description
, loc.location_name            AS org_location
, ohl.org_level                       AS org_level
, ohl.org_hierarchy_name          AS org_hierarchy_name
FROM si_org_entities org
, si_org_locations loc
, si_org_hierarchy_levels ohl
, si_org_entities porg
WHERE org.location_id = loc.location_id (+)
AND SYSDATE BETWEEN org.start_date AND Nvl(org.end_date, SYSDATE)
AND org.org_hierarchy_id = ohl.org_hierarchy_id
AND org.parent_org_entity_id = porg.org_entity_id (+)
and ohl.org_level <> 0
AND org.ORG_ENTITY_ID in(select distinct ORGANIZATION_ID from MS_RSK_RISK_ASSESSMENT_ORG where organization_id is not null and object_id in (select object_id from ms_rsk_risk_assessment where scenario_name in (select trim(column_value) from table(cast((MS_APPS_UTILITIES.SPLIT(:2) from dual) as ms_varchar2_array)) where column_value is not null)))
AND ORG.ORG_ENTITY_ID IN ( 
SELECT ORG_ENTITY_ID FROM
SI_ORG_ROLES
WHERE ORG_ROLE_ID IN (
select ORG_ROLE_ID from SI_USER_ORG_ROLES
where user_id=(SELECT USER_ID FROM SI_USERS_t WHERE USER_NAME =:1)))
and org.org_entity_id in (select owner_organization from ms_rsk_scenarios_orn where scenario_id in (select trim(column_value) from table(cast((select MS_APPS_UTILITIES.SPLIT(:2) from dual) as ms_varchar2_array)) where column_value is not null)
union 
 select decode(assessed_by,1,org.org_entity_id||'','-1') from ms_rsk_scenarios_scn
)
AND org.ORG_ENTITY_ID IN 
  (SELECT ORGANIZATION_id
  from ms_rsk_organization_rollup
  WHERE SCENARIO_ID IN
    (SELECT TRIM(COLUMN_VALUE)
    FROM TABLE(CAST(
      (SELECT MS_APPS_UTILITIES.SPLIT(:2) FROM DUAL
      ) AS ms_varchar2_array))
    WHERE column_value IS NOT NULL
    )
  )
Order By org.org_entity_name
