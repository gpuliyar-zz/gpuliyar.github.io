SQL 
SELECT PLAN_ID,
  plan_name,
  ASSMNT_ID,
  ASSMNT_URL,
  Assigned_On,
  Due_Date,
  Over_Due_By,
  Assessors,
  Approver
FROM
  (SELECT plan_id,
    ms_rsk_rpt_utils.build_form_url ('MS_RSK_SCHEDULE_RISK_ASSMN', plan_pid, plan_instance_id, 'N', plan_name ) AS plan_name,
    assmnt_id,
    ms_rsk_rpt_utils.build_form_url ('MS_RSK_RISK_ASSESSMENT', assmnt_pid, assmnt_instance_id, 'N', assmnt_id ) AS assmnt_url,
    assmnt_assigned_on                                                                                          AS Assigned_On,
    assmnt_due_date                                                                                             AS Due_Date,
    to_number((TRUNC(sysdate) - TRUNC(assmnt_due_date)))                                                        AS Over_Due_By,
    assmnt_assessor AS Assessors,
    assmnt_approver                                                                                             AS Approver
  FROM ms_rsk_assmnt_rpt_dtls
  WHERE assmnt_assessed_on IS NULL
  )
WHERE over_due_by >0
and :1 in (SELECT COLUMN_VALUE FROM TABLE (MS_APPS_UTILITIES.SPLIT(nvl(Assessors,:1))))
ORDER BY Over_Due_By DESC
