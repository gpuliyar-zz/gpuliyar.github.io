SQL 
SELECT   risk_score, COUNT (risk_id),level_num,scenario_id,assessment_type
    FROM (SELECT a.risk_id, a.residual_rollup_rating,
                 a.inherent_rollup_rating, a.scenario_id, a.assessment_type,
                 a.version_no, b.level_num, e.risk_score
            FROM ms_rsk_risk_rollup a,
                 ms_rsk_risk_levels b,
                 ms_rsk_scenarios_scn c,
                 ms_rsk_assessed_score e                 
           WHERE a.risk_id = b.risk_id
             AND a.scenario_id = c.scenario_id
             --and c.default_scenario = '1' -- for default scenario
             AND b.level_num = :1
             AND a.scenario_id = :2
             AND a.assessment_type = :3
             AND a.version_no = (SELECT MAX (d.version_no)
                                   FROM ms_rsk_risk_rollup d
                                  WHERE d.risk_id = a.risk_id AND d.scenario_id = a.scenario_id 
AND d.assessment_type = a.assessment_type)
             AND a.residual_rollup_rating BETWEEN e.minimum_value
                                              AND e.maximum_value)
GROUP BY risk_score,level_num,scenario_id,assessment_type
