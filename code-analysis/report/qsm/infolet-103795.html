SELECT ISSUE_ID as ISSUE_ID , 
ISSUE_SOURCE_REF as ISSUE_SOURCE_REF,
 ISSUE_TITLE||' (Issue ID:'||substr(ISSUE_ID,7)||')' as ISSUE_TITLE,
 ISSUE_STATUS as ISSUE_STATUS, 
 CREATED_ON as CREATED_ON, 
 ISSUE_DUE_DATE as ISSUE_DUE_DATE, 
 ms_apps_utilities.get_user_full_name(ISSUE_OWNER) as ISSUE_OWNER,
 OVERDUE_YN as OVERDUE_YN, 
OPEN_YN as OPEN_YN  ,
W.LINKED_QUESTIONS as LINKED_QUESTIONS
FROM (
SELECT distinct(X.FINDING_ID) ISSUE_ID, 
X.FINDING_FK ISSUE_SOURCE_REF, 
X.ISSUE_TITLE, 
NVL((select ISSUE_STATUS from MS_QSM_ISM_RPT_V WHERE ISSUE_SOURCE_REF  = FINDING_FK AND ISSUE_TITLE=X.ISSUE_TITLE AND ROWNUM <=1 ),'Open')
ISSUE_STATUS, 
X.DD_CREATED_ON CREATED_ON, 
X.ISSUE_DUE_DATE,
X.ISSUE_OWNER,
   Y.OVERDUE_YN OVERDUE_YN,
   'Y' OPEN_YN   
   
   FROM MS_QSM_RESP_ISSUE X,MS_QSM_ISM_RPT_V Y,MS_QSM_RESP_ISSUE_LQS Z 
   WHERE X.FINDING_FK IN (SELECT DISTINCT QUE_RESPONSE_ID FROM MS_QSM_QUESTNR_RESP_DUP_V WHERE QUE_REQUEST_ID=:1)
   AND X.FINDING_FK = Y.ISSUE_SOURCE_REF(+)
    and x.issue_title = y.issue_title(+)
   AND NVL(Y.Issue_Status,'X') <> 'Closed'
   and X.FINDING_ID  = Z.FINDING_ID  (+)
   
union all

SELECT FINDING_ID, FINDING_FK ISSUE_SOURCE_REF, ISSUE_TITLE, NVL( (select ISSUE_STATUS from MS_QSM_ISM_RPT_V WHERE ISSUE_SOURCE_REF  = FINDING_FK AND ISSUE_TITLE=X.ISSUE_TITLE AND ROWNUM <=1),'Open')  ISSUE_STATUS, DD_CREATED_ON CREATED_ON, ISSUE_DUE_DATE, ISSUE_OWNER,
   NULL    OVERDUE_YN,NULL OPEN_YN
   FROM MS_QSM_RESP_ISSUE X
   WHERE X.FINDING_FK = :2   
   
UNION ALL

SELECT a.ISSUE_ID,a.ISSUE_SOURCE_REF,a.ISSUE_TITLE,a.ISSUE_STATUS,a.CREATED_ON,a.ISSUE_DUE_DATE,a.ISSUE_OWNER,a.OVERDUE_YN,a.OPEN_YN 
FROM MS_QSM_ISM_RPT_V A,MS_QSM_OVERDUE_ISSUE_LOG B
WHERE A.ISSUE_ID=B.ISSUE_ID
AND B.SOURCE=:1
AND B.CLOSURE_STATUS IS NULL

UNION ALL

SELECT a.ISSUE_ID,a.ISSUE_SOURCE_REF,a.ISSUE_TITLE,a.ISSUE_STATUS,a.CREATED_ON,a.ISSUE_DUE_DATE,a.ISSUE_OWNER,NULL as OVERDUE_YN, NULL as OPEN_YN 
FROM MS_QSM_ISM_RPT_V A,MS_QSM_OVERDUE_ISSUE_LOG B
WHERE A.ISSUE_ID=B.ISSUE_ID
AND B.SOURCE_REFERENCE=:2
)  A,
 (
 
SELECT (select QST_NAME from MS_QSM_QUESTION z where X.LINKED_QUESTIONS = Z.QST_ID_PK ) AS LINKED_QUESTIONS, 
        y.finding_id AS finding_id,  x.dd_status_flag, LINKED_QUESTIONS as qst_id
FROM   MS_QSM_RESP_ISSUE_LQS x,  MS_QSM_RESP_ISSUE y  
WHERE x.finding_id(+) = Y.FINDING_ID
and exists (SELECT DISTINCT 1 FROM MS_QSM_QUESTNR_RESP_DUP_V 
            WHERE (QUE_REQUEST_ID=:1 or QUE_RESPONSE_ID = :2) and resp_status in (6,7,8,9) 
            and QUE_RESPONSE_ID = y.FINDING_FK) 
) W
WHERE a.iSSUE_ID =W.finding_id
ORDER BY ISSUE_TITLE,qst_id
