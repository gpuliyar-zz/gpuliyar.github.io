Name MS_ISM_ACTION_LIST (101202)
SQL 
SELECT action_id, action_name, locale_id, role_activity, previous_stage,
       current_stage
  FROM (SELECT a.action_id, b.action_name, b.locale_id, a.role_activity,
               a.current_stage, a.previous_stage, a.active, a.created_by,
               a.creation_date, a.last_update_date, a.last_updated_by,
               a.stage_type, a.status
          FROM ms_ism_stage_action_mapping a, ms_ism_actions_tl_v b
         WHERE a.action_id = b.action_id
          and Previous_Stage = :2
          and Current_Stage = :3
         and exists (Select 1 From Ms_Ism_Org_Flow_Up_V
                            Where User_Name = :4 And Activity_Name = a.Role_Activity)
         and a.active = 'Y'
          and (ms_ism_helper.Get_Action_List(a.action_id,nvl(:1,''),nvl(:6,''),:2,:3,decode(nvl(:5,''),'NIL','',:5),:4,:7,:8)=1))
         order by action_id
