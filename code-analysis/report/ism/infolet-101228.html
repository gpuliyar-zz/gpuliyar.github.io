SQL 
Select Issue_Source_Type,lvl,Period, Sum(Decode(No_Of_Issue,'Y',1,'N',0,0)) As No_Of_Issue,total,NVL(:2,'') AS Owner_Org
from (
SELECT 
ms_ism_rpt_utils.fn_get_source_type_dis_val(ISSUE_SOURCE_TYPE_ADT,source_type_stored,:1) as issue_source_type,lvl,
(select replace(period,'X',ms_apps_utilities.get_parameter_value (1, 'MS_ISM_Configuration', 'Enable_Ageing_Due_In_Months')) 
from Ms_Ism_Ageing_Table_Tl_V
where locale_id=(SELECT locale_id
      FROM si_users_t us, 
           si_locales lc
     WHERE us.user_name =:1
       AND lc.locale_string = us.locale)
and period_stored=lvl)  AS period,
       DECODE(lvl, 1, OVERDUE_THIS_MONTH,
                           2, OD_PAST_THREE_MONTH,
                           3, OD_BEYOND_THREE_MONTH,
                           4, DUE_THIS_MONTH,
                           5, DUE_WITHIN_X_MONTH,
                           6, Due_Beyond_X_Month,
                           'N/A') As No_Of_Issue,
      null as total
FROM   ms_ism_issue_reporting_v,
       (Select Level As Lvl From Dual Connect By Level <= 6)
       where (exists (SELECT 1
			  FROM dual
			 WHERE EXISTS (
					SELECT 1
					  FROM si_user_org_act_view a
					 WHERE a.user_name = :1
					   AND a.activity_name = 'ISM - View All Issues')) 
      OR (exists (SELECT 1
			  FROM dual
			 WHERE EXISTS (
					SELECT 1
					  FROM si_user_org_act_view a
					 WHERE a.user_name =:1
					   AND a.activity_name = 'ISM - View Issue')) AND exists(SELECT 1
          FROM si_user_flow_down_org_act_view b
         WHERE b.user_name = :1
           AND (  (b.activity_name = 'ISM - View Issue By  Issue Owner Org' AND b.org_entity_id = issue_owner_org)
               OR (b.activity_name = 'ISM - View Issue By Issue Approver Org' AND b.org_entity_id 
               IN (issue_approverorg, act_plan_approverorg, final_iss_approverorg))
               )))
      OR (TGT_OBJ_ID is not null and 0 <> (SELECT COUNT (CASE WHEN INSTR (TGT_OBJ_ID || ',', org_entity_id || ',') > 0 THEN 1 ELSE NULL END)
          FROM si_user_flow_down_org_act_view
         WHERE user_name =:1
           AND activity_name = 'ISM - View Issue By Related To Org')))
       and (:2 Is Null Or Issue_Owner_Org =:2)
       )
group by issue_source_type,period,total,lvl
