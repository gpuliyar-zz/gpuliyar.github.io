Name MS_ISM_Issue_By_Type (101318)
SQL 
SELECT ms_ism_rpt_utils.fn_get_lov_dis_val ('MS ISM Issue Type', b.ISSUE_TYPE, :1) AS ISSUE_TYPE,
   COUNT (1) AS Issue_Count,
   NULL AS dummy1
FROM MS_ISM_LAST_ONE_YEAR_ISSUE a,
   MS_ISM_ISSUE_IST b
WHERE a.issue_id = b.issue_id
AND ((1 = ms_ism_security_pkg.fn_ism_view_issue(:1) 
           AND 1 = ms_ism_security_pkg.fn_ism_view_issue_org(:1,a.ownerorg, a.approverorg, a.act_plan_approverorg, a.final_iss_approverorg)) 
           OR 1 = ms_ism_security_pkg.fn_ism_view_issue_all(:1)
           OR 1 = decode(a.orb_tgt_obj_id,NULL,-1,ms_ism_security_pkg.fn_ism_view_issue_orb(:1,a.orb_tgt_obj_id)))
AND a.ISSUE_STATUS_STORED NOT  IN('1','5','6','7')
AND To_Date (a.ISSUE_CREATED_ON, 'DD/MM/RRRR')>=Add_Months(To_Date (Sysdate, 'DD/MM/RRRR'), -12)
GROUP BY b.ISSUE_TYPE
