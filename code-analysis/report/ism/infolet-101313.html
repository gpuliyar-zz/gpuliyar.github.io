SQL 
SELECT b.Issue_Source    AS Issue_Source,
   b.Total_Count         AS Total_Count,
   b.Dummy1              AS Dummy1,
   b.Issue_Source_Stored AS Issue_Source_Stored,
   b.Source_Type         AS Source_Type
FROM
   (SELECT MS_ISM_HELPER.get_Issue_Source_Title (a.issue_source_type, a.issue_source_type_adt,a.issue_source)       AS Issue_Source,
      COUNT(1)                                                                              AS Total_Count ,
      NULL                                                                                  AS Dummy1,
      a.issue_source                                                                        AS Issue_Source_Stored,
      ms_ism_rpt_utils.fn_get_lov_dis_val('MS ISM Source Type',a.issue_source_type,:1) AS Source_Type,
      dense_rank() over(order by COUNT(1) DESC) rn
   FROM ms_ism_issue a
   WHERE a.issue_source_type_adt='MS_RSK_RISK_ASSESSMENT'
   AND Issue_Source        IS NOT NULL
   AND a.ISSUE_STATUS NOT  IN('1','5','6','7','10','None')
   AND reached_manage       = 'Y'
   AND ((1 = ms_ism_security_pkg.fn_ism_view_issue(:1) 
           AND 1 = ms_ism_security_pkg.fn_ism_view_issue_org(:1,a.issue_ownerorg,a.issue_approverorg,a.act_plan_approverorg, a.final_iss_approverorg)) 
           OR 1 = ms_ism_security_pkg.fn_ism_view_issue_all(:1)
           OR 1 = decode((SELECT MS_JOIN(tgt_obj_id) as tgt_obj_id
                                       FROM ms_ism_orb_relation_orb
                                      WHERE rel_source_id = a.issue_id
                                        AND additional_column4 = 'SI_ORG_ENTITIES_T'
                                       GROUP BY rel_source_id),NULL,-1,ms_ism_security_pkg.fn_ism_view_issue_orb(:1,(SELECT MS_JOIN(tgt_obj_id) as tgt_obj_id
                                       FROM ms_ism_orb_relation_orb
                                      WHERE rel_source_id = a.issue_id
                                        AND additional_column4 = 'SI_ORG_ENTITIES_T'
                                       GROUP BY rel_source_id))))
   GROUP BY a.issue_source_type,
   a.issue_source_type_adt,
        a.issue_source
   ) b
WHERE b.rn <=10
