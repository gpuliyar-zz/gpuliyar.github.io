Name MS_ISM_Issue_By_GRC_Object (101311)
SQL 
select b.tgt_obj_name,b.tgt_obj_id,b.Overdue As Overdue,b.DueBetween As DueBetween,b.Ongoing As Ongoing,b.Total_Count As Total_Count
from (
select b.tgt_obj_name,b.tgt_obj_id,
ms_ism_helper.get_issue_count_for_grc_object(b.tgt_obj_id,'2','OVERDUE_BY',:1) As Overdue,
ms_ism_helper.get_issue_count_for_grc_object(b.tgt_obj_id,'2','DUE_BETWEEN',:1) As DueBetween,
ms_ism_helper.get_issue_count_for_grc_object(b.tgt_obj_id,'2','ONGOING',:1) As Ongoing,
ms_ism_helper.get_issue_count_for_grc_object(b.tgt_obj_id,'2','TOTAL_COUNT',:1) As Total_Count,
dense_rank() over(order by ms_ism_helper.get_issue_count_for_grc_object(b.tgt_obj_id,'2','TOTAL_COUNT',:1) desc) rn
from ms_ism_issue a,ms_ism_orb_relation_orb b
where a.issue_id=b.rel_source_id 
and b.rel_config_id='2'
and a.issue_status not in('10','1','5','6','7','None')
group by b.tgt_obj_id ,b.tgt_obj_name) b
where b.rn <=5
