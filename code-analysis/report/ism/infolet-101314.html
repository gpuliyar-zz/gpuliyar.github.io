Name MS_ISM_Issue_By_Severity (101314)
SQL 
SELECT ms_ism_rpt_utils.fn_get_lov_dis_val('MS ISM Risk Rating',A.ISSUE_RISK_RATING_STR,:1) 
AS Issue_Rating,
   COUNT (1)               AS Issue_Count,
   NULL                    AS dummy1
FROM MS_ISM_LAST_ONE_YEAR_ISSUE a
WHERE a.issue_risk_rating IS NOT NULL
AND a.ISSUE_STATUS_STORED NOT  IN ('1','5','6','7','10')
AND (exists (SELECT 1
			  FROM dual
			 WHERE EXISTS (
					SELECT 1
					  FROM si_user_org_act_view a
					 WHERE a.user_name = :1
					   AND a.activity_name = 'ISM - View All Issues')) 
      OR (exists (SELECT 1
			  FROM dual
			 WHERE EXISTS (
					SELECT 1
					  FROM si_user_org_act_view a
					 WHERE a.user_name =:1
					   AND a.activity_name = 'ISM - View Issue')) AND exists(SELECT 1
          FROM si_user_flow_down_org_act_view a
         WHERE a.user_name = :1
           AND (  (a.activity_name = 'ISM - View Issue By  Issue Owner Org' AND a.org_entity_id = a.ownerorg)
               OR (a.activity_name = 'ISM - View Issue By Issue Approver Org' AND a.org_entity_id 
               IN (a.approverorg, a.act_plan_approverorg, a.final_iss_approverorg))
               )))
      OR (ORB_TGT_OBJ_ID is not null and 0 <> (SELECT COUNT (CASE WHEN INSTR (ORB_TGT_OBJ_ID || ',', org_entity_id || ',') > 0 THEN 1 ELSE NULL END)
          FROM si_user_flow_down_org_act_view
         WHERE user_name =:1
           AND activity_name = 'ISM - View Issue By Related To Org')))
AND To_Date (a.ISSUE_CREATED_ON, 'DD/MM/RRRR')>=Add_Months(To_Date (Sysdate, 'DD/MM/RRRR'), -12)
GROUP BY A.ISSUE_RISK_RATING_STR
