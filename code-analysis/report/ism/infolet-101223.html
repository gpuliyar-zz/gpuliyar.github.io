SELECT a.issue_id,
       comments AS action_comments,
       a.action_due_date,
       action_id,
       action_owner,
       action_per_complete,
       action_priority,
       action_status,
       report_json,
       action_type,
       reopen_action,
       action_title,
       action_owner_stored,
       action_description
  FROM (
    SELECT a.issue_id,
           a.action_due_date,
           a.action_id,
           ms_apps_utilities.get_user_full_name (a.action_owner) AS action_owner,
           DECODE (action_status, '10', '', action_per_complte) AS action_per_complete,
           ms_ism_lov_sto_disp_value_pkg.get_display_value_lov ('MS_ISM', 'MS ISM Priority', a.action_priority, NULL, usr.locale) AS action_priority,
           ms_ism_lov_sto_disp_value_pkg.get_display_value_lov ('MS_ISM', 'MS_ISM_FORM_STATUS', a.action_status, NULL, usr.locale) AS action_status,
           NULL AS report_json,
           ms_ism_lov_sto_disp_value_pkg.get_display_value_lov ('MS_ISM', 'MS ISM Action Item Type', NVL (b.action_type, '1'), NULL, usr.locale) AS action_type,
           ms_ism_helper.get_link_url_for_reopen_act (a.action_id, :2, a.action_status, a.action_title) AS reopen_action,
           a.action_title AS action_title,
           a.action_owner AS action_owner_stored,
           REGEXP_REPLACE (a.action_details, '<[^>]+>', NULL) AS action_description,
           c.instance_id comments_iid,
           MAX (c.instance_id) OVER (PARTITION BY c.action_id, c.issue_id) max_comments_iid,
           c.comments
      FROM ms_ism_action a, 
           ms_ism_issue_iap b,
           ms_ism_comments_log c,
           (SELECT locale FROM si_users_t a WHERE a.user_name = :2) usr
     WHERE a.issue_id = b.issue_id(+)
       AND a.action_id = b.issue_action_id(+)
       AND a.issue_id = :1
       AND a.issue_id = c.issue_id(+)
       AND a.action_id = c.action_id(+)) a
 WHERE a.comments_iid = a.max_comments_iid or a.max_comments_iid is null
