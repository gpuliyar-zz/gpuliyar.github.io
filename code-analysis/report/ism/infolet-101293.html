Name MS_ISM_ISSUE_DRILLDOWN (101293)
SQL 
SELECT SUBSTR (c.issue_title, 1, 30) display_name,
       c.issue_id object_id,
       c.dd_object_type object_type,
       'Y' AS over_ride_action,
          'proc='
       || TO_CHAR (c.dd_process_instance_id)
       || '|instid='
       || (c.dd_instance_id)
       || '|emd='
       || '4'
       || '|edit_flag='
       || NVL (c.dd_edit_flag, 'Y')
          AS view_form_action_parameters
  FROM  (SELECT b.issue_related_issues 
                FROM ms_ism_issue_ris b
               WHERE b.issue_id = :1) a, ms_ism_issue c
  WHERE c.issue_id = a.issue_related_issues
UNION ALL
SELECT /*+ index (act MS_ISM_ACTION_PK )*/SUBSTR (act.action_title, 1, 30) display_name,
       act.action_id object_id,
       act.dd_object_type object_type,
       'Y' AS over_ride_action,
       'proc=' || TO_CHAR(act.dd_process_instance_id) || '|instid=-1|edit_flag=' || NVL (act.dd_edit_flag, 'Y') || '|emd=' || '5' AS view_form_action_parameters
  FROM ms_ism_action act
 WHERE act.action_id = act.action_id 
 and act.issue_id = :1
ORDER BY 3, 1
