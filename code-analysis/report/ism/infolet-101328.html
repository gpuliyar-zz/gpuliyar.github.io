SELECT  A.ACTION_TITLE  AS ACTION_TITLE,
       ms_ism_rpt_utils.fn_get_local_status(A.ACTION_STATUS,:1) AS ACTION_STATUS,
       MS_ISM_UTILITIES.GET_USER_FULL_NAME (A.ACTION_OWNER, '') AS ACTION_OWNER,
       B.ISSUE_TITLE AS ISSUE_TITLE,
       A.ACTION_DUE_DATE,
       B.ISSUE_DUE_DATE,
       greatest( (TO_DATE (SYSDATE, 'DD/MM/RRRR')- TO_DATE (A.ACTION_DUE_DATE, 'DD/MM/RRRR'))
,0)
          AS Days_past_due,
       A.NO_OF_EXTENSIONS AS NO_OF_EXTENSIONS
,
(MS_ISM_RPT_UTILS.fn_get_report_json
                               ((select metric_id from si_metrics_t where metric_name='MS_ISM_ACTION'),
                                 a.dd_process_instance_id,
                                 a.dd_instance_id,
                                '''flag=1'
                                ||CHR(38)
                                ||'emd=3'

                                || 'wrapper= No''') )As report_json,
                                action_title as action_title_adt,
                                a.action_id as action_id,
                                a.action_owner as action_owner_stored,
                                MS_ISM_UTILITIES.GET_USER_FULL_NAME (A.ACTION_OWNER, '') AS ACTION_OWNER_ADT,
                                a.action_delegatee,
                                  MS_ISM_UTILITIES.GET_USER_FULL_NAME (a.action_delegatee, '') AS delegatee_full_name,
				b.issue_owner as issue_owner_stored,
				MS_ISM_UTILITIES.GET_USER_FULL_NAME (b.issue_owner, '') as issue_owner_full_name,
  ms_ism_helper.get_comma_sep_value_for_mlov ('OTHER_IMPACT','MS_ISM_ACTION_IOR',a.action_id,'ORG') AS IMPACTED_ORG
FROM MS_ISM_ACTION A, MS_ISM_ISSUE B
 WHERE A.ISSUE_ID = B.ISSUE_ID and (A.ACTION_OWNER=:1 OR A.ACTION_DELEGATEE=:1)
AND A.ACTION_STATUS IN ('2','5','6','16','21')
 ORDER BY A.DD_CREATED_ON desc
Description This infolet provides the details of all the actions raised in the system where the logged in user is same as the action owner identified for those particular actions. All the actions that are assigned to the logged in user are to be pulled up in this report (including the actions that are in pending approval and pending review and closed stage).
