Name MS_ISM_Issue_By_Bizz_Unit (101310)
SQL 
select b.Issue_ownerorg As Issue_ownerorg,b.ONGOING As ONGOING,b.DUE_BETWEEN_ONE_TIR As DUE_BETWEEN_ONE_TIR,b.DUE_BETWEEN_TIR_X As DUE_BETWEEN_TIR_X,b.OVERDUE_BY As OVERDUE_BY,b.Total_Count As Total_Count
from (
select 
(SELECT org_entity_name
    FROM si_org_entities_t
    WHERE org_entity_id = a.issue_ownerorg
    )  As Issue_ownerorg,
ms_ism_helper.fn_get_iss_count_on_iss_due_dt(a.ISSUE_OWNERORG,'ONGOING',:1) As ONGOING,
ms_ism_helper.fn_get_iss_count_on_iss_due_dt(a.ISSUE_OWNERORG,'DUE_BETWEEN1AND30',:1) As DUE_BETWEEN_ONE_TIR,
ms_ism_helper.fn_get_iss_count_on_iss_due_dt(a.ISSUE_OWNERORG,'DUE_BETWEEN30ANDX',:1) As DUE_BETWEEN_TIR_X,
ms_ism_helper.fn_get_iss_count_on_iss_due_dt(a.ISSUE_OWNERORG,'OVERDUE_BY',:1) As OVERDUE_BY,
ms_ism_helper.fn_get_iss_count_on_iss_due_dt(a.ISSUE_OWNERORG,'',:1) As Total_Count,
dense_rank() over(order by ms_ism_helper.fn_get_iss_count_on_iss_due_dt(a.ISSUE_OWNERORG,'',:1) desc) rn
from ms_ism_issue a
where a.issue_status not in('10','1','5','6','7','None') AND a.reached_manage = 'Y'
AND ((1 = ms_ism_security_pkg.fn_ism_view_issue(:1) 
           AND 1 = ms_ism_security_pkg.fn_ism_view_issue_org(:1,a.issue_ownerorg,a.issue_approverorg,a.act_plan_approverorg, a.final_iss_approverorg)) 
           OR 1 = ms_ism_security_pkg.fn_ism_view_issue_all(:1)
           OR 1 = decode((SELECT MS_JOIN(tgt_obj_id) as tgt_obj_id
                                       FROM ms_ism_orb_relation_orb
                                      WHERE rel_source_id = a.issue_id
                                        AND additional_column4 = 'SI_ORG_ENTITIES_T'
                                       GROUP BY rel_source_id),NULL,-1,
                                       ms_ism_security_pkg.fn_ism_view_issue_orb(:1,(SELECT MS_JOIN(tgt_obj_id) as tgt_obj_id
                                        FROM ms_ism_orb_relation_orb
                                      WHERE rel_source_id = a.issue_id
                                        AND additional_column4 = 'SI_ORG_ENTITIES_T'
                                       GROUP BY rel_source_id))))
group by a.issue_ownerorg) b
where b.rn <=5
