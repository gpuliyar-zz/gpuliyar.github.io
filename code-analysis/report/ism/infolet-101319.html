select ISSUE_ID,ACTION_ID,USER_NAME,USER_FULL_NAME ,

'8','Final Closure','14','Cancel Issue with Approval','No Stage Detected')  as COMMENT_STAGE

WHERE ISSUE_ID = :1
and action_id is null
and (user_actions is null or user_actions= '2')

select ISSUE_ID,ACTION_ID,USER_NAME,USER_FULL_NAME ,
COMMENTS||'*' as COMMENTS,ENTERED_DATE, COMMENTS_COLUMN_NAME ,PROCESS_INSTANCE_ID,

'8','Final Closure','14','Cancel Issue with Approval','No Stage Detected')  as COMMENT_STAGE
FROM MS_ISM_COMMENTS_LOG     
WHERE ISSUE_ID = :1
and user_actions= '1' and (user_name=:2 
or (select issue_owner from ms_ism_issue where issue_id=:1)=:2)
order by entered_date desc
