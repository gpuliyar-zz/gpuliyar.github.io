Name MS_ISM_ISSUE_ACTIONS (101282)
SQL 
select a.action_id 
        ,d.action_name as action
    from MS_ISM_STAGE_ACTION_MAPPING a
        ,MS_ISM_STAGES b
        ,MS_ISM_STAGES c
        ,MS_ISM_ACTIONS d
   where A.STAGE_TYPE = B.STAGE_TYPE
     and a.previous_stage = B.STAGE_ID
     AND A.STAGE_TYPE = C.STAGE_TYPE
     AND A.CURRENT_STAGE = C.STAGE_ID
     and A.STAGE_TYPE = D.STAGE_TYPE
     and A.ACTION_ID = D.ACTION_ID
     and A.ACTIVE ='Y'
     AND B.ACTIVE = 'Y'
     AND D.ACTIVE = 'Y'
     AND a.stage_type = 1 -- For Issue
     and a.current_stage = nvl(:1,1)
     and A.PREVIOUS_STAGE =  nvl(:2,0)
    and a.role_activity in (SELECT activity_name
                            FROM ms_qms_flow_down_org_v fdo
                            Where user_name = nvl(:3,user_name )
                             And activity_name like 'ISM%')
