Name MS_ISM_ISSUE_TEN_ORG_RATING (101300)
SQL 
SELECT ISSUE_OWNER_ORG,HIGH1,MEDIUM,LOW,RN,RATING0,RATING1,RATING2 FROM
(SELECT ISSUE_OWNER_ORG,HIGH1,MEDIUM,LOW,
dense_rank() over(order by TOTAL DESC) AS RN,RATING0,RATING1,RATING2 FROM
(select ISSUE_OWNER_ORG 
      ,Sum(Case When rating_stored_COUNT='1' Then Cnt Else null End) AS High1
      ,sum(case when rating_stored_COUNT='2' then cnt else null end) AS Medium
      ,Sum(Case When rating_stored_COUNT='3' Then Cnt Else null End) AS Low
     , Sum(Case When rating_stored_COUNT='Total' Then Cnt Else 0 End) AS Total
      ,Max(Case When rating_stored_COUNT='1' Then ms_ism_rpt_utils.fn_get_lov_dis_val('MS ISM Risk Rating',RATING_STORED,:1) End) Rating0
      ,max(case when rating_stored_COUNT='2' then ms_ism_rpt_utils.fn_get_lov_dis_val('MS ISM Risk Rating',RATING_STORED,:1) end) Rating1
      ,max(case when rating_stored_COUNT='3' then ms_ism_rpt_utils.fn_get_lov_dis_val('MS ISM Risk Rating',RATING_STORED,:1) end) Rating2
  from (Select ISSUE_OWNER_ORG
               ,Nvl(RATING_STORED,'Total') AS rating_stored_COUNT
               ,count(ISSUE_ID) cnt,RATING_STORED
            From ms_ism_detailed_issue_list A
            Where (1=Ms_Ism_Security_Pkg.Fn_Ism_Securities(Issue_Id,:1))
           AND ISSUE_STATUS_stored NOT IN('10','1','5','6','7','None') AND reached_manage = 'Y'
           Group By Rollup(ISSUE_OWNER_ORG,RATING_STORED))
            Where ISSUE_OWNER_ORG Is Not Null
   Group By ISSUE_OWNER_ORG))
   WHERE RN<=10
Description This infolet will list open issues with past due date by owner org and rating
