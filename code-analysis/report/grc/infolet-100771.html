SELECT asstSumm.OBJECT_ID AS Asset_id,  
  asstSumm.OBJECT_NAME as Asset_name,
  asstMlsSumm.ASSET_TYPE as Asset_Type,
  asstMlsSumm.OBJECT_LEVEL as Asset_Level,
  asstSumm.OBJECT_PARENT_NAME as Asset_Parent, 
  asstMlsSumm.BUSINESS_CRITICALITY as Business_criticality, 
  'MS_GRC_ASSET' as Additional_column3,
  NULL as form_url,
  'MS_GRC_ASSET' as ADDITIONAL_COLUMN4,
  asstSumm.upper_object_name as upper_object_name
FROM MS_GRC_ASSET_SUMMARY asstSumm,
	   MS_GRC_ASSET_MLS_SUMMARY asstMlsSumm
	 where asstSumm.OBJECT_ID = asstMlsSumm.OBJECT_ID  
	 and asstMlsSumm.LOCALE_ID = (select LOCALE_ID from SI_USERS_T T1 ,SI_LOCALES T2 where USER_NAME = :1 and T1.LOCALE = T2.LOCALE_STRING)
	and exists   (SELECT COLUMN_VALUE
    FROM TABLE(CAST(ms_grc_orb_report_filter.Fn_Asset_Orb_Filters(:1,:3,:2,:5,:4,:7,:6) AS ms_varchar2_array))
  where COLUMN_VALUE=asstSumm.OBJECT_ID
) 
and TRUNC(sysdate) between NVL(TRUNC(ASSTSUMM.VALID_FROM),TRUNC(sysdate)) and NVL(TRUNC(ASSTSUMM.VALID_UNTIL),TRUNC(sysdate))
