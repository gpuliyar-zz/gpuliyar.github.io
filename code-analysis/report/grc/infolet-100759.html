SELECT distinct Datas AS CATEGORY_ID ,
  display_order,
  MSL_NAME AS Displaying
FROM
  ( 
  With usr_locale_id as (select locale_id from si_locales where locale_string = (select locale from si_users_t where user_name = :1)),
   full_data AS
  (SELECT CATEGORY_NAME,
display_order,
MSL_NAME,
all_cnt,
ctg_cnt,
DUE_CNT,
DUE_FLG,
DIFF_FLG,
diff_cnt
  FROM
(SELECT
 CASE
    WHEN DUE_FLG='O'
  THEN 'Overdue'
   ELSE CATEGORY_NAME
      END AS CATEGORY_NAME,
      CASE
        WHEN due_flg='O'
        THEN 2
        WHEN CATEGORY_NAME ='Others'
        THEN 98
        ELSE display_order
      END AS display_order,
      CASE
        WHEN due_flg='O'
        THEN (select category_mls_name from MS_GRC_APP_MYTASK_MLS t 
where t.category_name='Overdue' 
and t.local_id = (select locale_id from usr_locale_id))
        WHEN Category_name ='Others'
        THEN (select category_mls_name from MS_GRC_APP_MYTASK_MLS t 
where t.category_name='Others' 
and t.local_id = (select locale_id from usr_locale_id))
        ELSE MSL_NAME
      END AS MSL_NAME,
      all_cnt,
      ctg_cnt,
      due_flg,
      CASE
        WHEN due_flg='O'
        THEN COUNT(CATEGORY_NAME) over(PARTITION BY due_flg)
      END DUE_CNT ,
      DIFF_FLG,
      CASE
        WHEN DIFF_FLG='R'
        THEN COUNT(CATEGORY_NAME) OVER(PARTITION BY DIFF_FLG)
      END DIFF_cnt ,
      row_number() over(PARTITION BY CATEGORY_NAME, due_flg ORDER BY assignment_id) row_index
    FROM
      (SELECT b.assignment_id,
        NVL(C.CATEGORY_NAME,'Others') CATEGORY_NAME,
        COUNT(A.METRIC_ID) over(PARTITION BY '')                            AS all_cnt,
        COUNT(A.METRIC_ID) over(PARTITION BY NVL(C.CATEGORY_NAME,'Others')) AS ctg_cnt,
        CASE
          WHEN (TRUNC(DUE_DATE) - SYSDATE) <=-1
 and status!=8
          THEN 'O'
          ELSE 'N'
        END DUE_FLG,
        CASE
          WHEN (TO_NUMBER(TO_CHAR(SYSDATE, 'J'))- TO_NUMBER(TO_CHAR(B.CREATION_DATE,'J'))) <=3
          THEN 'R'
          ELSE 'O'
        END diff_flg,
        C.ORDER_1           AS display_order,
        E.CATEGORY_MLS_NAME AS MSL_NAME
      FROM SI_METRICS_T A
      INNER JOIN SI_EVENT_ASSIGNMENTS B
      ON A.METRIC_ID =B.INFOLET_ID
      AND STATUS IN (4,5,8,1,9,10)
      LEFT JOIN MS_GRC_APP_MYTASK_FORM_MAP D
      ON A.FORM_NAME =D.METRIC_NAME
      LEFT JOIN MS_GRC_APP_MYTASK_CTGRY C
      ON C.CATEGORY_NAME=D.CATEGORY_NAME
      INNER JOIN SI_USERS_T F
      ON F.USER_NAME =:1
      AND F.USER_ID  =B.USER_ID
      LEFT JOIN SI_LOCALES G
      ON F.LOCALE =G.LOCALE_STRING
      LEFT JOIN MS_GRC_APP_MYTASK_MLS E
      ON C.CATEGORY_NAME=E.CATEGORY_NAME
      AND G.LOCALE_ID   =E.LOCAL_ID
      ))
  WHERE row_index=1
  )
SELECT category_name
  ||' ('
  ||
  CASE
    WHEN DUE_FLG ='O'
    THEN DUE_CNT
 ||'/'
 ||ALL_CNT
    ELSE TO_CHAR(ctg_cnt)
  END
  ||')' AS DATAS,
  DISPLAY_ORDER,
  MSL_NAME
  ||' ('
  ||
  CASE
   WHEN due_flg ='O'
   THEN DUE_CNT
||'/'
||ALL_CNT
ELSE TO_CHAR(ctg_cnt)
END
||')' AS MSL_NAME
FROM full_data
UNION ALL
SELECT 'All'
||' ('
||ALL_CNT
||')' AS DATAS,
0   AS display_order,
  (select category_mls_name from MS_GRC_APP_MYTASK_MLS t 
where t.category_name='All' 
and t.local_id = (select locale_id from usr_locale_id))
 ||' ('
 ||ALL_CNT
 ||')' AS MSL_NAME
FROM FULL_DATA
WHERE ROWNUM=1
UNION ALL
SELECT 
(select category_mls_name from MS_GRC_APP_MYTASK_MLS t

and t.local_id = (select locale_id from usr_locale_id))
 ||' ('
 ||DIFF_cnt
 ||')' AS DATAS,
 1     AS DISPLAY_ORDER,
 'New Tasks in Past 3 Days'
 ||' ('
 ||DIFF_cnt
 ||')' AS MSL_NAME
