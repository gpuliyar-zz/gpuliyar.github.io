Name MS_GRC_Yes_No_Infolet (101100)
SQL 
Select Stored_Value,
       Displayed_Value
From   MS_GRC_LOV_ORG_VALUES_V
where (trim(LOV_NAME) = 'MS GRC Yes No') and locale_id =
(select locale_id from si_locales a,si_users_t b where locale=locale_string and user_name = :1)
order by display_order asc
