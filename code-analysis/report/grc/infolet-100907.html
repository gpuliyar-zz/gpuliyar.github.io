Name MS_GRC_GET_OBJECTIVE_PARENT (100907)
SQL 
select distinct a.OBJECT_NAME as OBJ_NAME,
'Level '||a.objective_level AS obj_level,
  a.DD_OBJECT_TYPE   as OBJ_TYPE,
a.OBJECT_ID as OBJ_ID,
a.Objective_Type as Objective_Type,c.locale_id as locale_id
FROM MS_GRC_OBJECTIVES a, Ms_Grc_Objectives_Org b,si_locales c,
si_users_t d where 
c.locale_string = d.locale
AND d.user_name = :3
and a.object_id = b.object_id
and OBJECTIVE_TYPE = :1
AND a.DD_OBJECT_TYPE = 'MS_GRC_OBJECTIVES'
and TRUNC(sysdate)  >= TRUNC(NVL(VALID_FROM,sysdate))
And Trunc(Sysdate)  <= Trunc(Nvl(Valid_Until,Sysdate))
And A.Objective_Level = (:2 -1)
And ms_grc_utilities.MS_GRC_SECURITY_OBJ(a.dd_object_type, a.restrict_access_to, b.OWNER_ORGANIZATIONS, :3, 'VIEW',:1) = 'ACCESS'
order by obj_name
