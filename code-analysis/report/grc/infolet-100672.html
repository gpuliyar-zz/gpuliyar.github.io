Name MS GRC Product Type (100672)
SQL 
Select c.type_id as type_id,
       c.type_name as type_name,
       c.category_id as category_id,a.locale_id as locale_id From   MS_GRC_Product_Category_type c,
si_locales a,si_users_t b 
where 
c.locale_id= a.locale_id
and a.locale_string = b.locale and b.user_name = :1 and  category_id=:2
order by Type_name asc
