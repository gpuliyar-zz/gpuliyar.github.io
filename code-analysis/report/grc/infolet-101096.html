Name MS_GRC_STDT_ORB_Rpt_Inf (101096)
SQL 
SELECT STDT_SUMM.OBJECT_ID AS standard_id,
  STDT_SUMM.OBJECT_NAME    AS standard_name,
 MLS_SUMM.OBJECT_LEVEL as standard_Level,
  STDT_SUMM.PARENT as standard_parent,
  MLS_SUMM.std_type                                                  AS standard_type,
  'MS_GRC_STANDARD'                                         AS Additional_column3,
  NULL AS form_url,
   'MS_GRC_STANDARD'                                         AS Additional_column4,
STDT_SUMM.upper_object_name as upper_object_name
FROM MS_GRC_STANDARD_SUMMARY STDT_SUMM,	MS_GRC_STANDARD_MLS_SUMMARY MLS_SUMM
where STDT_SUMM.OBJECT_ID = MLS_SUMM.OBJECT_ID
and MLS_SUMM.LOCALE_ID = (select LOCALE_ID from SI_USERS_T T1 ,SI_LOCALES T2 where USER_NAME = :1 and T1.LOCALE = T2.LOCALE_STRING)
AND EXISTS
  (SELECT COLUMN_VALUE
  FROM TABLE( CAST (ms_grc_orb_report_filter.fn_standard_orb_filter(:1,:2,:3,:4,:5,:6) AS ms_varchar2_array))
  WHERE COLUMN_VALUE = STDT_SUMM.OBJECT_ID
  )
AND TRUNC(SYSDATE) BETWEEN NVL(TRUNC(STDT_SUMM.VALID_FROM),TRUNC(SYSDATE)) AND NVL(TRUNC(STDT_SUMM.VALID_UNTIL),TRUNC(SYSDATE))
order by upper_object_name
