SQL 
SELECT rsk_summ.Object_Id  AS Risk_id,  
   rsk_summ.Object_Name     AS risk_name, 
   mls_Summ.Object_Level as object_level,
  rsk_summ.OBJECT_PARENT as object_parent,
  mls_Summ.CAT_NAME_AGG AS RISK_CATEGORIES ,
   'MS_GRC_RISK'                                         AS Additional_column_3,
  NULL AS form_url,
 'MS_GRC_RISK'                                         AS Additional_column_4,
 rsk_summ.upper_object_name as upper_object_name
FROM MS_GRC_RISK_OBJ_SUMMARY rsk_summ,
    MS_GRC_RISK_OBJ_MLS_SUMMARY mls_Summ
	where rsk_summ.OBJECT_ID = mls_Summ.OBJECT_ID
and mls_Summ.LOCALE_ID = (select LOCALE_ID from SI_USERS_T T1 ,SI_LOCALES T2 where USER_NAME = :1 and T1.LOCALE = T2.LOCALE_STRING)
and rsk_summ.OBJECT_ID IN (SELECT COLUMN_VALUE
  FROM TABLE(CAST(ms_grc_orb_report_filter.Fn_Risk_Orb_Filter(:1,:2,:3,:4,:5,:6)AS ms_varchar2_array))
 )
AND TRUNC(SYSDATE) BETWEEN NVL(TRUNC(rsk_summ.VALID_FROM),TRUNC(SYSDATE)) AND NVL(TRUNC(rsk_summ.VALID_UNTIL),TRUNC(SYSDATE)) order by upper_object_name
