SQL 
 SELECT
MREF_SUMM.object_id as Model_reference_id,
MREF_SUMM.OBJECT_NAME as Model_reference_Name, 
MLS_SUMM.mod_type as Type,
MLS_SUMM.mod_source as Source,
'MS_GRC_MODEL_REFERENCE'  as Additional_column_3,
NULL AS form_url,
'MS_GRC_MODEL_REFERENCE'  as Additional_column_4,
MREF_SUMM.upper_object_name as upper_object_name
FROM MS_GRC_MOD_REF_SUMMARY MREF_SUMM,MS_GRC_MOD_REF_MLS_SUMMARY MLS_SUMM
where MREF_SUMM.OBJECT_ID = MLS_SUMM.OBJECT_ID
and MLS_SUMM.LOCALE_ID = (select LOCALE_ID from SI_USERS_T T1 ,SI_LOCALES T2 where USER_NAME = :1 and T1.LOCALE = T2.LOCALE_STRING)
AND EXISTS
(
SELECT COLUMN_VALUE
FROM 
TABLE( CAST(ms_grc_orb_report_filter.fn_MODEL_REF_orb_filters(:1, :2,:3,:4,:5)AS ms_varchar2_array) ) 
WHERE COLUMN_VALUE = MREF_SUMM.OBJECT_ID
)
AND TRUNC(SYSDATE) BETWEEN NVL(TRUNC(MREF_SUMM.VALID_FROM),TRUNC(SYSDATE)) AND NVL(TRUNC(MREF_SUMM.VALID_UNTIL),TRUNC(SYSDATE)) order by upper_object_name
