SQL 
SELECT contSumm.OBJECT_ID AS ID,
  contSumm.OBJECT_NAME    AS NAME,
  mlsSumm.object_level    AS OBJECT_LEVEL,
  (SELECT object_name
  FROM MS_GRC_COMPLIANCE_OBJE comp
  WHERE comp.object_id    = contSumm.parent
  AND comp.dd_object_type = 'MS_GRC_CONTROL'
  )                          AS OBJECT_PARENT,
  mlsSumm.Key_Control        AS KEY_CONTROL,
  mlsSumm.control_frequency  AS CONTROL_FREQUENCY,
  mlsSumm.control_purpose    AS CONTROL_PURPOSE,
  mlsSumm.control_priority   AS CONTROL_PRIORITY,
  mlsSumm.control_nature     AS CONTROL_NATURE,
  'MS_GRC_CONTROL'           AS Additional_column3,
  NULL                       AS form_url,
  'MS_GRC_CONTROL'           AS Additional_column4,
  contSumm.UPPER_OBJECT_NAME AS UPPER_OBJECT_NAME
FROM MS_GRC_CONTROL_SUMMARY contSumm,
  MS_GRC_CONTROL_MLS_SUMMARY mlsSumm
WHERE contSumm.OBJECT_ID = mlsSumm.OBJECT_ID
AND mlsSumm.LOCALE_ID    =
  (SELECT LOCALE_ID
  FROM SI_USERS_T T1 ,
    SI_LOCALES T2
  WHERE USER_NAME = :1
  AND T1.LOCALE   = T2.LOCALE_STRING
  )
AND EXISTS
  (SELECT COLUMN_VALUE
  FROM TABLE(CAST(Ms_Grc_Orb_Report_Filter.Fn_Control_Orb_Filters(:1,:3,:2,:4,:5,:6,:7,:8,:9) AS ms_varchar2_array))
  WHERE COLUMN_VALUE=contSumm.OBJECT_ID
  )
AND TRUNC(SYSDATE) BETWEEN NVL(TRUNC(contSumm.VALID_FROM),TRUNC(SYSDATE)) AND NVL(TRUNC(contSumm.VALID_UNTIL),TRUNC(SYSDATE)) order by upper_object_name
