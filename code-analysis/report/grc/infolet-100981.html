SQL 
select
  ProcSumm.Object_Id  AS Process_ID,
  ProcSumm.OBJECT_NAME AS Process_Name,
  ProcMlsSumm.Object_Level as Process_Level,
  ProcSumm.OBJ_PARENT AS Process_Parent,
  ProcMlsSumm.PROCESS_TYPE     AS Process_Type,
  ProcMlsSumm.PROCESS_CATEGORY AS Process_category,
  PROCMLSSUMM.BUSINESS_CRITICALITY as BUSINESS_CRITICALITY,
   'MS_GRC_PROCESS' as ADDITIONAL_COLUMN3,
   null as FORM_URL,
  'MS_GRC_PROCESS' as ADDITIONAL_COLUMN4,
    ProcSumm.UPPER_OBJECT_NAME AS UPPER_OBJECT_NAME
from MS_GRC_PROCESS_SUMMARY PROCSUMM,
    MS_GRC_PROCESS_MLS_SUMMARY PROCMLSSUMM
where PROCSUMM.OBJECT_ID = PROCMLSSUMM.OBJECT_ID
and PROCMLSSUMM.LOCALE_ID = (select LOCALE_ID from SI_USERS_T T1 ,SI_LOCALES T2 where USER_NAME = :1 and T1.LOCALE = T2.LOCALE_STRING)
and exists (
                SELECT COLUMN_VALUE
  from table(cast(MS_GRC_ORB_REPORT_FILTER.FN_PROCESS_ORB_FILTERS(:1,:2, :3, :7, :6, :4,:5) as MS_VARCHAR2_ARRAY))
  where COLUMN_VALUE=PROCSUMM.OBJECT_ID
)
and TRUNC(sysdate) between NVL(TRUNC(ProcSumm.VALID_FROM),TRUNC(sysdate)) and NVL(TRUNC(ProcSumm.VALID_UNTIL),TRUNC(sysdate))
