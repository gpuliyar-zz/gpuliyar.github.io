Name MS_GRC_REF_ORB_Rpt_Inf (101017)
SQL 
SELECT REF_SUMM.OBJECT_ID as Reference_ID,
  REF_SUMM.OBJECT_NAME as Reference_Name,
  MLS_SUMM.REFERENCE_TYPE AS REFERENCE_TYPE,
  MLS_SUMM.REFERENCE_CATEGORY AS REFERENCE_CATEGORY,
  MLS_SUMM.REFERENCE_URL_TYPE AS REFERENCE_URL_TYPE,
  'MS_GRC_REFERENCE' as Additional_column3,
 NULL as form_url,
  'MS_GRC_REFERENCE' as Additional_column4,
REF_SUMM.UPPER_OBJECT_NAME as upper_object_name
FROM MS_GRC_REFERENCE_SUMMARY REF_SUMM,MS_GRC_REFERENCE_MLS_SUMMARY MLS_SUMM
where REF_SUMM.OBJECT_ID = MLS_SUMM.OBJECT_ID
and MLS_SUMM.LOCALE_ID = (select LOCALE_ID from SI_USERS_T T1 ,SI_LOCALES T2 where USER_NAME = :1 and T1.LOCALE = T2.LOCALE_STRING)
AND EXISTS
    (SELECT column_value
    FROM TABLE(CAST(ms_grc_orb_report_filter.fn_reference_orb_filters(:1,:2,:3,:5,:4,:6)as ms_varchar2_array))
  WHERE column_value = REF_SUMM.OBJECT_ID
  )
AND TRUNC(SYSDATE) BETWEEN NVL(TRUNC(REF_SUMM.VALID_FROM),TRUNC(SYSDATE)) AND NVL(TRUNC(REF_SUMM.VALID_UNTIL),TRUNC(SYSDATE))
