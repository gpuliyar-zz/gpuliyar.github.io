Name MS_GRC_GET_LIBRARY_TYPE (100905)
SQL 
SELECT REPLACE(REPLACE(FORM_TITLE,'(',''),')',''),
  object_type
FROM MS_GRC_ORB_CONFIG_FORM_TITLE
WHERE object_type IN
  (SELECT DEST_OBJ_TYPE
  FROM ms_grc_rel_defn
  WHERE src_obj_type=ms_apps_utilities.get_parameter_value ( 1, 'MS GRC Mapping Report Parameter', 'Library Object' )
  AND ENABLED       = 'Y'
  )
AND LOCALE =
  ( SELECT LOCALE FROM SI_USERS_T WHERE USER_NAME=:1
  )
union all
select
ms_apps_utilities.get_display_value (
               ms_grc_utilities.active_ent_id,
               'MS_GRC_ALL',
               'All',
               1,
               (select LOCALE_ID from SI_LOCALES where LOCALE_STRING = (
            select LOCALE from SI_USERS_T where USER_NAME=:1 ))
               ) AS FORM_TITLE,'All' as object_type  FROM DUAL
