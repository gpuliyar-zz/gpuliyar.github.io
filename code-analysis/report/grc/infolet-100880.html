SQL 
SELECT   OBJECT_ID,
         OBJECT_NAME,
         VALID_FROM,
         VALID_UNTIL,
         RESTRICT_ACCESS_TO,
         DD_PROCESS_INSTANCE_ID,
         DD_INSTANCE_ID,
         OWNER_ORGANIZATIONS,
         DISPLAY_NAME,
         DD_OBJECT_TYPE OBJECT_TYPE,ATO_ORGANIZATION,RISKS,
         OVER_RIDE_ACTION,
         EXPIRED_FLAG,EDIT_ACTION_PARAMETERS
         ,OBJECT_ID AS CONTROL_ID
  FROM   (SELECT   OBJECT_ID,
                   OBJECT_NAME,
                   SUBSTR (OBJECT_NAME, 1, 30) DISPLAY_NAME,
                   DD_OBJECT_TYPE OBJECT_TYPE,
                   'Y' AS OVER_RIDE_ACTION,
                   MS_GRC_UTILITIES.INCLUDE_EXPIRED (VALID_FROM, VALID_UNTIL)
                      AS EXPIRED_FLAG,
                   VALID_FROM,
                   VALID_UNTIL,
                   RESTRICT_ACCESS_TO,
                   DD_OBJECT_TYPE,
                   DD_PROCESS_INSTANCE_ID,'proc='|| TO_CHAR(DD_PROCESS_INSTANCE_ID)||'|instid='||TO_CHAR(DD_INSTANCE_ID)||'|edit_flag='||NVL(DD_EDIT_FLAG,'Y')  AS EDIT_ACTION_PARAMETERS,
                   DD_INSTANCE_ID,
                   (SELECT   MS_CONCAT (D.OWNER_ORGANIZATIONS)
                      FROM   MS_GRC_COMPLIANCE_OBJE_ogn D
                     WHERE   A.OBJECT_ID = D.OBJECT_ID)
                      AS OWNER_ORGANIZATIONS, (SELECT MS_CONCAT(RISKS) FROM MS_GRC_COMPLIANCE_OBJE_RIS C WHERE A.OBJECT_ID = C.OBJECT_ID) RISKS,
         (SELECT MS_CONCAT(ATO_ORGANIZATION) FROM MS_GRC_COMPLIANCE_OBJE_ORG B WHERE A.OBJECT_ID = B.OBJECT_ID) ATO_ORGANIZATION
            FROM   MS_GRC_COMPLIANCE_OBJE A) T1
 WHERE   DD_OBJECT_TYPE='MS_GRC_CONTROL' AND ms_grc_utilities.ms_grc_SECURITY(dd_object_type,
                                                   restrict_access_to,
                                                   t1.OWNER_ORGANIZATIONS,
                                                   :1,
                                                   'VIEW') = 'ACCESS'
 AND TRUNC(sysdate) BETWEEN NVL(TRUNC(valid_from),TRUNC(sysdate)) AND NVL(TRUNC(valid_until),TRUNC(sysdate))
                                                   order by OBJECT_NAME asc
