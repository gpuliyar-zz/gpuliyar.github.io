  (SELECT C.ADDITIONAL_COLUMN3,

    c.tgt_obj_type

    ms_grc_orb_relations b,
    ms_grc_orb_relations_orb c
  WHERE c.rel_config_id IN
    (SELECT rel_id
    FROM ms_grc_rel_defn d
    WHERE d.src_obj_type = ms_apps_utilities.get_parameter_value (1, 'MS GRC Mapping Report Parameter', 'Library Object' )
    )
  AND c.rel_config_id = a.rel_id
  AND b.rel_source_id = c.rel_source_id
  ) t1,
  (SELECT object_name,
    object_id,
    dd_object_Type
  FROM ms_grc_all_published_objects_v

  ) t2
where T2.OBJECT_ID = T1.OBJECT_ID(+)

  ||'%')
AND :3 IS NOT NULL
