Name MS_GRC_REGB_ORB_Rpt_Inf (101019)
SQL 
SELECT REG_SUMM.OBJECT_ID as Regulatory_Body_ID,
  REG_SUMM.OBJECT_NAME as Regulatory_Body_Name,
   'MS_GRC_REGULATORY_BODY' as Additional_column3,
NULL as form_url,
 'MS_GRC_REGULATORY_BODY' as Additional_column4,
REG_SUMM.UPPER_OBJECT_NAME as upper_object_name
FROM MS_GRC_REG_SUMMARY	REG_SUMM,MS_GRC_REG_MLS_SUMMARY MLS_SUMM
where REG_SUMM.OBJECT_ID = MLS_SUMM.OBJECT_ID
and MLS_SUMM.LOCALE_ID = (select LOCALE_ID from SI_USERS_T T1 ,SI_LOCALES T2 where USER_NAME = :1 and T1.LOCALE = T2.LOCALE_STRING)
AND EXISTS
    (SELECT column_value
    FROM TABLE(CAST(ms_grc_orb_report_filter.fn_regulatory_body_orb_filter(:1,:2,:3) as ms_varchar2_array))
    WHERE column_value = REG_SUMM.OBJECT_ID
    )
AND TRUNC(SYSDATE) BETWEEN NVL(TRUNC(REG_SUMM.VALID_FROM),TRUNC(SYSDATE)) AND NVL(TRUNC(REG_SUMM.VALID_UNTIL),TRUNC(SYSDATE))
