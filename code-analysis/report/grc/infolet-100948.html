SQL 
select OBJC_SUMM.object_id AS object_id
,OBJC_SUMM.OBJECT_NAME AS OBJECT_NAME
,(SELECT object_name
    FROM MS_GRC_OBJECTIVES objc
    WHERE objc.object_id = OBJC_SUMM.parent
    ) AS PARENT_NAME
,MLS_SUMM.OBJECTIVE_TYPE AS OBJECTIVE_TYPE
,MLS_SUMM.COSO_TYPE AS COSO_TYPE
,TO_CHAR(MLS_SUMM.OBJECTIVE_TYPE) AS additional_details
,'MS_GRC_OBJECTIVES' AS ADDITIONAL_COLUMN3,
NULL AS FORM_URL 
,'MS_GRC_OBJECTIVES' AS ADDITIONAL_COLUMN4,
OBJC_SUMM.UPPER_OBJECT_NAME as upper_object_name
from MS_GRC_OBJECTIVE_SUMMARY OBJC_SUMM,MS_GRC_OBJECTIVE_MLS_SUMMARY MLS_SUMM
where OBJC_SUMM.OBJECT_ID = MLS_SUMM.OBJECT_ID
and MLS_SUMM.LOCALE_ID = (select LOCALE_ID from SI_USERS_T T1 ,SI_LOCALES T2 where USER_NAME = :1 and T1.LOCALE = T2.LOCALE_STRING)
AND EXISTS
    (SELECT COLUMN_VALUE 
    from table (CAST(ms_grc_orb_report_filter.Fn_Objectives_ORB_Filters(:1,:2,:3,:4,:5,:6) as ms_varchar2_array)) 
    WHERE COLUMN_VALUE = OBJC_SUMM.OBJECT_ID
    )
AND TRUNC(sysdate) BETWEEN NVL(TRUNC(OBJC_SUMM.valid_from),TRUNC(sysdate)) AND NVL(TRUNC(OBJC_SUMM.valid_until),TRUNC(sysdate))
