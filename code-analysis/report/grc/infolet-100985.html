SELECT prdSumm.OBJECT_ID

,'MS_GRC_PRODUCT' AS ADDITIONAL_COLUMN3
,NULL AS FORM_URL 
,'MS_GRC_PRODUCT' AS ADDITIONAL_COLUMN4,
prdSumm.UPPER_OBJECT_NAME AS UPPER_OBJECT_NAME
  from MS_GRC_PRODUCT_SUMMARY prdSumm,
    MS_GRC_PRODUCT_MLS_SUMMARY mlsSumm
	where prdSumm.OBJECT_ID = mlsSumm.OBJECT_ID
	and mlsSumm.LOCALE_ID = (select LOCALE_ID from SI_USERS_T T1 ,SI_LOCALES T2 where USER_NAME = :1 and T1.LOCALE = T2.LOCALE_STRING)
and exists  (SELECT COLUMN_VALUE 
    from table (cast(MS_GRC_ORB_REPORT_FILTER.FN_PRODUCT_ORB_FILTERS(:1,:2,:3,:4,:5,:6,:7,:8,:9)as MS_VARCHAR2_ARRAY)) 
     where COLUMN_VALUE=prdSumm.OBJECT_ID)
AND TRUNC(sysdate) BETWEEN NVL(TRUNC(prdSumm.valid_from),TRUNC(sysdate)) AND NVL(TRUNC(prdSumm.valid_until),TRUNC(sysdate))
