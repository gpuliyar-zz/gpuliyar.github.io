Name MS_GRC_CURRENCY_UPLOAD_PID (100819)
SQL 
SELECT max_id as UPLOAD_PROCESS_INSTANCE_ID
from (
SELECT  SUCCESS_TEMPLATE, ERROR_TEMPLATE,UPLOAD_PROCESS_INSTANCE_ID, max(UPLOAD_PROCESS_INSTANCE_ID) over(PARTITION BY FORM_NAME) AS max_id
  FROM MS_APPS_DU_UPLOAD_STATUS_LOG
 WHERE FORM_NAME = 'MS_GRC_CURRENCY_UPLOAD'
   AND SUCCESS_TEMPLATE IS NOT NULL
   AND ERROR_TEMPLATE IS NOT NULL
   )
where ROWNUM = :1
