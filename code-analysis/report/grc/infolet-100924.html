SQL 
SELECT MS_JOIN(obj_type
  ||'^*^'
  || NVL(rep_title_rt2,report_title)
  ||'$#$'
  || nvl2(rep_title_rt2,NVL(rpt_disp_name_rr ,rpt_link_disp_name),rpt_link_disp_name)
  ||'@#@'
  ||img_path
  ||'/'
  ||img_file_name
  || '%#%'
  || rel_id
  || '&#&'
  || nvl2(rep_title_rt2,NVL(rep_title_rt2,form_titls),form_titls)) AS report_detail
FROM
  (SELECT obj_type,rep_title_rt2,report_title,rpt_disp_name_rr || decode(MANDATORY_YN,'Y','*') as rpt_disp_name_rr,rpt_link_disp_name || decode(MANDATORY_YN,'Y','*') as rpt_link_disp_name,img_path,img_file_name,form_titls,rel_id,MANDATORY_YN
  FROM si_reports rt ,
    (SELECT pp.report_id as rep_id,
      pp.report_title as rep_title_rt2,
      rr.form_title as form_title,
      rr.rpt_link_display_name as rpt_disp_name_rr
    FROM si_reports_tl pp ,
      si_reports_t qq ,
      (SELECT b.locale_id,
        b.LOCALE_STRING
      FROM si_users_t a ,
        si_locales b
      WHERE user_name = :1
      AND a.LOCALE    =b.LOCALE_STRING
      ) kk ,
      MS_GRC_ORB_CONFIG_FORM_TITLE rr
    WHERE pp.report_id = qq.report_id
    AND pp.locale_id   = kk.locale_id
    AND qq.report_name = rr.report_name(+)
    AND rr.locale      = kk.locale_string
    )rt_2,
    (SELECT p.rel_id as rel_id,
      p.dest_obj_type as dest_obj_type,
      p.orb_report_name as orb_rep_name,
      p.object_type as obj_type,
      p.rpt_link_display_name as rpt_link_disp_name,
      p.form_title as form_titls,
      p.IMAGE_PATH as img_path,
      p.IMAGE_FILE_NAME as img_file_name
     ,p.MANDATORY_YN as MANDATORY_YN
    FROM
      (SELECT rel_id,
        dest_obj_type,
        b.report_name AS orb_report_name,
        b.object_type,
        b.rpt_link_display_name,
        b.form_title,
        b.IMAGE_PATH,
        b.IMAGE_FILE_NAME
,
       a.MANDATORY_YN 
      FROM ms_grc_rel_defn a,
        MS_GRC_ORB_CONFIG_FORM_TITLE b,
        si_locales d,SI_USERS_T usr
      WHERE b.locale      = d.locale_string
      AND src_obj_type    = :2
      AND d.locale_string = USR.LOCALE
      AND usr.user_name = :1
      AND a.dest_obj_type = b.object_Type
     AND a.enabled = 'Y'
      )p
    ) tt
  WHERE tt.orb_rep_name = rt.report_name
  AND rt.report_id         = rt_2.rep_id (+)
  AND rt.CREATED_BY        =
    (SELECT USER_ID FROM SI_USERS_T WHERE USER_NAME = 'pfadmin'
    )
  ORDER BY rpt_link_disp_name
  )
