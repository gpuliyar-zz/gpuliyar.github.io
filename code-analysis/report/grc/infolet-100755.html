SQL 
SELECT
AOC_SUMM.object_id as Area_of_Compliance_ID,
AOC_SUMM.OBJECT_NAME as Area_of_Compliance_Name,	
MLS_SUMM.aoc_type as aoc_type,	
'MS_GRC_AREA_OF_COMPLIANCE'  as Additional_column_3,
NULL as form_url,
'MS_GRC_AREA_OF_COMPLIA'  as Additional_column_4,
AOC_SUMM.UPPER_OBJECT_NAME as UPPER_OBJECT_NAME
FROM MS_GRC_AOC_SUMMARY AOC_SUMM,MS_GRC_AOC_MLS_SUMMARY MLS_SUMM
where AOC_SUMM.OBJECT_ID = MLS_SUMM.OBJECT_ID
and MLS_SUMM.LOCALE_ID = (select LOCALE_ID from SI_USERS_T T1 ,SI_LOCALES T2 where USER_NAME = :1 and T1.LOCALE = T2.LOCALE_STRING)
AND EXISTS
(
	SELECT COLUMN_VALUE
	FROM 
	TABLE( CAST(ms_grc_orb_report_filter.fn_AOC_ORB_filters(:1, :3,:2,:4)AS ms_varchar2_array) ) 
	WHERE COLUMN_VALUE = AOC_SUMM.OBJECT_ID
)
AND TRUNC(SYSDATE) BETWEEN NVL(TRUNC(AOC_SUMM.VALID_FROM),TRUNC(SYSDATE)) AND NVL(TRUNC(AOC_SUMM.VALID_UNTIL),TRUNC(SYSDATE))
