select form_name,metric_id,locale_id,metric_title from (
WITH form_det AS (
  SELECT metric_name,metric_id,form_name,metric_title
  FROM   si_metrics_t
  WHERE metric_name in ('MS_GRC_AREA_OF_COMPLIANCE',
'MS_GRC_ASSET_CLASS',
'MS_GRC_ASSET',
'MS_GRC_CONTROL',
'MS_GRC_FINANCIAL_ACCOUNTS',
'MS_GRC_FUNCTION',
'MS_GRC_MODEL_REFERENCE',
'MS_GRC_OBJECTIVES',
'MS_GRC_ORG_PROFILE',
'MS_GRC_PROCESS',
'MS_GRC_PRODUCT',
'MS_GRC_REFERENCE',
'MS_GRC_REGULATORY_BODY',
'MS_GRC_REQUIREMENT',
'MS_GRC_RISK',
'MS_GRC_STANDARD'))
select form_name,metric_title,metric_id,b.locale_id,b.locale_string 
from form_det met,si_locales b 
where LOCALE_STRING ='en_US'
union all
select a.form_name ,a.metric_title,a.metric_id,c.locale_id,c.locale_string 
from si_metrics_tl a,form_det b,si_locales c 
where c.LOCALE_STRING <> 'en_US'
and ENABLED_FLAG='Y'
and a.LOCALE_ID = C.LOCALE_ID
and a.metric_id=b.metric_id) order by form_name
