SQL 
SELECT REQ_SUMM.OBJECT_ID                                          AS requirement_id,
  REQ_SUMM.OBJECT_NAME                                             AS Requirement_Name,
  REQ_SUMM.OBJECT_PARENT                                           AS requirement_Parent,
  REQ_SUMM.Section_no                                              AS section_citation,
  MLS_SUMM.OBJECT_LEVEL                                            AS requirement_Level,
  MLS_SUMM.key_requirement                                         AS key_requirement,
  MLS_SUMM.frequency                                               AS frequency,
  'MS_GRC_REQUIREMENT' AS Additional_column3,
  NULL AS form_url,
'MS_GRC_REQUIREMENT' AS Additional_column4,
REQ_SUMM.UPPER_OBJECT_NAME AS UPPER_OBJECT_NAME
FROM MS_GRC_REQ_SUMMARY REQ_SUMM,MS_GRC_REQ_MLS_SUMMARY MLS_SUMM
where REQ_SUMM.OBJECT_ID = MLS_SUMM.OBJECT_ID
and MLS_SUMM.LOCALE_ID = (select LOCALE_ID from SI_USERS_T T1 ,SI_LOCALES T2 where USER_NAME = :1 and T1.LOCALE = T2.LOCALE_STRING)
AND EXISTS
  (SELECT COLUMN_VALUE
  FROM TABLE(CAST(ms_grc_orb_report_filter.fn_Requirement_orb_filters(:1,:3,:2,:4,:5,:6,:7,:8) AS ms_varchar2_array)) 
  WHERE COLUMN_VALUE = REQ_SUMM.OBJECT_ID
  )
AND TRUNC(SYSDATE) BETWEEN NVL(TRUNC(REQ_SUMM.VALID_FROM),TRUNC(SYSDATE)) AND NVL(TRUNC(REQ_SUMM.VALID_UNTIL),TRUNC(SYSDATE)) order by upper_object_name
