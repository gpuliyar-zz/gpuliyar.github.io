SELECT   object_id,
         OBJECT_NAME,
         object_level,
         VALID_FROM,
         VALID_UNTIL,
         RESTRICT_ACCESS_TO,
         DD_PROCESS_INSTANCE_ID,
         DD_INSTANCE_ID,
         OWNER_ORGANIZATIONS,
         display_name,
         DD_OBJECT_TYPE object_type,
         OVER_RIDE_ACTION,
         EXPIRED_FLAG,EDIT_ACTION_PARAMETERS,OBJECT_ID
  FROM   (SELECT  a.OBJECT_ID,
                   a.OBJECT_NAME,
                   SUBSTR (a.object_name, 1, 30) display_name,
                   a.dd_object_type object_type,  b.object_level,
                   'Y' AS OVER_RIDE_ACTION,
                   ms_Grc_utilities.INCLUDE_eXPIRED (a.VALID_FROM, a.VALID_UNTIL)
                      AS EXPIRED_FLAG,
                   a.VALID_FROM,
                   a.VALID_UNTIL,
                  a.RESTRICT_ACCESS_TO,
                  a. DD_OBJECT_TYPE,
                   a.DD_PROCESS_INSTANCE_ID,'proc='|| TO_CHAR(a.dd_process_instance_id)||'|instid='||TO_CHAR(a.dd_instance_id)||'|edit_flag='||NVL(a.dd_edit_flag,'Y')  AS EDIT_ACTION_PARAMETERS,
                   a.DD_INSTANCE_ID,
                   (SELECT   ms_concat (d.OWNER_ORGANIZATIONS)
                      FROM   ms_grc_standard_Org d
                     WHERE   B.object_id = d.object_id)
                      AS OWNER_ORGANIZATIONS
                                  FROM   ms_grc_AREA_OF_COMPLIA a,MS_GRC_STANDARD B where a.object_id=b.object_id) t1
 WHERE   dd_object_type='MS_GRC_STANDARD' AND  ms_grc_utilities.ms_grc_SECURITY(dd_object_type,
                                                   restrict_access_to,
                                                   t1.OWNER_ORGANIZATIONS,
                                                   :1,
                                                   'VIEW') = 'ACCESS'
 AND TRUNC(sysdate) BETWEEN NVL(TRUNC(valid_from),TRUNC(sysdate)) AND NVL(TRUNC(valid_until),TRUNC(sysdate))
                                                   order by OBJECT_NAME ASC
