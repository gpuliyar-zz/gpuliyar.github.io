SELECT SUBSTR(object_name,1,30) display_name,
  object_id,
  object_type,
  over_ride_action,
  edit_action_parameters
FROM
  (SELECT ms_grc_utilities.prfn_return_object_name(i.tgt_obj_id, d.dest_obj_type) AS object_name,
    tgt_obj_id                                                                    AS object_id,
    d.dest_obj_type                                                               AS object_type,
    'Y'                                                                           AS over_ride_action,
    ms_grc_utilities.get_url(dest_obj_type, i.tgt_obj_id)                         AS edit_action_parameters
  FROM ms_grc_rel_instance i ,
    ms_grc_rel_defn d
  WHERE d.rel_id = i.rel_id
  AND SRC_OBJ_ID = :1
  ) A
  ORDER BY OBJECT_NAME
