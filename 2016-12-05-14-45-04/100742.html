Name MS_GRC_GET_DATA_PROCESS (100742)
SQL 
SELECT object_id, object_name, valid_from, valid_until, restrict_access_to,
       dd_process_instance_id, dd_instance_id, owner_organizations,
       display_name, dd_object_type object_type, ato_organization, risks,
       over_ride_action, expired_flag, edit_action_parameters, object_id,CORE_TYPE,OBJECT_LEVEL
  FROM (SELECT object_id, object_name,
               SUBSTR (object_name, 1, 30) display_name,
               dd_object_type object_type,                    
                                          'Y' AS over_ride_action,
               ms_grc_utilities.include_expired (valid_from,
                                                 valid_until
                                                ) AS expired_flag,
               valid_from, valid_until, restrict_access_to, dd_object_type,
               dd_process_instance_id,
                  'proc='
               || TO_CHAR (dd_process_instance_id)
               || '|instid='
               || TO_CHAR (dd_instance_id)
               || '|edit_flag='
               || NVL (dd_edit_flag, 'Y') AS edit_action_parameters,
               dd_instance_id,
               (SELECT ms_concat
                                (d.owner_organizations)
                  FROM ms_grc_core_object_ogn d
                 WHERE a.object_id = d.object_id) AS owner_organizations,
               (SELECT ms_concat (risk)
                  FROM ms_grc_core_object_ris c
                 WHERE a.object_id = c.object_id) risks,
               (SELECT ms_concat (ato_organization)
                  FROM ms_grc_core_object_org b
                 WHERE a.object_id = b.object_id) ato_organization,OBJECT_LEVEL,CORE_TYPE
          FROM ms_grc_core_object a) t1
 WHERE ms_grc_utilities.ms_grc_SECURITY(dd_object_type, restrict_access_to, t1.OWNER_ORGANIZATIONS, :1, 'VIEW') = 'ACCESS' and dd_object_type='MS_GRC_PROCESS'
 AND TRUNC(sysdate) BETWEEN NVL(TRUNC(valid_from),TRUNC(sysdate)) AND NVL(TRUNC(valid_until),TRUNC(sysdate)) 
                                                order by OBJECT_NAME asc
