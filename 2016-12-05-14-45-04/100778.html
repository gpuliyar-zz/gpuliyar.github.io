SQL 
select col,count(col) as count ,:1 from (select current_value,applies_to,
     case
when current_value >= critical_rising_thresh then 'Critical Rising' 
when (current_value < critical_rising_thresh AND current_value >= non_critical_rising_thresh) then 'Non-Critical Rising'
when (current_value > critical_dropping_thresh AND current_value <= non_critical_dropping_thresh) then 'Non-Critical Dropping'
when (current_value <= critical_dropping_thresh) then 'Critical Dropping'
end col  
from ms_grc_kpi_kri_data_en a, ms_grc_kpi_kri_Defn b where applies_to=:1
and a.object_id=b.object_id and provided_on in (select max(provided_on) from ms_grc_kpi_kri_data_en a,ms_grc_kpi_kri_defn b where a.object_id=b.object_id group by a.ati_object_id,a.object_id)
and ms_apps_security_pkg.check_object_access(b.dd_object_type, b.restrict_access_to, (SELECT MS_CONCAT(owner_organizations) FROM MS_gRC_KPI_KRI_DEFN_OGN C WHERE C.OBJECt_ID=B.OBJECT_ID), :2, 'VIEW') = 'ACCESS' )
group by col
