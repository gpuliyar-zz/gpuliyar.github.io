Name MS_GRC_GET_DATA_ORG_PROFILE (100741)
SQL 
SELECT   object_name display_name,
           object_id object_id,DD_OBJECT_TYPE,
           dd_object_type object_type,
           org_level org_level,RESTRICT_ACCESS_TO,
           'Y' AS over_ride_action,
              'proc='
           || TO_CHAR (dd_process_instance_id)
           || '|instid='
           || TO_CHAR (dd_instance_id)
           || '|edit_flag='
           || NVL (dd_edit_flag, 'Y')
              AS edit_action_parameters,
           Valid_From,
           Valid_Until,
             (SELECT   ms_concat (d.OWNER_ORGANIZATIONS)
                      FROM   ms_grc_org_profile_ORG d
                     WHERE   a.object_id = d.object_id)
                      AS OWNER_ORGANIZATIONS,
           ms_Grc_utilities.INCLUDE_eXPIRED (VALID_FROM, VALID_UNTIL)
              AS EXPIRED_FLAG,object_id
    FROM   ms_grc_org_profile a
     WHERE   ms_apps_security_pkg.check_object_access (dd_object_type,
                                                     restrict_access_to,
                                                    (SELECT   ms_concat (d.OWNER_ORGANIZATIONS)
                      FROM   ms_grc_org_profile_ORG d
                     WHERE   a.object_id = d.object_id),
                                                     :1,
                                                     'VIEW') = 'ACCESS'
 AND TRUNC(sysdate) BETWEEN NVL(TRUNC(valid_from),TRUNC(sysdate)) AND NVL(TRUNC(valid_until),TRUNC(sysdate)) 
ORDER BY  display_name
