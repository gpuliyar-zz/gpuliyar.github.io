SQL 
SELECT DISTINCT fdo.user_id,
        fdo.user_name,
        ms_apps_utilities.get_user_full_name(fdo.user_id) user_full_name
  FROM SI_USERS_T fdo,
        si_user_org_roles uorg
 WHERE fdo.user_id = uorg.user_id
   AND nvl(fdo.end_date, SYSDATE+1)>=SYSDATE
   AND :1 IS NOT NULL
ORDER BY fdo.user_name
