SELECT DISTINCT E.CATEGORY_MLS_NAME AS DISPLAYING,
C.ORDER_1 as displayorder
FROM SI_METRICS_T A
INNER JOIN SI_EVENT_ASSIGNMENTS B
ON A.METRIC_ID =B.INFOLET_ID
LEFT JOIN MS_GRC_APP_MYTASK_FORM_MAP D
ON A.FORM_NAME =D.METRIC_NAME
LEFT JOIN MS_GRC_APP_MYTASK_CTGRY C
ON C.CATEGORY_NAME=D.CATEGORY_NAME
LEFT JOIN MS_GRC_APP_MYTASK_MLS E
ON C.CATEGORY_NAME=E.CATEGORY_NAME
INNER JOIN SI_USERS_T F
ON F.USER_NAME =:1
AND F.USER_ID  =B.USER_ID
LEFT JOIN SI_LOCALES G
ON G.LOCALE_ID           =E.LOCAL_ID
AND F.LOCALE             =G.LOCALE_STRING
WHERE CATEGORY_MLS_NAME IS NOT NULL
UNION
SELECT 'Overdue' AS DISPLAYING, 0 as displayorder FROM DUAL
UNION
select 'Others' AS Displaying, 98 as displayorder from dual
UNION
SELECT 'All' AS DISPLAYING, 99 AS DISPLAYORDER FROM DUAL
order by displayorder
