SELECT a.displayed_value AS display_value, b.action_value AS stored_value
  FROM ms_qs_list_of_values_t a, ms_grc_kpi_kri_action_de b
 WHERE a.lov_id = (SELECT LOV_ID
                     FROM ms_qs_lov_names
                    WHERE lov_name = 'MS GRC KPI KRI Action for Data Entry')
   AND a.stored_value = b.action_value
   AND b.filter_cond =
          (CASE
              WHEN :1 IS NULL
               OR :1 NOT IN
                               ('DATA_APPROVER','DATA_DEFINITION_OWNER')
                 THEN 'INITIATE'
              ELSE 'NEXT'
           END
          )
