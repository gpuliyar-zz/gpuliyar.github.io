Name MS_GRC_Metrics_Trend_Dashboard_Infolet (100844)
SQL 
SELECT a.object_id,
  a.ati_object_id,
  MS_GRC_UTILITIES.GET_VALUE_IN_DATE_FORMAT (a.current_date_value,B.FREQUENCY) current_date_value,
  ROUND(a.CRITICAL_DROPPING_THRESH,2)     AS X_Critical_Dropping_Thresh,
  ROUND(a.CRITICAL_RISING_THRESH,2)       AS X_Critical_Rising_Thresh,
  ROUND(a.Non_CRITICAL_DROPPING_THRESH,2) AS X_Non_Critical_Dropping_Thresh,
  ROUND(a.NON_CRITICAL_RISING_THRESH,2)   AS X_Non_Critical_Rising_Thresh,
  ROUND(a.EXPECTED_VALUE,2)               AS X_Expected_Value,
  ROUND(a.current_value,2)                AS X_Xatest_value,
  a.current_date_value                    AS CURRENT_DATE
FROM ms_grc_kpi_kri_data_en a,
  ms_Grc_kpi_kri_Defn b
WHERE a.OBJECT_ID       =B.OBJECT_ID
AND CURRENT_DATE_VALUE >=MS_GRC_UTILITIES.GET_START_DATE_VAL(A.OBJECT_ID,A.ATI_OBJECT_ID,:3,MS_GRC_UTILITIES.GET_END_DATE_VAL(A.OBJECT_ID,A.ATI_OBJECT_ID,:4))
AND current_date_value <=ms_grc_utilities.get_end_Date_val(a.object_id,a.ati_object_id,:4) 
and ((trim(a.Object_ID) = :1)AND(trim(a.Ati_Object_Id) = nvl(:2,a.ati_object_id)))
ORDER BY 10
