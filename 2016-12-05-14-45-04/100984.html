Name MS_GRC_SHOW_PARENT_OBJECTS (100984)
SQL 
SELECT distinct object_name AS obj_name,
  'Level '
  ||object_level AS obj_level,
  DD_OBJECT_TYPE AS OBJ_TYPE,
  a.OBJECT_ID    AS OBJ_ID
FROM MS_GRC_CORE_OBJECT a,
  MS_GRC_CORE_OBJECT_OGN B
where a.OBJECT_ID          =B.OBJECT_ID
AND A.DD_OBJECT_TYPE = 'MS_GRC_PROCESS'
AND dd_PROCESS_INSTANCE_ID =
  (SELECT MAX(dd_PROCESS_INSTANCE_ID)
  FROM MS_GRC_CORE_OBJECT B
  WHERE B.dd_PROCESS_INSTANCE_ID =a.dd_PROCESS_INSTANCE_ID
  AND DD_OBJECT_TYPE             =a.DD_OBJECT_TYPE
  )
AND OBJECT_LEVEL    = (:1 -1)
AND TRUNC(sysdate) >= TRUNC(NVL(VALID_FROM,sysdate))
and TRUNC(sysdate) <= TRUNC(NVL(VALID_UNTIL,sysdate))
and 
  MS_APPS_SECURITY_PKG.CHECK_OBJECT_ACCESS(a.DD_OBJECT_TYPE, a.RESTRICT_ACCESS_TO,
  (select OWNER_ORG_IDS from MS_GRC_PROCESS_RV where OBJECT_ID=a.OBJECT_ID
  ),:2, 'VIEW') = 'ACCESS'
