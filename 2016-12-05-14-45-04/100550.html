SELECT t2.object_name AS NAME, t2.object_id AS ID,
       '['||tgt_obj_type ||'] '||substr(t1.tgt_obj_name,1,3600) AS mapped_name, t1.tgt_obj_id AS mapped_id,
       DECODE (NVL (t1.tgt_obj_name, 'n\a'), 'n\a', NULL, 'X') DATA,replace(replace(tgt_obj_Type,'(',''),')','') AS tgt_obj_Type,substr(t1.tgt_obj_name,1,3600) as mapped_name_only
  FROM (SELECT               --c.tgt_obj_type,a.dest_obj_type,c.rel_config_id,
               ms_grc_utilities.get_grc_object_name
                  (b.rel_source_object_id,
                   ms_apps_utilities.get_parameter_value
                                           (1,
                                            'MS GRC Mapping Report Parameter',
                                            'Library Object'
                                           )
                  ) object_name,
               b.rel_source_object_id AS object_id, c.tgt_obj_name,
               c.tgt_obj_id,c.tgt_obj_type
          FROM ms_grc_rel_defn a,
               ms_grc_orb_relations b,
               ms_grc_orb_relations_orb c
         WHERE c.rel_config_id IN (
                  SELECT rel_id
                    FROM ms_grc_rel_defn d
                   WHERE d.src_obj_type =
                            ms_apps_utilities.get_parameter_value
                                           (1,
                                            'MS GRC Mapping Report Parameter',
                                            'Library Object'
                                           ))
           AND c.rel_config_id = a.rel_id
           AND b.rel_source_id = c.rel_source_id) t1,
       (SELECT object_name,object_id,dd_object_Type
          FROM ms_grc_all_published_objects_v
         WHERE dd_object_type =
                  ms_apps_utilities.get_parameter_value
                                           (1,
                                            'MS GRC Mapping Report Parameter',
                                            'Library Object'
                                           )) t2
 WHERE t2.object_id = t1.object_id(+)
