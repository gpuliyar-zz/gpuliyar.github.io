Name MS GRC KPI KRI Assigned Data Entries Report Infolet (100517)
SQL 
SELECT a.object_name,
  a.object_id,
 a.event_id,
  (SELECT ms_grc_utilities.GET_STATUS_name(:1,a.OBJ_STATUS) FROM DUAL
  ) AS sTATUS,
  a.INDICATOR_MEASUREMENT,
  ms_APPS_utilities.get_DISPLAY_VALUE(MS_GRC_UTILITIES.ACTive_ENT_ID ,'MS GRC KI Applies To Items',a.applies_to ) applies_to,
  (SELECT ms_grc_kpi_kri_utilities.get_kpi_kri_ati_name (a.applies_to,a.ati_object_id )
  FROM DUAL
  )                   AS ASSESSMENT_FOR,
  (trunc(a.NEXT_SCHEDULED_DATE)) AS DUE_BY,
  DECODE(SIGN(TRUNC(SYSDATE- a.NEXT_SCHEDULED_DATE)),1,'Days Past Due is'
  ||TO_CHAR(TRUNC(SYSDATE) - TRUNC(a.NEXT_SCHEDULED_DATE),'99')
  ||' days',               -1,'Days Past Due is'
  ||TO_CHAR(TRUNC(SYSDATE) -TRUNC(a.NEXT_SCHEDULED_DATE),'99')
  ||' days', 0,'Days Past Due is'
  ||TO_CHAR(TRUNC(SYSDATE)-TRUNC(a.NEXT_SCHEDULED_DATE),'99')) AS past_due_date,
  (TRUNC(SYSDATE)         - TRUNC(a.NEXT_SCHEDULED_DATE))      AS days_past_due,
  a.assigned_on,
  a.current_date_value,
 a.previous_period,
  a.previous_value,
  a.process_instance_id,
  a.instance_id,
  ms_apps_utilities.get_display_value (MS_GRC_UTILITIES.ACTive_ENT_ID , 'MS GRC KI Frequency', a.frequency ) frequency,
  a.CRITICAL_RISING_THRESH,
  a.CRITICAL_DROPPING_THRESH,
  a.NON_CRITICAL_RISING_THRESH,
  a.NON_CRITICAL_DROPPING_THRESH,
  a.EXPECTED_VALUE,
  a.ATI_COMMENTS,
  b.dd_process_instance_id,b.dd_instance_id,
 (SELECT ms_Grc_utilities.Get_Edit_Url(a.DD_OBJECT_TYPE, a.PROCESS_INSTANCE_ID, a.iNSTANCE_ID,' ',a.event_id )
    FROM dual
    )        as EVENT_ID_URL,
(select ms_grc_utilities.fn_kpi_kri_preval_color_code(a.object_id,a.event_id) from dual) as color_code,ms_APPS_utilities.get_DISPLAY_VALUE(MS_GRC_UTILITIES.ACTive_ENT_ID ,'MS GRC KI Indicator Measurement',a.INDICATOR_MEASUREMENT) as indi_measure_dis_val
FROM ms_grc_kpi_kri_data_entry_v a,ms_grc_kpi_kri_defn b
WHERE a.process_instance_id IN
  (SELECT process_instance_id
  FROM si_event_assignments
  WHERE user_id   = ms_apps_utilities.get_user_id(:1)
  AND infolet_id IN
    (SELECT metric_id
    FROM si_metrics_t
    WHERE metric_name = 'MS_GRC_KPI_KRI_DATA_ENTRY'
    )
  AND a.instance_id =
    (SELECT MAX (instance_id)
    FROM ms_grc_kpi_kri_data_entry_v
    WHERE process_instance_id = a.process_instance_id
    )
  )
  and a.object_id=b.object_id
order by object_name,object_id
Description Infolet to show assigned data entries to an user
