Name MS GRC KPI KRI Trend Report Infolet (100527)
SQL 
SELECT object_name, ati_name, level1, ms_Grc_utilities.GET_VALUE_IN_DATE_FORMAT (date1, frequency) AS date1,

       my_count  FROM (SELECT DISTINCT object_name,
                        ms_grc_kpi_kri_utilities.get_kpi_kri_ati_name
                                                      (assement_for,
                                                       ati_obj_id
                                                      ) ati_name,
                        LEVEL AS level1,
                        ms_grc_kpi_kri_utilities.get_kpi_kri_curr_date
                                                         (:1,
                                                          LEVEL,
                                                          ms_grc_utilities.get_Start_Date_val(:1,ati_obj_id,:2,ms_grc_utilities.get_end_Date_val(:1,ati_obj_id,:3),frequency),
                                                          ms_grc_utilities.get_end_Date_val(:1,ati_obj_id,:3),
                                                          ati_obj_id
                                                         ) AS date1,
                        ms_grc_kpi_kri_utilities.get_kpi_kri_curr_value
                           (object_id,
                            ati_obj_id,
                            ms_grc_kpi_kri_utilities.get_kpi_kri_curr_date
                                                                   (object_id,
                                                                    LEVEL,
                                                                    ms_grc_utilities.get_Start_Date_val(:1,ati_obj_id,:2,ms_grc_utilities.get_end_Date_val(:1,ati_obj_id,:3),frequency),
                                                                    ms_grc_utilities.get_end_Date_val(:1,ati_obj_id,:3),
                                                                    ati_obj_id
                                                                   )
                           ) my_count,frequency


                   FROM (SELECT b.object_id, a.object_name, assement_for,
                                ati_obj_id,frequency
                           FROM ms_grc_kpi_kri_defn_ati b,
                                ms_grc_kpi_kri_defn a
                          WHERE a.object_id = b.object_id AND a.object_id =:1)
             CONNECT BY LEVEL <=
                           ms_apps_utilities.get_parameter_value
                                            (1,
                                             'MS GRC KPI KRI Set No Of Dates',
                                             'No_Of_Dates'
                                            ))
 WHERE my_count IS NOT NULL order by upper(ati_name),date1
