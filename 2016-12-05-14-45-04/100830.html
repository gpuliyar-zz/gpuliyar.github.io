Name MS_GRC_Metric_Trend_Assessment_For (100830)
SQL 
select distinct b.ati_object_id,applies_to,
case when applies_to='1'
then 'Organization'||'-'||ms_grc_kpi_kri_utilities.get_kpi_kri_ati_name(applies_to,ati_object_id) 
when applies_to='2'
then 'Risk'||'-'||ms_grc_kpi_kri_utilities.get_kpi_kri_ati_name(applies_to,ati_object_id) 
when applies_to='3'
then 'Core Object'||'-'||ms_grc_kpi_kri_utilities.get_kpi_kri_ati_name(applies_to,ati_object_id) 
when applies_to='4'
then 'Control'||'-'||ms_grc_kpi_kri_utilities.get_kpi_kri_ati_name(applies_to,ati_object_id)
end applies_to
from ms_grc_kpi_kri_defn a,ms_grc_kpi_kri_data_en b
where a.object_id=b.object_id
AND (ms_grc_metrics_reports.get_view_access_org(:1,b.ati_object_id,b.applies_to,b.object_id))='VIEW ACCESS'
order by b.applies_to
