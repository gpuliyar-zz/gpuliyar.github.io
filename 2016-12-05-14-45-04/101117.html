SELECT B.ASSIGNMENT_ID                                                                                     AS ASSIGNMENT_ID,
  C.CATEGORY_NAME                                                                                          AS CATEGORY_ID,
  ms_grc_apps_get_edit_url(B.INFOLET_ID,B.PROCESS_INSTANCE_ID,B.USER_ID,'Y', B.DUE_DATE,B.ASSIGNMENT_TEXT) AS ASSIGNMENT_TEXT,
  D.METRIC_NAME                                                                                            AS METRIC_NAME,
  B.CREATION_DATE                                                                                          AS CREATION_DATE,
  B.DUE_DATE                                                                                               AS DUE_DATE
FROM SI_METRICS_T A
INNER JOIN SI_EVENT_ASSIGNMENTS B
ON A.METRIC_ID =B.INFOLET_ID
AND STATUS IN (4,5,8,1,9,10)
LEFT JOIN MS_GRC_APP_MYTASK_FORM_MAP D
ON A.FORM_NAME =D.METRIC_NAME
LEFT JOIN MS_GRC_APP_MYTASK_CTGRY C
ON C.CATEGORY_NAME=D.CATEGORY_NAME
INNER JOIN SI_USERS_T F
ON F.USER_NAME =:1
AND F.USER_ID  =B.USER_ID
LEFT JOIN SI_LOCALES G
ON F.LOCALE =G.LOCALE_STRING
LEFT JOIN MS_GRC_APP_MYTASK_MLS E
ON C.CATEGORY_NAME                                                               =E.CATEGORY_NAME
AND G.LOCALE_ID                                                                  =E.LOCAL_ID
WHERE ( INSTR(:2,E.CATEGORY_MLS_NAME,1,1)                                        >0
OR ( INSTR(:2,'Overdue',1,1)                                                     >0
AND  (TRUNC(DUE_DATE) - SYSDATE) <=-1
 and status!=8
)
OR ( INSTR(:2,'All',1,1)                                                         >0 )
OR ( INSTR(:2,'Others',1,1)                                                      >0
AND E.CATEGORY_MLS_NAME                                                         IS NULL)
OR ( INSTR(:2, 'New Tasks',1,1)                                                  >0
AND (TO_NUMBER(TO_CHAR(SYSDATE, 'J'))- TO_NUMBER(TO_CHAR(B.CREATION_DATE,'J'))) <=3
AND (TO_NUMBER(TO_CHAR(SYSDATE, 'J'))- TO_NUMBER(TO_CHAR(B.CREATION_DATE,'J'))) >=0) )
ORDER BY B.DUE_DATE DESC NULLS LAST
