MS_AUD_ADHOC_RISK_PUB_V 

                                                
  CREATE OR REPLACE FORCE VIEW "AUDDEVSP3"."MS_AUD_ADHOC_RISK_PUB_V" ("RISK_ID", "RISK_NAME") AS 
  SELECT DISTINCT TO_CHAR(risk_id) AS risk_id,
    TO_CHAR(risk_name)             AS risk_name
  FROM
    (SELECT (SUBSTR (aps_risks_ss,0,INSTR (aps_risks_ss, ';#~', 1, 1)-1))                                                                                  AS risk_id ,
      SUBSTR(aps_risks_ss,Regexp_Instr(aps_risks_ss,';#~',1,1)       +3,(Regexp_Instr(aps_risks_ss,';#~',1,2)-3) -( Regexp_Instr(aps_risks_ss,';#~',1,1))) AS risk_name ,
      SUBSTR(aps_risks_ss,Regexp_Instr(aps_risks_ss,';#~',1,2)       +3,(Regexp_Instr(aps_risks_ss,';#~',1,3)-3) -( Regexp_Instr(aps_risks_ss,';#~',1,2))) AS risk_type
    FROM
      (SELECT audit_id ,
        regexp_substr(aps_risks,'[^|]+',1,level) aps_risks_ss
      FROM
        (SELECT audit_id,
          REPLACE(aps_risks,';#,','|') aps_risks
        FROM ms_aud_audit
        WHERE NULLIF (LENGTH(aps_risks), 0) IS NOT NULL
        )
        CONNECT BY level   <= regexp_count(aps_risks,'[^|]+')
      AND prior audit_id    = audit_id
      AND prior sys_guid() IS NOT NULL
      )
    )
  WHERE TO_CHAR(risk_type) ='A'
