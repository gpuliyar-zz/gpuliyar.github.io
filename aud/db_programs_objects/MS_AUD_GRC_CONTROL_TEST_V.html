MS_AUD_GRC_CONTROL_TEST_V 

                                                
  CREATE OR REPLACE FORCE VIEW "AUDDEVSP3"."MS_AUD_GRC_CONTROL_TEST_V" ("TEST_ID", "TEST_NAME", "TEST_DESCRIPTION", "TEST_DESCRIPTION_HTM", "CONTROL_ID", "CONTROL_NAME", "VALID_FROM", "VALID_UNTIL") AS 
  SELECT qst.object_id AS test_id, qst.object_name AS test_name,
          qst.question AS test_description,
          qst.question_htm AS test_description_htm,
          con.object_id AS control_id, con.object_name AS control_name,
          qst.valid_from as valid_from, qst.valid_until as valid_until
     FROM ms_grc_orb_relations_pub_v orb,
          ms_grc_control_pub_v con,
          ms_grc_question_proc_pub_v qst
    WHERE UPPER (src_obj_type) = 'MS_GRC_CONTROL'
      AND UPPER (tgt_obj_type) = 'MS_GRC_QUESTION_PROCEDURE'
      AND orb.tgt_obj_id = qst.object_id
      AND orb.src_obj_id = con.object_id
