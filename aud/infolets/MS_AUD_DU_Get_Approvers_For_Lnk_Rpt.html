                                                        SELECT  USER_NAME, USER_FULL_NAME, ACTIVITY_NAME
  FROM (       
        SELECT  user_name AS USER_NAME,
                ms_apps_utilities.get_user_full_name (user_name)
                   AS USER_FULL_NAME,
                activity_name AS ACTIVITY_NAME
           from ms_aud_flow_down_org_v
          where  (:2  = 'O' and org_entity_id = :1)
        or (:2  = 'N')
        OR (:2  = 'R'
               AND user_name IN
                      (SELECT team_member from ms_aud_audit_tmm
                                  WHERE audit_id = :3
                       UNION
                       SELECT rsm_team_member as team_member 
                         from ms_aud_audit_tem 
                        WHERE audit_id = :3)))
        WHERE trim(ACTIVITY_NAME) = 'AUD - Approve Tasks'          
        UNION      
        SELECT user_name AS user_name,
               ms_apps_utilities.get_user_full_name (user_name) AS USER_FULL_NAME,
               activity_name AS activity_name
          FROM MS_AUD_FLOW_DOWN_ORG_V
         WHERE  :2  = 'R'
               AND (user_name IN (SELECT COLUMN_VALUE
                                   FROM TABLE (SELECT ms_apps_utilities.
                                                      parse_string (
                                                         (SELECT LEAD_AUDITOR
                                                                 || ','
                                                                 || AUDIT_MANAGER
                                                                    AS USER_NAME
                                                            FROM ms_aud_audit
                                                           WHERE audit_id =
                                                                    :3),
                                                         ',')
                                                 FROM DUAL))
                   OR user_name IN (SELECT audit_manager_mlov AS team_member
                                     FROM ms_aud_audit_amm
                                    WHERE audit_id = :3
                                   UNION
                                   SELECT lead_auditor_mlov AS team_member
                                     FROM ms_aud_audit_alm
                                    WHERE audit_id = :3))
        AND trim(ACTIVITY_NAME) = 'AUD - Approve Tasks'
