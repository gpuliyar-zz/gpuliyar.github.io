                restrict_access_to, dd_process_instance_id, dd_instance_id,
                owner_organizations, display_name, dd_object_type object_type,
                over_ride_action, expired_flag, edit_action_parameters,
                object_id
           FROM (SELECT object_id, object_name,
                        SUBSTR (object_name, 1, 30) display_name,
                        dd_object_type object_type, 'Y' AS over_ride_action,
                        ms_grc_utilities.include_expired
                                                 (valid_from,
                                                  valid_until
                                                 ) AS expired_flag,
                        valid_from, valid_until, restrict_access_to,
                        dd_object_type, dd_process_instance_id,
                           'proc='
                        || TO_CHAR (dd_process_instance_id)
                        || '|instid='
                        || TO_CHAR (dd_instance_id)
                        || '|edit_flag='
                        || NVL (dd_edit_flag, 'Y') AS edit_action_parameters,
                        dd_instance_id,
                        (SELECT ms_concat
                                   (d.owner_organizations
                                   )
                           FROM ms_grc_objectives_org d
                          WHERE a.object_id = d.object_id)
                                                       AS owner_organizations
                   FROM ms_grc_objectives a) t1
       ORDER BY object_name ASC
