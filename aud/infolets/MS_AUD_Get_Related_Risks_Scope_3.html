                                                        select   '<input type=checkbox id="' || risk_id || '" value="' || htf.escape_sc(risk_name) || '" onClick=checkboxChange("' || risk_id || '")>
 </input>' as "Check Box", 
          risk_name as "Risk Name", 
          risk_id AS "Risk ID"
from
( 
    SELECT distinct a.object_name AS risk_name, a.object_id AS risk_id
      FROM ms_grc_risk_pub_v a, ms_aud_grc_risk_process_v b
     WHERE a.object_id = b.tgt_obj_id
       and b.src_obj_id in (select column_value
                              from table (ms_apps_utilities.parse_string (:1, ',')))         
       and b.tgt_obj_id in (select related_risks from ms_aud_audit_ris where audit_id = :2
                                          and ms_aud_utilities.fn_get_plan_scope_status_id(:2) = '0'
                            union
                           select risk_id
                                from ms_aud_plan_scope_rsk 
                               where audit_id = :2
                               and ms_aud_utilities.fn_get_plan_scope_status_id(:2) = '1' and rs_scope = '1')
    union
       SELECT risk_name,risk_id from (
       SELECT
        (SUBSTR (COLUMN_VALUE,0,INSTR (COLUMN_VALUE, ';#~', 1, 1)-1)) AS risk_id,
       SUBSTR(COLUMN_VALUE,Regexp_Instr(COLUMN_VALUE,';#~',1,1) +3,(Regexp_Instr(COLUMN_VALUE,';#~',1,2)-3) -( Regexp_Instr(COLUMN_VALUE,';#~',1,1))) AS risk_name,
       SUBSTR(COLUMN_VALUE,Regexp_Instr(COLUMN_VALUE,';#~',1,2) +3,(Regexp_Instr(COLUMN_VALUE,';#~',1,3)-3) -( Regexp_Instr(COLUMN_VALUE,';#~',1,2))) as risk_type,
       SUBSTR(COLUMN_VALUE,Regexp_Instr(COLUMN_VALUE,';#~',1,3) +3,(Regexp_Instr(COLUMN_VALUE,';#~',1,4)-3) -( Regexp_Instr(COLUMN_VALUE,';#~',1,3))) as aue_id
       FROM TABLE
          (SELECT ms_aud_utilities.parse_string (aps_risks, ';#,')
          FROM ms_aud_audit
          WHERE audit_id = :2
          AND aps_risks IS NOT NULL
          ))
        WHERE aue_id IN
        (SELECT COLUMN_VALUE
        FROM TABLE (CAST (
          (SELECT ms_apps_utilities.parse_string (:1, ',') FROM DUAL
          ) AS ms_varchar2_array ) 
        ))
        AND risk_type = 'A' 
)
where (upper(risk_name) like nvl(upper('%'||:3||'%'),upper(risk_name)) 
      OR upper(risk_id) LIKE nvl(upper('%'||:3||'%'),upper(risk_id))
      )
order by risk_name
