MS_AUD_RPT_Get_Action_Owner Report pfadmin 
                                                        select 
            ms_aud_utilities.get_select_url ( user_full_name,user_name,'RADIO') AS select_user
            ,user_id
            ,user_name
            , user_full_name
       from ( SELECT distinct user_id
               ,user_name
              ,ms_apps_utilities.get_user_full_name (user_name) AS USER_FULL_NAME
            FROM MS_AUD_FLOW_DOWN_ORG_V
            where activity_name = 'ISM - Implement Action'
            group by user_id, user_name
            )user_list
WHERE  (upper(user_name) like nvl(upper('%'||:1||'%'),upper(user_name)) 
      or upper(user_full_name) like nvl(upper('%'||:1||'%'),upper(user_full_name)))
