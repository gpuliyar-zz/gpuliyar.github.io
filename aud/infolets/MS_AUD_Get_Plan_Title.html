MS_AUD_Get_Plan_Title

Description This infolet is attached to Plan Title filed of Manage Audit,Copy Audit form.This infolet fetches all the active plan titles in the system

                                                        SELECT 'AD-ID' as plan_id, 
       'Adhoc Audit' AS plan_title,
       'NA' as start_date,
       'NA' as end_date,
       'NA' as organization
  FROM DUAL
UNION ALL
SELECT plan_id, plan_title,
                 to_char(start_date,'dd/mm/yyyy') as start_date,
                 to_char(end_date,'dd/mm/yyyy') as end_date,
                 to_char(organization) as organization
  FROM (SELECT   internal_plan_id AS plan_id, 
                 plan_title, 
                 MAX (version_no),
                 ms_aud_utilities. fn_get_org_display_values (hdr_organization) as organization,
                 start_date,
                 end_date
  FROM ms_aud_audit_plan 
           WHERE status = '3'
        GROUP BY internal_plan_id, plan_title,start_date,end_date,hdr_organization)
