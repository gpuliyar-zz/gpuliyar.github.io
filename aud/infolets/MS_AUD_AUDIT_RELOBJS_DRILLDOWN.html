                                                        select dd_obj_type as DISPLAY_NAME,
:1 as OBJECT_ID,
OBJECT_TYPE
FROM (
select 'Organizations' as dd_obj_type,
'MS_AUD_AUDIT_ORGS_DD' AS OBJECT_TYPE
FROM dual 
WHERE EXISTS 
(SELECT org_entity_name
FROM si_org_entities_t a
 WHERE EXISTS (
WITH org_list AS
 (SELECT TRIM (SUBSTR (txt,
 INSTR (txt, ',', 1, LEVEL) + 1,
 INSTR (txt, ',', 1, LEVEL + 1)
 - INSTR (txt, ',', 1, LEVEL)
 - 1
)
) AS org_id
FROM (SELECT ',' || ms_aud_utilities.mlov_values(audit_id,'audit_id','manage_organizations','ms_aud_audit_org')|| ',' AS txt
FROM ms_aud_audit aud
 WHERE audit_id = :1)
CONNECT BY LEVEL <=
LENGTH (txt) - LENGTH (REPLACE (txt, ',', ''))
- 1)
SELECT DISTINCT org_id
 FROM org_list
WHERE org_id = A.org_entity_id))
UNION ALL
select 'Auditable Entities' as dd_obj_type,
'MS_AUD_AUDIT_AES_DD' as OBJECT_TYPE
FROM dual
WHERE EXISTS 
(SELECT object_name 
FROM ms_grc_core_object a
 WHERE EXISTS (
WITH ae_list AS
 (SELECT TRIM (SUBSTR (txt,
 INSTR (txt, ',', 1, LEVEL) + 1,
 INSTR (txt, ',', 1, LEVEL + 1)
 - INSTR (txt, ',', 1, LEVEL)
 - 1
)
) AS ae_id
FROM (SELECT ',' || ms_aud_utilities.mlov_values (audit_id,'audit_id','core_objects','ms_aud_audit_aue') || ',' AS txt
FROM ms_aud_audit aud
 WHERE audit_id = :1)
CONNECT BY LEVEL <=
LENGTH (txt) - LENGTH (REPLACE (txt, ',', ''))
- 1)
SELECT DISTINCT ae_id
 FROM ae_list
WHERE ae_id = a.object_id))
UNION ALL
select 'Risks' as dd_obj_type,
'MS_AUD_AUDIT_RISKS_DD' as OBJECT_TYPE
FROM dual
WHERE EXISTS
(SELECT object_name 
FROM ms_grc_risk a
 WHERE EXISTS (
WITH risk_list AS
 (SELECT TRIM (SUBSTR (txt,
 INSTR (txt, ',', 1, LEVEL) + 1,
 INSTR (txt, ',', 1, LEVEL + 1)
 - INSTR (txt, ',', 1, LEVEL)
 - 1
)
) AS risk_id
FROM (SELECT ',' || ms_aud_utilities.mlov_values(audit_id,'audit_id','related_risks','ms_aud_audit_ris') || ',' AS txt
FROM ms_aud_audit aud
 WHERE audit_id = :1)
CONNECT BY LEVEL <=
LENGTH (txt) - LENGTH (REPLACE (txt, ',', ''))
- 1)
SELECT DISTINCT risk_id
 FROM risk_list
WHERE risk_id = a.object_id))
UNION ALL SELECT 'Audit Plan/Scope' AS dd_obj_type,'MS_AUD_PLAN_SCOPE_DD' AS OBJECT_TYPE FROM dual WHERE EXISTS (SELECT 1 FROM ms_aud_plan_scope A WHERE A.AUDIT_ID = :1 AND A.FORM_TYPE='P')
UNION ALL
select 'Control Tests' as dd_obj_type,
'MS_AUD_AUDIT_TCS_DD' as OBJECT_TYPE
FROM dual 
WHERE EXISTS
(SELECT A.task_name
from
MS_AUD_AUDIT_TASK a
where 
task_type = '4'
AND a.audit_id =  :1)
UNION ALL
select 'Other Tasks' as dd_obj_type,
'MS_AUD_AUDIT_TASKS_DD' as OBJECT_TYPE
FROM dual
WHERE EXISTS
(SELECT A.task_name 
from MS_AUD_AUDIT_TASK a
WHERE 
A.task_type = '2'
AND A.audit_id =  :1)
UNION ALL
select 'Checklist Tasks' as dd_obj_type,
'MS_AUD_CHECKLIST_TASK_DD' as OBJECT_TYPE
FROM dual 
WHERE EXISTS
(SELECT A.task_name
from
MS_AUD_AUDIT_TASK a
WHERE 
task_type = '3'
AND A.audit_id =  :1)
UNION ALL
select 'Draft Issues' as dd_obj_type,
'MS_AUD_AUDIT_ISSUES_DD' as OBJECT_TYPE
FROM dual 
WHERE EXISTS
(SELECT A.fn_id
FROM
MS_AUD_DRAFT_ISSUES_fnd a
WHERE A.audit_id =  :1)
UNION ALL
select 'Reports' as dd_obj_type,
'MS_AUD_AUDIT_REPORTS_DD' as OBJECT_TYPE
FROM dual 
WHERE EXISTS
(SELECT substr(decode(report_type,1,a.ar_report,nvl(a.pfr_updated_final_report,a.pfr_final_report)),1,instr(decode(report_type,1,a.ar_report,nvl(a.pfr_updated_final_report,a.pfr_final_report)),'#')-1)
FROM
MS_AUD_AUDIT_REPORT a
WHERE 
A.audit_id =  :1)
UNION ALL
select 'Reported Issues' as dd_obj_type,
'MS_AUD_AUDIT_REP_ISSUES_DD' as OBJECT_TYPE
from dual 
WHERE 'Y'=ms_aud_utilities.get_int_parameter_value('Is_ISM_Integrated')
AND EXISTS 
(SELECT a.fn_finding_title
FROM ms_aud_audit_task_fnd A
     , MS_AUD_DRAFT_ISSUES_fnd b
     , MS_ISM_ISSUE d
 where a.task_id = b.TASK_ID
   AND b.audit_id = d.issue_source
   and b.ISSUE_STATUS = 28
   and a.fn_issue_resolution=1
   and a.FN_FINDING_TITLE = d.ISSUE_TITLE
   AND b.audit_id = :1
)
)
