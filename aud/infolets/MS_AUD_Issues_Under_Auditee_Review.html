MS_AUD_Rpt_Issues_Under_Auditee_Review Report SYSTEMI 
                                                        SELECT DISTINCT 
      CASE
      WHEN TRIM(MS_AUD_UTILITIES.FN_SEND_ISSUES_REV_EDIT (:1,A.PROCESS_INSTANCE_ID)) <> 'N'
        then '<a target="_self" '
                || 'href="javascript' || CHR (58)|| 'callreportdetails(''Pushinfolet?id='
                || ms_apps_utilities.get_infolet_id('MS_AUD_AUDIT_ISSUE_REV_TASK')
                || CHR (38)|| 'proc=' || a.process_instance_id || CHR (38)|| 'flag=1'|| CHR (38)
                || 'emd=4'
                || CHR (38)|| 'x_edit_mode=Yes'|| CHR (38)
                || 'audit_id='|| a.audit_id || CHR (38)|| 'wrapper=yes'')">'
                || B.AUDIT_TITLE|| '</a>'
      Else      
      '<a target="_self" '
                || 'href="javascript'|| CHR (58) || 'callreportdetails(''Pushinfolet?id='
                || ms_apps_utilities.get_infolet_id('MS_AUD_AUDIT_ISSUE_REV_TASK')
                || CHR (38)|| 'proc='|| a.process_instance_id
                || CHR (38)|| 'flag=1' || CHR (38)|| 'emd=2'
                || CHR (38) || 'x_edit_mode=No'|| CHR (38)
                || 'audit_id='|| a.audit_id|| CHR (38)
                || 'wrapper=no'')">'|| b.audit_title|| '</a>'
                 END AS audit_title,
         a.management_response_due_date AS Mgmt__response_due_date,
       ms_aud_utilities.fn_get_mlov_aud_man_disp
          (NVL (a.lead_auditor,
                (ms_aud_utilities.mlov_values (a.audit_id,
                                               'audit_id',
                                               'LEAD_AUDITOR_MLOV',
                                               'ms_aud_audit_alm'
                                              )))) AS lead_auditor,
       ms_aud_utilities.fn_get_mlov_aud_man_disp
          (NVL (a.audit_manager,
                (ms_aud_utilities.mlov_values (a.audit_id,
                                               'audit_id',
                                               'AUDIT_MANAGER_MLOV',
                                               'ms_aud_audit_amm'
                                              )))) AS audit_manager,
       (SELECT ms_aud_utilities.fn_get_mlov_aud_man_disp(issue_reviewer) FROM ms_aud_audit_issue_rev_task_v
 WHERE audit_id = A.AUDIT_ID AND process_instance_id = A.PROCESS_INSTANCE_ID
   AND instance_id = (SELECT MAX (instance_id) FROM ms_aud_audit_issue_rev_task_v
            WHERE audit_id = A.AUDIT_ID and process_instance_id = A.PROCESS_INSTANCE_ID
              AND instance_id <>(SELECT MAX (instance_id) FROM ms_aud_audit_issue_rev_task_v
                       WHERE audit_id =A.AUDIT_ID  AND process_instance_id = A.PROCESS_INSTANCE_ID))
   AND instance_rec_num = 1)AS auditee,ms_aud_utilities.fn_get_report_status(a.status) AS current_status
  FROM ms_aud_audit_issue_rev_task_v a,
                ms_aud_audit b
 WHERE instance_id = (SELECT MAX (instance_id)
                        FROM ms_aud_audit_issue_rev_task_v
                       WHERE audit_id = a.audit_id and process_instance_id=a.process_instance_id)
                       and A.AUDIT_ID=B.AUDIT_ID
   AND a.instance_rec_num = 1
   AND review_type = 2
   AND A.STATUS IN (26,29)
   and b.status not in (4)
   and (:1=issue_reviewer or :1=a.lead_auditor or :1=a.audit_manager or :1=delegatee)
