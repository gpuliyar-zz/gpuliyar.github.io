                                                        SELECT  
       a.audit_id as audit_id,
       a.fn_id as fn_id,
       a.fn_finding_title as Finding_Title,
       (select status_name from ms_aud_status where status_id = s.status_id) as Status,
       (select first_name || ' ' || last_name FROM si_users WHERE user_name = c.Task_Owner ) as Task_Owner,
       COALESCE(c.task_name,'Draft Issue') as Task_Name,
       null as ex_col_1,
       null as ex_col_2
FROM ms_aud_draft_issues_fnd b, ms_aud_audit_task_fnd a, ms_aud_audit_task c,ms_aud_trans_status s
 WHERE b.audit_id = a.audit_id
   AND b.task_id = a.task_id
   AND b.fn_id = a.fn_id
   and c.audit_id = a.audit_id
   and b.audit_id = c.audit_id
   and a.task_id= c.task_id
   AND a.audit_id = :1
   and a.audit_id = s.ref_pk_id
   and a.fn_id = s.ref_secondary_id
   and s.status_id IN (23, 25, 27, 28)  
   AND a.fn_issue_resolution IN (1,2)
 order by b.fnd_orderkey
