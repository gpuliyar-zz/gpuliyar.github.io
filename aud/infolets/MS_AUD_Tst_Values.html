                                                        select object_name as test_name, b.risk_ctrl_id as test_id
from ms_grc_question_proc a,
(select ms_aud_utilities.fn_get_test_display_values(COLUMN_VALUE) as test_id, column_value as risk_ctrl_id
                    from table (ms_apps_utilities.parse_string(:1,','))) b
where a. object_id = b.test_id
union
select a.ts_test_name as test_name, b.risk_ctrl_id as test_id
from ms_aud_plan_scope_tst a,
(select ms_aud_utilities.fn_get_test_display_values(COLUMN_VALUE) as test_id, column_value as risk_ctrl_id
                    from table (ms_apps_utilities.parse_string(:1,','))) b
where a.test_id = b.test_id
order by test_name
