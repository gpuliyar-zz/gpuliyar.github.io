                                                        Select Stored_Value,
       Displayed_Value,
       Lov_Name,
       Locale_Id,
       Enterprise_Id,
       Lov_Id,
       Lov_Value_Id,
       Start_Date,
       End_Date
From   Ms_Qs_Lov_Org_Values_v
where  (trim(LOV_Name) = 'MS AUD Task Type')
        And(trim(Stored_Value) IN ('2','3'))
        And :1 in (4,5)
UNION
Select Stored_Value,
       Displayed_Value,
       Lov_Name,
       Locale_Id,
       Enterprise_Id,
       Lov_Id,
       Lov_Value_Id,
       Start_Date,
       End_Date
From   Ms_Qs_Lov_Org_Values_v
where  (trim(LOV_Name) = 'MS AUD Task Type')
        And(trim(Stored_Value) IN ('2','3','4'))
        And :1 in (1,2,3)  
order by displayed_value
