MS_AUD_Prior_finding Report pfadmin 
                                                        select select_finding
     , audit_id
     , fn_id
     , fn_ref_id
     , finding_title
     , Detail
     , issue_owner_name
     , issue_owner
     , issue_rating
     , ex_col_1
     , ex_col_2
     , ex_col_3
     , ex_col_4
 from (select null as seq_no,
       ms_aud_utilities.get_select_url (fn_finding_title,drft.fn_id,'RADIO') AS select_finding,
       drft.audit_id as audit_id,
       drft.fn_id as fn_id,
       drft.fn_ref_id as fn_ref_id, 
       fn_finding_title as finding_title, 
       REGEXP_REPLACE(fn_finding_details, '<[^>]+>', NULL) as Detail,
       fn_issue_owner as issue_owner_name,
       MS_APPS_UTILITIES.get_user_full_name(fn_issue_owner) as issue_owner,
       ms_apps_utilities.get_display_value(MS_APPS_UTILITIES.GET_ENTERPRISE_ID(MS_APPS_UTILITIES.GET_INFOLET_ID('MS_AUD_AUDIT_TASK')),'MS AUD Audit Task Issue Rating',fn_issue_rating) as issue_rating, 
       null as ex_col_1,
       null as ex_col_2,
       null as ex_col_3,
       null as ex_col_4 
FROM ms_aud_draft_issues_fnd drft, ms_aud_audit_task_fnd task, ms_aud_audit aud
where drft.audit_id = task.audit_id
      and drft.task_id = task.task_id 
      and drft.fn_id = task.fn_id 
      and task.audit_id = aud.audit_id
      and aud.status = 8
      and AUD.CONDUCTED_BY_ORG = :1 and RESTRICT_ACCESS_TO!='R')
WHERE  (upper(finding_title) like nvl(upper('%'||:2||'%'),upper(finding_title)) 
      or upper(fn_id) like nvl(upper('%'||:2||'%'),upper(fn_id))
       or upper(Detail) like nvl(upper('%'||:2||'%'),upper(Detail)))
