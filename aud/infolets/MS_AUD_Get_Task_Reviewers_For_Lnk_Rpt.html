                                                        select  ms_aud_utilities.get_select_url(user_full_name,user_name,'RADIO') as "Check Box", 
          user_full_name as "User Full Name", 
          user_name as "User Name"
from (SELECT  USER_NAME, USER_FULL_NAME, ACTIVITY_NAME
  FROM (       -- Owner Organization
        SELECT  user_name AS USER_NAME,
                ms_apps_utilities.get_user_full_name (user_name)
                   AS USER_FULL_NAME,
                activity_name AS ACTIVITY_NAME
           FROM MS_AUD_FLOW_DOWN_ORG_V
          WHERE  :1  = 'O' AND org_entity_id = :3 
        UNION
       -- No restriction
        SELECT user_name AS user_name,
               ms_apps_utilities.get_user_full_name (user_name)
                  AS USER_FULL_NAME,
               activity_name AS activity_name
          FROM MS_AUD_FLOW_DOWN_ORG_V
         WHERE  :1  = 'N'
        UNION
        -- Team Mambers
        SELECT user_name AS user_name,
               ms_apps_utilities.get_user_full_name (user_name)
                  AS USER_FULL_NAME,
               activity_name AS activity_name
          FROM MS_AUD_FLOW_DOWN_ORG_V
         WHERE  :1  = 'R'
               AND user_name IN
                      (SELECT team_member from ms_aud_audit_tmm
                                  WHERE audit_id = :2
                       UNION
                       SELECT rsm_team_member as team_member 
                         FROM ms_aud_audit_tem 
                        WHERE audit_id = :2))
        WHERE trim(ACTIVITY_NAME) = 'AUD - Review Tasks'          
        UNION
        --Lead Auditor and Auditor Manager
        SELECT user_name AS user_name,
               ms_apps_utilities.get_user_full_name (user_name)
                  AS USER_FULL_NAME,
               activity_name AS activity_name
          FROM MS_AUD_FLOW_DOWN_ORG_V
         WHERE  :1  = 'R'
               AND (user_name IN (SELECT COLUMN_VALUE
                                   FROM TABLE (SELECT ms_apps_utilities.
                                                      parse_string (
                                                         (SELECT LEAD_AUDITOR
                                                                 || ','
                                                                 || AUDIT_MANAGER
                                                                    AS USER_NAME
                                                            FROM ms_aud_audit
                                                           WHERE audit_id =
                                                                    :2),
                                                         ',')
                                                 FROM DUAL))
                   OR user_name IN (SELECT audit_manager_mlov AS team_member
                                     FROM ms_aud_audit_amm
                                    WHERE audit_id = :2
                                   UNION
                                   SELECT lead_auditor_mlov AS team_member
                                     FROM ms_aud_audit_alm
                                    WHERE audit_id = :2))
        AND trim(ACTIVITY_NAME) = 'AUD - Review Tasks')
where (upper(user_full_name) like nvl(upper('%'||:4||'%'),upper(user_full_name)) 
      or upper(user_name) like nvl(upper('%'||:4||'%'),upper(user_name))
      )
order by user_full_name
