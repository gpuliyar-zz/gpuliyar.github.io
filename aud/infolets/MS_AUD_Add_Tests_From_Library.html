                                                        SELECT '<input type=checkbox id="' || test_id || '" value="' || htf.escape_sc(test_name) || '" onClick=checkboxChange("' || test_id || '")>
 </input>' AS select_test, 
 control_id as ctrl_id, control_name as ctrl_name, test_id as tst_id, test_name as tst_name, View_Form as view_tst_form, attachments as attached_files
  from (select distinct a.control_id, a.control_name, a.test_id, a.test_name, 
        ms_aud_utilities_2.fn_get_control_tst_url_scope(a.test_id,'TST') as View_Form, b.attachments
        FROM ms_aud_risk_control_test_v a, ms_Grc_question_proc b
        where a.test_id = b.object_id
		and a.control_id = :1 and nvl(trunc(a.valid_until),trunc(sysdate)) >= trunc(sysdate) 
		and (:2 is null or (:2 is not null and a.test_id not in (select column_value
                                FROM TABLE(ms_apps_utilities.parse_string(:2, ',')))))) 
where (upper(test_name) like nvl(upper('%'||:3||'%'),upper(test_name)) 
        or upper(test_id) like nvl(upper('%'||:3||'%'),upper(test_id))
        )
