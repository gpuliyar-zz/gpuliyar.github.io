                                                        SELECT  null AS VIEW_ACCESS
      , ms_aud_utilities.FN_GET_CURRENT_AUDIT_URLS('MS_AUD_MANAGE_AUDIT',:1,a.audit_id,a.audit_title,a.dd_process_instance_id) AS audit_title
      , a.audit_id AS audit_id
      , ms_aud_utilities.FN_GET_STATUS_DISPLAY_NAME(a.audit_status) AS audit_status
      , NULL AS audit_result
      , NVL(ms_apps_utilities.get_user_full_name (a.lead_auditor),ms_aud_utilities.fn_get_mlov_lead_aud_disp(ms_aud_utilities.fn_get_mlov_values(audit_id,'lead_auditor_mlov','ms_aud_audit_alm'))) AS lead_auditor
      , NVL(ms_apps_utilities.get_user_full_name (a.audit_manager),ms_aud_utilities.fn_get_mlov_aud_man_disp(ms_aud_utilities.fn_get_mlov_values(audit_id,'audit_manager_mlov','ms_aud_audit_amm'))) AS audit_manager
      , audit_start_date AS audit_start_date
      , ms_aud_utilities.FN_GET_CURRENT_AUDIT_URLS('MS_AUD_PLAN_SCOPE',:1,a.audit_id,a.audit_title,a.dd_process_instance_id) AS audit_plan_scope
      , ms_aud_utilities.FN_GET_SETUP_AUDITOR_CRT_TASKS(a.setup_auditors_create_tasks,a.audit_id,:1) AS create_tasks
      , ms_aud_utilities.fn_get_current_audit_urls(null,null,a.audit_id,null,null) AS view_tasks
      , ms_aud_utilities.FN_GET_CURRENT_AUDIT_URLS('MS_AUD_DRAFT_ISSUES',:1,a.audit_id,a.audit_title,a.dd_process_instance_id) AS draft_issue
      , ms_aud_utilities.FN_GET_CURRENT_AUDIT_URLS('MS_AUD_AUDIT_ISSUE_REV_TASK',:1,a.audit_id,a.audit_title,a.dd_process_instance_id) AS send_issues_for_review
      , ms_aud_utilities.FN_GET_CURRENT_AUDIT_URLS('MS_AUD_AUDIT_REPORT',:1,a.audit_id,a.audit_title,a.dd_process_instance_id) AS audit_report     ,ms_aud_utilities.get_code_description('MS AUD Audit Type',audit_type) as audit_type , 
ms_aud_utilities.get_code_description('MS AUD Classification',audit_classification) as audit_classification,
ms_aud_utilities.FN_GET_CURRENT_AUD_TEAM_LINK(a.audit_id,:1) AS manage_team
       FROM ms_aud_audit a
      WHERE (TRUNC (a.audit_start_date) <=
                TRUNC ((  SYSDATE
                        + (SELECT ms_apps_utilities.get_parameter_value (ms_apps_utilities.get_enterprise_id(ms_apps_utilities.GET_INFOLET_ID('MS_AUD_MANAGE_AUDIT')) ,
                                                                         'MS_AUD_SYSTEM_PARAMETERS',
                                                                         'MS_OnGoingAudits_window'
                                                                        )
                             FROM DUAL)
                       ))
            )
        --AND (a.status is null or a.status not in ('4','8','44'))
        AND a.status not in ('4','8','44','2','31')
        AND MS_AUD_AUDIT_SECURITY.CHECK_TEAM_MEMBER(a.audit_id,:1) > 0
   ORDER BY a.audit_id DESC
