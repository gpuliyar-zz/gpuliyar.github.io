                                                        SELECT     
         process_name, process_id, core_type,
         risk_name AS "Risk Name", risk_id AS "Risk ID",
         org_entity_name AS "Organization Name", org_entity_id AS "Organization ID",process_level
    FROM (SELECT object_name AS process_name, object_id AS process_id,
                 ms_aud_utilities.fn_get_ae_obj_name(DD_OBJECT_TYPE) AS core_type,
                 NULL AS risk_name, NULL AS risk_id,
                 NULL AS org_entity_name, NULL AS org_entity_id,nvl(OBJECT_LEVEL,1) as process_level
            FROM ms_aud_grc_core_object_pub_v
           WHERE (:2 = 5 OR :2 = 1)
             AND  trunc(nvl(valid_from,sysdate))<= trunc(sysdate) AND NVL(TRUNC(valid_until),TRUNC(SYSDATE)) >= TRUNC(SYSDATE)
             AND ms_apps_security_pkg.check_object_access
                                                         (dd_object_type,
                                                          restrict_access_to,
                                                          owner_organizations,
                                                          :1,
                                                          'VIEW'
                                                         ) = 'ACCESS')
ORDER BY process_name
