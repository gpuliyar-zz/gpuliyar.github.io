                                                        SELECT a.audit_title, 
       a.audit_manager,
       (select user_full_name from ms_aud_users u where u.user_name = a.audit_manager) audit_manager_disp,
       a.lead_auditor,
       (select user_full_name from ms_aud_users u where u.user_name = a.lead_auditor)  lead_auditor_disp,
       a.conducted_by_org,
       (select distinct org_entity_name from si_org_entities_t where org_entity_id=a.conducted_by_org) conducted_by_org_disp,
       TRUNC(a.audit_start_date) AS audit_date,
       (select ms_concat(audit_manager_mlov) from ms_aud_audit_amm amm where amm.audit_id=a.audit_id) audit_manager_mlov_sto,
       (select ms_concat(u.first_name ||' '|| u.last_name)
            from si_users_t u
                ,ms_aud_audit_amm amm 
            where amm.audit_id = a.audit_id
               and amm.audit_manager_mlov = u.user_name
          ) audit_manager_mlov_disp,
       (select ms_concat(lead_auditor_mlov) from ms_aud_audit_alm alm where alm.audit_id=a.audit_id) lead_auditor_mlov_sto,
       (select ms_concat(u.first_name ||' '|| u.last_name)
            from si_users_t u
                ,ms_aud_audit_alm alm 
            where alm.audit_id = a.audit_id
               and alm.lead_auditor_mlov = u.user_name
       ) lead_auditor_mlov_disp,
       COALESCE(a.audit_classification,'01') AS audit_classification
  FROM ms_aud_audit a
 WHERE a.audit_id =:1
