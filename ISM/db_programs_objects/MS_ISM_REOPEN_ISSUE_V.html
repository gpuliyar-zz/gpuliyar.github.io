MS_ISM_REOPEN_ISSUE_V 

                                                
  CREATE OR REPLACE FORCE VIEW "NOVT_DEV"."MS_ISM_REOPEN_ISSUE_V" ("ISSUE_LINK", "ISSUE_TITLE", "ISSUE_OWNER", "ISSUE_APPROVER", "SOURCE_TYPE", "ISSUE_RATING", "ISSUE_DUE_DATE", "ISSUE_CLOSED_ON", "ISSUE_ID", "PID", "INSTANCEID", "ACTIONS", "REOPEN_COUNT", "NO_OF_ACTIONS", "ISSUE_OWNERORG_DISP", "ISSUE_OWNERORG", "ISSUE_APPRORG_DISP", "ISSUE_APPRORG") AS 
  SELECT
    (SELECT ''
      || a.issue_title
      || ''
    FROM DUAL
    )                                                                                                      AS issue_link,
    a.issue_title                                                                                          AS issue_title,
    ms_apps_utilities.get_user_full_name (a.issue_owner)                                                   AS issue_owner,
    ms_apps_utilities.get_user_full_name (a.issue_approver)                                                AS issue_approver,
    ms_apps_utilities.get_display_value (100000, 'MS ISM Source Type', a.issue_source_type, 1, NULL, NULL) AS source_type,
    ms_apps_utilities.get_display_value (100000, 'MS ISM Risk Rating', a.issue_risk_rating, 1, NULL, NULL) AS issue_rating,
    a.issue_due_date                                                                                       AS issue_due_date,
    a.issue_closed_on                                                                                      AS issue_closed_on,
    a.issue_id                                                                                             AS issue_id,
    a.dd_process_instance_id                                                                               AS pid,
    a.dd_instance_id                                                                                       AS instanceid,
    NULL                                                                                                   AS actions,
    NVL(a.reopen_issue,0)                                                                                  AS reopen_count,
    (SELECT COUNT (1) FROM ms_ism_action WHERE issue_id = a.issue_id
    ) AS no_of_actions,
    (SELECT org_entity_name
    FROM si_org_entities_t
    WHERE org_entity_id = a.issue_ownerorg
    )                AS issue_ownerorg_disp,
    a.issue_ownerorg AS issue_ownerorg,
    (SELECT org_entity_name
    FROM si_org_entities_t
    WHERE org_entity_id = a.issue_approverorg
    )                   AS issue_approrg_disp,
    a.issue_approverorg AS issue_approrg
  FROM ms_ism_issue a
  WHERE a.issue_status = '10'
