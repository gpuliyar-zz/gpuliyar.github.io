MS_ISM_DETAILED_ISSUE_LIST_V 

                                                
  CREATE OR REPLACE FORCE VIEW "NOVT_DEV"."MS_ISM_DETAILED_ISSUE_LIST_V" ("ISSUE_TITLE", "ISSUE_STATUS", "ISSUE_STATUS_STORED", "ISSUE_OWNER", "ISSUE_RATING", "PRIORITY", "SOURCE_TYPE", "ISSUE_DUE_DATE", "ISSUE_INITIATOR", "FIRST_IDENTIFIED_ON", "ISSUE_ID", "ISSUE_DETAILS", "NO_OF_ACTIONS", "ISSUE_OWNER_ORG", "ISSUE_APPROVERORG", "OWNERORG", "APPROVERORG", "PROGRESS_STATUS", "PID", "INSTANCEID", "CORE_OBJECTS", "AREA_OF_COMPLIANCE", "CONTROLS", "RISKS", "ORG", "STANDARD", "ASSET", "ASSET_CLASS", "FINANCIAL_ACCOUNT", "FUNCTIONOBJ", "FRAMEWORK_REFERENCE", "OBJECTIVE", "PROCESS", "REFERENCE", "REGULATORY_BOD", "REQUIREMENT", "QUESTION_PROCEDURE", "EXCEPTIONOBJ", "SUPPLIER", "AUDITABLE_ENTITY", "ISSUE_APPROVER", "ISSUE_RISK_IMPACT", "ISSUE_RISK_LIKLIHOOD", "ISSUE_SOURCE", "ISSUE_RISK_LOSS_GAIN", "ISSUE_RISK_LOSS_GAIN_AMT", "EXCEPTION_TYPE", "REACHED_MANAGE", "ADHOC_SRC_DETAILS", "NO_OF_EXTENSIONS", "REPEAT_ISSUE", "REOPEN_ISSUE", "ISSUE_TYPE", "RELATED_ISSUES", "DD_CREATED_ON", "ISSUE_SOURCE_STORED") AS 
  SELECT
    (SELECT ''
      || REPLACE(REPLACE(a.issue_title,'<', '&'||'lt;'), '>', '&'||'gt;') 
      || ''
    FROM DUAL
    )                                                                                                      AS issue_title,
    ms_ism_helper.get_status_display_name(a.issue_status)                                                  AS issue_status,
    a.issue_status                                                                                         AS issue_status_stored,
    ms_apps_utilities.get_user_full_name (a.issue_owner)                                                   AS issue_owner,
    ms_apps_utilities.get_display_value (100000, 'MS ISM Risk Rating', a.issue_risk_rating, 1, NULL, NULL) AS issue_rating,
    ms_apps_utilities.get_display_value (100000, 'MS ISM Priority', a.issue_risk_priority, 1, NULL, NULL)  AS priority,
    ms_apps_utilities.get_display_value (100000, 'MS ISM Source Type', a.issue_source_type, 1, NULL, NULL) AS source_type,
    a.issue_due_date                                                                                       AS issue_due_date,
    ms_apps_utilities.get_user_full_name (a.dd_created_by)                                                 AS issue_initiator,
    a.issue_identified_on                                                                                  AS first_identified_on,
    a.issue_id                                                                                             AS issue_id,
    REGEXP_REPLACE (a.issue_details, '<[^>]+>', NULL)                                                      AS issue_details,
    (SELECT COUNT (1) FROM ms_ism_action WHERE issue_id = a.issue_id
    ) AS no_of_actions,
    (SELECT org_entity_name
    FROM si_org_entities_t
    WHERE org_entity_id = a.issue_ownerorg
    ) AS issue_owner_org,
    (SELECT org_entity_name
    FROM si_org_entities_t
    WHERE org_entity_id = a.issue_approverorg
    ) AS issue_approverorg,
    a.issue_ownerorg,
    a.issue_approverorg,
    ms_apps_utilities.get_display_value (100000, 'MS ISM Update Status', a.progress_status, 1, NULL, NULL)                        AS progress_status,
    a.dd_process_instance_id                                                                                                      AS pid,
    a.dd_instance_id                                                                                                              AS instanceid,
    ms_ism_helper.get_coma_seprtd_orb_value ('Core Object(s)',a.issue_id)                                                         AS Core_Objects,
    ms_ism_helper.get_coma_seprtd_orb_value ('Area(s) of Compliance',a.issue_id)                                                  AS Area_of_Compliance,
    ms_ism_helper.get_coma_seprtd_orb_value ('Control(s)',a.issue_id)                                                             AS Controls,
    ms_ism_helper.get_coma_seprtd_orb_value ('Risk(s)',a.issue_id)                                                                AS Risks,
    ms_ism_helper.get_coma_seprtd_orb_value ('Organization(s)',a.issue_id)                                                        AS Org,
    ms_ism_helper.get_coma_seprtd_orb_value ('Standard(s)',a.issue_id)                                                            AS Standard,
    ms_ism_helper.get_coma_seprtd_orb_value ('Asset(s)',a.issue_id)                                                               AS Asset,
    ms_ism_helper.get_coma_seprtd_orb_value ('Asset Class(es)',a.issue_id)                                                        AS Asset_Class,
    ms_ism_helper.get_coma_seprtd_orb_value ('Financial Account(s)',a.issue_id)                                                   AS Financial_Account,
    ms_ism_helper.get_coma_seprtd_orb_value ('Function(s)',a.issue_id)                                                            AS FunctionObj,
    ms_ism_helper.get_coma_seprtd_orb_value ('Framework Reference(s)',a.issue_id)                                                 AS Framework_Reference,
    ms_ism_helper.get_coma_seprtd_orb_value ('Objective(s)',a.issue_id)                                                           AS Objective,
    ms_ism_helper.get_coma_seprtd_orb_value ('Process(s)',a.issue_id)                                                             AS Process,
    ms_ism_helper.get_coma_seprtd_orb_value ('Reference(s)',a.issue_id)                                                           AS REFERENCE,
    ms_ism_helper.get_coma_seprtd_orb_value ('Regulatory Bod(ies)',a.issue_id)                                                    AS Regulatory_Bod,
    ms_ism_helper.get_coma_seprtd_orb_value ('Requirement(s)',a.issue_id)                                                         AS Requirement,
    ms_ism_helper.get_coma_seprtd_orb_value ('Question(s)/Procedure(s)',a.issue_id)                                               AS Question_Procedure,
    ms_ism_helper.get_coma_seprtd_orb_value ('Exception(s)',a.issue_id)                                                           AS ExceptionObj,
    ms_ism_helper.get_coma_seprtd_orb_value ('Supplier(s)',a.issue_id)                                                            AS Supplier,
    ms_ism_helper.get_coma_seprtd_orb_value ('Auditable Entity(s)',a.issue_id)                                                    AS Auditable_Entity,
    ms_apps_utilities.get_user_full_name (a.ISSUE_APPROVER)                                                                       AS ISSUE_APPROVER,
    ms_apps_utilities.get_display_value (100000, 'MS ISM Impact', a.issue_risk_impact, 1, NULL, NULL)                             AS issue_risk_impact,
    ms_apps_utilities.get_display_value (100000, 'MS ISM Likelihood', a.issue_risk_liklihood, 1, NULL, NULL)                      AS issue_risk_liklihood,
    DECODE (a.issue_source_type, '0', a.adhoc_source, MS_ISM_HELPER.get_Issue_Source_Title (a.issue_source, a.issue_source_type)) AS ISSUE_SOURCE,
    ms_apps_utilities.get_display_value (100000, 'MS ISM Loss Gain', a.issue_risk_loss_gain, 1, NULL, NULL)                       AS issue_risk_loss_gain,
    issue_risk_loss_gain_amt,
    ms_apps_utilities.get_display_value (100000, 'MS ISM Classification Type', a.TYPE_OF_EXECUTION, 1, NULL, NULL) AS EXCEPTION_TYPE,
    a.reached_manage                                                                                               AS reached_manage,
    REGEXP_REPLACE (a.ADHOC_SRC_DETAILS, '<[^>]+>', NULL)                                                          AS ADHOC_SRC_DETAILS,
    a.NO_OF_EXTENSIONS                                                                                             AS NO_OF_EXTENSIONS,
    ms_apps_utilities.get_display_value (100000, 'MS_ISM_YES_NO', a.REPEAT_ISSUE, 1, NULL, NULL)                   AS REPEAT_ISSUE,
    a.REOPEN_ISSUE                                                                                                 AS REOPEN_ISSUE,
    ms_ism_helper.get_comma_sep_value_for_mlov ('ISSUE_TYPE','MS_ISM_ISSUE_IST',a.issue_id,'MS ISM Issue Type')    AS ISSUE_TYPE,
    ms_ism_helper.get_comma_sep_related_issues(a.issue_id)                                                      AS RELATED_ISSUES,
    dd_created_on                                                                                               As DD_CREATED_ON,
    ISSUE_SOURCE                                                                                                As ISSUE_SOURCE_STORED 
  FROM ms_ism_issue a
