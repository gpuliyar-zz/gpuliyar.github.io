MS_ISM_DETAILED_ACTION_LIST_V 

                                                
  CREATE OR REPLACE FORCE VIEW "ISM_61_SP2_DEV"."MS_ISM_DETAILED_ACTION_LIST_V" ("ACTION_TITLE", "ACTION_STATUS", "ACTION_OWNER", "PRIORITY", "ACTION_DUE_DATE", "ACTION_COMPLETED_ON", "ISSUE_TITLE", "ISSUE_OWNER", "ACTION_TYPE", "PERCENTAGE_COMPLETED", "PID", "INSTANCEID", "OWNER_ORGANIZATION", "OWNER_ORG_ID", "ISSUE_ID", "APPROVER_ORG", "APPROVER_ORG_ID", "ACTION_APPROVER", "FINAL_APPROVER", "ACTION_START_DATE", "ACTION_DETAILS", "ACTION_CLOSED_ON", "ACTION_WORK_DONE", "ACTION_RESULTS", "NO_OF_EXTENSIONS", "REOPEN_ACTION") AS 
  SELECT
    (SELECT ''
      || REPLACE(REPLACE(a.action_title,'<', '&'||'lt;'), '>', '&'||'gt;')
      || ''
    FROM DUAL
    )                                                                                                    AS action_title,
    ms_ism_helper.get_status_display_name(a.action_status)                                               AS action_status,
    ms_apps_utilities.get_user_full_name (a.action_owner)                                                AS action_owner,
    ms_apps_utilities.get_display_value (100000, 'MS ISM Risk Rating', a.action_priority, 1, NULL, NULL) AS priority,
    a.action_due_date                                                                                    AS action_due_date,
    NULL                                                                                                 AS action_completed_on,
    a.issue_title                                                                                        AS issue_title,
    ms_apps_utilities.get_user_full_name (a.issue_owner)                                                 AS issue_owner,
    ms_apps_utilities.get_display_value (100000, 'MS ISM Action Item Type',
    (SELECT action_type FROM ms_ism_issue_iap WHERE issue_action_id=a.action_id
    ), 1, NULL, NULL) AS Action_Type,
    --a.action_per_complte AS percentage_completed,
    a.remove_delegation      AS percentage_completed, --remove delegation has been used to capture percentage complete value as it will get nullified on chnage of actions
    a.dd_process_instance_id AS pid,
    a.dd_instance_id         AS instanceid,
    (SELECT org_entity_name
    FROM si_org_entities_t
    WHERE org_entity_id = b.ISSUE_OWNERORG
    )                AS owner_organization,
    b.ISSUE_OWNERORG AS owner_organization_id,
    a.issue_id       AS issue_id,
    (SELECT org_entity_name
    FROM si_org_entities_t
    WHERE org_entity_id = b.issue_approverorg
    )                                                         AS issue_approverorg,
    b.issue_approverorg                                       AS issue_approverorg_id,
    ms_apps_utilities.get_user_full_name (a.ACTION_APPROVER1) AS Action_Approver,
    ms_apps_utilities.get_user_full_name (a.ACTION_APPROVER2) AS Final_Approver,
    a.ACTION_START_DATE                                       AS Action_start_date,
    REGEXP_REPLACE (a.ACTION_DETAILS, '<[^>]+>', NULL)        AS Action_Details,
    a.ACTION_CLOSED_ON                                        AS Action_Closed_On,
    REGEXP_REPLACE (a.ACTION_WORK_DONE, '<[^>]+>', NULL)      AS Work_Done,
    REGEXP_REPLACE (a.ACTION_RESULTS, '<[^>]+>', NULL)        AS Results,
    a.NO_OF_EXTENSIONS                                        AS No_Of_Extensions,
    a.REOPEN_ACTION                                           AS Reopen_Action
  FROM ms_ism_action a,
    ms_ism_issue b
  WHERE a.issue_id = b.issue_id
