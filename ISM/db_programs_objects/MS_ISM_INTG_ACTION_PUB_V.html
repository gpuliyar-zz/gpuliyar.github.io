MS_ISM_INTG_ACTION_PUB_V 

                                                
  CREATE OR REPLACE FORCE VIEW "NOVT_DEV"."MS_ISM_INTG_ACTION_PUB_V" ("ACTION_ID", "ISSUE_ID", "ACTION_TITLE", "ACTION_STATUS", "ACTION_DUE_DATE", "CREATED_BY", "CREATED_ON", "LAST_MODIFIED_BY", "LAST_MODIFIED_ON", "ACTION_START_DATE", "ACTION_CLOSED_ON", "OBJECT_TYPE", "ACTION_OWNER", "ACTION_DELEGATEE", "ACTION_APPROVER1", "ACTION_APPROVER2", "NO_OF_REOPEN", "PRIORITY", "ACTION_TYPE", "PERCENTAGE_COMPLETED", "ACTION_DETAILS", "WORK_DONE", "RESULTS") AS 
  Select A.Action_Id  As Action_Id,
 A.Issue_Id As Issue_Id,
    a.action_title  AS action_title,
    Ms_Ism_Helper.Get_Status_Display_Name(A.Action_Status) As Action_Status,
    A.action_Due_Date As action_Due_Date,
    A.Dd_Created_By As Created_By,
    A.Dd_Created_On As Created_On,
    A.Dd_Modified_By as Last_Modified_By,
    A.Dd_Modified_On As Last_Modified_On,
    A.Action_Start_Date As Action_Start_Date,
    A.action_Closed_On As action_Closed_On,
    A.Dd_Object_Type As Object_Type,
    A.Action_Owner As Action_Owner,
    A.Action_Delegatee As Action_Delegatee,
    (Select Decode(Final_Action_Approval, '4',A.Issue_Owner, A.Action_Approver1)
    From Ms_Ism_Issue Where Issue_Id = A.Issue_Id)As Action_Approver1,
    A.Action_Approver2 As Action_Approver2,
    A.Reopen_Action As Reopen_Count,
    ms_apps_utilities.get_display_value (100000, 'MS ISM Priority', a.action_priority, 1, NULL, NULL)  AS priority,
    Ms_Apps_Utilities.Get_Display_Value (100000, 'MS ISM Action Item Type',
      (Select Action_Type From Ms_Ism_Issue_Iap Where Issue_Action_Id=A.Action_Id), 1, Null, Null) As Action_Type,
    A.Remove_Delegation      As Percentage_Completed,
    Regexp_Replace (A.action_details, '<[^>]+>', Null)      As Action_Details,
    Regexp_Replace (A.Action_Work_Done, '<[^>]+>', Null)      As Work_Done,
    REGEXP_REPLACE (a.ACTION_RESULTS, '<[^>]+>', NULL)        AS Results
  From Ms_Ism_action A
