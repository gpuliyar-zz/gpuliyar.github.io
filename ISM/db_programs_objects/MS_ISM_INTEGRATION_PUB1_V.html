MS_ISM_INTEGRATION_PUB1_V 

                                                
  CREATE OR REPLACE FORCE VIEW "ISM_61_SP2_DEV"."MS_ISM_INTEGRATION_PUB1_V" ("ISSUE_ID", "ISSUE_TITLE", "ISSUE_STATUS", "ISSUE_DUE_DATE", "ISSUE_SOURCE_TYPE", "ISSUE_SOURCE_REF", "CREATED_ON", "ISSUE_CLOSED_ON", "OBJECT_TYPE", "RELATED_RISKS", "RELATED_AREA_OF_COMP", "RELATED_CONTROLS", "RELATED_CORE_PROCESSES", "RELATED_ORGS") AS 
  SELECT a.issue_id                                                                                                                                    AS issue_id,
    a.issue_title                                                                                                                                      AS issue_title,
    a.issue_status                                                                                                                                     AS issue_status,
    a.issue_due_date                                                                                                                                   AS issue_due_date,
    DECODE (a.issue_source_type, 1, 'Audit', 2, 'Test Execution', 3, 'Risk Assessment', 4, 'Surveys', 5, 'Internal Loss Event', 6, 'GRC Intelligence') AS issue_source_type,
    a.issue_source_ref                                                                                                                                 AS issue_source_ref,
    a.dd_created_on                                                                                                                                    AS created_on,
    a.issue_closed_on                                                                                                                                  AS issue_closed_on,
    a.dd_object_type                                                                                                                                   AS object_type,
    (SELECT XMLAGG (XMLELEMENT ("elem", r.tgt_obj_name
      || ',')). EXTRACT ('/elem/text()').getclobval ()
    FROM ms_ism_orb_relation_orb r
    WHERE r.rel_source_id = a.issue_id
    AND r.tgt_obj_type = 'Risk(s)'
    ) AS related_risks,
    (SELECT XMLAGG (XMLELEMENT ("elem", r.tgt_obj_name
      || ',')). EXTRACT ('/elem/text()').getclobval ()
    FROM ms_ism_orb_relation_orb r
    WHERE r.rel_source_id = a.issue_id
    AND r.tgt_obj_type = 'Area(s) of Compliance'
    ) AS related_area_of_comp,
    (SELECT XMLAGG (XMLELEMENT ("elem", r.tgt_obj_name
      || ',')). EXTRACT ('/elem/text()').getclobval ()
    FROM ms_ism_orb_relation_orb r
    WHERE r.rel_source_id = a.issue_id
    AND r.tgt_obj_type = 'Control(s)'
    ) AS related_controls,
    (SELECT XMLAGG (XMLELEMENT ("elem", r.tgt_obj_name
      || ',')). EXTRACT ('/elem/text()').getclobval ()
    FROM ms_ism_orb_relation_orb r
    WHERE r.rel_source_id = a.issue_id
    AND r.tgt_obj_type = 'Core Object(s)'
    ) AS related_core_processes,
    (SELECT XMLAGG (XMLELEMENT ("elem", r.tgt_obj_name
      || ',')). EXTRACT ('/elem/text()').getclobval ()
    FROM ms_ism_orb_relation_orb r
    WHERE r.rel_source_id = a.issue_id
    AND r.tgt_obj_type = 'Organization(s)'
    ) AS related_orgs
  FROM ms_ism_issue a
  WHERE NVL (issue_source_type, '0') <> 0
