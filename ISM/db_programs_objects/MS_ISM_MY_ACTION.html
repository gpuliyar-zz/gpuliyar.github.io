MS_ISM_MY_ACTION 

                                                
  CREATE OR REPLACE FORCE VIEW "ISM_61_SP2_DEV"."MS_ISM_MY_ACTION" ("ACTION_TITLE", "ACTION_STATUS", "ACTION_OWNER", "ISSUE_TITLE", "ISSUE_SOURCE", "ACTION_DUE_DATE", "ISSUE_DUE_DATE", "DAYS_PAST_DUE", "NO_OF_EXTENSION", "USER_NAME", "ACTION_DELEGATEE", "ISSUE_ID", "DD_CREATED_ON") AS 
  SELECT (SELECT ms_ism_helper.
               get_link_url_for_my_actions ('ACTION',
                                            A.DD_PROCESS_INSTANCE_ID,
                                            A.ACTION_STATUS,
                                            A.ACTION_TITLE)
          FROM DUAL)
          AS ACTION_TITLE,
       ms_ism_helper.get_status_display_name(A.ACTION_STATUS) AS ACTION_STATUS,
       MS_APPS_UTILITIES.GET_USER_FULL_NAME (A.ACTION_OWNER) AS ACTION_OWNER,
       (SELECT ms_ism_helper.
               get_link_url_for_my_actions ('ISS',
                                            B.DD_PROCESS_INSTANCE_ID,
                                            B.ISSUE_STATUS,
                                            B.ISSUE_TITLE)
          FROM DUAL)
          AS ISSUE_TITLE,
       decode(a.issue_source_type,'0',a.delegatee_comments,MS_ISM_HELPER.get_Issue_Source_Title(a.issue_source,a.issue_source_type))  AS ISSUE_SOURCE,
       A.ACTION_DUE_DATE,
       B.ISSUE_DUE_DATE,
       NVL ((SELECT (TO_DATE (SYSDATE, 'DD/MM/RRRR')- TO_DATE (A.ACTION_DUE_DATE, 'DD/MM/RRRR'))AS days_past_due
             FROM DUAL
            WHERE (TO_DATE (SYSDATE, 'DD/MM/RRRR')
                   - TO_DATE (A.ACTION_DUE_DATE, 'DD/MM/RRRR')) > 0),
          0)
          As Days_Past_Due,
      --decode((TO_DATE (SYSDATE, 'DD/MM/RRRR')- TO_DATE (A.ACTION_DUE_DATE, 'DD/MM/RRRR')),sign(TO_DATE (SYSDATE, 'DD/MM/RRRR')- TO_DATE (A.ACTION_DUE_DATE, 'DD/MM/RRRR'))=1,(TO_DATE (SYSDATE, 'DD/MM/RRRR')- TO_DATE (A.ACTION_DUE_DATE, 'DD/MM/RRRR')),0) as due_days,
      A.NO_OF_EXTENSIONS AS NO_OF_EXTENSIONS,
       A.ACTION_OWNER AS USER_NAME,
       A.ACTION_DELEGATEE,
       A.Issue_Id,
       a.dd_created_on
  From Ms_Ism_Action A, Ms_Ism_Issue B
 Where A.Issue_Id = B.Issue_Id 
and action_status != '20'
