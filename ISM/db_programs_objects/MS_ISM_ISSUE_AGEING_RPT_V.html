MS_ISM_ISSUE_AGEING_RPT_V 

                                                
  CREATE OR REPLACE FORCE VIEW "NOVT_DEV"."MS_ISM_ISSUE_AGEING_RPT_V" ("ORG_ID", "ORG_NAME", "ISSUE_SOURCE_TYPE", "OVER_DUE", "DUE_THIS_MONTH", "OVERDUE_PAST_3_MONTH", "DUE_BEYOND_3_MONTH", "DUE_NEXT_X_MONTH", "DUE_BEYOND_X_MONTH", "TOTAL", "OWNER_ORG", "COL1", "COL2", "COL3") AS 
  select ISSUE_OWNER_ORG as Org_id, Org_name as Org_name,
  null as ISSUE_SOURCE_TYPE,
  NVL(SUM(OVERDUE_THIS_MONTH),0)    as OVER_DUE,
  NVL(SUM(DUE_THIS_MONTH),0)        as DUE_THIS_MONTH,
  NVL(SUM(OD_PAST_THREE_MONTH),0)   as OVERDUE_PAST_3_MONTH,
  NVL(SUM(OD_BEYOND_THREE_MONTH),0) as DUE_BEYOND_3_MONTH,
  NVL(SUM(DUE_WITHIN_X_MONTH),0)    as DUE_NEXT_X_MONTH,
  NVL(SUM(DUE_BEYOND_X_MONTH),0)    as DUE_BEYOND_X_MONTH,
  SUM((nvl(OVERDUE_THIS_MONTH,0)+nvl(DUE_THIS_MONTH,0)+nvl(OD_PAST_THREE_MONTH,0)+nvl(OD_BEYOND_THREE_MONTH,0)+nvl(DUE_WITHIN_X_MONTH,0)+nvl(DUE_BEYOND_X_MONTH,0)))
  ||'$==$'
  ||ms_apps_utilities.get_parameter_value (1, 'MS_ISM_Configuration', 'Enable_Ageing_Due_In_Months')
AS total,
  null AS Owner_Org,
  (select count(issue_id) from MS_ISM_ISSUE where ISSUE_OWNERORG=a.ISSUE_OWNER_ORG) AS Col1,
  2          AS Col2,
  3          AS COL3
FROM(
SELECT ISSUE_OWNER_ORG, (select org_entity_name from SI_ORG_ENTITIES_T where org_entity_id in (ISSUE_OWNER_ORG)) as Org_name,
    DECODE(OVERDUE_THIS_MONTH,'Y',COUNT(OVERDUE_THIS_MONTH))       AS OVERDUE_THIS_MONTH,
    DECODE(DUE_THIS_MONTH,'Y',COUNT(DUE_THIS_MONTH))               AS DUE_THIS_MONTH,
    DECODE(OD_PAST_THREE_MONTH,'Y',COUNT(OD_PAST_THREE_MONTH))     AS OD_PAST_THREE_MONTH,
    DECODE(OD_BEYOND_THREE_MONTH,'Y',COUNT(OD_BEYOND_THREE_MONTH)) AS OD_BEYOND_THREE_MONTH,
    DECODE(DUE_WITHIN_X_MONTH,'Y',COUNT(DUE_WITHIN_X_MONTH))       AS DUE_WITHIN_X_MONTH,
    DECODE(DUE_BEYOND_X_MONTH,'Y',COUNT(DUE_BEYOND_X_MONTH))       AS DUE_BEYOND_X_MONTH
  FROM MS_ISM_ISSUE_RPT_TBL
GROUP BY ISSUE_OWNER_ORG,
    OVERDUE_THIS_MONTH,
    DUE_THIS_MONTH,
    OD_PAST_THREE_MONTH,
    OD_BEYOND_THREE_MONTH,
    DUE_WITHIN_X_MONTH,
    DUE_BEYOND_X_MONTH) a
    GROUP BY ISSUE_OWNER_ORG, Org_name
