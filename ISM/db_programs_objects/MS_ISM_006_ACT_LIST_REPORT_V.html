MS_ISM_006_ACT_LIST_REPORT_V 

                                                
  CREATE OR REPLACE FORCE VIEW "NOVT_DEV"."MS_ISM_006_ACT_LIST_REPORT_V" ("ACTION_TITLE", "ACTION_STATUS", "ACTION_OWNER", "PRIORITY", "ACTION_DUE_DATE", "ACTION_COMPLETED_ON", "ISSUE_TITLE", "ISSUE_OWNER", "PERCENTAGE_COMPLETED", "PID", "INSTANCEID", "OWNER_ORGANIZATION", "ISSUE_ID", "APPROVER_ORG") AS 
  SELECT
    (SELECT ''
      || REPLACE(REPLACE(a.action_title,'<', '&'||'lt;'), '>', '&'||'gt;') 
      || ''
    FROM DUAL
    )                                                                                                    AS action_title,
    ms_ism_helper.get_status_display_name(a.action_status)                                               AS action_status,
    ms_apps_utilities.get_user_full_name (a.action_owner)                                                AS action_owner,
    ms_apps_utilities.get_display_value (100000, 'MS ISM Risk Rating', a.action_priority, 1, NULL, NULL) AS priority,
    a.action_due_date                                                                                    AS action_due_date,
    NULL                                                                                                 AS action_completed_on,
    a.issue_title                                                                                        AS issue_title,
    ms_apps_utilities.get_user_full_name (a.issue_owner)                                                 AS issue_owner,
    --a.action_per_complte AS percentage_completed,
    a.remove_delegation      AS percentage_completed, --remove delegation has been used to capture percentage complete value as it will get nullified on chnage of actions
    a.dd_process_instance_id AS pid,
    a.dd_instance_id         AS instanceid,
    b.ISSUE_OWNERORG         AS owner_organization,
    a.issue_id               AS issue_id,
    b.ISSUE_APPROVERORG      AS approver_org
    --start of bug - 39313 , c.IAR_TITLE as RISK End of Bug - 39313
  FROM ms_ism_action a,
    ms_ism_issue b              -- start of bug - 39313, MS_ISM_ISSUE_IAR c End of Bug - 39313
  WHERE a.issue_id      = b.issue_id -- start of bug - 39313 AND b.issue_id = c.issue_id End of Bug - 39313
  AND b.issue_status NOT IN ('1', 'Issue Reviewed', '5', '7', '6')
