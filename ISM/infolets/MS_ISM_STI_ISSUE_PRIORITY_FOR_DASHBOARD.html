                                                        Select Nvl(a.Issue_Risk_Priority,
           'Unknown'),
       Count(1),
       Nvl(Ms_Apps_Utilities.Get_Display_Value(100000,
     'MS ISM Priority',
     Nvl(a.Issue_Risk_Priority,
         'Unknown'),
     1,
     Null,
     Null),
           'Unknown') Issue_Priority,
       Decode(a.Issue_Status,
'Action Plan Clarify','Open','Approve Action Plan','Open','Cancel Clarify','Open','Clarify Issue','Open','Closed','Closed','Final Approval Clarify',
'Open','Issue Cancelled','Cancelled','Issue Closed','Closed','Issue Reviewed','Open','Manage Issue','Open','Monitor Issue','Open',
'New','Open','Open','Open','Open Issue','Open','Pending Approval','Open','Pending Final Approval','Open','Pending Review','Open',
'Request to Cancel','Open','Request to Close','Open','Open') Issue_Status,
       :2 As Dummy1,
       Null As Dummy2,
       Null As Dummy3,
       Null As Dummy4
From   Ms_Ism_Issue a
Where  a.Issue_Status Not In ('Pending Approval','New', 'Pending Review')
And    Decode(a.Issue_Status,'Action Plan Clarify','Open','Approve Action Plan','Open','Cancel Clarify','Open','Clarify Issue','Open','Closed','Closed',
'Final Approval Clarify','Open','Issue Cancelled','Cancelled','Issue Closed','Closed','Issue Reviewed','Open','Manage Issue','Open','Monitor Issue','Open',
'New','Open','Open','Open','Open Issue','Open','Pending Approval','Open','Pending Final Approval','Open','Pending Review','Open','Request to Cancel','Open',
'Request to Close','Open','Open') = :1
And    ( 1 = ms_ism_helper.fn_ism_securities(a.issue_id, :2) ) AND a.reached_manage = 'Y'
Group  By Nvl(a.Issue_Risk_Priority,
'Unknown'),
          Decode(a.Issue_Status,'Action Plan Clarify','Open','Approve Action Plan','Open','Cancel Clarify','Open','Clarify Issue','Open','Closed','Closed',
          'Final Approval Clarify','Open','Issue Cancelled','Cancelled','Issue Closed','Closed','Issue Reviewed','Open','Manage Issue','Open','Monitor Issue',
          'Open','New','Open','Open','Open','Open Issue','Open','Pending Approval','Open','Pending Final Approval','Open','Pending Review','Open',
          'Request to Cancel','Open','Request to Close','Open','Open')
