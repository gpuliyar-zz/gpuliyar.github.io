                                                        SELECT  (SELECT '<a target="_self" href="javascript'||CHR(58)||'callreportdetails(''Pushinfolet?id='||ms_apps_utilities.get_infolet_id('MS_STI_ISSUE')||CHR(38)||'proc='||a.dd_process_instance_id|| CHR (38)|| 'flag=1'|| CHR (38)|| 'emd=1'|| CHR (38)|| 'x_edit_mode=Yes' || CHR (38) || 'wrapper=no'')">'||a.issue_title|| '</a>'  from dual ) as Issue_Title
     , a.issue_status as Issue_Status
     , ms_apps_utilities.get_user_full_name(a.issue_owner) as Issue_Owner
     ,  MS_APPS_UTILITIES.GET_DISPLAY_VALUE(100000,'MS ISM STI Risk Rating',a.issue_risk_rating,1,null,null) as Issue_Rating, a.issue_due_date as Issue_Due_Date
     , ms_apps_utilities.get_user_full_name(a.dd_created_by) as Issue_Initiator
     , a.issue_identified_on as First_Identified_On 
     , a.issue_id as Issue_Id
     , a.issue_details as issue_details
     , (select count(1) from MS_ISM_Action where issue_id = a.issue_id) as No_Of_Actions
     , (select org_entity_name from si_org_entities_t where org_entity_id = a.issue_ownerorg) as issue_owner_org
     , MS_APPS_UTILITIES.GET_DISPLAY_VALUE(100000,'MS ISM Update Status',a.progress_status,1,null,null) as PROGRESS_STATUS
     , a.dd_process_instance_id as pid
     , a.dd_instance_id as instanceid,b.ISSUE_OWN_CLOSE_DATE,b.ISSUE_FINAL_CLOSE_DATE
    ,NULL as Dummy1,NULL as Dummy2,NULL as DUMMY3,NULL as DUMMY4,null as dummy5
  FROM ms_ism_issue a, ms_sti_issue b
WHERE a.issue_id = b.issue_id 
 and a.issue_ownerorg = NVL (:1, a.issue_ownerorg)
 and (
          (:2 is null)
             or
          (:2 is not null and upper(a.issue_title) like '%' || upper(:2) || '%')   
       )
and a.issue_owner = nvl(:3,a.issue_owner)
 and (
         :4 is null
                or
         (:4 is not null and a.issue_risk_rating = :4)
       ) 
      and (
          (:5 is null)
             or
          (:5 is not null and a.issue_status = :5)   
       )
       and issue_status not in ('New',
        'New - Pending Review',
        'New - Reviewed',
        'New - Issue Cancelled',
        'New - Pending Approval',
        'New - Clarify by Initiator')
       order by upper(a.ISSUE_TITLE)
