                                                        select case 
when Lead(status,1) over (order by x) is NULL and user_action <>'Approve and Close Issue'
    then 'Closed'
else
    'Closed'
end,  
x,stage_name,status, user_action,action_title, stage_completed_on, stage_completed_by,issue_owner_org,
issue_approverorg,issue_type,issue_id,issue_title,issue_due_date,STAGE_ROLE,ORGANIZATION,
comments,sign_role,sign_timestamp,
issue_rating,source_type
from
(SELECT rownum x,
            (SELECT    '<a target="_self" href="javascript'
                 || CHR (58)
                 || 'callreportdetails(''Pushinfolet?id='
                 || a.metric_id
                 || CHR (38)
                 || 'proc='
                 || a.process_instance_id
                 || CHR (38)
                 || 'instan_id='
                 || a.instance_id
                 || 'flag=1'
                 || CHR (38)
                 || 'emd=1'
                 || CHR (38)
                 || 'x_edit_mode=No'
                 || CHR (38)
                 || 'wrapper=no'')">'
                 || b.stage_name
                 || '</a>'
            FROM DUAL) as stage_name,
              DECODE (c.action_name,
               'Save and Continue Working', 'Open',
               'Update Status','Open',
               'Delegate Issue','Open',
               'Due Date Change Request','Open',
               'Reassign Owner','Open',
               'Remove Delegation','Open',
               'Delegate Action','Open',
               'Action Due Date Change Request','Open',
               'Update Delegation','Open',
               'Closed'
              ) AS status,d.issue_id,d.issue_title,d.issue_due_date,
              ms_apps_utilities.get_display_value (100000, 'MS ISM Risk Rating', d.issue_risk_rating, 1, NULL, NULL) AS issue_rating,
              ms_apps_utilities.get_display_value (100000, 'MS ISM Source Type', d.issue_source_type, 1, NULL, NULL) AS source_type,
              (SELECT org_entity_name FROM si_org_entities_t WHERE org_entity_id = d.issue_ownerorg) AS issue_owner_org,
              (SELECT org_entity_name  FROM si_org_entities_t WHERE org_entity_id = d.issue_approverorg) AS issue_approverorg,
              ms_apps_utilities.get_display_value (100000, 'MS ISM Issue Type', d.issue_type, 1, NULL, NULL) AS issue_type,
       c.action_name as user_action,a.action_title, stage_completed_on, stage_completed_by,
       DECODE (a.issue_stage, '1','Issue Initiator',
                              '2','Issue Reviewer',
                              '3','Issue Approver',
                              '4','Issue Owner',
                              '5','Issue Approver',
                              '7',DECODE (a.USER_ACTIONS,a.stage_completed_by,'Issue Owner','Delegatee'),
                              '8','Final Approval',
                              '20',DECODE (a.USER_ACTIONS,a.stage_completed_by,'Action Owner','Delegatee'),
                              '21','Action Reviewer',
                              '22','Action Approver',
                              '23','Action Approver' ) as STAGE_ROLE,a.ORGANIZATION,a.comments,a.sign_role,a.sign_timestamp
  FROM (SELECT DISTINCT instance_id, process_instance_id,metric_id,comments, issue_stage, status,
                        user_actions, action_title, stage_completed_on,
                        stage_completed_by, stage_role, ORGANIZATION,
                        sign_role, sign_timestamp
                   FROM ms_ism_comments_log
                  WHERE issue_id = :1
               ORDER BY instance_id) a,
               ms_ism_issue_v d,
       ms_ism_stages b,
       ms_ism_actions c
WHERE a.issue_stage = b.stage_id AND a.status = c.action_id
AND a.instance_id = d.instance_id 
AND a.process_instance_id = d.process_instance_id
AND d.instance_rec_num = 1
) a
order by rownum
