                                                        select a.issue_ownerorg,a.Adhoc As Adhoc,a.Audits As Audits,a.TestExecution As TestExecution,a.RiskAssessment As RiskAssessment,a.Surveys As Surveys,
a.InternalLossEvent As InternalLossEvent,a.GRCIntelligence As GRCIntelligence,a.Dummy1 As Dummy1,a.Total_Count As Total_Count,a.sourcetype0 As sourcetype0,
a.sourcetype1 As sourcetype1,a.sourcetype2 As sourcetype2,a.sourcetype3 As sourcetype3,a.sourcetype4 As sourcetype4,a.sourcetype5 As sourcetype5,a.sourcetype6 As sourcetype6,
a.sourcetype7 As sourcetype7
from (select b.issue_ownerorg,
      ms_ism_helper.get_source_count('0',b.issue_ownerorg,:1,'MS_ISM_OUTSTANDING_ISSUE') As Adhoc,
      ms_ism_helper.get_source_count('1',b.issue_ownerorg,:1,'MS_ISM_OUTSTANDING_ISSUE') As Audits,
      ms_ism_helper.get_source_count('2',b.issue_ownerorg,:1,'MS_ISM_OUTSTANDING_ISSUE') As TestExecution,
      ms_ism_helper.get_source_count('3',b.issue_ownerorg,:1,'MS_ISM_OUTSTANDING_ISSUE') As RiskAssessment,
      ms_ism_helper.get_source_count('4',b.issue_ownerorg,:1,'MS_ISM_OUTSTANDING_ISSUE') As Surveys,
      ms_ism_helper.get_source_count('5',b.issue_ownerorg,:1,'MS_ISM_OUTSTANDING_ISSUE') As InternalLossEvent,
      ms_ism_helper.get_source_count('6',b.issue_ownerorg,:1,'MS_ISM_OUTSTANDING_ISSUE') As GRCIntelligence,
      ms_ism_helper.get_source_count('7',b.issue_ownerorg,:1,'MS_ISM_OUTSTANDING_ISSUE') As Dummy1,
      ms_ism_helper.get_source_count('TOTAL_COUNT',b.issue_ownerorg,:1,'MS_ISM_OUTSTANDING_ISSUE') As Total_Count,
      ms_apps_utilities.get_display_value (100000, 'MS ISM Source Type','0', 1, NULL, NULL)  AS sourcetype0,
      ms_apps_utilities.get_display_value (100000, 'MS ISM Source Type','1', 1, NULL, NULL)  AS sourcetype1,
      ms_apps_utilities.get_display_value (100000, 'MS ISM Source Type','2', 1, NULL, NULL)  AS sourcetype2,
      ms_apps_utilities.get_display_value (100000, 'MS ISM Source Type','3', 1, NULL, NULL)  AS sourcetype3,
      ms_apps_utilities.get_display_value (100000, 'MS ISM Source Type','4', 1, NULL, NULL)  AS sourcetype4,
      ms_apps_utilities.get_display_value (100000, 'MS ISM Source Type','5', 1, NULL, NULL)  AS sourcetype5,
      ms_apps_utilities.get_display_value (100000, 'MS ISM Source Type','6', 1, NULL, NULL)  AS sourcetype6,
      ms_apps_utilities.get_display_value (100000, 'MS ISM Source Type','7', 1, NULL, NULL)  AS sourcetype7,
      dense_rank() over(order by ms_ism_helper.get_source_count('TOTAL_COUNT',b.issue_ownerorg,:1,'MS_ISM_OUTSTANDING_ISSUE') desc) rn
      from MS_ISM_OUTSTANDING_ISSUE b where  (1=ms_ism_helper.fn_ism_securities(b.issue_id,:1))     group by b.issue_ownerorg ) a
where a.rn <=10
