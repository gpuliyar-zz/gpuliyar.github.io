                                                        SELECT  a.ISSUE_ID
 , ms_ism_helper.fn_get_act_latest_comments(a.action_id) as ACTION_COMMENTS
 ,a.ACTION_DUE_DATE
 , ACTION_ID
 , MS_APPS_UTILITIES.GET_USER_FULL_NAME(a.ACTION_OWNER) as Action_Owner
 , decode(Action_Status,'10','', ACTION_PER_COMPLTE) as Action_Per_Complete
 , MS_APPS_UTILITIES.GET_DISPLAY_VALUE(100000,'MS ISM Priority',a.ACTION_PRIORITY,1,null,null) as Action_Priority
 , DECODE(ACTION_STATUS,'2','Open','16','Pending Closure','5','Pending Review','21','Reopened','6','Pending Approval','Closed') ACTION_STATUS
, (SELECT    '<a target="_self" href="javascript'
                  || CHR (58)
                  || 'callreportdetails(htmlEncode(''Pushinfolet?id='
                  --|| ms_apps_utilities.get_infolet_id ('MS_ISM_ACTION')
                  || ms_ism_utilities.get_ism_infolet_id ('ACTION')
                  || CHR (38)
                  || 'proc='
                  || a.dd_process_instance_id
                  || CHR (38)
                  || 'instid='
                  || a.dd_instance_id
                  || CHR (38)
                  || 'flag=1'
                  || CHR (38)
                  || 'emd=1'
                  || CHR (38)
                  || 'x_edit_mode=Yes'
                  || CHR (38)
                  || 'wrapper=no''))">'
                  || HTF.ESCAPE_SC(a.action_title
)
                  || '</a>'
             FROM DUAL) as ACTION_TITLE
,Ms_Apps_Utilities.Get_Display_Value(100000,'MS ISM Action Item Type', nvl(B.Action_Type,'1')
,1,Null,Null) as Action_Type ,MS_ISM_HELPER.get_link_url_for_reopen_act(a.action_id,:2,a.action_status,a.action_title) as Reopen_Action
  FROM MS_ISM_ACTION a,
 MS_ISM_ISSUE_IAP B
WHERE a.ISSUE_ID = B.ISSUE_ID(+)
And A.Action_Id  = B.Issue_Action_Id(+)
and  a.issue_id=:1
