MS_AUD_Rpt_Detailed_Loi Report SYSTEMI 
                                                        SELECT a.audit_id AS audit_id ,
  a.audit_title
  ||' ('
  ||a.audit_id
  ||')'                  AS audit_title,
  a.issue_title          as issue_title,
  c.task_name            as task_name,
  a.issue_status         as issue_status,
  a.issue_due_date       AS issue_due_date,
  a.issue_owner          as issue_owner,
  a.issue_approver       as issue_approver,
  a.issue_disposition    AS issue_disposition,
  a.issue_rating         as issue_rating,
  b.audit_start_date     as audit_start_date,
  (select status_name from ms_aud_status 
  where status_id = b.audit_status) as audit_status,
  a.lead_auditor         as lead_auditor,
  a.AUDIT_MANAGER        AS AUDIT_MANAGER,  
  a.priority             as priority,
  a.issue_created_by     as issue_created_by,
  a.issue_id             as consolidated_issue,
  a.issue_progress       AS issue_progress,
  a.no_of_actions        AS no_of_actions,
  a.auditable_entities   AS auditable_entities,
  a.risks                as risks,
  A.COMPLIANCE_CNTRL     AS COMPLIANCE_CNTRL,
  a.AREA_compliance      AS AREA_compliance,
  a.related_to_org       AS related_to_org,
  A.FN_ORGANIZATION      AS FN_ORGANIZATION,
  a.audit_classification as audit_classification,
  a.AUDIT_TYPE           AS AUDIT_TYPE,
  b.conducted_by_org     as conducted_by_org
from table (ms_aud_utilities.get_issue_data) a,
  ms_aud_audit b, ms_aud_audit_task c
where a.audit_id = b.audit_id
and b.audit_id = c.audit_id
and c.task_id = :2
AND ms_aud_audit_security.check_view_access(b.audit_id,b.dd_object_type,b.restrict_access_to,''
  ||b.conducted_by_org
  ||'',:1,'VIEW') = 'ACCESS'
