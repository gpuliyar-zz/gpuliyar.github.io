                                                        SELECT DISTINCT a1.audit_id,   
                   '<a target="_blank" href="Reportsexportwordpopup?resourcename=Audit_Details_Report_New'   
                || CHR (38)
                || 'process_id='   
                || a1.dd_process_instance_id
                || CHR (38)
                || 'aud_id='   
                || a1.audit_id   
                || '" Audit Title="' || DBMS_XMLGEN.CONVERT(a1.audit_title)||'" >'   
                || ' '   
                ||DBMS_XMLGEN.CONVERT( a1.audit_title) || ' [' || a1.audit_id || ']'   
                || '</a>' AS audit_title,   
                a1.audit_type, a1.audit_classification, a1.audit_start_date,   
                a1.conducted_by_org, a1.plan_title, a1.plan_id,   
                CASE a1.status  WHEN '3' THEN 'In Process'   
                   WHEN '4' THEN 'Canceled'   
                   WHEN '8' THEN 'Closed'   
                   WHEN '20' THEN 'Future Audit'   
                   ELSE 'Status not Found'   
                END,   
                c1.km_milestones, to_char(TRUNC (c1.km_proposed_date),'MM/DD/YYYY'),   
                to_char(TRUNC (c1.km_actual_date),'MM/DD/YYYY'),NVL (TO_CHAR (TRUNC (c1.km_proposed_date), 'MM/DD/YYYY'), '    		 -')   
                || CHR (10) || NVL (TO_CHAR (TRUNC (c1.km_actual_date), 'MM/DD/YYYY'), '      		-'),   
                'Planned' || CHR (10) || 'Actual',   
                CASE   
                   WHEN TRUNC (c1.km_proposed_date) < TRUNC (c1.km_actual_date) THEN 'Delayed'   
                   WHEN TRUNC (c1.km_proposed_date) >= TRUNC (c1.km_actual_date) THEN 'On-Time'   
                   ELSE 'Not-Applicable'   
                END   
 , C1.KM_EVENT_IN_SYSTEM   
,nvl((trunc(c1.km_actual_date))-trunc((c1.km_proposed_date)),0) date_color   
           FROM ms_aud_audit a1, ms_aud_audit_kms c1   
          WHERE 1 = 1   
            AND a1.audit_id = c1.audit_id   
            AND NVL (TRIM (a1.plan_title), '1') =   
                              NVL (TRIM (:1), NVL (TRIM (a1.plan_title), '1'))   
            AND NVL (TRIM (a1.conducted_by_org), '0') =   
                        NVL (TRIM (:2), NVL (TRIM (a1.conducted_by_org), '0'))   
            AND NVL (TRIM (a1.audit_type), '0') =   
                              NVL (TRIM (:4), NVL (TRIM (a1.audit_type), '0'))   
            AND NVL (TRIM (a1.audit_classification), '0') =   
                    NVL (TRIM (:3), NVL (TRIM (a1.audit_classification), '0'))   
            AND TRUNC (a1.audit_start_date) <= TRUNC (SYSDATE)   
            AND (:5 IS NULL OR TRUNC (a1.audit_start_date) > TO_DATE (:5))   
            AND (:6 IS NULL OR TRUNC (a1.audit_start_date) < TO_DATE (:6))   
            AND (   :7 IS NULL   
                 OR CASE a1.status   
                       WHEN '3'   
                          THEN 'In Process'   
                       WHEN '4'   
                          THEN 'Canceled'   
                       WHEN '8'   
                          THEN 'Closed'   
                       WHEN '20'   
                          THEN 'Future Audit'   
                       ELSE 'Status not Found'   
                    END = :7   
                )   
            AND ms_aud_audit_security.check_view_access(a1.audit_id,a1.dd_object_type,a1.restrict_access_to,   
                                                        a1.conducted_by_org,:9,'VIEW') = 'ACCESS'   
AND (   (:8 IS NULL AND a1.milestone_template = '100')   
                    OR (    :8 IS NOT NULL   
                        AND a1.milestone_template = :8   
                       )   
                ) AND a1.status NOT IN ('2','31','51') and a1.approve_flag = '1'
