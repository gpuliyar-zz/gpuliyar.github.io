          a. audit_title AS audit_title_link,
        a.plan_title AS plan_title,
        a.audit_type AS audit_type,
          a.audit_classification AS audit_category,
         a.audit_status  AS audit_status,
       'core_objects' AS core_objects,
        'related_risks' AS risks,
         'organizations' AS organizations,
         a.lead_auditor as lead_auditor,
         a.audit_manager AS audit_manager,
         a.dd_created_by AS created_by,
        a.audit_start_date AS audit_start_date,
          'view tasks' AS view_tasks, restrict_access_to,
        'team_member' AS team_member,
         a.conducted_by_org,
         a.audit_classification,
         a.audit_status,b.audit_rating
    FROM ms_aud_audit a ,
    ms_aud_audit_report b
    where a.audit_id=b.audit_id
    and audit_rating=:1
    and audit_classification=:2
    and conducted_by_org=:3 and
ms_aud_audit_security.check_view_access 
(a.audit_id,a.dd_object_type,a.restrict_access_to,a.conducted_by_org,:4,'VIEW') = 
'ACCESS' 
    ORDER BY audit_title
