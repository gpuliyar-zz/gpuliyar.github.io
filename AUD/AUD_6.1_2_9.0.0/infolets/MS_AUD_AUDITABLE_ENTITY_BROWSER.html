                                                        SELECT   a.dd_object_type AS dd_object_type, a.object_id AS object_id, a.object_name AS object_name, a.object_level AS object_level, a.PARENT AS PARENT,
         a.description AS description, a.owner_organizations AS owner_organizations, a.owners AS owners, a.valid_until AS valid_until, a.valid_from AS valid_from,
         a.CATEGORY AS CATEGORY, a.TYPE AS process_type,
         (SELECT ms_grc_utilities.get_edit_url(a.dd_object_type,
                                              a.dd_process_instance_id,
                                              a.dd_instance_id,
                                              a.dd_edit_flag,
                                              a.object_name
                                             ) FROM DUAL) AS edit_url,
         ms_grc_utilities.include_expired (a.valid_from,
                                           a.valid_until
                                          ) AS expired_flag,
         (SELECT ms_grc_utilities.get_status_name (:1,
                                                   a.status
                                                  )
            FROM DUAL) AS status,
         a.related_risks AS related_risks, a.related_controls AS related_controls, a.related_areas_of_compliance AS related_areas_of_compliance,
         a.related_requirements AS related_requirements, a.related_framework_references AS related_framework_references,
         a.related_objectives AS related_objectives, a.related_functions AS related_functions, a.related_references AS related_references,
         a.related_financial_accounts AS related_financial_accounts, a.dd_process_instance_id AS dd_process_instance_id, a.dd_instance_id AS dd_instance_id,
         a.dd_edit_flag AS dd_edit_flag, SUBSTR (a.object_name, 1, 30) display_name,
         a.dd_object_type AS object_type, 'Y' AS over_ride_action,
            'proc='
         || TO_CHAR (a.dd_process_instance_id)
         || '|instid='
         || TO_CHAR (a.dd_instance_id)
         || '|edit_flag='
         || NVL (a.dd_edit_flag, 'Y') AS edit_action_parameters,
         a.applies_to_organizations AS applies_to_organizations, a.core_object_type AS core_object_type
    FROM ms_aud_grc_process_rv a
   WHERE (TRUNC(a.valid_from) <= TRUNC(SYSDATE) OR TRUNC(a.valid_from) IS NULL)
         AND ms_apps_security_pkg.check_object_access (a.dd_object_type,
                                                   a.restrict_access_to,
                                                   a.owner_org_ids,
                                                   :1,
                                                   'VIEW'
                                                  ) = 'ACCESS'
ORDER BY a.object_name, a.object_id
