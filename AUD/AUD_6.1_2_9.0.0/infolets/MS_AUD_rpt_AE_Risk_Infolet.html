                                                        SELECT a.object_id as object_id, 
   (SELECT displayed_value
          FROM ms_qs_list_of_values lovvalue,ms_qs_lov_names lovname,si_metrics_t m
          WHERE lovvalue.lov_id = lovname.lov_id
              AND lovvalue.org_entity_id = 1
              AND lovname.lov_name in('MS_AUD_AE_Process_Type','MS GRC Process Type')
              AND lovvalue.locale_id = nvl(cast(si_db_locale_sv.getvalue as number),1009) 
              and stored_value=a.core_type
              and lovvalue.enterprise_id=m.enterprise_id
              and m.metric_name = a.dd_object_type) object_type,
               '<a target="_self" '|| 'href="javascript'
                || CHR (58)|| 'callreportdetails(''Pushinfolet?id='
                || (SELECT metric_id FROM si_metrics_t WHERE metric_name = a.dd_object_type)
                || CHR (38)|| 'proc='|| a.dd_process_instance_id|| CHR (38)
                || 'instid='|| a.dd_instance_id|| CHR (38)|| 'edit_flag=N'
                || CHR (38)|| 'flag=2'|| CHR (38)|| 'emd=1'')">'
                || SUBSTR ('' || a.object_name || '', 1, 3800)
                || '</a>' AS object_name ,
               trim(substr(ms_aud_advisor_utilities.fn_get_rating(ms_aud_utilities.get_risk_score (a.object_id,:1)),1,
              instr(ms_aud_advisor_utilities.fn_get_rating(ms_aud_utilities.get_risk_score (a.object_id,:1)),' '))) AS risk_rating,
              (SELECT TRUNC (MAX (dd_modified_on))
                   FROM ms_aud_audit WHERE audit_id IN (
                           SELECT audit_id FROM ms_aud_audit_aue b
                            WHERE b.core_objects = a.object_id)
                    AND audit_status = '8') AS last_audited_on,
                (SELECT TRUNC (MIN (audit_start_date))
                   FROM ms_aud_audit WHERE audit_id IN (
                           SELECT audit_id FROM ms_aud_audit_aue b
                            WHERE b.core_objects = a.object_id)
                    AND audit_status = '33') AS next_audit_date,
                ms_aud_utilities.get_plan_years (a.object_id) AS plan_years,
                NVL((SELECT COUNT(b.audit_id) FROM ms_aud_audit_aue b,ms_aud_audit aud
                     WHERE b.core_objects = a.object_id and aud.status not in (4,8) and aud.audit_id=b.audit_id),0) AS no_of_audits_till_date,
                ms_aud_advisor_utilities.fn_get_open_issue(a.object_id) AS open_issues,
                (SELECT displayed_value
          FROM ms_qs_list_of_values lovvalue,ms_qs_lov_names lovname ,si_metrics_t m
          WHERE lovvalue.lov_id = lovname.lov_id
              AND lovvalue.org_entity_id = 1
              AND lovname.lov_name in('MS GRC Core Category','MS_AUD_AE_Category_Type')
              AND lovvalue.locale_id = nvl(cast(si_db_locale_sv.getvalue as number),1009) 
              and stored_value=a.core_category
              and lovvalue.enterprise_id=m.enterprise_id
              and m.metric_name = a.dd_object_type) AS core_category ,
                ms_apps_utilities.get_user_full_name(d.obj_created_by) AS created_by,
                TRUNC (d.dd_created_on) AS creation_date
             FROM ms_aud_grc_core_object_pub_v a, ms_grc_core_object d
          WHERE a.object_id = d.object_id
         AND ms_apps_security_pkg.check_object_access(a.dd_object_type, a.restrict_access_to, a.owner_organizations, :2, 'VIEW') = 'ACCESS'
         AND  TRUNC(NVL(a.VALID_UNTIL,SYSDATE)) >= TRUNC(SYSDATE)   
         ORDER BY ms_aud_utilities.get_risk_score (a.object_id,:1) desc NULLS LAST
