MS_AUD_Rpt_Reopen_Audits Report SYSTEMI 
                                                        SELECT a.audit_id as audit_id, 
      ms_aud_utilities_2.fn_get_data_correction_url('MS_AUD_MANAGE_AUDIT',a.audit_id,a.dd_process_instance_id,a.dd_instance_id,'12',null,a.audit_title,:7) AS audit_title,
       DECODE(a.audit_status, 58, ms_aud_utilities.fn_get_status_display_name('58'),ms_aud_utilities.fn_get_status_display_name(a.status)) AS status,
       ms_aud_utilities.fn_get_org_display_value(a.conducted_by_org) as conducted_by_org,
       ms_aud_utilities.fn_get_plan_title(a.plan_title) as plan_title,
       ms_aud_utilities.get_code_description ('MS AUD Audit Type',a.audit_type) AS audit_type, 
       ms_aud_utilities.get_code_description ('MS AUD Classification',a.audit_classification) AS audit_classification,
       nvl(ms_apps_utilities.get_user_full_name (a.lead_auditor),ms_aud_utilities.fn_get_mlov_lead_aud_disp(ms_aud_utilities.fn_get_mlov_values(audit_id,'lead_auditor_mlov','ms_aud_audit_alm'))) AS lead_auditor,
       nvl(ms_apps_utilities.get_user_full_name (a.audit_manager),ms_aud_utilities.fn_get_mlov_aud_man_disp(ms_aud_utilities.fn_get_mlov_values(audit_id,'audit_manager_mlov','ms_aud_audit_amm'))) AS audit_manager,
       a.audit_start_date AS audit_start_date,
       a.audit_closure_date as audit_closure_date,
       ms_aud_utilities_2.fn_get_data_correction_url('MS_AUD_PLAN_SCOPE',a.audit_id,(select dd_process_instance_id from ms_aud_plan_scope where audit_id=a.audit_id and form_type='P'),(select dd_instance_id from ms_aud_plan_scope where audit_id=a.audit_id and form_type='P'),'12',null,(ms_aud_utilities.fn_get_plan_scope_status(a.audit_id)),:7) as audit_plan_scope,
 a.last_reopened_date as last_reopened_date,
       num_of_audit_reopen As num_of_audit_reopen,
       ms_aud_utilities.fn_get_task_count(a.audit_id,:7) as view_tasks,
       ms_aud_utilities_2.fn_get_data_correction_url('MS_AUD_DRAFT_ISSUES',a.audit_id,(select dd_process_instance_id from ms_aud_draft_issues where audit_id=a.audit_id),(select dd_instance_id from ms_aud_draft_issues where audit_id=a.audit_id),'12',null,ms_aud_utilities.fn_get_draft_issue_count(a.audit_id),:7) as draft_issues,
       NULL AS actions,
       ms_aud_utilities.fn_get_reopen_audit_url('MS_AUD_MANAGE_AUDIT',a.audit_id,a.dd_process_instance_id,null,'11','Yes',a.audit_title,:7) AS reopen_audit,
       ms_aud_utilities_2.fn_get_data_correction_url('MS_AUD_AUDIT_REPORT',a.audit_id,(select max(dd_process_instance_id) from ms_aud_audit_report where audit_id=a.audit_id),(select max(dd_instance_id) from ms_aud_audit_report where audit_id=a.audit_id),'12',null,'Final Report',:7) as last_published_report,
 a.chk_user_for_reopen_activity as chk_user_for_reopen_activity 
  FROM ms_aud_audit A
    WHERE a.status = 8
 
AND TRUNC(a.audit_start_date) 
            BETWEEN TRUNC(add_months(SYSDATE,-(ms_apps_utilities.get_parameter_value(ms_apps_utilities.get_enterprise_id(ms_apps_utilities.get_infolet_id('MS_AUD_MANAGE_AUDIT')),
            'MS_AUD_SYSTEM_PARAMETERS','Audits closed in past X months')))) 
            AND TRUNC(SYSDATE) 
      AND NVL(a.conducted_by_org,1) = NVL(:1, NVL(a.conducted_by_org,1))
      AND (:2 IS NULL OR (:2 IS NOT NULL  AND (LOWER(a.audit_title) LIKE'%' || LOWER (:2) || '%' )))
 
AND (:3 IS NULL OR (:3 IS NOT NULL AND (a.lead_auditor = :3) OR :3 IN (SELECT lead_auditor_mlov FROM ms_aud_audit_alm WHERE audit_id = a.audit_id))) 
AND (:4 IS NULL OR (:4 IS NOT NULL AND (a.audit_manager = :4) OR :4 IN (SELECT audit_manager_mlov FROM ms_aud_audit_amm WHERE audit_id = a.audit_id))) 
      AND (:5 IS NULL OR (:5 IS NOT NULL AND (TRUNC(a.audit_closure_date) >= :5))) 
      AND (:6 IS NULL OR (:6 IS NOT NULL AND (TRUNC (a.audit_closure_date) <= :6)))
      AND (ms_aud_audit_security.check_audit_lead_manager(a.audit_id,:7) > 0 OR 
            ms_aud_audit_security.check_restricted_view_access (:7,null) = 1)
ORDER BY a.dd_process_instance_id DESC
