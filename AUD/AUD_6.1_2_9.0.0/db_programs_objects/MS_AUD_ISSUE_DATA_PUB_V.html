MS_AUD_ISSUE_DATA_PUB_V 

                                                
  CREATE OR REPLACE FORCE VIEW "AUDDEVSP3"."MS_AUD_ISSUE_DATA_PUB_V" ("ISSUE_ID", "ISSUE_TITLE", "OBJECT_ID", "ISSUE_DETAILS", "ISSUE_STATUS", "ISSUE_STATUS_STORED", "ISSUE_DUE_DATE", "ISSUE_IDENTIFIED_ON", "ISSUE_PROGRESS", "PROGRESS_STATUS", "PROGRESS_STATUS_STORED", "ISSUE_SOURCE_TYPE", "ISSUE_SOURCE", "ISSUE_SOURCE_STORED", "ISSUE_SOURCE_REF", "CREATED_BY", "CREATED_BY_FULL_NAME", "CREATED_ON", "LAST_MODIFIED_BY", "MODIFIED_BY_FULL_NAME", "LAST_MODIFIED_ON", "ISSUE_CLOSED_ON", "OBJECT_TYPE", "ISSUE_OWNER", "ISSUE_OWNER_FULL_NAME", "ISSUE_OWNERORG", "ISSUE_OWNERORG_STORED", "ISSUE_DELEGATEE", "ISSUE_DLGT_FULL_NAME", "INITIAL_APPROVER", "INITIAL_APPR_FULL_NAME", "INITIAL_APPROVER_ORG", "INITIAL_APPR_ORG_STORED", "ACTION_PLAN_APPROVER", "ACT_PLAN_APPR_FULL_NAME", "ACT_PLAN_APPROVER_ORG", "ACT_PLAN_APPR_ORG_STORED", "FINAL_APPROVER", "FINAL_APPR_FULL_NAME", "FINAL_APPROVER_ORG", "FINAL_APPR_ORG_STORED", "NO_OF_REOPEN", "ISSUE_RATING", "ISSUE_RATING_STORED", "PRIORITY", "PRIORITY_STORED", "NO_OF_ACTIONS_TRIGGERED", "ACTION_APPROVAL_STORED", "IMPACT", "IMPACT_STORED", "LIKELIHOOD", "LIKELIHOOD_STORED", "LOSS_GAIN", "LOSS_GAIN_STORED", "LOSS_GAIN_AMOUNT", "EXCEPTION_TYPE", "EXCEPTION_TYPE_STORED", "REACHED_TO_MANAGE", "REPEAT_ISSUE", "REPEAT_ISSUE_STORED", "ISSUE_TYPE", "ISSUE_TYPE_STORED", "RELATED_ISSUES", "RELATED_RISKS", "RELATED_AREA_OF_COMP", "RELATED_CONTROLS", "RELATED_CORE_PROCESSES", "RELATED_ORGS", "RELATED_STANDARD", "RELATED_ASSET", "RELATED_ASSET_CLASS", "RELATED_FINANCIAL_ACOUNT", "RELATED_FUNCTION", "RELATED_FRAMEWORK_REF", "RELATED_OBJECTIVE", "RELATED_PROCESS", "RELATED_REFERENCE", "RELATED_REGULATORY_BODY", "RELATED_REQUIREMENT", "RELATED_QUESTION_PROCEDURE", "RELATED_SUPPLIER", "RELATED_AUDITABLE_ENTITY", "RELATED_PRODUCT", "DD_PROCESS_INSTANCE_ID", "DD_INSTANCE_ID", "DD_METRIC_ID", "ISSUE_LINK", "ISSUE_POPUP_LINK") AS 
  SELECT
NULL AS ISSUE_ID,
A.FN_FINDING_TITLE AS  ISSUE_TITLE,
ar.fn_core_objs  as OBJECT_ID,
NULL AS ISSUE_DETAILS,
(SELECT s.status_name
                 FROM ms_aud_trans_status ts
                ,ms_aud_status s
                 WHERE ts.ref_pk_id = a.audit_id 
                   AND ts.ref_secondary_id = b.fn_id --fn_rec.draft_fn_id
               and s.status_id = ts.status_id
              ) AS ISSUE_STATUS,
NULL AS ISSUE_STATUS_STORED,
A.FN_ISSUE_DUE_DATE AS ISSUE_DUE_DATE,
NULL AS ISSUE_IDENTIFIED_ON,
case when A.FN_ISSUE_RESOLUTION in('2','3')
                THEN 'NA'
               end  AS ISSUE_PROGRESS,
NULL AS PROGRESS_STATUS,
NULL AS PROGRESS_STATUS_STORED,
NULL AS ISSUE_SOURCE_TYPE,
NULL AS ISSUE_SOURCE,
NULL AS ISSUE_SOURCE_STORED,
NULL AS ISSUE_SOURCE_REF,
null AS CREATED_BY,
null AS CREATED_BY_FULL_NAME,
NULL AS CREATED_ON,
NULL AS LAST_MODIFIED_BY,
NULL AS MODIFIED_BY_FULL_NAME,
NULL AS LAST_MODIFIED_ON,
NULL AS ISSUE_CLOSED_ON,
NULL AS OBJECT_TYPE,
A.FN_ISSUE_OWNER  AS ISSUE_OWNER,
(SELECT first_name || ' ' || last_name
  FROM si_users_T
 WHERE user_name = A.FN_ISSUE_OWNER)AS ISSUE_OWNER_FULL_NAME,
       (SELECT ORG_ENTITY_NAME
          FROM SI_ORG_ENTITIES_T
         WHERE ORG_ENTITY_ID = A.FN_OWNER_ORGANIZATION) AS ISSUE_OWNERORG,
A.FN_OWNER_ORGANIZATION AS ISSUE_OWNERORG_STORED,
NULL AS ISSUE_DELEGATEE,
NULL AS ISSUE_DLGT_FULL_NAME,
A.FN_ISSUE_APPROVER AS INITIAL_APPROVER,
       (SELECT FIRST_NAME || ' ' || LAST_NAME
          FROM SI_USERS
         WHERE USER_NAME = A.FN_ISSUE_APPROVER) AS INITIAL_APPR_FULL_NAME,
       (select org_entity_name
          from si_org_entities_t
         where org_entity_id = A.FN_APPROVER_ORG) AS INITIAL_APPROVER_ORG,
A.FN_APPROVER_ORG AS INITIAL_APPR_ORG_STORED,
NULL AS ACTION_PLAN_APPROVER,
NULL AS ACT_PLAN_APPR_FULL_NAME,
NULL AS ACT_PLAN_APPROVER_ORG,
NULL AS ACT_PLAN_APPR_ORG_STORED,
NULL AS FINAL_APPROVER,
NULL AS FINAL_APPR_FULL_NAME,
NULL AS FINAL_APPROVER_ORG,
NULL AS FINAL_APPR_ORG_STORED,
NULL AS NO_OF_REOPEN,
       (SELECT displayed_value
          FROM ms_qs_list_of_values lovvalue, ms_qs_lov_names lovname
         WHERE LOVVALUE.LOV_ID = LOVNAME.LOV_ID
           AND LOVVALUE.ORG_ENTITY_ID = 1 --I_ORG_ENTITY_ID
           AND lovvalue.enterprise_id = (SELECT ENTERPRISE_ID FROM SI_ENT)
           AND LOVVALUE.STORED_VALUE = A.FN_ISSUE_RATING
           AND lovname.enterprise_id = (SELECT ENTERPRISE_ID FROM SI_ENT)
           AND LOVNAME.LOV_NAME = 'MS AUD Audit Task Issue Rating'
           AND LOVVALUE.LOCALE_ID =
               NVL(CAST(SI_DB_LOCALE_SV.GETVALUE AS NUMBER), 1009)) AS ISSUE_RATING,
A.FN_ISSUE_RATING AS ISSUE_RATING_STORED,
null AS PRIORITY,
null AS PRIORITY_STORED,
(SELECT COUNT(ac_id)
                  FROM ms_aud_audit_task_act
                  WHERE ac_finding_id = A.fn_id
)  AS NO_OF_ACTIONS_TRIGGERED,
NULL AS ACTION_APPROVAL_STORED,
NULL AS IMPACT,
NULL AS IMPACT_STORED,
NULL AS LIKELIHOOD,
NULL AS LIKELIHOOD_STORED,
NULL AS LOSS_GAIN,
NULL AS LOSS_GAIN_STORED,
NULL AS LOSS_GAIN_AMOUNT,
NULL AS EXCEPTION_TYPE,
NULL AS EXCEPTION_TYPE_STORED,
NULL AS REACHED_TO_MANAGE,
NULL AS REPEAT_ISSUE,
NULL AS REPEAT_ISSUE_STORED,
A.FN_FINDING_TYPE AS ISSUE_TYPE,
NULL AS ISSUE_TYPE_STORED,
NULL AS RELATED_ISSUES,
(select ms_concat(fn_related_risks) from MS_AUD_AUDIT_TASK_RSK where FN_ID = a.fn_id) AS RELATED_RISKS,
null AS RELATED_AREA_OF_COMP,
null AS RELATED_CONTROLS,
(SELECT MS_CONCAT(AE.OBJECT_NAME)
  FROM MS_AUD_GRC_CORE_OBJECT_PUB_V ae, MS_AUD_DRAFT_ISSUES_FCO B
 WHERE AE.OBJECT_ID = B.FN_CORE_OBJS
   AND b.FN_ID = A.FN_ID)   AS RELATED_CORE_PROCESSES,
(SELECT ms_concat(org_entity_name)
                  FROM si_org_entities_t 
                     ,MS_AUD_AUDIT_TASK_ORG 
                  WHERE org_entity_id =FN_ORGANIZATIONS
                     and FN_ID =a.fn_id
) AS RELATED_ORGS,
NULL AS RELATED_STANDARD,
NULL AS RELATED_ASSET,
NULL AS RELATED_ASSET_CLASS,
NULL AS RELATED_FINANCIAL_ACOUNT,
NULL AS RELATED_FUNCTION,
NULL AS RELATED_FRAMEWORK_REF,
NULL AS RELATED_OBJECTIVE,
NULL AS RELATED_PROCESS,
NULL AS RELATED_REFERENCE,
NULL AS RELATED_REGULATORY_BODY,
NULL AS RELATED_REQUIREMENT,
NULL AS RELATED_QUESTION_PROCEDURE,
NULL AS RELATED_SUPPLIER,
NULL AS RELATED_AUDITABLE_ENTITY,
NULL AS RELATED_PRODUCT,
NULL AS DD_PROCESS_INSTANCE_ID,
NULL AS DD_INSTANCE_ID,
NULL AS DD_METRIC_ID,
NULL AS ISSUE_LINK,
NULL AS ISSUE_POPUP_LINK
FROM ms_aud_draft_issues_fnd b,
       ms_aud_audit_task_fnd   a,
       ms_aud_draft_issues_fco ar,
       ms_aud_trans_status     st
 WHERE b.audit_id = a.audit_id
   AND b.task_id = a.task_id
   AND b.fn_id = a.fn_id
   AND AR.FN_ID = B.FN_ID
   and a.fn_id = ar.fn_id
   and ar.audit_id = st.ref_pk_id
   and st.status_id = '28'
   AND ST.REF_SECONDARY_ID = A.FN_ID
   AND A.fn_issue_resolution = '1'
