MS_AUD_RISK_CONTROL_TEST_APS_V 

                                                
  CREATE OR REPLACE FORCE VIEW "AUDDEVSP3"."MS_AUD_RISK_CONTROL_TEST_APS_V" ("AUDIT_ID", "CONTROL_ID", "TS_TEST_NAME", "CTL_CONTROL_NAME", "RISK_ID", "RS_RISK_NAME", "TEST_ID", "RISK_CONTROL", "RISK_CONTROL_TEST", "APPLIES_TO_CONTROLS", "CONTROL_ACTION", "CONTROL_OBJECT_TYPE", "DD_PROCESS_INSTANCE_ID", "RISK_MAPPING", "RISK_TYPE") AS 
  SELECT  rsk.audit_id
        , control_id
        ,ts_test_name 
       ,ctl_control_name
       ,risk_id
       ,rs_risk_name
       ,test_id
       ,risk_id ||'~'|| control_id as risk_control
       ,risk_id ||'~'|| control_id||'~'||test_id risk_control_test
       ,null as applies_to_controls
       ,null as control_action
       ,null as control_object_type
       ,null as dd_process_instance_id
       ,null as risk_mapping
       ,null as risk_type
from ms_aud_plan_scope_rsk rsk, ms_aud_plan_scope_ctl ctl, ms_aud_plan_scope_tst tst
where rsk.RS_ID = ctl.ct_rs_id and ctl.ct_id = tst.TS_CT_ID AND RSK.AUDIT_ID=CTL.AUDIT_ID AND CTL.AUDIT_ID=TST.AUDIT_ID
