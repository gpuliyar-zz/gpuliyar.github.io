MS_AUD_HEADER_PUB_V 

                                                
  CREATE OR REPLACE FORCE VIEW "AUDDEVSP3"."MS_AUD_HEADER_PUB_V" ("AUDIT_ID", "AUDIT_TITLE", "AUDIT_START_DATE", "CONDUCTED_BY_ORGANIZATION_STO", "CONDUCTED_BY_ORGANIZATION_DISP", "AUDIT_CLASSIFICATION_STO", "AUDIT_CLASSIFICATION_DISP", "AUDIT_TYPE_STO", "AUDIT_TYPE_DISP", "AUDIT_STATUS_STO", "AUDIT_STATUS_DISP", "LEAD_AUDITOR_STO", "LEAD_AUDITOR_DISP", "AUDIT_MANAGER", "AUDIT_MANAGER_DISP", "LEAD_AUDITOR_MLOV_STO", "LEAD_AUDITOR_MLOV_DISP", "AUDIT_MANAGER_MLOV_STO", "AUDIT_MANAGER_MLOV_DISP", "REFERENCE_ID", "SOURCE_MODULE") AS 
  select a.AUDIT_ID AUDIT_ID,
         a.AUDIT_TITLE AUDIT_TITLE,
         a.AUDIT_START_DATE AUDIT_START_DATE,
         a.CONDUCTED_BY_ORG CONDUCTED_BY_ORGANIZATION_STO,
         MS_AUD_UTILITIES.FN_GET_ORG_DISPLAY_VALUE (a.CONDUCTED_BY_ORG) CONDUCTED_BY_ORGANIZATION_DISP,
         a.audit_classification Audit_Classification_Sto,
          (SELECT code_description
          FROM ms_aud_lov_config
          where CODE_GROUP = 'MS AUD Classification'
          AND code_code    = a.audit_classification
          ) AUDIT_CLASSIFICATION_DISP,
         a.audit_type Audit_Type_Sto,
          (SELECT code_description
          FROM ms_aud_lov_config
          where CODE_GROUP = 'MS AUD Audit Type'
          AND code_code    = a.audit_type
          ) AUDIT_TYPE_DISP,
          a.AUDIT_STATUS AUDIT_STATUS_STO,
          MS_AUD_UTILITIES.FN_GET_STATUS_DISPLAY_NAME (a.AUDIT_STATUS) AUDIT_STATUS_DISP,
          a.LEAD_AUDITOR LEAD_AUDITOR_STO,
          MS_APPS_UTILITIES.GET_USER_FULL_NAME (a.LEAD_AUDITOR) LEAD_AUDITOR_DISP,
          a.AUDIT_MANAGER AUDIT_MANAGER,
          MS_APPS_UTILITIES.GET_USER_FULL_NAME (a.AUDIT_MANAGER) AUDIT_MANAGER_DISP,
          MS_AUD_UTILITIES.FN_GET_MLOV_VALUES (a.AUDIT_ID, 'lead_auditor_mlov', 'ms_aud_audit_alm' ) LEAD_AUDITOR_MLOV_STO,
          MS_AUD_UTILITIES.FN_GET_MLOV_LEAD_AUD_DISP (MS_AUD_UTILITIES.FN_GET_MLOV_VALUES (a.AUDIT_ID, 'lead_auditor_mlov', 'ms_aud_audit_alm' ) ) LEAD_AUDITOR_MLOV_DISP,
          MS_AUD_UTILITIES.FN_GET_MLOV_VALUES (a.AUDIT_ID, 'audit_manager_mlov', 'ms_aud_audit_amm' ) AUDIT_MANAGER_MLOV_STO,
          MS_AUD_UTILITIES.FN_GET_MLOV_AUD_MAN_DISP (MS_AUD_UTILITIES.FN_GET_MLOV_VALUES (a.AUDIT_ID, 'audit_manager_mlov', 'ms_aud_audit_amm' ) ) AUDIT_MANAGER_MLOV_DISP,
          (SELECT SOURCE_ID
          from MS_AUD_TRIGGER_API_LOG
          where audit_id = a.audit_id) REFERENCE_ID,
          (SELECT SOURCE_MODULE
          from MS_AUD_TRIGGER_API_LOG
          where audit_id = a.audit_id) Source_module
  FROM ms_aud_audit a
