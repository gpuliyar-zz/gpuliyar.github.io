                                                        SELECT   OBJECT_NAME AS DISPLAY_NAME, OBJECT_ID AS OBJECT_ID,
         DD_OBJECT_TYPE  OBJECT_TYPE, 'Y' AS OVER_RIDE_ACTION, MANUAL_ENTRY_AUTOMATED,RISING_DROPPING_INDICATOR,KPI_KRI_FLAG,(SELECT MS_CONCAT(OWNER_ORGANIZATIONS) FROM MS_MET_DEFINITION_OGN C WHERE C.OBJECT_ID=A.OBJECT_ID) AS OWNER_ORGANIZATIONS,
            'PROC='
         || TO_CHAR (DD_PROCESS_INSTANCE_ID)
         || '|INSTID='
         || TO_CHAR (DD_INSTANCE_ID)
         || '|EDIT_FLAG='
         || NVL (DD_EDIT_FLAG, 'Y') AS EDIT_ACTION_PARAMETERS,         VALID_FROM,VALID_UNTIL,MS_MET_UTILITIES.INCLUDE_EXPIRED(VALID_FROM,VALID_UNTIL) AS EXPIRED_FLAG
    FROM MS_MET_DEFINITION A
  WHERE  MS_MET_UTILITIES.MS_MET_SECURITY(DD_OBJECT_TYPE, RESTRICT_ACCESS_TO, (SELECT MS_CONCAT(OWNER_ORGANIZATIONS) FROM MS_MET_DEFINITION_OGN B WHERE B.OBJECT_ID=A.OBJECT_ID), :1, 'VIEW') = 'ACCESS'
 AND TRUNC(SYSDATE) BETWEEN NVL(TRUNC(VALID_FROM),TRUNC(SYSDATE)) AND NVL(TRUNC(VALID_UNTIL),TRUNC(SYSDATE))
  ORDER BY OBJECT_NAME
