                                                        SELECT '<INPUT TYPE=CHECKBOX ID="' || OBJECT_ID ||
       '" VALUE="' || OBJECT_NAME || '"ONCLICK=CHECKBOXCHANGE("' ||
       OBJECT_ID || '")></INPUT>' AS CHECKBOX,
       OBJECT_ID, SUBSTR(OBJECT_NAME,1,3700) AS OBJECT_NAME FROM (
  SELECT OBJECT_ID, SUBSTR(OBJECT_NAME,1,3700) AS OBJECT_NAME, DD_OBJECT_TYPE, RESTRICT_ACCESS_TO, (SELECT MS_CONCAT(
                     OWNER_ORGANIZATIONS)
             FROM MS_GRC_RISK_OGN T
            WHERE A.OBJECT_ID = T.OBJECT_ID) OWNER_ORGANIZATIONS,VALID_FROM,VALID_UNTIL  FROM MS_GRC_RISK A
  WHERE :1 = 2
  UNION ALL
  SELECT OBJECT_ID, SUBSTR(OBJECT_NAME,1,3700) AS OBJECT_NAME, DD_OBJECT_TYPE, RESTRICT_ACCESS_TO, (SELECT MS_CONCAT(
                     OWNER_ORGANIZATIONS)
             FROM MS_GRC_CORE_OBJECT_OGN T
            WHERE B.OBJECT_ID = T.OBJECT_ID) OWNER_ORGANIZATIONS,VALID_FROM,VALID_UNTIL  FROM MS_GRC_CORE_OBJECT B
  WHERE :1 = 3
  UNION ALL
  SELECT OBJECT_ID, SUBSTR(OBJECT_NAME,1,3700) AS OBJECT_NAME, DD_OBJECT_TYPE, RESTRICT_ACCESS_TO, (SELECT MS_CONCAT(
                     OWNER_ORGANIZATIONS)
             FROM MS_GRC_COMPLIANCE_OBJE_OGN T
            WHERE C.OBJECT_ID = T.OBJECT_ID) OWNER_ORGANIZATIONS,VALID_FROM,VALID_UNTIL  FROM MS_GRC_COMPLIANCE_OBJE C
  WHERE :1 = 4
  )
  WHERE  SYSDATE BETWEEN NVL (VALID_FROM, SYSDATE) AND NVL (VALID_UNTIL,
                                                            SYSDATE
                                                           )
     AND
  MS_MET_UTILITIES.MS_MET_SECURITY(DD_OBJECT_TYPE,
                                                  RESTRICT_ACCESS_TO,
                                                  OWNER_ORGANIZATIONS,
                                                  :2,
                                                  'VIEW') = 'ACCESS'
  ORDER BY OBJECT_NAME
