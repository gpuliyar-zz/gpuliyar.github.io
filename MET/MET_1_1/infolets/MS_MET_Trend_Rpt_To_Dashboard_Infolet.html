                                                        SELECT NAME,OBJECT_ID,OBJECT_NAME,ATI_OBJECT_ID,FREQUENCY,ROUND(EXPECTED_VALUE,2) AS EXPECTED_VALUE,ROUND(PREVIOUS_VALUE,2) AS PREVIOUS_VALUE,ROUND(CURRENT_VALUE,2) AS CURRENT_VALUE,CURRENT_DATE_VALUE,CHANGE,ROUND(LAST_3ASMNT,2) AS LAST_3ASMNT,ROUND(LAST_6ASMNT,2) AS LAST_6ASMNT,ROUND(LAST_12ASMNT,2) AS LAST_12ASMNT, MS_APPS_UTILITIES.GET_DISPLAY_VALUE(MS_MET_UTILITIES.ACTIVE_ENT_ID,'MS MET KI APPLIES TO ITEMS',APPLIES_TO)  AS APPLIES_TO ,RISK_CATEGORIES,
MS_APPS_UTILITIES.GET_DISPLAY_VALUES (MS_MET_UTILITIES.ACTIVE_ENT_ID,'ORGENTITY',ASSESSED_ORGS,','),
(SELECT MS_MET_UTILITIES.FN_KPI_KRI_PREVAL_COLOR_CODE
                                         (OBJECT_ID,
                                          EVENT_ID
                                         )
          FROM DUAL) AS PREV_COLOR_CODE,
          (SELECT MS_MET_UTILITIES.FN_KPI_KRI_CURVAL_COLOR_CODE
                                           (OBJECT_ID,
                                            EVENT_ID
                                           )
          FROM DUAL) AS CURR_COLOR_CODE,
          (SELECT MS_MET_UTILITIES.FN_KPI_KRI_CURVAL_COLOR_CODE
                                           (OBJECT_ID,
                                            EVENT_ID
                                           )
          FROM DUAL) AS CHANGE_COLOR_CODE,METRIC_DEFN_URL,CHANGE_NUMBER
  FROM
(SELECT NAME,OBJECT_ID,OBJECT_NAME,ATI_OBJECT_ID,FREQUENCY,EXPECTED_VALUE,PREVIOUS_VALUE,CURRENT_VALUE,CURRENT_DATE_VALUE,CHANGE,LAST_3ASMNT,LAST_6ASMNT,LAST_12ASMNT,APPLIES_TO,EVENT_ID,(SELECT MS_MET_REPORTBROWSER_FILTERS.GET_RISK_CATEGORY(:1,
								  RISK_CATEGORIES) FROM DUAL ) AS RISK_CATEGORIES,ASSESSED_ORGS, METRIC_DEFN_URL,CHANGE_NUMBER FROM MS_MET_DBOARD_ORG_V
WHERE ms_met_security.check_met_form_access(DD_OBJECT_TYPE, RESTRICT_ACCESS_TO, OWNER_ORGANIZATIONS, :1, 'VIEW') = 'ACCESS'
UNION
SELECT NAME,OBJECT_ID,OBJECT_NAME,ATI_OBJECT_ID,FREQUENCY,EXPECTED_VALUE,PREVIOUS_VALUE,CURRENT_VALUE,CURRENT_DATE_VALUE,CHANGE,LAST_3ASMNT,LAST_6ASMNT,LAST_12ASMNT,APPLIES_TO,EVENT_ID,(SELECT MS_MET_REPORTBROWSER_FILTERS.GET_RISK_CATEGORY(:1,
								  RISK_CATEGORIES) FROM DUAL ) AS RISK_CATEGORIES,ASSESSED_ORGS,METRIC_DEFN_URL,CHANGE_NUMBER  FROM MS_MET_DBOARD_RISK_V
WHERE ms_met_security.check_met_form_access(DD_OBJECT_TYPE, RESTRICT_ACCESS_TO, OWNER_ORGANIZATIONS, :1, 'VIEW') = 'ACCESS'
UNION
SELECT NAME,OBJECT_ID,OBJECT_NAME,ATI_OBJECT_ID,FREQUENCY,EXPECTED_VALUE,PREVIOUS_VALUE,CURRENT_VALUE,CURRENT_DATE_VALUE,CHANGE,LAST_3ASMNT,LAST_6ASMNT,LAST_12ASMNT,APPLIES_TO,EVENT_ID,(SELECT MS_MET_REPORTBROWSER_FILTERS.GET_RISK_CATEGORY(:1,
								  RISK_CATEGORIES) FROM DUAL ) AS RISK_CATEGORIES,ASSESSED_ORGS,METRIC_DEFN_URL,CHANGE_NUMBER  FROM MS_MET_DBOARD_COROBJ_V
WHERE ms_met_security.check_met_form_access(DD_OBJECT_TYPE, RESTRICT_ACCESS_TO, OWNER_ORGANIZATIONS, :1, 'VIEW') = 'ACCESS'
UNION
SELECT NAME,OBJECT_ID,OBJECT_NAME,ATI_OBJECT_ID,FREQUENCY,EXPECTED_VALUE,PREVIOUS_VALUE,CURRENT_VALUE,CURRENT_DATE_VALUE,CHANGE,LAST_3ASMNT,LAST_6ASMNT,LAST_12ASMNT,APPLIES_TO,EVENT_ID,(SELECT MS_MET_REPORTBROWSER_FILTERS.GET_RISK_CATEGORY(:1,
								  RISK_CATEGORIES) FROM DUAL ) AS RISK_CATEGORIES,ASSESSED_ORGS,METRIC_DEFN_URL,CHANGE_NUMBER  FROM MS_MET_DBOARD_CTRL_V
WHERE ms_met_security.check_met_form_access(DD_OBJECT_TYPE, RESTRICT_ACCESS_TO, OWNER_ORGANIZATIONS, :1, 'VIEW') = 'ACCESS')
WHERE CURRENT_DATE_VALUE >=MS_MET_UTILITIES.GET_START_DATE_VAL(OBJECT_ID,ATI_OBJECT_ID,:3,MS_MET_UTILITIES.GET_END_DATE_VAL(OBJECT_ID,ATI_OBJECT_ID,:2))
AND CURRENT_DATE_VALUE <=MS_MET_UTILITIES.GET_END_DATE_VAL(OBJECT_ID,ATI_OBJECT_ID,:2) ORDER BY NAME
