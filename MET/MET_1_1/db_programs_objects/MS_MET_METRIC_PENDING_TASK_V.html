MS_MET_METRIC_PENDING_TASK_V 

                                                
  CREATE OR REPLACE FORCE VIEW "METRICS_DEV"."MS_MET_METRIC_PENDING_TASK_V" ("OBJECT_ID", "OBJECT_NAME_URL", "OBJECT_NAME", "EVENT_ID", "DD_OBJECT_TYPE", "STATUS", "STATUS_STORED", "DUE_BY", "PAST_DUE_DATE", "PREVIOUS_VALUE", "CURRENT_VALUE", "INDI_MEASURE_DIS_VAL", "INDI_MEASURE_STORED", "FREQUENCY", "VALIDITY", "VALID_FROM", "STORED_RISK_CATEGORY", "RISK_CATEGORY", "ASSESSMENT_TYPE", "ASSESSMENT_TYPE_STORED", "ASSESSMENT_FOR", "ASSESSMENT_FOR_STORED", "METRIC_TYPE", "METRIC_TYPE_STORED", "DATA_COLL_TYPE", "CRITICAL_RISING_THRESH", "CRITICAL_DROPPING_THRESH", "NON_CRITICAL_RISING_THRESH", "NON_CRITICAL_DROPPING_THRESH", "EXPECTED_VALUE", "OWN_ORGNIZATIONS", "OWNERS", "PROCESS_INSTANCE_ID", "INSTANCE_ID", "CURRENT_ASSIGNEE_STORED", "CURRENT_ASSIGNEE", "PREV_CLR", "CUR_CLR") AS 
  SELECT b.object_id,
    NULL AS object_name_url,
    a.object_name,
    a.event_id,
    --(SELECT MS_MET_utilities.Get_Edit_Url(a.DD_OBJECT_TYPE, a.PROCESS_INSTANCE_ID, a.iNSTANCE_ID,' ',a.event_id)
    --FROM dual  )
    A.dd_object_type,
    DECODE( A.OBJ_STATUS,'M_REASSIGN','Data Entry - Pending','M_CLR_PEND','Data Entry - Pending', MS_APPS_UTILITIES.GET_DISPLAY_VALUE (MS_MET_UTILITIES.ACTIVE_ENT_ID ,'MS_MET_Metrics_Status_Custom', A.OBJ_STATUS)) AS STATUS,
    DECODE(a.OBJ_STATUS,'M_CLR_PEND','MET_PEND','M_REASSIGN','MET_PEND',a.OBJ_STATUS)                                                                                                                                 AS status_stored,
    (SELECT TRUNC(a.NEXT_SCHEDULED_DATE)FROM dual
    ) AS DUE_BY,
    (SELECT TRUNC(SYSDATE - a.NEXT_SCHEDULED_DATE)FROM dual
    ) AS past_due_date,
    a.previous_value,
    a.current_value,
    ms_APPS_utilities.get_DISPLAY_VALUE(MS_MET_UTILITIES.ACTive_ENT_ID ,'MS MET KI Indicator Measurement',A.INDICATOR_MEASUREMENT) AS indi_measure_dis_val,
    A.INDICATOR_MEASUREMENT                                                                                                        AS indi_measure_stored,
    MS_APPS_UTILITIES.GET_DISPLAY_VALUE (MS_MET_UTILITIES.ACTIVE_ENT_ID , 'MS MET KI Frequency', A.FREQUENCY )                     AS FREQUENCY,
    B.VALID_UNTIL                                                                                                                  AS VALIDITY,
    b.valid_from                                                                                                                   AS valid_from,
    a.Risk_category                                                                                                                AS stored_risk_category,
    (SELECT MS_MET_utilities.get_risk_category_comma_sep(a.Risk_category)
    FROM dual
    )                                                                                                              AS Risk_Category,
    MS_APPS_UTILITIES.GET_DISPLAY_VALUE(MS_MET_UTILITIES.ACTIVE_ENT_ID,'MS MET KI Applies To Items',A.APPLIES_TO ) AS ASSESSMENT_TYPE,
    a.applies_to                                                                                                   as assessment_type_stored,
    (SELECT MS_MET_OBJECT_utilities.get_metrics_ati_name (A.applies_to,A.ati_object_id )
    FROM DUAL
    )               AS ASSESSMENT_FOR,
    a.ati_object_id AS assessment_for_stored,
    (SELECT ms_apps_utilities.get_display_value(MS_MET_UTILITIES.ACTive_ENT_ID,'MS MET KI Type',b.kpi_kri_flag)
    FROM dual
    )                                                                                                                                           AS metric_type,
    b.kpi_kri_flag                                                                                                                              AS metric_type_stored,
    NVL(ms_apps_utilities.get_display_value (MS_MET_UTILITIES.ACTive_ENT_ID,'MS MET KI ATI Manual Automated',b.MANUAL_ENTRY_AUTOMATED ),'Both') AS data_coll_type,
    a.CRITICAL_RISING_THRESH,
    a.CRITICAL_DROPPING_THRESH,
    a.NON_CRITICAL_RISING_THRESH,
    a.NON_CRITICAL_DROPPING_THRESH,
    a.EXPECTED_VALUE,
    (SELECT (RTRIM(XMLAGG(XMLELEMENT(e,
      (SELECT org_entity_name
      FROM SI_ORG_ENTITIES_T
      WHERE org_entity_id = g.owner_organizations
      )
      || ',' )).EXTRACT ('//text()'),','))
    FROM MS_MET_DEFINITION_ogn g
    WHERE G.OBJECT_ID = A.OBJECT_ID
    ) AS OWN_ORGNIZATIONS,
    (SELECT (RTRIM(XMLAGG(XMLELEMENT(E,
      (SELECT FIRST_NAME||' '||LAST_NAME FROM SI_USERS_T WHERE user_name=g.owners
      )
      || ',' )).EXTRACT ('//text()'),',' ))
    FROM MS_MET_DEFINITION_own g
    WHERE g.object_id = a.object_id
    ) AS owners,
    a.process_instance_id,
    a.instance_id,
    (SELECT (RTRIM(XMLAGG(XMLELEMENT(e,
      (SELECT user_name FROM si_users_t WHERE user_id=c.user_id
      )
      || ',' )).EXTRACT ('//text()'),',' ))
    FROM si_event_assignments c
    WHERE c.process_instance_id = a.process_instance_id
    ) AS current_assignee_stored,
    (SELECT (RTRIM(XMLAGG(XMLELEMENT(e,
      (SELECT first_name||' '||last_name FROM si_users_t WHERE user_id=c.user_id
      )
      || ',' )).EXTRACT ('//text()'),',' ))
    FROM si_event_assignments c
    WHERE c.process_instance_id = A.process_instance_id
    ) AS current_assignee,
    (SELECT MS_MET_UTILITIES.FN_KPI_KRI_PREVAL_COLOR_CODE(B.OBJECT_ID, A.EVENT_ID)
    FROM DUAL
    ) AS PREV_CLR,
    (SELECT MS_MET_Utilities.Fn_Kpi_Kri_Curval_Color_Code(b.object_id, a.event_id)
    FROM Dual
    ) AS cur_Clr
  FROM MS_MET_data_entry_v A,
    MS_MET_DEFINITION b
  WHERE a.process_instance_id IN
    (SELECT process_instance_id
    FROM si_event_assignments
      --WHERE user_id   = ms_apps_utilities.get_user_id(:LOGIN_ID)
    WHERE infolet_id IN
      (SELECT metric_id
      FROM si_metrics_t
      WHERE metric_name = 'MS_MET_DATA_ENTRY'
      )
    AND a.instance_id =
      (SELECT MAX (instance_id)
      FROM MS_MET_data_entry_v
      WHERE process_instance_id = a.process_instance_id
      )
    )
  AND A.OBJECT_ID=B.OBJECT_ID
