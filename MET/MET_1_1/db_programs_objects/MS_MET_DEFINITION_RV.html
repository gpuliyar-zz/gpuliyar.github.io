MS_MET_DEFINITION_RV 

                                                
  CREATE OR REPLACE FORCE VIEW "METRICS_DEV"."MS_MET_DEFINITION_RV" ("OBJECT_ID", "OBJECT_NAME", "OWNERS", "OWNER_ORG_IDS", "VALID_UNTIL", "VALID_FROM", "OBJ_STATUS", "OBJECT_URL", "MANUAL_ENTRY_AUTOMATED", "INDICATOR_MEASUREMENT", "ASSEMENT_TYPE", "FREQUENCY", "DESCRIPTION", "DD_OBJECT_TYPE", "DISPLAY_NAME", "DD_PROCESS_INSTANCE_ID", "DD_INSTANCE_ID", "DD_EDIT_FLAG", "RESTRICT_ACCESS_TO", "ASSEMENT_FOR", "OBJ_MODIFIED_ON", "OBJ_MODIFIED_BY", "KPI_KRI_FLAG", "RISK_CATEGORIES", "ATI_OWNER", "RELEATED_CONTROLS", "RELEATED_CORE_OBJECTS", "RELEATED_ORGANIZATIONS", "RELEATED_RISKS", "OVER_RIDE_ACTION", "EDIT_ACTION_PARAMETERS", "EXPIRED_FLAG") AS 
  SELECT object_id,
    trim(object_name) object_name,
    (SELECT ms_concat(usr.first_name
      || ' '
      || usr.last_name) AS owners
    FROM ms_met_definition_own d,
      si_users_t usr ,
      si_ent e
    WHERE usr.user_name   = d.owners
    AND usr.enterprise_id = e.ENTERPRISE_ID
    AND a.OBJECT_ID       = D.OBJECT_ID
    ) AS OWNERS,
    (SELECT ms_concat(owner_organizations)
    FROM ms_met_definition_ogn d
    WHERE a.object_id = d.object_id
    ) AS owner_org_ids,
    valid_until,
    valid_from,
    obj_status,
    ms_met_utilities.get_edit_url (dd_object_type, dd_process_instance_id, dd_instance_id, NVL (dd_edit_flag, 'Y'), a.OBJECT_NAME) AS OBJECT_URL,
    manual_entry_automated,
    /*ms_apps_utilities.get_display_value ( ms_met_utilities.active_ent_id, 'MS MET KI Manual Automated', manual_entry_automated )*/
    indicator_measurement,
    /*ms_apps_utilities.get_display_value ( ms_met_utilities.active_ent_id, 'MS MET KI Indicator Measurement', indicator_measurement )*/
    assement_for AS assement_type ,
    /* ms_apps_utilities.get_display_value ( ms_met_utilities.active_ent_id, 'MS MET KI Applies To Items', assement_for )*/
    frequency,
    /*ms_apps_utilities.get_display_value ( ms_met_utilities.active_ent_id, 'MS MET KI Frequency', frequency )*/
    description,
    dd_object_type,
    SUBSTR (trim(object_name), 1, 30) display_name,
    dd_process_instance_id,
    dd_instance_id,
    NVL (dd_edit_flag, 'Y') AS dd_edit_flag,
    restrict_access_to,
    (SELECT ms_concat(ms_met_object_utilities.get_metrics_ati_name ( a.assement_for, k.ati_obj_id ))
    FROM ms_met_definition_ati k
    WHERE a.object_id = k.object_id
    )                                     AS ASSEMENT_FOR,
    TO_CHAR(OBJ_MODIFIED_ON,'MM/DD/YYYY') AS OBJ_MODIFIED_ON,
    OBJ_MODIFIED_by,
    KPI_KRI_FLAG,
    /*ms_apps_utilities.get_display_value ( ms_met_utilities.active_ent_id, 'MS MET KI Type', KPI_KRI_FLAG )*/
    (
    SELECT ms_concat(risk_category)--ms_concat(ms_apps_utilities.get_display_values ( ms_met_utilities.active_ent_id, 'MS MET Risk Category', risk_category, ',' ))
    FROM MS_MET_DEFINITION_RCG e
    WHERE e.object_id = a.object_id
    ) AS RISK_CATEGORIES,
    (SELECT ms_concat(usr.first_name
      || ' '
      || usr.last_name)
    FROM MS_MET_DEFINITION_ATI s,
      si_users_t usr ,
      si_ent e
    WHERE usr.user_name   = s.ati_owner
    AND usr.enterprise_id = e.ENTERPRISE_ID
    AND a.OBJECT_ID       = s.OBJECT_ID
    ) AS ATI_OWNER,
    (SELECT MS_GRC_FN_CONCAT(orb.tgt_obj_id)
    FROM ms_met_orb_relations_pub_v orb,
      ms_grc_compliance_obje base
    WHERE src_obj_type   = 'MS_MET_DEFINITION'
    AND orb.tgt_obj_type = 'MS_GRC_CONTROL'
    AND orb.src_obj_id   = a.object_id
    AND orb.tgt_obj_id   = base.object_id
    ) AS releated_controls,
    (SELECT MS_GRC_FN_CONCAT(orb.tgt_obj_id)
    FROM ms_met_orb_relations_pub_v orb,
      ms_grc_core_object base
    WHERE src_obj_type   = 'MS_MET_DEFINITION'
    AND orb.tgt_obj_type = 'MS_GRC_PROCESS'
    AND orb.src_obj_id   = a.object_id
    AND orb.tgt_obj_id   = base.object_id
    ) AS releated_core_objects,
    (SELECT MS_GRC_FN_CONCAT(orb.tgt_obj_id)
    FROM ms_met_orb_relations_pub_v orb,
      si_org_entities_t base
    WHERE src_obj_type   = 'MS_MET_DEFINITION'
    AND orb.tgt_obj_type = 'MS_GRC_ORGANIZATION'
    AND orb.src_obj_id   = a.object_id
    AND orb.tgt_obj_id   = base.org_entity_id
    ) AS releated_organizations,
    (SELECT MS_GRC_FN_CONCAT(orb.tgt_obj_id)
    FROM ms_met_orb_relations_pub_v orb,
      ms_grc_risk base
    WHERE src_obj_type   = 'MS_MET_DEFINITION'
    AND orb.tgt_obj_type = 'MS_GRC_RISK'
    AND orb.src_obj_id   = a.object_id
    AND orb.tgt_obj_id   = base.object_id
    ) releated_risks,
    'Y' AS OVER_RIDE_ACTION,
    'proc='
    || TO_CHAR(dd_process_instance_id)
    ||'|instid='
    ||TO_CHAR(dd_instance_id)
    ||'|edit_flag='
    ||NVL(dd_edit_flag,'Y')                                      AS EDIT_ACTION_PARAMETERS,
    ms_met_utilities.include_expired(a.valid_from,a.valid_until) as expired_flag
  FROM ms_met_definition a
