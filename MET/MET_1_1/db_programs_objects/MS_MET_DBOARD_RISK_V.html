MS_MET_DBOARD_RISK_V 

                                                
  CREATE OR REPLACE FORCE VIEW "METRICS_DEV"."MS_MET_DBOARD_RISK_V" ("NAME", "OBJECT_ID", "OBJECT_NAME", "ATI_OBJECT_ID", "FREQUENCY", "EXPECTED_VALUE", "PREVIOUS_VALUE", "CURRENT_VALUE", "CURRENT_DATE_VALUE", "CHANGE", "LAST_3ASMNT", "LAST_6ASMNT", "LAST_12ASMNT", "APPLIES_TO", "DD_OBJECT_TYPE", "RESTRICT_ACCESS_TO", "OWNER_ORGANIZATIONS", "EVENT_ID", "RISK_CATEGORIES", "ASSESSED_ORGS", "METRIC_DEFN_URL", "CHANGE_NUMBER") AS 
  SELECT b.object_name AS NAME,
    a.de_object_id,
    c.object_name AS object_name,
    a.ati_object_id,
    ms_apps_utilities.get_display_value (ms_met_object_utilities.get_ent_id (), 'MS GRC KI Frequency', frequency ) frequency,
    a.expected_value,
    a.previous_value,
    a.current_value,
    a.current_date_value,
    ms_met_object_utilities.for_breached_rpt (current_value, previous_value )                AS CHANGE,
    ms_met_object_utilities.fn_avg (a.current_date_value, a.ati_object_id, a.de_object_id, 3 )  AS last_3asmnt,
    ms_met_object_utilities.fn_avg (a.current_date_value, a.ati_object_id, a.de_object_id, 6 )  AS last_6asmnt,
    ms_met_object_utilities.fn_avg (a.current_date_value, a.ati_object_id, a.de_object_id, 12 ) AS last_12asmnt,
    applies_to,
    b.dd_object_type,
    b.restrict_access_to,
    (SELECT ms_concat (owner_organizations)
    FROM ms_met_definition_ogn d
    WHERE a.de_object_id = d.object_id
    ) AS owner_organizations,
    event_id,
    (SELECT ms_concat(risk_category)
    FROM ms_met_definition_RCG e
    WHERE e.object_id = b.object_id
    )                                                                                                                                  AS RISK_CATEGORIES,
    g.ati_assessed_orgs                                                                                                                AS ASSESSED_ORGS,
    Ms_met_Utilities.Get_Edit_Url(b.DD_OBJECT_TYPE, b.dd_process_instance_id,b.dd_instance_id , NVL(b.dd_edit_flag,'Y'),b.object_name) AS Metric_defn_url,
    ROUND((current_value - previous_value),2)                                                                                          AS Change_number
  FROM ms_met_data_entry a
    /*,
    ms_met_definition b,
    ms_grc_risk c,
    ms_met_definition_aor g,
    ms_met_definition_ati t*/
  JOIN ms_met_definition b
  ON a.de_object_id = b.object_id
  JOIN ms_grc_risk c
  ON c.object_id = a.ati_object_id
  JOIN ms_met_definition_ati t
  ON t.object_id   = a.de_object_id
  AND t.ati_obj_id = a.ati_object_id
  LEFT JOIN ms_met_definition_aor g
  ON g.ati_object_id_pk     = t.ati_object_id_pk
  WHERE current_date_value IN
    (SELECT MAX (current_date_value)
    from ms_met_data_entry
    WHERE de_object_id   = a.de_object_id
    AND ati_object_id = a.ati_object_id
    )
    /*AND a.de_object_id = b.object_id
    AND c.object_id = a.ati_object_id
    and t.object_id= g.object_id
    and t.ati_object_id_pk=g.ati_object_id_pk
    and a.ati_object_id=t.ati_obj_id
    AND b.object_id =g.object_id(+)*/
  AND applies_to = '2'
