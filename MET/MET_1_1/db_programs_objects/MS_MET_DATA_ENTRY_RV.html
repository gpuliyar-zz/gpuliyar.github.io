MS_MET_DATA_ENTRY_RV 

                                                
  CREATE OR REPLACE FORCE VIEW "METRICS_DEV"."MS_MET_DATA_ENTRY_RV" ("METRIC_DEFINITION_ID", "METRIC_DEFN_NAME", "EVENT_ID", "STATUS", "ASSESSMENT_TYPE", "DUE_BY", "ASSIGNED_TO", "INDICATOR_MEASUREMENT", "OWNER_ORG_IDS", "RESTRICT_ACCESS_TO", "DATA_ENTRY_OWNER", "DATA_ENTRY_APPROVER", "DATA_DEFINITION_OWNER", "CURRENT_VALUE", "NON_CRITICAL_RISING_THRESH", "CRITICAL_RISING_THRESH", "NON_CRITICAL_DROPPING_THRESH", "CRITICAL_DROPPING_THRESH", "CURRENT_DATE_VALUE", "ATI_OBJECT_ID", "ASSESSMENT_FOR", "ATI_COMMENTS", "EXPECTED_VALUE", "FREQUENCY", "RISING_DROPPING_INDICATOR", "PROCESS_INSTANCE_ID", "DD_OBJECT_TYPE", "PREVIOUS_VALUE", "PAST_DUE_DATE", "DAYS_PAST_DUE", "DD_EDIT_FLAG", "METRIC_EVENT_ID", "COLOR_CODE", "CURRENT_PERIOD_AS_TEXT", "PREVIOUS_PERIOD", "CLR_CODE", "DATA_COLLECTION_TYPE", "PROVIDED_ON", "CREATION_DATE", "FREQUENCY_STOREDVAL", "NEXT_SCHEDULED_DATE", "PID_DEFN", "INSTANCEID_DEFN", "DD_EDIT_FLAG_DEFN", "DEFN_OBJECT_TYPE", "DEFN_PID", "DEFN_INSTID", "DATA_EN_PID", "DATA_INST_ID", "DATA_EN_OBJECT_TYPE", "DATA_EN_EDIT_FLAG", "RISK_CATEGORIES", "EVENT_URL", "EDIT", "ASSESSED_ORGS", "ASSESSED_ORG_IDS") AS 
  SELECT B.object_id AS METRIC_DEFINITION_ID ,
    b.object_name    AS METRIC_DEFN_NAME,
    A.EVENT_ID       AS EVENT_ID,
    (
    CASE
      WHEN a.CURRENT_DATE_VALUE < TRUNC(sysdate)
      AND a.OBJ_STATUS         <> 'MET_COM'
      THEN 'M_DUE'
      ELSE a.OBJ_STATUS
    END )                AS status,
    a.applies_to         AS ASSESSMENT_TYPE,
    A.CURRENT_DATE_VALUE AS DUE_BY,
    (SELECT (RTRIM(XMLAGG(XMLELEMENT(e,
      (SELECT user_name FROM si_users_t WHERE user_id=c.user_id
      )
      || ',' )).EXTRACT ('//text()'),',' ))
    FROM si_event_assignments c
    WHERE c.process_instance_id = A.process_instance_id
    )                       AS assigned_to,
    b.indicator_measurement AS INDICATOR_MEASUREMENT,
    a.OWNER_ORGANIZATIONS   AS owner_org_ids,
    a.RESTRICT_ACCESS_TO,
    (SELECT ms_concat(DISTINCT(usr.user_name)) AS DATA_ENTRY_OWNER
    FROM si_users_t usr,
      si_ent e
    WHERE usr.user_name = a.DATA_ENTRY_OWNER
    AND a.object_id     = b.object_id
    AND e.enterprise_id = usr.enterprise_id
    ) AS DATA_ENTRY_OWNER,
    (SELECT ms_concat(DISTINCT(usr.user_name)) AS DATA_ENTRY_APPROVER
    FROM si_users_t usr,
      si_ent e
    WHERE usr.user_name = a.data_entry_approver
    AND a.object_id     = b.object_id
    AND e.enterprise_id = usr.enterprise_id
    ) AS DATA_ENTRY_APPROVER,
    /* (
    SELECT ms_concat(distinct(usr.user_name)) AS data_defn_owners
    FROM MS_MET_DEFINITION_own d, si_users_t usr, si_ent e
    WHERE usr.user_name = d.owners
    AND usr.enterprise_id = e.ENTERPRISE_ID
    AND a.OBJECT_ID = D.OBJECT_ID
    and a.object_id = b.object_id
    )*/
    a.DEFINITION_OWNERS                     AS DATA_DEFINITION_OWNER,
    ROUND(a.current_value,2)                AS current_value,
    ROUND(a.non_critical_rising_thresh,2)   AS non_critical_rising_thresh,
    ROUND(a.critical_rising_thresh,2)       AS critical_rising_thresh,
    ROUND(a.non_critical_dropping_thresh,2) AS non_critical_dropping_thresh,
    ROUND(a.critical_dropping_thresh,2)     AS critical_dropping_thresh,
    a.current_date_value,
    a.ati_object_id,
    (SELECT ms_met_object_utilities.get_metrics_ati_name (a.applies_to, a.ati_object_id )
    FROM DUAL
    ) AS ASSESSMENT_FOR,
    a.ati_comments,
    ROUND(a.expected_value,2) AS expected_value,
    a.frequency,
    a.rising_dropping_indicator,
    a.PROCESS_INSTANCE_ID,
    --  a.INSTANCE_ID, --a.dd_instance_id,
    a.dd_object_type,
    ROUND(a.Previous_Value,2) AS PREVIOUS_VALUE,
    DECODE(SIGN(TRUNC(A.CURRENT_DATE_VALUE -sysdate)),1,'Days Past Due is'
    ||TO_CHAR(TRUNC(A.CURRENT_DATE_VALUE ) - TRUNC(sysdate),'99')
    ||' days',                             -1,'Days Past Due is'
    ||TO_CHAR(TRUNC(A.CURRENT_DATE_VALUE)  -TRUNC(sysdate),'99')
    ||' days', 0,'Days Past Due is'
    ||TO_CHAR(TRUNC(A.CURRENT_DATE_VALUE)-TRUNC(sysdate),'99')) AS past_due_date,
    (TRUNC(A.CURRENT_DATE_VALUE)         - TRUNC(sysdate))      AS days_past_due,
    NVL(a.LATEST_FLAG,'Y')                                      AS DD_EDIT_FLAG, -- NVL (a.dd_edit_flag, 'Y')                                   AS DD_EDIT_FLAG,
    a.event_id                                                  AS metric_event_id,
    (SELECT ms_met_utilities.FN_KPI_KRI_preval_COLOR_CODE(B.object_id,a.event_id)
    FROM dual
    ) AS color_code,
    (SELECT ms_met_utilities.GET_VALUE_IN_DATE_FORMAT(current_date_value, B.FREQUENCY)
    FROM dual
    ) AS CURRENT_PERIOD_AS_TEXT,
    (SELECT ms_met_utilities.get_value_in_date_format(previous_period, B.FREQUENCY)
    FROM dual
    ) AS PREVIOUS_PERIOD,
    (SELECT ms_met_utilities.FN_KPI_KRI_CURVAL_color_code(B.object_id,a.event_id)
    FROM dual
    )                        AS clr_code,
    b.manual_entry_automated AS data_collection_type,
    A.PROVIDED_ON,
    a.CREATION_DATE,
    B.FREQUENCY              AS FREQUENCY_STOREDVAL,
    B.NEXT_SCHEDULED_DATE    AS NEXT_SCHEDULED_DATE,
    B.dd_PROCESS_INSTANCE_ID AS PID_DEFN,
    B.dd_instance_id         AS INSTANCEID_DEFN,
    NVL(b.dd_edit_flag,'Y')  AS DD_EDIT_FLAG_DEFN,
    B.DD_OBJECT_TYPE         AS DEFN_OBJECT_TYPE,
    b.dd_process_instance_id AS defn_pid,
    b.dd_instance_id         AS defn_instid,
    a.PROCESS_INSTANCE_ID    AS data_en_pid,         --a.dd_process_instance_id AS data_en_pid,
    a.INSTANCE_ID            AS data_inst_id,        --a.dd_instance_id         AS data_inst_id,
    a.dd_object_type         AS data_en_object_type, --a.dd_edit_flag           AS data_en_edit_flag,
    a.LATEST_FLAG            AS data_en_edit_flag,
    (SELECT ms_concat(risk_category)--ms_concat(ms_apps_utilities.get_display_values ( ms_met_utilities.active_ent_id, 'MS GRC Risk Category', risk_category, ',' ))
    FROM MS_MET_DEFINITION_RCG e
    WHERE e.object_id = b.object_id
    ) AS RISK_CATEGORIES,
    (SELECT ms_met_utilities.Get_Edit_Url(a.dd_object_type, a.process_instance_id,a.instance_id,NVL(a.LATEST_FLAG,'Y'),A.EVENT_ID)
    FROM dual
    ) AS EVENT_URL,--changed dd_edit_flag to LATEST_FLAG, instance_id,process_instance_id
    (SELECT ms_met_utilities.Get_Edit_Url(B.DD_OBJECT_TYPE, b.dd_process_instance_id, B.dd_instance_id , NVL(b.dd_edit_flag,'Y'),b.object_name)
    FROM Dual
    )                      AS Edit,
    a.ATI_ASSESSED_ORGS    AS assessed_orgs,
    a.ATI_ASSESSED_ORG_IDS AS assessed_org_ids
  FROM
    /*ms_grc_kpi_kri_data_en*/
    ms_met_data_entry_v a,
    ms_met_definition b
  WHERE a.instance_id =
    (SELECT MAX (instance_id)
    FROM ms_met_data_entry_v
    WHERE process_instance_id = a.process_instance_id
    )
  AND a.object_id       = b.object_id
  AND a.OBJ_STATUS NOT IN ('M_REJ','M_REASSIGN')
  ORDER BY metric_defn_name,
    due_by
