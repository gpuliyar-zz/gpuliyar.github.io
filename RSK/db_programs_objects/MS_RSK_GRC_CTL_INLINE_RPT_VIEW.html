MS_RSK_GRC_CTL_INLINE_RPT_VIEW 

                                                
  CREATE OR REPLACE FORCE VIEW "NOVT_DEV"."MS_RSK_GRC_CTL_INLINE_RPT_VIEW" ("CHECKBOX", "OBJID_ROWNUM", "OBJECT_ID", "OBJECT_NAME", "C_TYPE", "C_KEY", "C_CATEGORY") AS 
  SELECT     ''   AS checkbox,
          a.object_id || '$' || ROWNUM AS objid_rownum, a.object_id, a.object_name,
          ms_apps_utilities.get_display_values
                                         ('100000',
                                          'MS_GRC_COMPLIANCE_TYPE',
                                          a.compliance_type,','
                                         ) AS c_type,
                ms_apps_utilities.get_display_value
                                    ('100000',
                                     'MS_GRC_COMPLIANCE_KeyControl',
                                     decode(a.key_compliance,'yes','1', 'no','2',null)
                                    ) AS c_key,
         MS_APPS_UTILITIES.GET_DISPLAY_VALUES
                                 ('100000',
                                  'MS_GRC_COMPLIANCE_CATEGORY',
                                  (SELECT MS_CONCAT(P.PURPOSE)  FROM ms_grc_compliance_obje O, MS_GRC_CONTROL_PUR P
where  O.OBJECT_ID=P.OBJECT_ID and O.OBJECT_ID=a.OBJECT_ID),','
                                 ) AS c_category
     FROM ms_grc_compliance_obje a where nvl(VALID_UNTIL,sysdate)>=sysdate
