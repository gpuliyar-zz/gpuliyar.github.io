MS_RSK_CORE_OBJECTS_PR_V 

                                                
  CREATE OR REPLACE FORCE VIEW "NOVT_DEV"."MS_RSK_CORE_OBJECTS_PR_V" ("PROCESS_NAME", "PROCESS_ID", "OWNER_ORGANIZATIONS", "VALID_UNTIL", "VALID_FROM", "DD_OBJECT_TYPE") AS 
  SELECT case
         when b.object_name is null then
          case
         when c.object_name is null then
          d.object_name
         else
          c.object_name
       end else b.object_name end as process_name,
       a.object_id AS process_id,
       A.ato_organization AS OWNER_ORGANIZATIONS,
       b.valid_until AS valid_until,
       B.VALID_FROM AS VALID_FROM,
       case
         when B.DD_OBJECT_TYPE is null then
          case
         when c.dd_object_type is null then
          d.dd_object_type
         ELSE
          C.DD_OBJECT_TYPE
       end else b.dd_object_type end as DD_OBJECT_TYPE
  FROM ms_grc_core_object_org_pr_vw a,
       ms_grc_core_object        b,
       ms_grc_objectives         c,
       ms_grc_product            d
 WHERE a.object_id = b.object_id(+)
   AND A.OBJECT_ID = C.OBJECT_ID(+)
   AND A.OBJECT_ID = D.OBJECT_ID(+)
UNION
 SELECT a.object_name as process_name,
       a.object_id AS process_id,
       b.OWNER_ORGANIZATIONS AS OWNER_ORGANIZATIONS,
       a.valid_until AS valid_until,
       a.VALID_FROM AS VALID_FROM,
        a.DD_OBJECT_TYPE AS DD_OBJECT_TYPE
  FROM MS_GRC_CORE_OBJECT        A ,
  MS_GRC_CORE_OBJECT_OGN B
WHERE A.OBJECT_ID = B.OBJECT_ID(+)
