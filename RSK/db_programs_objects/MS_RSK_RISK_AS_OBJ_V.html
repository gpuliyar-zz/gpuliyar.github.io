MS_RSK_RISK_AS_OBJ_V 

                                                
  CREATE OR REPLACE FORCE VIEW "NOVT_DEV"."MS_RSK_RISK_AS_OBJ_V" ("OBJECT_ID", "OBJECT_NAME", "OWNERS", "OWNER_ORGANIZATIONS", "RESTRICT_ACCESS_TO", "OBJECT_PARENT", "CATEGORY_RISK", "TYPE_RISK", "DD_PROCESS_INSTANCE_ID", "DD_INSTANCE_ID", "ASSOCIATED_OBJECTS", "OBJECT_SEQUENCE", "OBJECT_TYPE", "ASSOCIATED_REPORT", "DD_OBJECT_TYPE", "NO_OF_OBJECTS", "RISK_LEVEL") AS 
  SELECT Risk."OBJECT_ID",
    Risk."OBJECT_NAME",
    Risk."OWNERS",
    Risk."OWNER_ORGANIZATIONS",
    Risk."RESTRICT_ACCESS_TO",
    Risk."OBJECT_PARENT",
    Risk."CATEGORY_RISK",
    Risk."TYPE_RISK",
    Risk."DD_PROCESS_INSTANCE_ID",
    Risk."DD_INSTANCE_ID",
    Risk."ASSOCIATED_OBJECTS",
    Risk."OBJECT_SEQUENCE",
    Risk."OBJECT_TYPE",
    Risk."ASSOCIATED_REPORT",
    Risk."DD_OBJECT_TYPE",
    (LENGTH(associated_objects)-LENGTH(REPLACE(associated_objects,',')))+1 AS no_of_objects,
    Risk.Risk_Level
  FROM
    (SELECT Rsk.object_id ,
      OBJECT_NAME,
      OWNERS ,
      OWNER_ORGANIZATIONS ,
      RESTRICT_ACCESS_TO ,
      object_parent,
      C.CATEGORY_RISK ,
      TYPE_RISK ,
      DD_PROCESS_INSTANCE_ID,
      DD_INSTANCE_ID ,
      associated_objects ,
      OBJECT_SEQUENCE,
      object_type,
      associated_report ,
      dd_object_type,
      Risk_Level
    FROM
      ( WITH Z AS
      (SELECT
        /*+ materialize */
        V.object_id ,
        OBJECT_NAME,
        OWNERS ,
        OWNER_ORGANIZATIONS ,
        RESTRICT_ACCESS_TO ,
        object_parent,
        CATEGORY_RISK ,
        TYPE_RISK ,
        DD_PROCESS_INSTANCE_ID,
        DD_INSTANCE_ID ,
        dd_object_type,
        object_level
      FROM ms_grc_risk_pub_v v
      )
    SELECT object_id ,
      OBJECT_NAME,
      OWNERS ,
      OWNER_ORGANIZATIONS ,
      RESTRICT_ACCESS_TO ,
      OBJECT_PARENT,
      CATEGORY_RISK ,
      TYPE_RISK ,
      DD_PROCESS_INSTANCE_ID,
      DD_INSTANCE_ID ,
      MS_RSK_LND_REPORTING_UTILS.ms_rsk_get_orb_relations(dd_object_type,'MS_GRC_CONTROL',OBJECT_ID ) AS associated_objects ,
      2                                                                                     AS OBJECT_SEQUENCE,
      'Compliance Object'                                                                   AS object_type,
      'MS_GRC_Control_Report'                                                               AS associated_report ,
      dd_object_type,
      Object_Level AS Risk_Level
    FROM Z
    UNION ALL
    SELECT object_id ,
      OBJECT_NAME,
      OWNERS ,
      OWNER_ORGANIZATIONS ,
      RESTRICT_ACCESS_TO ,
      OBJECT_PARENT,
      CATEGORY_RISK ,
      TYPE_RISK ,
      DD_PROCESS_INSTANCE_ID,
      DD_INSTANCE_ID ,
      MS_RSK_LND_REPORTING_UTILS.ms_rsk_get_orb_relations(dd_object_type,'MS_GRC_AREA_OF_COMPLIANCE',OBJECT_ID) AS associated_objects ,
      1                                                                                               AS OBJECT_SEQUENCE,
      'Area Of Compliance'                                                                            AS object_type,
      'MS_GRC_Area_Of_Compliance_Report'                                                              AS ASSOCIATED_REPORT ,
      dd_object_type,
      Object_Level AS Risk_Level
    FROM Z
    UNION ALL
    SELECT object_id ,
      OBJECT_NAME,
      OWNERS ,
      OWNER_ORGANIZATIONS ,
      RESTRICT_ACCESS_TO ,
      OBJECT_PARENT,
      CATEGORY_RISK ,
      TYPE_RISK ,
      DD_PROCESS_INSTANCE_ID,
      DD_INSTANCE_ID ,
      MS_RSK_LND_REPORTING_UTILS.ms_rsk_get_orb_relations(dd_object_type,'MS_GRC_PROCESS',OBJECT_ID) AS associated_objects ,
      3                                                                                    AS OBJECT_SEQUENCE ,
      'Core Object'                                                                        AS object_type,
      'MS_GRC_Process_Report'                                                              AS associated_report ,
      dd_object_type,
      Object_Level AS Risk_Level
    FROM Z
    UNION ALL
    SELECT object_id ,
      OBJECT_NAME,
      OWNERS ,
      OWNER_ORGANIZATIONS ,
      RESTRICT_ACCESS_TO ,
      OBJECT_PARENT,
      CATEGORY_RISK ,
      TYPE_RISK ,
      DD_PROCESS_INSTANCE_ID,
      DD_INSTANCE_ID ,
      MS_RSK_LND_REPORTING_UTILS.ms_rsk_get_orb_relations(dd_object_type,'MS_GRC_REQUIREMENT',OBJECT_ID) AS associated_objects ,
      4                                                                                        AS OBJECT_SEQUENCE ,
      'Requirement'                                                                            AS object_type,
      'MS_GRC_Requirement_Report'                                                              AS associated_report ,
      dd_object_type,
      object_level AS risk_level
    FROM z
      ) rsk,
      ms_grc_risk_cat c
    where c.object_id = rsk.object_id
    ) RISK
