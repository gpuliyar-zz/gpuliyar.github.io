MS_RSK_CONTROL_AS_OBJ_V 

                                                
  CREATE OR REPLACE FORCE VIEW "NOVT_DEV"."MS_RSK_CONTROL_AS_OBJ_V" ("OBJECT_ID", "OBJECT_NAME", "OWNERS", "OWNER_ORGANIZATIONS", "RESTRICT_ACCESS_TO", "OBJECT_PARENT", "DD_PROCESS_INSTANCE_ID", "DD_INSTANCE_ID", "ASSOCIATED_OBJECTS", "OBJECT_SEQUENCE", "OBJECT_TYPE", "ASSOCIATED_REPORT", "DD_OBJECT_TYPE", "NO_OF_OBJECTS", "RISK_LEVEL") AS 
  SELECT CONTROL."OBJECT_ID",
    CONTROL."OBJECT_NAME",
    CONTROL."OWNERS",
    CONTROL."OWNER_ORGANIZATIONS",
    CONTROL."RESTRICT_ACCESS_TO",
    CONTROL."OBJECT_PARENT",
    CONTROL."DD_PROCESS_INSTANCE_ID",
    CONTROL."DD_INSTANCE_ID",
    CONTROL."ASSOCIATED_OBJECTS",
    CONTROL."OBJECT_SEQUENCE",
    CONTROL."OBJECT_TYPE",
    CONTROL."ASSOCIATED_REPORT",
    CONTROL."DD_OBJECT_TYPE",
    (LENGTH(ASSOCIATED_OBJECTS)-LENGTH(REPLACE(ASSOCIATED_OBJECTS,',')))+1 AS NO_OF_OBJECTS,
    CONTROL.Control_Level
    FROM
    (SELECT Rsk.object_id ,
      OBJECT_NAME,
      OWNERS ,
      OWNER_ORGANIZATIONS ,
      RESTRICT_ACCESS_TO ,
      object_parent,
      DD_PROCESS_INSTANCE_ID,
      DD_INSTANCE_ID ,
      associated_objects ,
      OBJECT_SEQUENCE,
      object_type,
      associated_report ,
      DD_OBJECT_TYPE,
      Control_Level
    FROM
      ( WITH Z AS
      (SELECT
        /*+ materialize */
        V.object_id ,
        OBJECT_NAME,
        OWNERS ,
        OWNER_ORGANIZATIONS ,
        RESTRICT_ACCESS_TO ,
        OBJECT_PARENT,
        DD_PROCESS_INSTANCE_ID,
        DD_INSTANCE_ID ,
        dd_object_type,
        object_level
      FROM MS_GRC_CONTROL_PUB_V v
      )
    SELECT object_id ,
      OBJECT_NAME,
      OWNERS ,
      OWNER_ORGANIZATIONS ,
      RESTRICT_ACCESS_TO ,
      OBJECT_PARENT,
      DD_PROCESS_INSTANCE_ID,
      DD_INSTANCE_ID ,
      MS_RSK_LND_REPORTING_UTILS.ms_rsk_get_orb_relations(dd_object_type,'MS_GRC_CONTROL',OBJECT_ID ) AS associated_objects ,
      2                                                                                     AS OBJECT_SEQUENCE,
      'Compliance Object'                                                                   AS object_type,
      'MS_GRC_Control_Report'                                                               AS associated_report ,
      dd_object_type,
      Object_Level AS Control_Level
    FROM Z
    UNION ALL
    SELECT object_id ,
      OBJECT_NAME,
      OWNERS ,
      OWNER_ORGANIZATIONS ,
      RESTRICT_ACCESS_TO ,
      OBJECT_PARENT,
      DD_PROCESS_INSTANCE_ID,
      DD_INSTANCE_ID ,
      MS_RSK_LND_REPORTING_UTILS.ms_rsk_get_orb_relations(dd_object_type,'MS_GRC_AREA_OF_COMPLIANCE',OBJECT_ID) AS associated_objects ,
      1                                                                                               AS OBJECT_SEQUENCE,
      'Area Of Compliance'                                                                            AS object_type,
      'MS_GRC_Area_Of_Compliance_Report'                                                              AS ASSOCIATED_REPORT ,
      dd_object_type,
      Object_Level AS Control_Level
    FROM Z
    UNION ALL
    SELECT object_id ,
      OBJECT_NAME,
      OWNERS ,
      OWNER_ORGANIZATIONS ,
      RESTRICT_ACCESS_TO ,
      OBJECT_PARENT,
      DD_PROCESS_INSTANCE_ID,
      DD_INSTANCE_ID ,
      MS_RSK_LND_REPORTING_UTILS.ms_rsk_get_orb_relations(dd_object_type,'MS_GRC_PROCESS',OBJECT_ID) AS associated_objects ,
      3                                                                                    AS OBJECT_SEQUENCE ,
      'Core Object'                                                                        AS object_type,
      'MS_GRC_Process_Report'                                                              AS associated_report ,
      DD_OBJECT_TYPE,
      Object_Level AS Control_Level
    FROM Z
    UNION ALL
    SELECT object_id ,
      OBJECT_NAME,
      OWNERS ,
      OWNER_ORGANIZATIONS ,
      RESTRICT_ACCESS_TO ,
      OBJECT_PARENT,
      DD_PROCESS_INSTANCE_ID,
      DD_INSTANCE_ID ,
      MS_RSK_LND_REPORTING_UTILS.ms_rsk_get_orb_relations(dd_object_type,'MS_GRC_REQUIREMENT',OBJECT_ID) AS associated_objects ,
      4                                                                                        AS OBJECT_SEQUENCE ,
      'Requirement'                                                                            AS object_type,
      'MS_GRC_Requirement_Report'                                                              AS associated_report ,
      DD_OBJECT_TYPE,
      object_level AS Control_Level
    FROM z
      ) rsk
    ) CONTROL
