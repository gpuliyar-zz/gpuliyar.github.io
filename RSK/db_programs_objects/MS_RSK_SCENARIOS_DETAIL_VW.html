  CREATE OR REPLACE FORCE VIEW "NOVT_DEV"."MS_RSK_SCENARIOS_DETAIL_VW" ("OBJECT_NAME", "SCENARIO_ID", "VALID_FROM", "SCENARIO_ALGORITHM", "VALID_UNTIL", "ADDITIONAL_INFORMATION", "ASSESSED_BY", "CONTINUOUS_ASSESSMENT", "CALCULATE_RESIDUAL_SCORES", "DEFAULT_SCENARIO", "ASSESSMENT_TYPE", "OPERAND_FOR_STAN_SCORIN_SCENAR", "OBJECT_ID", "DD_STATUS_FLAG", "OWNER_ORGANIZATION", "SCN_RISK_CATEGORIES") AS 
  Select b."OBJECT_NAME",b."SCENARIO_ID",b."VALID_FROM",b."SCENARIO_ALGORITHM",b."VALID_UNTIL",b."ADDITIONAL_INFORMATION",b."ASSESSED_BY",b."CONTINUOUS_ASSESSMENT",b."CALCULATE_RESIDUAL_SCORES",b."DEFAULT_SCENARIO",b."ASSESSMENT_TYPE",b."OPERAND_FOR_STAN_SCORIN_SCENAR",b."OBJECT_ID",b."DD_STATUS_FLAG",(Select ms_concat(distinct c.owner_organization)
            from Ms_Rsk_Scenarios_Orn c
            Where c.scenario_id = b.scenario_id)owner_organization,
            (Select ms_concat(distinct d.risk_categories )
             From  Ms_Rsk_Scenarios_Rsc d
            Where b.scenario_id = d.scenario_id) Scn_risk_categories
 From Ms_Rsk_Scenarios_Scn b
