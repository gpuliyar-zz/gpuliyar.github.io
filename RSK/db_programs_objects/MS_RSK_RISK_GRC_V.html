MS_RSK_RISK_GRC_V 

                                                
  CREATE OR REPLACE FORCE VIEW "NOVT_DEV"."MS_RSK_RISK_GRC_V" ("OBJECT_ID", "OBJECT_NAME", "OBJECT_TYPE", "OBJECT_SEQUENCE", "DESCRIPTION", "DESCRIPTION_HTM", "OWNERS", "OWNER_ORGANIZATIONS", "RESTRICT_ACCESS_TO", "OBJ_STATUS", "VALID_FROM", "VALID_UNTIL", "OBJECT_LEVEL", "OBJECT_PARENT", "CATEGORY_RISK", "TYPE_RISK", "RTF_ATTACH", "DD_OBJECT_TYPE", "ATO_OWNER", "ATO_COMMENTS", "ATO_ORGANIZATION") AS 
  select 
Ca.OBJECT_ID,
Ca.OBJECT_NAME,
'Risk Object' Object_Type,
'2' Object_Sequence,
DESCRIPTION,
DESCRIPTION_HTM,
OWNERS,
OWNER_ORGANIZATIONS,
RESTRICT_ACCESS_TO,
OBJ_STATUS,
VALID_FROM,
VALID_UNTIL,
OBJECT_LEVEL,
OBJECT_PARENT,
CATEGORY_RISK,
TYPE_RISK,
RTF_ATTACH,
DD_OBJECT_TYPE,
Ato_Owner,
Ato_Comments,
Ato_Organization
from Ms_Grc_Risk_Pub_v Ca

LEFT OUTER JOIN
    (
    SELECT
      /*+ Index(b MS_GRC_COMPLIANCE_OBJE_ORG_IX)*/
      (Xmlagg(Xmlelement(a,Nvl2(Ato_Owner, Ato_Owner
      || ',', Ato_Owner))).Extract('//text()')).GetClobVal() Ato_Owner,
      (Xmlagg(Xmlelement(a,Nvl2(Ato_Comments,Ato_Comments
      || ',',Ato_Comments))).Extract('//text()')).GetClobVal() Ato_Comments,
      (Xmlagg(Xmlelement(a,Nvl2(Ato_Organization,Ato_Organization
      || ',',Ato_Organization))).Extract('//text()')).GetClobVal() Ato_Organization,
      b.Object_Id
    FROM
      (SELECT SRC_OBJ_ID AS object_id ,
        TGT_OBJ_id       AS ATO_ORGANIZATION ,
        NULL             AS Ato_Owner ,
        NULL             AS Ato_Comments
      FROM MS_GRC_ORB_RELATIONS_PUB_V
      WHERE SRC_OBJ_TYPE='MS_GRC_RISK'
      AND TGT_OBJ_TYPE  ='MS_GRC_ORGANIZATION'
      ) b
    GROUP BY b.Object_Id
    ) Cb
  ON CA.OBJECT_ID = CB.OBJECT_ID
