  CREATE OR REPLACE FORCE VIEW "NOVT_DEV"."MS_RSK_RISK_ASSMNT_CTL_TAB_VW" ("OBJECT_ID", "ASSESS_CONTROLS_FROM", "SHOW_DETAILS_CTL", "CTRL_FLAG", "SCORE_CTL", "WEIGHTING_CTL", "CTL_ID", "AD_HOC_CONTROL_NAME", "CTL_DUMMY2", "CONTROL_NAME", "CTRL_RISK_ID", "CTRL_PROC_ID", "CTRL_ORG_ID", "CONTROL_CATEGORY", "KEY_CONTROL", "CONTROL_TYPE", "CONTROL_EFFECTIVENESS_RATING") AS 
  Select b.Object_id,b.assess_controls_from ,b.show_details_ctl , b.ctrl_flag , b.score_ctl , b.weighting_ctl,
        b.ctl_id,b.ad_hoc_control_name,b.ctl_dummy2,b.control_name,b.ctrl_risk_id,b.ctrl_proc_id,b.ctrl_org_id,
       ms_apps_utilities.get_display_value((Select enterprise_id from si_ent),
                                           'MS_GRC_COMPLIANCE_CATEGORY',
                                           b.CONTROL_CATEGORY
                                          ) CONTROL_CATEGORY,
       ms_apps_utilities.get_display_value((Select enterprise_id from si_ent),
                                           'MS GRC Yes No',
                                           b.KEY_CONTROL
                                          ) KEY_CONTROL,

       ms_apps_utilities.get_display_value((Select enterprise_id from si_ent),
                                           'MS_GRC_COMPLIANCE_TYPE',
                                           b.CONTROL_TYPE
                                          ) CONTROL_TYPE,
       (Select Display_value from Ms_Rsk_Control_Effectiveness where store_value = b.control_effectiveness_rating) as control_effectiveness_rating

from Ms_Rsk_Risk_Assessment_Ctl b
