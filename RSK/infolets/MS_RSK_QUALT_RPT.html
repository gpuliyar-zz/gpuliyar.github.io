                                                        SELECT aa.object_id, 
          (SELECT ms_rsk_utils.Get_Edit_Url
                            (dd_object_type,
                             dd_process_instance_id,
                             dd_instance_id,
                             'Y',
                             object_name
                            )
          FROM DUAL) AS object_name, 
          decode(PROGRESS_STATUS,'P', 'Published',PROGRESS_STATUS) PROGRESS_STATUS,
          (SELECT ms_concat
                     (ms_apps_utilities.get_display_value
                                          ((SELECT enterprise_id
                      FROM si_ent_active_v),
                                           'MS GRC Risk Category',
                                           risk_category
                                          )
                     )
             FROM MS_RSK_QUALT_ASSMNT_FA_rst ff
            WHERE aa.object_id = ff.object_id) AS risk_category,
          REGEXP_REPLACE (DESCRIPTION, '<[^>]+>'),
         (SELECT ms_concat
                     (ms_apps_utilities.get_display_value
                                          ((SELECT enterprise_id
                      FROM si_ent_active_v),
                                           'ORGENTITY',
                                           owner_organizations
                                          )
                     )
             FROM MS_RSK_QUALT_ASSMNT_FA_OWs dd
            WHERE aa.object_id = dd.object_id) AS owner_organizations,
        (SELECT ms_concat
                      (ms_apps_utilities.get_user_full_name(OWNERS)
                      )
             FROM MS_RSK_QUALT_ASSMNT_FA_OWN gg
            WHERE aa.object_id = gg.object_id) AS owners,
          valid_from, 
          valid_until, 
          NULL AS col1, 
          NULL AS col2, 
          NULL AS col3,
          NULL AS col4,
          Ms_Rsk_Utils.include_expired (valid_from,
                                         valid_until
                                        ) AS expired_flag, 
          (SELECT ms_apps_utilities.get_display_value
                       ((SELECT enterprise_id
                           FROM si_ent_active_v),
                        'MS RSK List of Response Type',
                        RESPONSE_TYPE
                       )
          FROM DUAL) AS RESPONSE_TYPE,
          Response_type AS Response_type_stored,
          (SELECT ms_concat (risk_category)
             FROM MS_RSK_QUALT_ASSMNT_FA_rst hh
            WHERE aa.object_id = hh.object_id) AS risk_category_stored,
          object_name AS Object_name_stored,
          (SELECT ms_concat (owner_organizations)
             FROM MS_RSK_QUALT_ASSMNT_FA_OWs ee
            WHERE ee.object_id = aa.object_id) AS owner_organizations_stored,
       (SELECT ms_concat (OWNERS)
             FROM MS_RSK_QUALT_ASSMNT_FA_OWN ii
            WHERE aa.object_id = ii.object_id) AS owner_stored
FROM MS_RSK_QUALT_ASSMNT_FA aa
 WHERE ms_apps_security_pkg.check_object_access (dd_object_type,
                                                   restrict_access_to,
                                                   (SELECT ms_concat (owner_organizations)
             FROM MS_RSK_QUALT_ASSMNT_FA_OWs z
            WHERE z.object_id = aa.object_id),
                                                   :1,
                                                   'VIEW'
                                                  ) = 'ACCESS'
order by aa.object_id
