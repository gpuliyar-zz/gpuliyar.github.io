MS_RSK_HEATMAP_COUNT_RPT_RES Report SYSTEMI 
                                                        SELECT COUNT( org_id)          AS org_count,
  COUNT(core_object_id)       AS cor_count,
  COUNT( risk_id)              AS rsk_count,
  ms_concat(DISTINCT org_id)           AS total_ord_id,
  ms_concat(DISTINCT org_name)         AS total_org_name,
  ms_concat(DISTINCT core_object_id)   AS total_core_object_id,
  ms_concat(DISTINCT core_object_name) AS total_core_object_name,
  ms_concat(DISTINCT risk_id)          AS total_risk_id,
  ms_concat(DISTINCT risk_name)        AS total_risk_name,
  x_axis_residual,
  y_axis_residual,
  MS_RSK_HEATMAP.GET_RATING_FOR_AXIS(SCENARIO_ID,2,y_axis_residual) AS SEVERITY,
  MS_RSK_HEATMAP.GET_RATING_FOR_AXIS(SCENARIO_ID,1,x_axis_residual) AS LIKELIHOOD,
  RESIDUAL_RATING,
  SCENARIO_ID                               AS TOTAL_SCENARIO_ID,
  COUNT(DISTINCT risk_id)                   AS std_count,
  MS_CONCAT(DISTINCT SF1_FCT_RAT_OBJECT_ID) AS FACTOR1,
  MS_CONCAT(DISTINCT SF2_FCT_RAT_OBJECT_ID) AS FACTOR2,
  MS_CONCAT(DISTINCT ASSESSMENT_TYPE)       AS Ass_type,
NULL       AS X_AXIS_SCORE,
   NULL     AS Y_AXIS_SCORE,
  'RES'                                     AS REPORT_TYPE
FROM MS_RSK_HM_MASTER
WHERE scenario_id = NVL(:2,scenario_id)
AND flag          = NVL(:3,flag)
AND Instr(','
  ||(NVL(:4,level_num))
  ||',',','
  ||level_num
  ||',') >0
AND Instr(','
  ||(NVL(:5, org_id))
  ||',',','
  ||org_id
  ||',') >0
AND Instr(','
  ||(NVL(:6, core_object_id))
  ||',',','
  ||core_object_id
  ||',') >0
AND instr(','
  ||(NVL(:7, risk_id))
  ||',',','
  ||risk_id
  ||',')       >0
AND ((rISK_ID IN
  (SELECT object_id
  FROM TABLE(ms_rsk_rpt_utils.rsk_hm_Return_objecT_id(SCENARIO_ID,:1 ,'1,2,3,4',flag) )
  ))
OR (CORE_OBJECT_ID IN
  (SELECT object_id
  FROM TABLE(ms_rsk_rpt_utils.rsk_hm_Return_objecT_id(SCENARIO_ID,:1 ,'1,2,3,4',flag) )
  ))
OR(MS_RSK_RPT_USER_SECURITY.GET_ACCESS(:1,ORG_ID,null,'MS_RSK_RISK_ASSESSMENT',null)=1)
OR (FLAG='S' AND MS_RSK_RPT_USER_SECURITY.GET_ACCESS(:1,MS_RSK_LND_REPORTING_UTILS.MS_RSK_GET_PLAN_ASSESSED_ORGS(MS_RSK_HEATMAP.GET_ASSMNT_FROM_HEATMAP(RISK_ID,CORE_OBJECT_ID,ORG_ID)),MS_RSK_LND_REPORTING_UTILS.MS_RSK_GET_PLAN_SECURITY_ORGS(MS_RSK_LND_REPORTING_UTILS.MS_RSK_GET_PLAN_ID(MS_RSK_HEATMAP.GET_ASSMNT_FROM_HEATMAP(RISK_ID,CORE_OBJECT_ID,ORG_ID))),'MS_RSK_RISK_ASSESSMENT',MS_RSK_RPT_USER_SECURITY.GET_RISK_OWNERS(RISK_ID))=1))
GROUP BY SCENARIO_ID,
  X_AXIS_RESIDUAL,
  Y_AXIS_RESIDUAL,
  RESIDUAL_RATING,
  SF1_FCT_RAT_OBJECT_ID,
  SF2_FCT_RAT_OBJECT_ID,
  ASSESSMENT_TYPE
