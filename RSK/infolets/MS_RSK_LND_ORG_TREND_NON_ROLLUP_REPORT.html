SELECT
DISTINCT
asm.scenario_name AS scenerio_id,
per.object_name AS scenario_name,
org.organization_name,
ROUND(org.org_previous_score_inh),
ROUND(org.org_assess_inh_hidn),
ROUND(org.org_previous_score_res),
ROUND(org.org_assess_res_hidn),
TRIM(SUBSTR(org.org_assessed_rating_inh, 0, DECODE(INSTR(org.org_assessed_rating_inh, '['),0,LENGTH(org.org_assessed_rating_inh),INSTR(org.org_assessed_rating_inh, '[')-1))) AS org_assessed_rating_inh,
TRIM(SUBSTR(org.ORG_PREVIOUS_RATING_INH, 0, DECODE(INSTR(org.ORG_PREVIOUS_RATING_INH, '['),0,LENGTH(org.ORG_PREVIOUS_RATING_INH),INSTR(org.ORG_PREVIOUS_RATING_INH, '[')-1))) AS org_previous_rating_inh,
TRIM(SUBSTR(org.org_assessed_rating_res, 0, DECODE(INSTR(org.org_assessed_rating_res, '['),0,LENGTH(org.org_assessed_rating_res),INSTR(org.org_assessed_rating_res, '[')-1))) AS org_assessed_rating_res,
TRIM(SUBSTR(org.ORG_PREVIOUS_RATING_RES, 0, DECODE(INSTR(org.ORG_PREVIOUS_RATING_RES, '['),0,LENGTH(org.ORG_PREVIOUS_RATING_RES),INSTR(org.ORG_PREVIOUS_RATING_RES, '[')-1))) AS org_previous_rating_res,
CASE
WHEN round(org.org_assess_res_hidn) > ROUND(org.org_previous_score_res) THEN MS_RSK_RPT_UTILS.build_image_url('Upward')
WHEN round(org.org_assess_res_hidn) < ROUND(org.org_previous_score_res) THEN MS_RSK_RPT_UTILS.build_image_url('Downward')
WHEN org.org_previous_score_res IS NULL THEN NULL
ELSE ms_rsk_rpt_utils.build_image_url('No Change')
END AS residual_trend,
CASE
WHEN round(org.org_assess_res_hidn) > round(org.org_previous_score_res) THEN 'Upward'
WHEN round(org.org_assess_res_hidn) < round(org.org_previous_score_res) THEN 'Downward'
WHEN org.org_previous_score_res IS NULL THEN NULL
ELSE 'No Change'
END AS Trend,
org.object_id AS maxobject_id,
org.organization_id,
MS_CONCAT(DISTINCT prc.process_id),
MS_CONCAT(DISTINCT rsk.risk_id),
MS_RSK_LND_REPORTING_UTILS.get_parent_org_name(org.organization_id) AS parent_org,
'View' as View_Assessments,
asm.risk_assessment_type,
MS_RSK_LND_REPORTING_UTILS.get_parent_org_id(org.organization_id) AS dummy1,
asm.scenario_name AS I_PERSPECTIVE_NAME,
org.organization_name AS I_ORG_NAME,
MS_CONCAT(DISTINCT prc.process_id) AS I_ASSESSABLE_ENTITY_NAME,
MS_CONCAT(DISTINCT rsk.risk_id) AS I_RISK_NAME,
NULL AS Dummy3
FROM
 ms_rsk_risk_assessment_org  org,
ms_rsk_risk_assessment_prc prc,
ms_rsk_risk_assessment_rsk rsk,
ms_rsk_risk_assessment asm,
ms_rsk_scenarios_scn per
WHERE
 org.organization_id = rsk.rsk_custom_dummy1
AND NVL(org.organization_id,'xxx') = DECODE(asm.risk_assessment_type,2,prc.prc_custom_dummy2,3,NVL(org.organization_id,'xxx'))
AND org.organization_id IS NOT NULL
AND asm.scenario_name = :1
AND asm.scenario_name = per.scenario_id
and asm.object_id=prc.object_id
and asm.object_id=rsk.object_id
and asm.object_id=org.object_id
AND (org.organization_id,org.object_id) IN
(SELECT
org.organization_id,
MAX(ass.object_id) maxobject_id
FROM
ms_rsk_risk_assessment ass,
ms_rsk_risk_assessment_org org
WHERE
ass.object_id     = org.object_id
AND ass.scenario_name = :1
GROUP BY
ass.scenario_name,
org.organization_id
    )
and MS_RSK_RPT_USER_SECURITY.GET_ACCESS(:2,ORG.ORGANIZATION_ID,MS_RSK_LND_REPORTING_UTILS.MS_RSK_GET_PLAN_SECURITY_ORGS(asm.schedule_risk_id),'MS_RSK_RISK_ASSESSMENT',null)    =1
GROUP BY
asm.scenario_name,
org.object_id,
org.organization_id,
org.organization_name,
org.org_previous_score_inh,
org.org_assess_inh_hidn,
org.org_previous_score_res,
org.org_assess_res_hidn,
org.org_assessed_rating_inh,
org.org_previous_rating_inh,
org.org_assessed_rating_res,
org.org_previous_rating_res,
asm.risk_assessment_type,
per.object_name,
per.ram_profile
ORDER BY
7 desc,
3
