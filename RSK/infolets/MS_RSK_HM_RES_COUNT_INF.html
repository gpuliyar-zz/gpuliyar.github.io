                                                        select count( org_id), count( core_object_id), count( risk_id), 
ms_concat( org_id) as total_ord_id, 
ms_concat( org_name) as total_org_name, 
ms_concat( core_object_id) as total_core_object_id,
ms_concat( core_object_name) as total_core_object_name,
ms_concat( risk_id) as total_risk_id,
ms_concat( risk_name) as total_risk_name,
x_axis_residual, y_axis_residual,
ms_rsk_heatmap.get_yaxis_label(y_axis_residual) as severity,
ms_rsk_heatmap.get_xaxis_label(x_axis_residual) as likelihood,
residual_rating,
ms_concat(distinct scenario_id) as total_scenario_id 
from MS_RSK_HM_MASTER
where scenario_id = nvl(:1,scenario_id)
and flag = nvl(:6,flag)
and Instr(','||(nvl(:2,level_num))||',',','||level_num||',') >0
and Instr(','||(nvl(:3, org_id))||',',','||org_id||',') >0
AND Instr(','||(nvl(:4, core_object_id))||',',','||core_object_id||',') >0
and instr(','||(nvl(:5, risk_id))||',',','||risk_id||',') >0
and 
 ((rISK_ID IN (select object_id FROM TABLE(ms_rsk_rpt_utils.rsk_hm_Return_objecT_id(SCENARIO_ID,:7 ,'1,2,3,4',flag) )))
 OR
 (CORE_OBJECT_ID IN (select object_id FROM TABLE(ms_rsk_rpt_utils.rsk_hm_Return_objecT_id(SCENARIO_ID,:7 ,'1,2,3,4',flag) )))
 OR
 (org_id IN (select object_id FROM TABLE(ms_rsk_rpt_utils.rsk_hm_Return_objecT_id(SCENARIO_ID,:7 ,'1,2,3,4',flag) ))))
group by x_axis_residual, y_axis_residual, residual_rating
