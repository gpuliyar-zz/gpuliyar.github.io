                                                        SELECT
    DISTINCT
    (CASE
        WHEN b.risk_assessment_methodology IN (3,5)
        THEN (SELECT a.object_id FROM ms_rsk_risk_rule_build a WHERE a.object_id = b.scenario_algorithm)
        ELSE (SELECT a.object_id FROM ms_rsk_risk_ram a WHERE a.object_id = b.ram_profile)
    END) AS object_id,
    (CASE
        WHEN b.risk_assessment_methodology IN (3,5)
        THEN (SELECT a.rrb_temp3 FROM ms_rsk_risk_rule_build a WHERE a.object_id = b.scenario_algorithm)
        ELSE (SELECT a.hide_section FROM ms_rsk_risk_ram a WHERE a.object_id = b.ram_profile)
    END) ||'<>'|| NVL(b.manual_refresh,'1') AS rsa_flag
FROM
    ms_rsk_scenarios_scn  b,
    ms_rsk_scenarios      c
WHERE
    b.object_id   = c.object_id
AND b.scenario_id = :1
AND :2 IS NOT NULL
