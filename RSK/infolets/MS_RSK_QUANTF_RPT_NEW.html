                                                        SELECT object_id, 
          (SELECT ms_rsk_utils.Get_Edit_Url
                            (dd_object_type,
                             dd_process_instance_id,
                             dd_instance_id,
                             'Y',
                             object_name
                            )
          FROM DUAL) AS object_name, 
          PROGRESS_STATUS,
          (SELECT ms_rsk_utils.Get_Rsk_Categ_disp_values (RISK_CATEGORY,',')
          FROM DUAl) AS RISK_CATEGORY,
          DESCRIPTION,
         (Select  substr((Trim(Xmlagg(Xmlelement(a, Nvl2(Org_Entity_Name,Org_Entity_Name||',',Org_Entity_Name))).Extract('//text()'))),1,(instr(Trim(Xmlagg(Xmlelement(a, Nvl2(Org_Entity_Name,Org_Entity_Name||',',Org_Entity_Name))).Extract('//text()')),',',-1)-1))
        From   Table(Ms_Apps_Utilities.Parse_String(Owner_Organizations,',')) a,
               Si_Org_Entities_t
        Where  a.Column_Value = Org_Entity_Id)  AS OWNER_ORGANIZATIONS,
        (Select substr((Trim(Xmlagg(Xmlelement(a, Nvl2(Ms_Apps_Utilities.Get_User_Full_Name(Ms_Apps_Utilities.Get_User_Id(User_Name)),
                                              Ms_Apps_Utilities.Get_User_Full_Name(Ms_Apps_Utilities.Get_User_Id(User_Name))||',',
                                              Ms_Apps_Utilities.Get_User_Full_Name(Ms_Apps_Utilities.Get_User_Id(User_Name))))).Extract('//text()'))),
                                              
                                              1,
                                              
                                              (instr(Trim(Xmlagg(Xmlelement(a, Nvl2(Ms_Apps_Utilities.Get_User_Full_Name(Ms_Apps_Utilities.Get_User_Id(User_Name)),
                                              Ms_Apps_Utilities.Get_User_Full_Name(Ms_Apps_Utilities.Get_User_Id(User_Name))||',',
                                              Ms_Apps_Utilities.Get_User_Full_Name(Ms_Apps_Utilities.Get_User_Id(User_Name))))).Extract('//text()')),',',-1)-1))
                                              
        From   Table(Ms_Apps_Utilities.Parse_String(Owners, ',')) v,
               Si_Users_t
        Where  v.Column_Value = User_Name)  AS OWNERS,
          valid_from, 
          valid_until, 
          NULL AS col1, 
          NULL AS col2, 
          NULL AS col3,
          NULL AS col4,
          Ms_Rsk_Utils.include_expired (valid_from,
                                         valid_until
                                        ) AS expired_flag, 
          (SELECT ms_apps_utilities.get_display_value
                       ((SELECT enterprise_id
                           FROM si_ent_active_v),
                        'MS RSK List of Response Type',
                        RESPONSE_TYPE
                       )
          FROM DUAL) AS RESPONSE_TYPE,
          Response_type AS Response_type_stored,
          Risk_Category AS Risk_Category_stored,
          object_name AS Object_name_stored,
          OWNER_ORGANIZATIONS as OWNER_ORGANIZATIONS_STORED,
          OWNERS AS OWNERS_STORED
FROM MS_RSK_QUALT_ASSMNT_FA
 WHERE ms_apps_security_pkg.check_object_access (dd_object_type,
                                                   restrict_access_to,

                                                   :1,

                                                  ) = 'ACCESS'
