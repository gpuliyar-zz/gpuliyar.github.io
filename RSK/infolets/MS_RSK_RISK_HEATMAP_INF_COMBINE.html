  (SELECT Ms_Apps_Utilities.Get_Display_Values (Ms_Grc_Utilities.Active_Ent_Id, 'MS RSK List of Assessment Types', Assessment_Type, ',')
  FROM DUAL
  ) AS ASSESSMENT_TYPE,  SCENARIO_NAME,  ORG_ID,  ORG_NAME,  CORE_OBJECT_ID,  CORE_OBJECT_NAME,  RISK_ID,  RISK_NAME,  PREV_INHERENT_SCORE,
  CURR_INHERENT_SCORE,  SHORT_NAME,  TREND_VALUE_INHERENT,  CREATED_BY,  CREATION_DATE,  LAST_UPDATED_BY,  LAST_UPDATE_DATE,  X_AXIS_INHERENT,
  Y_AXIS_INHERENT,  BUBBLE_SIZE_INHERENT,  BUBBLE_SHAPE_INHERENT,  BUBBLE_COLOR_INHERENT,  TREND_INHERENT,  CURR_RESIDUAL_SCORE,  PREV_RESIDUAL_SCORE,
  LEVEL_NUM,  X_AXIS_RESIDUAL,  Y_AXIS_RESIDUAL,  TREND_RESIDUAL,  TREND_VALUE_RESIDUAL,  INHERENT_RATING,  RESIDUAL_RATING,  BUBBLE_SIZE_RESIDUAL,
  BUBBLE_SHAPE_RESIDUAL,  BUBBLE_COLOR_RESIDUAL,  SCENARIO_ID,  PREV_RESIDUAL_RATING,  PREV_INHERENT_RATING,
  (SELECT DECODE( TREND_SIGN_INHERENT,1,'<img border="0" src="/images/icons/Upward.png">', -1,'<img border="0" src="/images/icons/Downward.png">', 0,'<img border="0" src="/images/icons/No_Change.png">', 2, NULL, TREND_SIGN_INHERENT)
  FROM DUAL
  ) AS TREND_SIGN_INHERENT,
  (SELECT DECODE( TREND_SIGN_RESIDUAL,1,'<img border="0" src="/images/icons/Upward.png">', -1,'<img border="0" src="/images/icons/Downward.png">', 0,'<img border="0" src="/images/icons/No_Change.png">', 2, NULL, TREND_SIGN_RESIDUAL)
  FROM DUAL
  ) AS TREND_SIGN_RESIDUAL,
  FLAG,
  'View' AS VIEW_ASSESSMENT_ORG,  'View' AS VIEW_ASSESSMENT_ASS,  'View' AS VIEW_ASSESSMENT_RSK,  'View' AS VIEW_ASSESSMENT_ST,  ASSESSMENT_TYPE AS ASSESSMENT_TYPE1,
  (SELECT ms_apps_utilities.get_display_values (MS_GRC_UTILITIES.active_ent_id, 'MS_GRC_LEVEL', LEVEL_NUM, ',')
  FROM DUAL
  ) AS LEVEL_NAME,
  RISK_CATEGORIES,  OBJECT_OWNERS,  (SELECT Ms_Rsk_Rpt_Utils.build_form_Url_Heatmap_ass_ent('MS_GRC_PROCESS', NULL, NULL,Core_Object_Id,'PRC','N',Core_Object_Name )
  FROM DUAL
  ) AS Core_Object_Url,
  (SELECT Ms_Rsk_Rpt_Utils.Build_Form_Url_Heatmap('MS_GRC_RISK', NULL, NULL,Risk_Id,'RSK','N',Risk_Name )
  FROM DUAL
  ) AS Risk_Name_Url,
  RMC_INH_RANKING,
  RMC_RES_RANKING,
  X_AXIS_CO_ORD_INH_LABEL,  X_AXIS_CO_ORD_INH_SCORE,  X_AXIS_CO_ORD_RES_LABEL,  X_AXIS_CO_ORD_RES_SCORE,
  Y_AXIS_CO_ORD_INH_LABEL,  Y_AXIS_CO_ORD_INH_SCORE,  Y_AXIS_CO_ORD_RES_LABEL,  Y_AXIS_CO_ORD_RES_SCORE,
  SF1_PREV_RAT,  SF1_CHNGE,  SF2_PREV_RAT,
  SF2_CHNGE,  SF1_FCT_RAT_OBJECT_ID,  SF2_FCT_RAT_OBJECT_ID,  SF1_NAME,  SF2_NAME,  SF1_PREV_RAT_RES,  SF1_CHNGE_RES,  SF2_PREV_RAT_RES,
  SF2_CHNGE_RES,  SF1_TREND_RES,  SF2_TREND_RES,  SF1_TREND,  SF2_TREND,  CONFIG_PROFILE_ID
FROM Ms_Rsk_Hm_Master
WHERE Instr(','||NVL((SELECT Ms_Concat(Object_Id) FROM TABLE(Ms_Rsk_Rpt_Utils.Rsk_Hm_Return_Object_Id(Scenario_Id,:1,'1,2,3,4',Flag) )),'OBJ')||',',','||NVL(Risk_Id,'RISK')||',') >0
OR Instr(','||NVL((SELECT Ms_Concat(Object_Id) FROM TABLE(Ms_Rsk_Rpt_Utils.Rsk_Hm_Return_Object_Id(Scenario_Id,:1,'1,2,3,4',Flag) )),'OBJ')||',',','||NVL(Core_Object_Id,'PRCS')||',') >0
or ( MS_RSK_RPT_USER_SECURITY.GET_ACCESS(:1,ORG_ID,null,'MS_RSK_RISK_ASSESSMENT',null)=1)
OR ( FLAG ='S' AND MS_RSK_RPT_USER_SECURITY.GET_ACCESS(:1,MS_RSK_LND_REPORTING_UTILS.MS_RSK_GET_PLAN_ASSESSED_ORGS(MS_RSK_HEATMAP.GET_ASSMNT_FROM_HEATMAP(RISK_ID,CORE_OBJECT_ID,ORG_ID)),MS_RSK_LND_REPORTING_UTILS.MS_RSK_GET_PLAN_SECURITY_ORGS(MS_RSK_LND_REPORTING_UTILS.MS_RSK_GET_PLAN_ID(MS_RSK_HEATMAP.GET_ASSMNT_FROM_HEATMAP(RISK_ID,CORE_OBJECT_ID,ORG_ID))),'MS_RSK_RISK_ASSESSMENT',MS_RSK_RPT_USER_SECURITY.GET_RISK_OWNERS(RISK_ID))=1)
