                                                        select distinct object_id, 
                  Object_name , 
                  trunc(due_date) due_by,
                  (select ms_concat( distinct organization_name)   
                   from ms_rsk_risk_assessment_v ab
                   where a.OBJECT_ID = ab.OBJECT_ID)organizations,
                   (select ms_concat( distinct process_name)   
                   from ms_rsk_risk_assessment_v ab
                   where a.OBJECT_ID = ab.OBJECT_ID)process,
                   (select ms_concat( distinct rsk_question_title )   
                   from ms_rsk_risk_assessment_v ab
                   where a.OBJECT_ID = ab.OBJECT_ID)Risks,
                   progress_status , 
                   (select object_name from MS_RSK_SCENARIOS_SCN where scenario_id = scenario_name) scenario_name, 
                   decode(risk_assessment_type, 1, 'Core Object - Risk', 2, 'Org - Core Object - Risk', 3, 'Org - Risk')risk_assessment_type,
                   null as var_name1,
                    null as var_name2,
                    null as var_name3,
                    null as start_date   
  from ms_rsk_risk_assessment_v a
  where a.instance_id =    (Select max(instance_id) 
                            from ms_rsk_risk_assessment_v mi
                            where a.OBJECT_ID = mi.OBJECT_ID)
  and a.process_instance_id = (Select max(process_instance_id) 
                               from ms_rsk_risk_assessment_v mp
                               where a.OBJECT_ID = mp.OBJECT_ID)
  and a.object_id in (select  distinct Object_id from ms_rsk_risk_assessment_v a
                            where a.dd_current_stage   <> 'PUBLISH' 
                            and a.instance_id =    (Select max(instance_id) 
                                               from ms_rsk_risk_assessment_v mi
                                               where a.OBJECT_ID = mi.OBJECT_ID)
                             and a.process_instance_id = (Select max(process_instance_id) 
                                                         from ms_rsk_risk_assessment_v mp
                                                         where a.OBJECT_ID = mp.OBJECT_ID))
