                                                        SELECT aa.object_id,
  (SELECT ms_rsk_utils.get_edit_url(dd_object_type,dd_process_instance_id,dd_instance_id,'Y',object_name)
  FROM DUAL
  ) AS object_name,
  DECODE (progress_status, 'P', 'Published',progress_status) progress_status,
get_Clob_Chunk(description) as "description",
  (SELECT ms_concat(ms_apps_utilities.get_display_value(
    (SELECT enterprise_id FROM si_ent_active_v
    ), 'ORGENTITY', owner_organizations))
  FROM ms_rsk_qnt_asmt_fctr_orz dd
  WHERE aa.object_id = dd.object_id
  ) AS owner_organizations,
  (SELECT ms_concat(ms_apps_utilities.get_user_full_name (owners))
  FROM ms_rsk_qnt_asmt_fctr_owr gg
  WHERE aa.object_id = gg.object_id
  ) AS owners,
  valid_from,
  valid_until,
  NULL  AS col1,
  NULL  AS col2,
  NULL  AS col3,
  NULL  AS col4,
  ms_rsk_utils.include_expired (valid_from, valid_until) AS expired_flag,
  (SELECT ms_apps_utilities.get_display_value (
    (SELECT enterprise_id FROM si_ent_active_v
    ), 'MS RSK Type of Factor', factor_type )
  FROM DUAL) AS factor_type_disp,  
 decode((SELECT ms_apps_utilities.get_display_value(
    (SELECT enterprise_id FROM si_ent_active_v
    ),'MS RSK QUANT OBJECT TYPE', aa.factor_classification)
  from DUAL
  ),'Sub Factor',null,(SELECT ms_apps_utilities.get_display_value (
    (SELECT ENTERPRISE_ID FROM SI_ENT_ACTIVE_V
    ), 'MS RSK List of Factor Categorization', case when control_Scoring = 1 then null else factor_contribution end)
  FROM DUAL))AS factor_contribution_disp,
  (SELECT ms_concat (ms_apps_utilities.get_display_value (
    (SELECT enterprise_id FROM si_ent_active_v
    ), 'MS GRC Risk Category', risk_category ) )
  FROM ms_rsk_qnt_asmt_fctr_rse ff
  WHERE aa.object_id = ff.object_id) AS risk_category_disp,
  (SELECT ms_concat (risk_category)
  FROM ms_rsk_qnt_asmt_fctr_rse hh
  WHERE aa.object_id = hh.object_id)  AS RISK_CATEGORY_STORED,
  case when control_scoring =1 then null else factor_contribution end AS factor_contribution_stored,
  factor_type    AS factor_type_stored,
  object_name  AS object_name_stored,
  (SELECT ms_concat (owner_organizations)
  FROM ms_rsk_qnt_asmt_fctr_orz ee
  WHERE ee.object_id = aa.object_id
  ) AS owner_organizations_stored,
  (SELECT ms_concat (owners)
  FROM ms_rsk_qnt_asmt_fctr_owr ii
  WHERE aa.object_id = ii.object_id) AS owner_stored,
  (SELECT ms_apps_utilities.get_display_value(
(SELECT enterprise_id FROM si_ent_active_v
),'MS_RSK_ENH_CURRENCY', aa.currency) 
FROM dual
) AS currency_disp ,
  CASE WHEN  AA.FACTOR_TYPE = 1 THEN (SELECT ms_apps_utilities.get_display_value(
(SELECT enterprise_id FROM si_ent_active_v
),'MS_RSK_QNT_ENH_INPUT_TYPE', aa.INPUT_TYPE)
 FROM dual) ELSE NULL END AS INPUT_TYPE_disp,
  (SELECT ms_apps_utilities.get_display_value(
    (SELECT enterprise_id FROM si_ent_active_v
    ),'MS_RSK_QNT_ENH_APPLICABLE_FOR', aa.APPLICABLE_FOR)
  FROM dual) AS APPLICABLE_FOR_disp ,
  (SELECT ms_apps_utilities.get_display_value(
    (SELECT enterprise_id FROM si_ent_active_v
    ),'MS RSK Yes No', aa.CONTROL_SCORING)
  FROM dual) AS CONTROL_SCORING_disp,
  initcap(DECODE(CONTROL_SCORING,1,DECODE(FACTOR_TYPE,1,NOT_APPLICABLE_SCR,NOT_APPLICABLE_RTG),2,DECODE(FACTOR_TYPE,1,NOT_APPLICABLE_SCR,NOT_APPLICABLE_MNR))) AS not_applicable_for
,
    (SELECT ms_apps_utilities.get_display_value(
    (SELECT enterprise_id FROM si_ent_active_v
    ),'MS RSK QUANT OBJECT TYPE', aa.factor_classification)
  FROM dual) As FACTOR_CLASSIFICATION_disp,
currency as currenc_stored,(DECODE(CONTROL_SCORING,1,DECODE(FACTOR_TYPE,1,NOT_APPLICABLE_SCR,NOT_APPLICABLE_RTG),2,DECODE(FACTOR_TYPE,1,NOT_APPLICABLE_SCR,NOT_APPLICABLE_MNR))) AS not_applicable_for_stored 
FROM ms_rsk_qnt_asmt_fctr aa
WHERE ms_apps_security_pkg.check_object_access (dd_object_type, restrict_access_to,
  (SELECT ms_concat (owner_organizations)
  FROM ms_rsk_qnt_asmt_fctr_orz z WHERE z.object_id = aa.object_id),:1,'VIEW' )   = 'ACCESS' ORDER BY aa.object_id
