                                                        SELECT (select object_name
          from ms_rsk_scenarios_scn
         WHERE scenario_id = a.scenario_id) as scenario_name,
       A.ASSESSMENT_ID,
       (select object_name from ms_grc_risk where object_id = a.risk_id) as risk,
       (SELECT object_name
          from ms_grc_core_object
         where object_id = a.core_object_id
        union all
        select object_name
          from ms_grc_objectives
         WHERE object_id = a.core_object_id
        union all
        select object_name
          from ms_grc_product
         where object_id = a.core_object_id) as core_object,
       (SELECT org_entity_name
          FROM si_org_entities_t
         where to_char(org_entity_id) = a.organization_id) as organization,
       a.x_inh,
       a.y_inh,
       a.x_res,
       a.y_res,
       nvl((SELECT percentage
             FROM Ms_Rsk_Org_Wtg_Org
            where org_pk =
                  (SELECT max(org_pk)
                     FROM MS_RSK_ORG_WTG_ORG
                    where child_org_id = a.organization_id)
              and child_org_id = a.organization_id),
           100) as org_wtg,
       a.inh_score as inh_risk_score,
       a.res_score as res_risk_score,
       A.SCENARIO_ID,
       A.RISK_ID,
       A.CORE_OBJECT_ID,
       a.organization_id
  FROM ms_rsk_risk_scrs a
