MS_RSK_RISK_ASSESSMENT_OPEN_CAL Report SYSTEMI 
                                                        SELECT next_scheduled_date + NVL (due_by, 0) startdate, object_name title,
       object_id || '[ Planned ]' description, 0 ismonthevent,
          'Pushinfolet?id='
       || ms_apps_utilities.get_infolet_id (dd_object_type)
       || CHR (38)
       || 'proc='
       || dd_process_instance_id
       || CHR (38)
       || 'instid='
       || dd_instance_id
       || CHR (38)
       || 'edit_flag='
       || 'Y'
       || CHR (38)
       || 'flag=1'
       || CHR (38)
       || 'emd=1' url,
       NULL end_date, NULL objid, NULL objtype
  FROM ms_rsk_schedule_risk_a a
 WHERE 1 = 1
--AND PROGRESS_STATUS ='Completed'
   AND SYSDATE BETWEEN NVL (valid_from, SYSDATE) AND NVL (valid_until,
                                                          SYSDATE)
   AND ms_apps_security_pkg.check_object_access
                                    (dd_object_type,
                                     restrict_access_to,
                                     (SELECT ms_concat (owner_organizations)
                                        FROM ms_rsk_schedule_risk_a_owo z
                                       WHERE a.object_id = z.object_id),
                                     :1,
                                     'VIEW'
                                    ) = 'ACCESS'
UNION
SELECT c.assessment_date                                            --due_date
                        ,
       c.object_name title, c.object_id || ' [ In Progress ]' description, 0,
          'Pushinfolet?id='
       || ms_apps_utilities.get_infolet_id (c.dd_object_type)
       || CHR (38)
       || 'proc='
       || c.process_instance_id
       || CHR (38)
       || 'instid='
       || c.instance_id
       || CHR (38)
       || 'edit_flag=N'
       || CHR (38)
       || 'flag=1'
       || CHR (38)
       || 'emd=1' url,
       NULL end_date, NULL objid, NULL objtype
  FROM ms_rsk_risk_assessment_v c, ms_rsk_schedule_risk_a x
 WHERE NOT EXISTS (SELECT object_id
                     FROM ms_rsk_risk_assessment b
                    WHERE c.object_id = b.object_id)
   AND instance_rec_num = 1
   AND instance_id = (SELECT MAX (instance_id)
                        FROM ms_rsk_risk_assessment_v d
                       WHERE d.process_instance_id = c.process_instance_id)
   AND 1 = 1
   AND x.object_id = c.schedule_risk_id
   AND ms_apps_security_pkg.check_object_access
                                    (x.dd_object_type,
                                     x.restrict_access_to,
                                     (SELECT ms_concat (owner_organizations)
                                        FROM ms_rsk_schedule_risk_a_owo z
                                       WHERE x.object_id = z.object_id),
                                     :1,
                                     'VIEW'
                                    ) = 'ACCESS'
