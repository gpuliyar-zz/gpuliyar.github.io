                                                        SELECT SCENARIO_ID,
  OBJECT_NAME_LINK,
  ASSESSMENT_TYPE,
  DEFAULT_SCENARIO,
  CALCULATE_RESIDUAL_SCORES,
  RISK_CATEGORY,
  VALID_FROM,
  VALID_UNTIL,
  EXPIRED_FLAG,
  SCN_RISK_CATEGORIES,
  SCENARIO_ID_STORED,
  DEFAULT_SCENARIO_STORED,
  OBJECT_NAME_STORED,
  GROUP_NAME_STORED,
  ASSESSMENT_TYPE_STORED,
  I_PERSPECTIVE_NAME,
  RAM_LINK,
  risa_name_link
,
risk_assessment_methodology
  
FROM
  ( SELECT DISTINCT A.SCENARIO_ID AS SCENARIO_ID,
    (SELECT ms_rsk_utils.Get_Edit_Url(dd_object_type,dd_process_instance_id,dd_instance_id,'N',e.object_name)
    FROM DUAL
    ) AS object_name_link,
    (SELECT ms_apps_utilities.get_display_value (
      (SELECT ENTERPRISE_ID FROM SI_ENT_ACTIVE_V
      ),'MS RSK List of Assessment Types',e.assessment_type)
    FROM DUAL
    ) AS assessment_type,
    (SELECT MS_APPS_UTILITIES.GET_DISPLAY_VALUE (
      (SELECT ENTERPRISE_ID FROM SI_ENT_ACTIVE_V
      ),'MS RSK Yes No', e.DEFAULT_SCENARIO )
    FROM DUAL
    ) AS DEFAULT_SCENARIO,
    (SELECT ms_apps_utilities.get_display_value(
      (SELECT ENTERPRISE_ID FROM SI_ENT_ACTIVE_V
      ),'MS RSK Yes No', e.calculate_residual_scores)
    FROM DUAL
    )                 AS calculate_residual_scores,
    D.RISK_CATEGORIES AS RISK_CATEGORY,
    TO_CHAR(trunc(e.VALID_FROM)) as VALID_FROM,
    TO_CHAR(TRUNC(E.VALID_UNTIL)) AS VALID_UNTIL,
    (SELECT Ms_Rsk_Utils.include_expired (e.valid_from, e.valid_until) FROM dual
    )                     AS expired_flag,
    a.Scn_risk_categories AS Scn_risk_categories,
    A.SCENARIO_ID         AS SCENARIO_ID_STORED,
    e.DEFAULT_SCENARIO      AS DEFAULT_SCENARIO_STORED,
    e.object_name           AS object_name_stored,
    GROUP_NAME            AS GROUP_NAME_STORED,
    e.ASSESSMENT_TYPE       AS ASSESSMENT_TYPE_STORED,
    e.OBJECT_NAME           AS I_PERSPECTIVE_NAME,
    (SELECT ms_apps_utilities.get_display_value(
      (SELECT enterprise_id FROM si_ent_active_v
      ),'MS_RSK_RISK_ASSESSMENT_METHODOLOGY', e.risk_assessment_methodology)
    FROM DUAL
    )                             AS risk_assessment_methodology,
    e.risk_assessment_methodology AS risk_ass_methodology_stored,
    (SELECT object_name FROM ms_rsk_risk_ram WHERE object_id=e.ram_profile
    ) AS ram_name,
    e.ram_profile,
    (SELECT ms_rsk_utils.Get_Edit_Url(dd_object_type,dd_process_instance_id,dd_instance_id,'N',object_name)
    FROM ms_rsk_risk_ram
    WHERE object_id= e.ram_profile
    ) AS ram_link,
    (SELECT object_name
    FROM ms_rsk_risk_rule_build
    WHERE object_id=e.scenario_algorithm
    ) AS risa_name,
    (SELECT ms_rsk_utils.Get_Edit_Url(dd_object_type,dd_process_instance_id,dd_instance_id,'N',object_name)
    FROM ms_rsk_risk_rule_build
    WHERE object_id= e.scenario_algorithm
    ) AS risa_name_link,
    e.scenario_algorithm
  FROM Ms_Rsk_Scenarios_detail_Vw a,
    ms_rsk_scenarios b,
    MS_RSK_SCENARIOS_RSC C,
    ms_rsk_scenarios_scn e,
    (SELECT SCENARIO_ID,
      ms_concat (ms_apps_utilities.get_display_value (
      (SELECT ENTERPRISE_ID FROM SI_ENT_ACTIVE_V
      ),'MS GRC Risk Category',RISK_CATEGORIES)) risk_categories
    FROM MS_RSK_SCENARIOS_RSC
    GROUP BY SCENARIO_ID
    ) d
  WHERE A.OBJECT_ID = B.OBJECT_ID
  AND e.scenario_id =a.scenario_id
  AND A.SCENARIO_ID = C.SCENARIO_ID
  AND C.SCENARIO_ID = D.SCENARIO_ID
  )
