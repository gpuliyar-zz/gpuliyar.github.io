                                                        SELECT DISTINCT B.OBJECT_ID,
  OBJECT_NAME,
  'Level '||OBJECT_LEVEL,
  OBJECT_PARENT,
  (SELECT f.object_name
  FROM MS_GRC_RISK_PUB_V F
  WHERE f.OBJECT_ID=B.OBJECT_PARENT
  ) AS RISK_parent,case when object_level in ('3','4') then ms_Rsk_rpt_utils.FN_GET_RISK_LEVEL1_PARENT(b.object_level,b.object_id) else null end as Level1_Risk
 FROM MS_GRC_RISK B,
--MS_GRC_CORE_OBJECT_RIS_VW C,
--MS_GRC_RISK_ORG_VW D,
MS_GRC_RISK_OGN E
WHERE B.OBJECT_ID=E.OBJECT_ID
AND B.OBJECT_ID IN
  (SELECT distinct RISK_ID
  FROM MS_RSK_RISK_ROLLUP
  WHERE SCENARIO_ID IN
    (SELECT TRIM(COLUMN_VALUE)
    FROM TABLE(CAST(
      (SELECT ms_apps_utilities.parse_string(:2,',') FROM DUAL
      ) AS ms_varchar2_array))
    WHERE column_value IS NOT NULL
    )
  )
AND MS_APPS_SECURITY_PKG.CHECK_OBJECT_ACCESS(DD_OBJECT_TYPE, 'O', E.OWNER_ORGANIZATIONS, :3, 'VIEW') = 'ACCESS'
AND B.OBJECT_LEVEL                                                                                           IN
  (SELECT trim(column_value)
  FROM TABLE(CAST(
    (SELECT ms_apps_utilities.parse_string(:1,',') FROM dual
    ) AS MS_VARCHAR2_ARRAY))
  WHERE COLUMN_VALUE IS NOT NULL
  )
