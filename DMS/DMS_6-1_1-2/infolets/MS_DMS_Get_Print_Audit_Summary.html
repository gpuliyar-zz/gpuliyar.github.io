                                                        SELECT dh.document_name, dh.version_number, dh.process_instance_id,
       dh.life_cycle_id, dh.current_stage_id stage_id,pl.max_copies,
       NVL ((SELECT   COUNT (pa.copy_number) total_copy
                 FROM si_dms_print_audit pa INNER JOIN ms_dms_documents_hdr dh1
                      ON pa.process_instance_id = dh1.process_instance_id
                WHERE dh1.document_object_id = :1
                  AND dh1.process_instance_id = dh.process_instance_id
                  AND pa.stage_id=dh.current_stage_id
             GROUP BY pa.process_instance_id), 0
           ) copies_printed
  FROM ms_dms_documents_hdr dh LEFT OUTER JOIN si_dms_printlayout pl
       ON dh.life_cycle_id = pl.lifecycle_id
     AND dh.current_stage_id = pl.stage_id
 WHERE dh.document_object_id = :1 AND dh.in_process_flag = 'Y'
UNION
 select document_name,revision_number,  a.audit_id as process_instance_id
  ,approval_process_id,0 stage_id,max_copies,
  (SELECT   COUNT (pa.copy_number) total_copy
                 FROM si_dms_print_audit pa INNER JOIN ms_dms_all_production_docs dh1
                      ON pa.process_instance_id = dh1.process_instance_id
                WHERE dh1.document_object_id= :1
                  AND dh1.process_instance_id = a.process_instance_id
                  and pa.audit_id in(( select max(audit_id) from ms_dms_doc_audit_history_v
                                   where process_instance_id=dh1.process_instance_id and document_object_id=dh1.document_object_id),dh1.audit_id)
                             AND pa.stage_id = 0
             GROUP BY pa.process_instance_id) copy_number
 from ms_dms_all_production_docs a  where document_object_id=:1
