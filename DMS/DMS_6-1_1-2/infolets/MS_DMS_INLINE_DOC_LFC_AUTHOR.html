                                                        select lut.user_name, lut.full_name user_full_name from 
(SELECT g.user_id FROM SI_DMS_LIFECYCLE_ROLE_GRPS lg,SI_DMS_GROUP_DETAILS g 
WHERE g.role_group_id =lg.role_group_id AND lifecycle_id=:1 
and exists (select 1 from SI_DMS_LIFECYCLE_STAGES where lifecycle_id=:1 and seq = 1 and stage_id = lg.stage_id)
AND (5 = :5 OR 6=:5)
union
select  a.user_id from MS_DMS_CATEGORY_USER_LIST_V a 
where a.CATEGORY_ID=:4 and SUBSTR(access_rights,15,1)=1 AND (5 <> :5 AND 6<>:5)) t_tab, ms_dms_local_users_t lut
where t_tab.user_id = lut.user_id and lut.valid_flg = 'Y'
UNION
select lut.user_name, lut.full_name from 
(SELECT :2 as user_name FROM si_dms_lifecycle_stages 
where lifecycle_id=:1 and add_owner_flag='Y' and seq=1 AND (5 = :5 OR 6=:5)
UNION
SELECT EXTRACT (VALUE (d), '//row/text()' ).getstringval () AS user_name
FROM (SELECT XMLTYPE('<rows><row>'||REPLACE (document_author||','||document_owner,',','</row><row>')||'</row></rows>') AS xmlval
FROM ms_dms_documents where document_name= :3 AND (5 = :5 OR 6=:5)) x,
TABLE (XMLSEQUENCE(EXTRACT(X.XMLVAL,'/rows/row'))) d) t_tab, ms_dms_local_users_t lut
where t_tab.user_name = lut.user_name and lut.valid_flg = 'Y'
UNION
SELECT B.USER_NAME,B.FULL_NAME
FROM MS_DMS_DOCUMENTS_HDR_ALL_V A ,MS_DMS_LOCAL_USERS_T B
WHERE A.document_name= :3
  AND A.OWNER_ID=b.USER_ID
  AND a.PROCESS_INSTANCE_ID=(SELECT MAX(PROCESS_INSTANCE_ID)
                              FROM MS_DMS_DOCUMENTS_HDR_ALL_V 
                             WHERE document_name= :3)
