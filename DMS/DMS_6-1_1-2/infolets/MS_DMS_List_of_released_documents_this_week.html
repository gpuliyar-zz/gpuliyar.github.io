                a.TIMESTAMP, 
                ms_dms_doc_url_for_report 
                                     (b.process_instance_id,
                                      a.audit_id,
                                      DECODE (a.obsolete_flag,
                                              'Y', 'Obsolete',
                                              'Production'
                                             ),
                                      b.document_attachment,
                                      b.document_name,
                                                        :1
                                     ) document_link,
                b.document_object_id, 'P' flag, ms_dms_utilities.ms_dms_get_val_for_locale(:1,'Attributes')  meta_data,
                b.document_number,
             ms_dms_utilities.MS_DMS_GET_VAL_FOR_LOV(:1, DECODE (a.document_status, 'Obsoleted', 'Obsolete', 
                                                            (CASE 
                                                             WHEN NVL(a.effective_date,TRUNC(SYSDATE)) > SYSDATE THEN 'Ready for Production'
                                                             WHEN NVL(a.effective_date,TRUNC(SYSDATE)) <= SYSDATE THEN  a.document_status
                                                             END)), 'MS_DMS_DOCUMENT_STATUS',1) as DOCUMENT_STATUS  
           FROM ms_dms_documents a, ms_dms_documents_hdr b
          WHERE TRUNC (a.TIMESTAMP) BETWEEN TRUNC (  SYSDATE
                                                   - (  TO_CHAR (SYSDATE, 'D')
                                                      - 1 
                                                     )
                                                  )
                                        AND TRUNC (SYSDATE)
            AND a.process_instance_id(+) = b.process_instance_id
            AND b.in_process_flag = 'N'
       ORDER BY document_name ASC
