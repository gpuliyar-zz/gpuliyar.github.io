                                                        SELECT MS_DMS_MLS_UTILITIES.MS_DMS_GET_METADATA_MLSVALUE(FORM_CODE,STAGE_ID,action_type,user_action_type,process_instance_id,:2) 
            action_type, 
         document_name, 
         VERSION_NUMBER, 
         CATEGORY, 
         SUB_CATEGORY1, 
         SUB_CATEGORY2, 
         SUB_CATEGORY3, 
         SUB_CATEGORY4, 
         Ms_Dms_Doc_Url_For_Report ( 
            process_instance_id, 
            audit_id, 
            DECODE (document_status_flag, 
                    'P', 'Production', 
                    'I', 'In Process', 
                    'O', 'Obsolete', 
                    'T', 'Terminated'), 
            document_attachment, 
            'View', 
            :2,
            document_object_id), 
         DECODE(life_cycle_id,NULL,' ',MS_DMS_MLS_UTILITIES.MS_DMS_GET_LIFECYCLE_MLSVALUE (life_cycle_id, NULL, 0, :2)) AS life_cycle_name, 
         Decode(a.form_code,'Review Complete, No UpVersion Required','',MS_DMS_MLS_UTILITIES.MS_DMS_GET_STAGE_MLSVALUE(stage_id, NULL, :2)) AS stage_name, 
         MS_APPS_UTILITIES.GET_USER_FULL_NAME (A.USER_ID), 
         CUR_DATE, 
         SIGNATURE_MEANING, 
         parent_audit_id, 
         process_instance_id, 
         effective_from, 
         comments, 
         sys_comment_code, 
         audit_id, 
         DECODE (form_code, 'MS DMS Virtual Document', 'Details') AS Details, 
         Document_object_id, 
         Document_object_type, 
         DECODE ( 
            document_object_type, 
            'V', DECODE (form_code, 
                         'MS DMS Create Document', 'View Details', 
                         'MS DMS Document Review', 'View Details')) 
            AS Details1, 
         MS_DMS_MLS_UTILITIES.MS_DMS_GET_CATEGORY_MLSVALUE (:2, CATEGORY_ID) 
         || (NVL2 ( 
                SUB_CATEGORY1, 
                '/' 
                || MS_DMS_MLS_UTILITIES. 
                   MS_DMS_GET_CATEGORY_MLSVALUE (:2, SUB_CATEGORY1_ID), 
                '')) 
         || (NVL2 ( 
                SUB_CATEGORY2, 
                '/' 
                || MS_DMS_MLS_UTILITIES. 
                   MS_DMS_GET_CATEGORY_MLSVALUE (:2, SUB_CATEGORY2_ID), 
                '')) 
         || (NVL2 ( 
                SUB_CATEGORY3, 
                '/' 
                || MS_DMS_MLS_UTILITIES. 
                   MS_DMS_GET_CATEGORY_MLSVALUE (:2, SUB_CATEGORY3_ID), 
                '')) 
         || (NVL2 ( 
                SUB_CATEGORY4, 
                '/' 
                || MS_DMS_MLS_UTILITIES. 
                   MS_DMS_GET_CATEGORY_MLSVALUE (:2, SUB_CATEGORY4_ID), 
                '')) 
            AS CATEGORY_PATH, 
         DOCUMENT_NUMBER,
         to_number(SUBSTR(version_number, 0, INSTR(version_number, '.')-1)) AS version1,
         to_number(SUBSTR(version_number, INSTR(version_number, '.')+1, instr(version_number, '.', 1, 2)-INSTR(version_number, '.')-1)) as version2,
         to_number(SUBSTR(version_number, INSTR(version_number, '.',1,2)+1, instr(version_number, '.', 1, 3)-INSTR(version_number, '.',1,2)-1)) as version3,
         TO_NUMBER(SUBSTR(VERSION_NUMBER, INSTR(VERSION_NUMBER,'.',1,3)+1)) AS VERSION4 ,
         DESCRIPTION,
         description_htm
    FROM MS_DMS_DOC_AUDIT_HIST_V A 
   WHERE DOCUMENT_OBJECT_ID = :1
   AND((a.form_code != 'Review Complete, No UpVersion Required'
              AND NVL (a.action_type, 'NULL') != 'RCNUR') OR (a.form_code = 'Review Complete, No UpVersion Required' AND (a.action_type IN               ('RCNUR', 'Change Request', 'Edit Attributes','Create Document', 'UpVersion Document')OR a.action_type IS NULL))) 
ORDER BY audit_id
