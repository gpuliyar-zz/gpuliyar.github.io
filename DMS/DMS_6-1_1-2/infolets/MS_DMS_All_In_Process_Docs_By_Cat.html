                                                        SELECT b.purpose as purpose,
         b.document_name AS document_name,
         b.document_number AS document_number,
         b.version_number as version_number,
         MS_DMS_MLS_UTILITIES.MS_DMS_GET_CATEGORY_MLSVALUE (:1, b.category_ID) AS category,
         MS_DMS_MLS_UTILITIES.MS_DMS_GET_CATEGORY_MLSVALUE (:1, b.sub_category1_ID) AS sub_category1,
         MS_DMS_MLS_UTILITIES.MS_DMS_GET_CATEGORY_MLSVALUE (:1, b.sub_category2_ID) AS sub_category2,
         MS_DMS_MLS_UTILITIES.MS_DMS_GET_CATEGORY_MLSVALUE (:1, b.sub_category3_ID) AS sub_category3,
         MS_DMS_MLS_UTILITIES.MS_DMS_GET_CATEGORY_MLSVALUE (:1, b.sub_category4_ID) AS sub_category4,
         NULL as Attachment,
         REPLACE (
            Ms_Dms_User_List (b.process_instance_id, current_stage_seq, 1),
            ',',
            ', ') as user_list,
         Ms_Dms_Doc_Url_For_nameReport (b.process_instance_id,
                                        a.audit_id,
                                        'In Process',
                                        b.document_attachment,
                                        b.document_name,
                                        1,
                                        :1) as url,
         b.document_object_id as document_object_id,
         'I' as Flag,
         ms_dms_utilities.ms_dms_get_val_for_locale (:1, 'Attributes')
           as Meta_Data,
		 a.category_id AS  category_id,
		 a.SUB_CATEGORY1_id AS SUB_CATEGORY1_id,
		 a.SUB_CATEGORY2_id AS SUB_CATEGORY2_id,
		 a.SUB_CATEGORY3_id AS SUB_CATEGORY3_id,
		 a.SUB_CATEGORY4_id AS SUB_CATEGORY4_id,
		 MS_APPS_UTILITIES.Get_Display_Value(ms_grc_utilities.active_ent_id, 'MS_DMS_INLINE_DOCUMENT_TYPE', b.doc_type,1,(select locale_id from si_locales aa , si_users_t bb where aa.locale_string=bb.locale and bb.user_name=:1)) as document_type
    FROM MS_DMS_DOC_AUDIT_HISTORY a,
         MS_DMS_DOCUMENTS_HDR b,
         MS_DMS_USER_GRP_CAT_RPT_V c
   WHERE     a.category_id = c.category_id
         AND NVL (a.SUB_CATEGORY1_id, '0') = NVL (c.SUBCATEGORY1_id, '0')
         AND NVL (a.SUB_CATEGORY2_id, '0') = NVL (c.SUBCATEGORY2_id, '0')
         AND NVL (a.SUB_CATEGORY3_id, '0') = NVL (c.SUBCATEGORY3_id, '0')
         AND NVL (a.SUB_CATEGORY4_id, '0') = NVL (c.SUBCATEGORY4_id, '0')
         AND b.category_id = c.category_id
         AND NVL (b.SUB_CATEGORY1_id, '0') = NVL (c.SUBCATEGORY1_id, '0')
         AND NVL (b.SUB_CATEGORY2_id, '0') = NVL (c.SUBCATEGORY2_id, '0')
         AND NVL (b.SUB_CATEGORY3_id, '0') = NVL (c.SUBCATEGORY3_id, '0')
         AND NVL (b.SUB_CATEGORY4_id, '0') = NVL (c.SUBCATEGORY4_id, '0')
         AND b.in_process_flag = 'Y'
         AND a.process_instance_id = b.process_instance_id
         AND c.user_name = :1
         AND audit_id = (SELECT MAX (audit_id)
                           FROM MS_DMS_DOC_AUDIT_HISTORY
                          where process_instance_id = b.process_instance_id)
ORDER BY UPPER (b.document_name)
