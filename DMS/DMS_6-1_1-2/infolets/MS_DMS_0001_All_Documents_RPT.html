DOCUMENT_OBJECT_ID, ms_dms_utilities.get_doc_status (flag, document_object_id, :1) FLAG, ms_dms_utilities.ms_dms_get_val_for_locale (:1, 'Attributes') as Attributes,MS_APPS_UTILITIES.Get_Display_Value(ms_grc_utilities.active_ent_id, 'MS_DMS_INLINE_DOCUMENT_TYPE', doc_type,1,(select locale_id from si_locales aa , si_users_t bb where aa.locale_string=bb.locale and bb.user_name=:1)) as document_type FROM (SELECT Ms_Dms_Doc_Url_For_nameReport ( process_instance_id, audit_id, DECODE (a.Document_Status, 'Production', 'Production', 'Obsoleted', 'Obsolete', NULL), document_attachment, document_name) AS DOCUMENT_NAME, DOCUMENT_NUMBER, REVISION_NUMBER, MS_DMS_MLS_UTILITIES.MS_DMS_GET_CATEGORY_MLSVALUE (:1, a.CATEGORY_ID) AS CATEGORY, MS_DMS_MLS_UTILITIES.MS_DMS_GET_CATEGORY_MLSVALUE (:1, SUB_CATEGORY1_ID) AS SUB_CATEGORY1, MS_DMS_MLS_UTILITIES.MS_DMS_GET_CATEGORY_MLSVALUE (:1, SUB_CATEGORY2_ID) AS SUB_CATEGORY2, MS_DMS_MLS_UTILITIES.MS_DMS_GET_CATEGORY_MLSVALUE (:1, SUB_CATEGORY3_ID) AS SUB_CATEGORY3, MS_DMS_MLS_UTILITIES.MS_DMS_GET_CATEGORY_MLSVALUE (:1, SUB_CATEGORY4_ID) AS SUB_CATEGORY4, DOCUMENT_OBJECT_ID, DECODE (a.Document_Status, 'Production', 'P', 'Obsoleted', 'O', NULL) flag, 'Attributes', a.doc_type FROM MS_DMS_DOCUMENTS a, MS_DMS_USER_GRP_CAT_RPT_V c WHERE     checked_out_flag = 'N' AND a.CATEGORY = c.CATEGORY AND NVL (a.SUB_CATEGORY1, '%') = NVL (c.SUBCATEGORY1, '%') AND NVL (a.SUB_CATEGORY2, '%') = NVL (c.SUBCATEGORY2, '%') AND NVL (a.SUB_CATEGORY3, '%') = NVL (c.SUBCATEGORY3, '%') AND NVL (a.SUB_CATEGORY4, '%') = NVL (c.SUBCATEGORY4, '%') AND c.user_name = :1 UNION SELECT (SELECT Ms_Dms_Doc_Url_For_nameReport (b.process_instance_id, a.audit_id, 'In Process', b.document_attachment, b.document_name) FROM DUAL) AS DOCUMENT_NAME, b.document_number, b.version_number, MS_DMS_MLS_UTILITIES.MS_DMS_GET_CATEGORY_MLSVALUE (:1,b.category_id) AS CATEGORY, MS_DMS_MLS_UTILITIES.MS_DMS_GET_CATEGORY_MLSVALUE (:1,b.sub_category1_id) AS SUB_CATEGORY1, MS_DMS_MLS_UTILITIES.MS_DMS_GET_CATEGORY_MLSVALUE (:1,b.sub_category2_id) AS SUB_CATEGORY2, MS_DMS_MLS_UTILITIES.MS_DMS_GET_CATEGORY_MLSVALUE (:1,b.sub_category3_id) AS SUB_CATEGORY3, MS_DMS_MLS_UTILITIES.MS_DMS_GET_CATEGORY_MLSVALUE (:1,b.sub_category4_id) AS SUB_CATEGORY4, b.document_object_id, DECODE (b.in_process_flag,  'Y', 'IN',  'T', 'T',  NULL) flag, 'Attributes' ,b.doc_type FROM MS_DMS_DOC_AUDIT_HISTORY a, MS_DMS_DOCUMENTS_HDR b, MS_DMS_USER_GRP_CAT_RPT_V c WHERE b.in_process_flag IN ('Y', 'T') AND a.process_instance_id = b.process_instance_id AND audit_id = (SELECT MAX (audit_id) FROM MS_DMS_DOC_AUDIT_HISTORY WHERE process_instance_id = b.process_instance_id) AND a.CATEGORY = c.CATEGORY AND NVL (a.SUB_CATEGORY1, '%') = NVL (c.SUBCATEGORY1, '%') AND NVL (a.SUB_CATEGORY2, '%') = NVL (c.SUBCATEGORY2, '%') AND NVL (a.SUB_CATEGORY3, '%') = NVL (c.SUBCATEGORY3, '%') AND NVL (a.SUB_CATEGORY4, '%') = NVL (c.SUBCATEGORY4, '%') AND b.CATEGORY = c.CATEGORY AND NVL (b.SUB_CATEGORY1, '%') = NVL (c.SUBCATEGORY1, '%') AND NVL (b.SUB_CATEGORY2, '%') = NVL (c.SUBCATEGORY2, '%') AND NVL (b.SUB_CATEGORY3, '%') = NVL (c.SUBCATEGORY3, '%') AND NVL (b.SUB_CATEGORY4, '%') = NVL (c.SUBCATEGORY4, '%') AND c.user_name = :1) a ORDER BY UPPER (DOCUMENT_NAME) ASC
