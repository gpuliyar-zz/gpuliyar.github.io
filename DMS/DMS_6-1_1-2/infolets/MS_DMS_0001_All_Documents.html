                                                        SELECT DOCUMENT_NAME,DOCUMENT_NUMBER,REVISION_NUMBER,CATEGORY,SUB_CATEGORY1,SUB_CATEGORY2,
  SUB_CATEGORY3,SUB_CATEGORY4,DOCUMENT_OBJECT_ID,FLAG,'Attributes' FROM
  (SELECT Ms_Dms_Doc_Url_For_nameReport 
  ( 
  process_instance_id, 
  audit_id, 
  DECODE(a.Document_Status,'Production','Production','Obsoleted','Obsoleted',NULL) ,
  document_attachment, 
  document_name 
  )  AS DOCUMENT_NAME,
      DOCUMENT_NUMBER,
      REVISION_NUMBER,
      a.CATEGORY,
      SUB_CATEGORY1,
      SUB_CATEGORY2,
      SUB_CATEGORY3,
      SUB_CATEGORY4,
      DOCUMENT_OBJECT_ID,
      DECODE(a.Document_Status,'Production','P','Obsoleted','O',NULL) flag,
       'Attributes'
  FROM MS_DMS_DOCUMENTS a,MS_DMS_USER_GRP_CAT_RPT_V c
  WHERE checked_out_flag = 'N' and
  a.CATEGORY        = c.CATEGORY
  AND NVL(a.SUB_CATEGORY1,'%')   = NVL(c.SUBCATEGORY1,'%')
  AND NVL(a.SUB_CATEGORY2,'%')   = NVL(c.SUBCATEGORY2,'%')
  AND NVL(a.SUB_CATEGORY3,'%')   = NVL(c.SUBCATEGORY3,'%')
  AND NVL(a.SUB_CATEGORY4,'%')   = NVL(c.SUBCATEGORY4,'%')
   AND c.user_name       = :1
  UNION
  SELECT  (SELECT Ms_Dms_Doc_Url_For_nameReport
  (
  b.process_instance_id,
  a.audit_id,
  'In Process',
  b.document_attachment,
  b.document_name
  )  FROM DUAL) AS DOCUMENT_NAME,b.document_number,
   b.version_number, NVL(b.category,''), NVL(b.sub_category1,''),
  NVL(b.sub_category2,''), NVL(b.sub_category3,''),
  NVL(b.sub_category4,''),
  b.document_object_id,
  DECODE(b.in_process_flag,'Y','I','T','T',NULL) flag,
   'Attributes' 
  FROM MS_DMS_DOC_AUDIT_HISTORY a,MS_DMS_DOCUMENTS_HDR b,MS_DMS_USER_GRP_CAT_RPT_V c
  WHERE b.in_process_flag IN ('Y','T') AND a.process_instance_id=b.process_instance_id 
  AND  audit_id=(SELECT MAX(audit_id) FROM   MS_DMS_DOC_AUDIT_HISTORY WHERE process_instance_id=b.process_instance_id)
  AND a.CATEGORY        = c.CATEGORY
  AND NVL(a.SUB_CATEGORY1,'%')   = NVL(c.SUBCATEGORY1,'%')
  AND NVL(a.SUB_CATEGORY2,'%')   = NVL(c.SUBCATEGORY2,'%')
  AND NVL(a.SUB_CATEGORY3,'%')   = NVL(c.SUBCATEGORY3,'%')
  AND NVL(a.SUB_CATEGORY4,'%')   = NVL(c.SUBCATEGORY4,'%')
  AND b.CATEGORY        = c.CATEGORY
  AND NVL(b.SUB_CATEGORY1,'%')   = NVL(c.SUBCATEGORY1,'%')
  AND NVL(b.SUB_CATEGORY2,'%')   = NVL(c.SUBCATEGORY2,'%')
  AND NVL(b.SUB_CATEGORY3,'%')   = NVL(c.SUBCATEGORY3,'%')
  AND NVL(b.SUB_CATEGORY4,'%')   = NVL(c.SUBCATEGORY4,'%')
  AND c.user_name       = :1)a
  ORDER BY UPPER(DOCUMENT_NAME) ASC
