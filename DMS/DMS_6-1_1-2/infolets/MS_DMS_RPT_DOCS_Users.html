                                                        SELECT count(1),doc_status,document_name,doc_num,category,stages,Due_By,MS_APPS_UTILITIES.GET_USER_FULL_NAME(usr_name) as usr_name,purpose,ver_num
FROM (
SELECT distinct b.process_instance_id pid,b.document_name,b.document_number doc_num,b.purpose,b.version_number ver_num,
             MS_DMS_MLS_UTILITIES.MS_DMS_GET_CATEGORY_MLSVALUE(:3, b.category_id) as category,
             MS_DMS_MLS_UTILITIES.MS_DMS_GET_LIFECYCLE_MLSVALUE(b.life_cycle_id,0,0,:3) life_cycle,ms_apps_utilities.get_user_name(a.user_id) usr_name,
             'Stage-'||( SELECT count(1)+1 cnt 
                   FROM ms_dms_documents_stage_dtl
                   WHERE  process_instance_id= a.process_instance_id
                   AND    stage_seq < a.stage_seq) stages,
             CASE 
                WHEN (to_date(SYSDATE,'DD/MM/YYYY') - to_date(NVL(a.assignment_creation_date,SYSDATE),'DD/MM/YYYY')) <= 0  THEN '(0 days)' 
             WHEN (to_date(SYSDATE,'DD/MM/YYYY') - to_date(NVL(a.assignment_creation_date,SYSDATE),'DD/MM/YYYY')) >  0  AND  
                    (to_date(SYSDATE,'DD/MM/YYYY') - to_date(NVL(a.assignment_creation_date,SYSDATE),'DD/MM/YYYY')) <= 7  THEN '(1-7 days)' 
                WHEN (to_date(SYSDATE,'DD/MM/YYYY') - to_date(NVL(a.assignment_creation_date,SYSDATE),'DD/MM/YYYY')) >  7  AND 
                   (to_date(SYSDATE,'DD/MM/YYYY') - to_date(NVL(a.assignment_creation_date,SYSDATE),'DD/MM/YYYY')) <= 21 THEN '(8-21 days)'  
                WHEN (to_date(SYSDATE,'DD/MM/YYYY') - to_date(NVL(a.assignment_creation_date,SYSDATE),'DD/MM/YYYY')) >  21 THEN '(>21 days)'
             END AS Due_By,
            'In-Process' doc_status 
      FROM  ms_dms_documents_stage_usr_dtl a,ms_dms_documents_hdr b,MS_DMS_USER_GRP_CAT_RPT_V  MDGCV
               WHERE    b.CATEGORY=MDGCV.CATEGORY and a.process_instance_id=b.process_instance_id 
                  AND  NVL(b.SUB_CATEGORY1,'%')=NVL(MDGCV.SUBCATEGORY1,'%')
                AND  NVL(b.SUB_CATEGORY2,'%')=NVL(MDGCV.SUBCATEGORY2,'%')
                 AND NVL(b.SUB_CATEGORY3,'%')=NVL(MDGCV.SUBCATEGORY3,'%')
                AND NVL(b.SUB_CATEGORY4,'%')=NVL(MDGCV.SUBCATEGORY4,'%')
          AND  b.category||','||SUB_CATEGORY1||','||SUB_CATEGORY2||','||SUB_CATEGORY3||','||SUB_CATEGORY4
      IN
       (select category||','||SUBCATEGORY1||','||SUBCATEGORY2||','||SUBCATEGORY3||','||SUBCATEGORY4 from  MS_DMS_USER_GRP_CAT_RPT_V
         WHERE user_name=:3)
           AND  a.process_instance_id     = b.process_instance_id 
             AND   a.current_process_status  = 'Y'
             AND b.document_object_type      = 'N'
   UNION
      SELECT DISTINCT process_instance_id pid,document_name,document_number doc_num,purpose,revision_number ver_num,
             MS_DMS_MLS_UTILITIES.MS_DMS_GET_CATEGORY_MLSVALUE(:3, MDD.category_id) as category,null,MDGCV.user_name usr_name,null ,null,
            'In Production' doc_status 
      FROM  ms_dms_documents MDD,MS_DMS_USER_GRP_CAT_RPT_V MDGCV
      WHERE checked_out_flag  IN ('N','I') 
      AND   document_object_type     = 'N'
      AND   obsolete_flag            = 'N' 
      and  MDD.CATEGORY_ID=MDGCV.CATEGORY_ID
                AND  NVL(MDD.SUB_CATEGORY1_ID,0)=NVL(MDGCV.SUBCATEGORY1_ID,0)
                AND  NVL(MDD.SUB_CATEGORY2_ID,0)=NVL(MDGCV.SUBCATEGORY2_ID,0)
                 AND NVL(MDD.SUB_CATEGORY3_ID,0)=NVL(MDGCV.SUBCATEGORY3_ID,0)
                AND NVL(MDD.SUB_CATEGORY4_ID,0)=NVL(MDGCV.SUBCATEGORY4_ID,0)
      AND  MDD.category||','||MDD.SUB_CATEGORY1||','||MDD.SUB_CATEGORY2||','||MDD.SUB_CATEGORY3||','||MDD.SUB_CATEGORY4
      IN
       (select category||','||SUBCATEGORY1||','||SUBCATEGORY2||','||SUBCATEGORY3||','||SUBCATEGORY4 from  MS_DMS_USER_GRP_CAT_RPT_V a
  WHERE a.user_name=:3)
     )
WHERE doc_status = :1
AND MS_APPS_UTILITIES.GET_USER_FULL_NAME(usr_name) = :2
GROUP BY document_name,doc_num,category,purpose,doc_status,ver_num,usr_name,stages,Due_By
