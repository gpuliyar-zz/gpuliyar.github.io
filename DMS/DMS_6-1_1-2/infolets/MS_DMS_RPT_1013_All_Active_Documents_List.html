                                                        SELECT 
        DOCUMENT_NAME, DOCUMENT_NUMBER, revision_NUMBER, 
       A.CATEGORY CATEGORY, SUB_CATEGORY1, SUB_CATEGORY2, 
           SUB_CATEGORY3, SUB_CATEGORY4, DOCUMENT_attachment_str, 
           Document_author , DOCUMENT_STATUS, EFFECTIVE_DATE, 
           DOCUMENT_attachment,'N/A',DOCUMENT_OBJECT_ID FROM 
        MS_DMS_AVAIL_PRO_Report_DOCS_V A, MS_DMS_USER_GRP_CAT_RPT_V B
      WHERE A.CATEGORY=B.CATEGORY
      AND NVL(a.SUB_CATEGORY1,'%')=NVL(B.SUBCATEGORY1,'%')
      AND NVL(a.SUB_CATEGORY2,'%')=NVL(B.SUBCATEGORY2,'%')
      AND NVL(a.SUB_CATEGORY3,'%')=NVL(B.SUBCATEGORY3,'%')
      AND NVL(a.sub_category4,'%')=NVL(B.SUBCATEGORY4,'%')
      AND USER_NAME=:1
