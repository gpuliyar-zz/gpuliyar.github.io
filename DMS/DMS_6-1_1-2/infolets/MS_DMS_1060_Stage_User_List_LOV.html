                                                        SELECT user_name FROM
(SELECT siu.user_name, UPPER(siu.user_name), dls.seq
   FROM  si_dms_user_group_users dugu, si_users siu,
         si_dms_lifecycle_role_grps dlrg, si_dms_lifecycle_stages dls, si_dms_lifecycles dlc
  WHERE siu.user_id = dugu.user_id
    AND siu.end_date is null
    AND dugu.user_group_id = dlrg.group_id
    AND dlrg.lifecycle_id = dlc.lifecycle_id
    AND dlc.lifecycle_id = dls.lifecycle_id
    AND dlrg.stage_id = dls.stage_id
    AND dlrg.lifecycle_id = :1
    AND dls.seq = :2
    AND dlc.STAGE_CATEGORY_USER_FLAG = 'N'
	UNION
 SELECT a.user_name, UPPER(a.user_name), b.seq
   FROM ms_dms_stage_users_v a, SI_DMS_LIFECYCLE_STAGES b, si_dms_lifecycles c
  WHERE a.lifecycle_id = b.lifecycle_id
    AND b.lifecycle_id = c.lifecycle_id
    AND a.stage_id = b.stage_id
    AND c.stage_category_user_flag ='Y'
    AND a.lifecycle_id = :1
UNION
 SELECT DISTINCT d.user_name, UPPER(d.user_name), e.seq
   FROM SI_DMS_ACCESS_GRP_CATS a, SI_DMS_CREATE_DOCUMENT_V b,
        SI_DMS_GROUP_DETAILS c, SI_USERS d,
        SI_DMS_LIFECYCLE_STAGES e, SI_DMS_LIFECYCLE_ROLE_GRPS f,
        si_dms_lifecycles g
 WHERE a.category = b.category
   AND c.user_id = d.user_id
   AND (a.subcategory1 IS NULL OR a.subcategory1 = b.sub_category1)
   AND (a.subcategory2 IS NULL OR a.subcategory2 = b.sub_category2)
   AND (a.subcategory3 IS NULL OR a.subcategory3 = b.sub_category3)
   AND (a.subcategory4 IS NULL OR a.subcategory4 = b.sub_category4)
   AND a.access_group_id = f.group_id
   AND f.role_group_id = c.role_group_id
   AND f.lifecycle_id = g.lifecycle_id
   AND E.LIFECYCLE_ID = g.lifecycle_id
   AND e.lifecycle_id = :1
   AND g.stage_category_user_flag ='Y'
   AND e.seq > (SELECT MIN(seq)
                  FROM SI_DMS_LIFECYCLE_STAGES
                 WHERE lifecycle_id = :1)
  AND b.process_instance_id = :3
 ORDER BY 3)
WHERE seq = :2
