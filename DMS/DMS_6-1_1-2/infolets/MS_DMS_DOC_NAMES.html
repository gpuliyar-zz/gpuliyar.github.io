                                                        select DISTINCT  
         hdr.document_name Document_Name
          FROM ms_dms_documents_hdr hdr,
      ms_dms_documents_dtl dtl,
      ( select distinct process_instance_id,
               user_assignment_id,
               document_object_id,
               stage_id,
               stage_seq  
        FROM  ms_dms_documents_stage_usr_dtl 
        where current_process_status='Y') stgdtl,
 si_event_assignments_v si,
 si_users_t ut , 
 SI_ENABLED_LOCALES sel ,  
 si_dms_lifecycle_role_grps sdlrg,
 SI_DMS_USER_GROUP_USERS sdugu,  
 SI_DMS_USER_GROUPS sdug,
 ms_dms_user_grp_cat_rpt_v  mdugcrv   
 where hdr.process_instance_id=dtl.process_instance_id  
 and   hdr.process_instance_id=si.process_instance_id  
 and   stgdtl.user_assignment_id = si.assignment_id  
 and   hdr.document_object_id = dtl.document_object_id  
 and   hdr.process_instance_id=stgdtl.process_instance_id  
 and   hdr.document_object_id=stgdtl.document_object_id  
 and   hdr.current_stage_id =stgdtl.stage_id  
 and   hdr.current_stage_seq=stgdtl.stage_seq  
 and   hdr.in_process_flag='Y'     
 and   sel.locale_string=ut.locale 
 and  hdr.life_cycle_id = sdlrg.lifecycle_id 
 and sdlrg.group_id =sdugu.user_group_id 
 and  sdugu.user_group_id=sdug.user_group_id
 and sdugu.user_id=ut.user_id
 and  hdr.category=mdugcrv.category
 and STGDTL.stage_id=sdlrg.stage_id
 and nvl(hdr.sub_category1,'X')  = NVL(mdugcrv.subcategory1,'X') 
 and NVL(hdr.sub_category2,'X')  = NVL(mdugcrv.subcategory2,'X') 
 and NVL(hdr.sub_category3,'X')  = NVL(mdugcrv.subcategory3,'X') 
 and NVL(hdr.sub_category4,'X')  = NVL(mdugcrv.subcategory4,'X') AND ut.user_name=:1
