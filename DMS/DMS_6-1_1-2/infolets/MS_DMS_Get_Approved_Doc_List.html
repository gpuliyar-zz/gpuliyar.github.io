                                                        SELECT 
  DOCUMENT_NAME,
  DOCUMENT_NUMBER,
  revision_number,
   MS_DMS_MLS_UTILITIES.MS_DMS_GET_CAT_MLS_LOCALE_ID (:1,a.CATEGORY_ID)      AS CATEGORY,
  MS_DMS_MLS_UTILITIES.MS_DMS_GET_CAT_MLS_LOCALE_ID (:1,a.SUB_CATEGORY1_ID) AS SUB_CATEGORY1,
  MS_DMS_MLS_UTILITIES.MS_DMS_GET_CAT_MLS_LOCALE_ID (:1,a.SUB_CATEGORY2_ID) AS SUB_CATEGORY2,
  MS_DMS_MLS_UTILITIES.MS_DMS_GET_CAT_MLS_LOCALE_ID (:1,a.SUB_CATEGORY3_ID) AS SUB_CATEGORY3,
  MS_DMS_MLS_UTILITIES.MS_DMS_GET_CAT_MLS_LOCALE_ID (:1,A.SUB_CATEGORY4_ID) AS SUB_CATEGORY4, 
 PURPOSE,
  description,
( '<a href="'
      || 'javascript'||CHR(58)||'openDMSDoc('''
      || REPLACE (REPLACE (document_attachment, '''', '\'''), '`', '\`')
      || ''','
      || process_instance_id
      || ','
      || audit_id
      || ','
      || ''''
	  || ( SELECT DECODE( pdf_convert_yn, 'Y', UPPER ('Production'),'E','Not Converted',NULL,'Yet Convert')
           from ms_dms_all_production_docs
          where audit_id = a.audit_id)
      || ''''
      || ','
      || chr (39)
     || ( select param_val from ms_dms_parameters
             where param_type = 'MS DMS Native Viewer Errors'
             and param_name = 'MS DMS Native Viewer Error Display_YN' )
      || CHR (39)
      || ','
      || chr (39)
     || ( select param_val from ms_dms_parameters
             where param_type = 'MS_DMS_PDF_CONVERSION_RQD_YN'
             and param_name = 'MS_DMS_PDF_CONVERSION_RQD_YN' )
      || '^'
     || ( select param_val from ms_dms_parameters
             where param_type = 'MS DMS Native Viewer Errors'
             and param_name = 'MS DMS Native Viewer Error Display_YN' )
      || '^'
     || ( select NVL(param_val,'Document is not yet converted.') from ms_dms_parameters
             where param_type = 'MS DMS Native Viewer Errors'
             and param_name = 'MS_DMS_CONV_INPROCESS_MSG' )
      || chr (39)
      || ')">'||( select v.displayed_value from ms_dms_list_of_values_v v,si_enabled_locales l,si_users_t t
                  where v.lov_name='MS_DMS_STATUS_LOV'
                  and v.stored_value='View'
                  and v.locale_id = l.locale_id
                  and t.user_name = :1
                )||'</a>'
   ) as document_attachment,
  (select full_name from ms_dms_local_users_t where user_name = LAST_USERNAME) AS LAST_USERNAME,
  TIMESTAMP,
  PROCESS_INSTANCE_ID,
  INSTANCE_ID,
  LAST_REVIEW_DATE,
  REVIEW_SCHEDULE_DAYS,
  key_words,
 (select full_name from ms_dms_local_users_t where user_name = DOCUMENT_AUTHOR) AS DOCUMENT_AUTHOR,
 (select full_name from ms_dms_local_users_t where user_name = DOCUMENT_OWNER) AS DOCUMENT_OWNER,
  EFFECTIVE_DATE,
  NEXT_REVIEW_DATE,
  initial_action_type,
 (select full_name from ms_dms_local_users_t where user_name = DOCUMENT_ORIGINATOR) AS DOCUMENT_ORIGINATOR,
  AUDIT_ID,
  DOCUMENT_OBJECT_ID,
  DOCUMENT_OBJECT_TYPE,
  DOCUMENT_STATUS,
  VIRTUAL_DOC_COMMENTS,
  author_id,
	 (SELECT DISPLAYED_VALUE
	FROM ms_dms_local_lov_t V,
		 MS_DMS_LOCAL_USERS_T D
	WHERE V.LOCALe_ID=D.LOCALe_ID and V.LOV_NAME='MS_DMS_STATUS_LOV'
	AND V.STORED_VALUE='Attributes' AND USER_NAME=:1) AS META_DATA,
  'P' as  FLAG  ,
 ( SELECT DISPLAYED_VALUE
	FROM ms_dms_local_lov_t V,
		 MS_DMS_LOCAL_USERS_T D
	WHERE V.LOCALE_ID=D.LOCALE_ID AND V.LOV_NAME='MS_DMS_INLINE_DOCUMENT_TYPE'
	and v.stored_value=a.doc_type and user_name=:1) AS DOCUMENT_TYPE
FROM MS_DMS_DOCUMENTS A,
MS_DMS_LOCAL_CAT_USER_ACR_T C
WHERE 
SELECTED_CATEGORY_ID=C.CATEGORY_ID
AND c.user_name               =:1
and a.checked_out_flag        in ('I', 'N')
AND OBSOLETE_FLAG              = 'N'
ORDER BY UPPER(DOCUMENT_NAME)
