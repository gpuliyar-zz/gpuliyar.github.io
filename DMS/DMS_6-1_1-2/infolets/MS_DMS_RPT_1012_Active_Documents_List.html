                                                        SELECT DISTINCT document_name, document_number, revision_number, 
ms_dms_mls_utilities.MS_DMS_GET_CATEGORY_MLSVALUE(:1,mdd.category_id) as category, 
ms_dms_mls_utilities.MS_DMS_GET_CATEGORY_MLSVALUE(:1,mdd.sub_category1_id) as sub_category1,
ms_dms_mls_utilities.MS_DMS_GET_CATEGORY_MLSVALUE(:1,mdd.sub_category2_id) as sub_category2, 
ms_dms_mls_utilities.MS_DMS_GET_CATEGORY_MLSVALUE(:1,mdd.sub_category3_id) as sub_category3, 
ms_dms_mls_utilities.MS_DMS_GET_CATEGORY_MLSVALUE(:1,mdd.sub_category4_id) as sub_category4,
    NULL  AS attachment,
    user_name,
    Ms_Dms_Doc_Url_For_nameReport
    (
    mdd.process_instance_id,
    mdd.audit_id,
    'Production',
    document_attachment,
    document_name
    ) AS document_link,
     ms_dms_utilities.ms_dms_get_val_for_locale(:1,decode(sign(sysdate-effective_date),-1,replace(doCument_status,'Production','Ready for Production'),Decode(document_status,'Production','Move To Production',document_status))) document_status,
    effective_date, mdd.DOCUMENT_OBJECT_ID, 'P' Flag, ms_dms_utilities.ms_dms_get_val_for_locale(:1,'Attributes') Meta_Data 
    FROM  MS_DMS_DOCUMENTS mdd, MS_DMS_USER_GRP_CAT_RPT_V  duv
    WHERE obsolete_flag = 'N' AND CHECKED_OUT_FLAG IN ('N','I')
    AND NVL(mdd.category, '%') LIKE NVL(duv.category, '%')
    AND NVL(mdd.sub_category1, '%') = NVL(duv.subcategory1, '%')
    AND NVL(mdd.sub_category2, '%') = NVL(duv.subcategory2, '%')
    AND NVL(mdd.sub_category3, '%') = NVL(duv.subcategory3, '%')
    AND NVL(mdd.sub_category4, '%') = NVL(duv.subcategory4, '%')
    AND USER_NAME=:1
