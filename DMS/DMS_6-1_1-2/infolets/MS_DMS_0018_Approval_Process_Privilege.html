       MS_DMS_MLS_UTILITIES.MS_DMS_GET_LIFECYCLE_MLSVALUE(e.lifecycle_id,0, NULL, :1) as lifecycle_name,
        MS_DMS_MLS_UTILITIES.MS_DMS_GET_STAGE_MLSVALUE(d.stage_id, NULL, :1) as STAGE_NAME,
       d.SEQ,
       decode(sum(substr(ACCESS_RIGHTS,7,1)),0,MS_DMS_MLS_UTILITIES.MS_DMS_get_val_for_lov(:1,'0', 'MS_DMS_YES_NO',1),MS_DMS_MLS_UTILITIES.MS_DMS_get_val_for_lov(:1,'1', 'MS_DMS_YES_NO',1)) read_flag,
       decode(sum(substr(ACCESS_RIGHTS,4,1)),0,MS_DMS_MLS_UTILITIES.MS_DMS_get_val_for_lov(:1,'0', 'MS_DMS_YES_NO',1),MS_DMS_MLS_UTILITIES.MS_DMS_get_val_for_lov(:1,'1', 'MS_DMS_YES_NO',1)) write_flag,
       decode(sum(substr(ACCESS_RIGHTS,5,1)),0,MS_DMS_MLS_UTILITIES.MS_DMS_get_val_for_lov(:1,'0', 'MS_DMS_YES_NO',1),MS_DMS_MLS_UTILITIES.MS_DMS_get_val_for_lov(:1,'1', 'MS_DMS_YES_NO',1)) comment_flag,
       decode(sum(substr(ACCESS_RIGHTS,7,1)),0,MS_DMS_MLS_UTILITIES.MS_DMS_get_val_for_lov(:1,'0', 'MS_DMS_YES_NO',1),MS_DMS_MLS_UTILITIES.MS_DMS_get_val_for_lov(:1,'1', 'MS_DMS_YES_NO',1)) delete_flag,
  MS_DMS_MLS_UTILITIES.MS_DMS_get_val_for_lov(:1,     d.APPROVAL_REQUIREMENT,'MS_DMS_APPROVAL_TYPES',1) Unanimous_Flag,
decode(d.EMAIL_FLAG,'Y',MS_DMS_MLS_UTILITIES.MS_DMS_get_val_for_lov(:1,'1', 'MS_DMS_YES_NO',1),MS_DMS_MLS_UTILITIES.MS_DMS_get_val_for_lov(:1,'0', 'MS_DMS_YES_NO',1)) email_flag
from    si_dms_GROUP_DETAILS a,
           SI_DMS_LIFECYCLE_ROLE_GRPS b,
          si_dms_roles c,si_dms_lifecycles E,
        SI_DMS_LIFECYCLE_STAGES D
where   E.lifecycle_id = D.lifecycle_id
    and     b.ROLE_group_ID =  a.ROLE_group_ID
    and     b.stage_id = d.stage_id
    and     d.lifecycle_id = e.lifecycle_id
    and     b.role_id = c.role_id
group by
    e.lifecycle_id,--approval_process_id,       
       d.STAGE_id,
       d.SEQ,
       d.APPROVAL_REQUIREMENT,
       email_flag
order by 1,2,4
