                                                        SELECT COUNT(1), Days
FROM (SELECT distinct a.document_name,
             CASE 
               WHEN MAX(TO_DATE(a.signed_on,'DD/MM/YYYY'))-MIN(TO_DATE(a.assignment_creation_date,'DD/MM/YYYY'))  <= 0  THEN '(0 days)'
               WHEN MAX(TO_DATE(a.signed_on,'DD/MM/YYYY'))-MIN(TO_DATE(a.assignment_creation_date,'DD/MM/YYYY'))   >  0  AND
                    MAX(TO_DATE(a.signed_on,'DD/MM/YYYY'))-MIN(TO_DATE(a.assignment_creation_date,'DD/MM/YYYY'))   <= 7  THEN '(1-7 days)'
               WHEN MAX(TO_DATE(a.signed_on,'DD/MM/YYYY'))-MIN(TO_DATE(a.assignment_creation_date,'DD/MM/YYYY'))   >  7  AND 
                    MAX(TO_DATE(a.signed_on,'DD/MM/YYYY'))-MIN(TO_DATE(a.assignment_creation_date,'DD/MM/YYYY'))  <= 21 THEN '(8-21 days)'
               WHEN MAX(TO_DATE(a.signed_on,'DD/MM/YYYY'))-MIN(TO_DATE(a.assignment_creation_date,'DD/MM/YYYY'))   >  21 THEN '(>21 days)'
             END AS days
      FROM   ms_dms_documents_stage_usr_dtl a
      WHERE  a.process_instance_id   in (select case
                                                    when a.process_instance_id < b.process_instance_id then max(a.process_instance_id)
                                                    when a.process_instance_id = b.process_instance_id then b.process_instance_id
                                                   end as pid
                                            from   ms_dms_documents b,MS_DMS_USER_GRP_CAT_RPT_V MDGCV
                                            where  a.document_name           = b.document_name
                                            AND    b.checked_out_flag        IN ('N','I')
                                            AND    b.obsolete_flag           ='N'
                                            and    b.initial_action_type is not null
                                             and  b.CATEGORY_ID=MDGCV.CATEGORY_ID
                                 AND  NVL(b.SUB_CATEGORY1_ID,0)=NVL(MDGCV.SUBCATEGORY1_ID,0)
                AND  NVL(b.SUB_CATEGORY2_ID,0)=NVL(MDGCV.SUBCATEGORY2_ID,0)
                 AND NVL(b.SUB_CATEGORY3_ID,0)=NVL(MDGCV.SUBCATEGORY3_ID,0)
                AND NVL(b.SUB_CATEGORY4_ID,0)=NVL(MDGCV.SUBCATEGORY4_ID,0)
                and MDGCV.user_name=:2
                                           group by b.process_instance_id)
      AND    a.document_object_type    = 'N'                                            
      AND    a.current_process_status  = 'N'
      AND    a.signed_on is not null
      AND    a.assignment_creation_date is not null
      AND    TO_CHAR(a.signed_on,'Mon-YYYY') in  (SELECT COLUMN_VALUE month_year
                                                         FROM   TABLE (get_age_month_year((SYSDATE-300),SYSDATE))
                                                         WHERE  ROWNUM <= :1)
      GROUP  BY a.process_instance_id,a.document_name
)
GROUP BY days ORDER BY DAYS
