                                                        SELECT '  ' AS lifecycle_name, 0 AS lifecycle_id
FROM DUAL
 WHERE NOT EXISTS (SELECT 1
                     FROM ms_dms_documents b
                    WHERE b.document_name = :1)UNION 
SELECT DISTINCT lifecycle_name,lifecycle_id FROM (SELECT DISTINCT lfc.lifecycle_name,  lfc.lifecycle_id FROM
 si_users siu,SI_ENT sie, SI_DMS_GROUP_DETAILS dsgrp,
 SI_DMS_ROLES dms_roles,  SI_DMS_LIFECYCLE_ROLE_GRPS lc_rl_grp, 
 SI_DMS_LIFECYCLE_STAGES stages, SI_DMS_LIFECYCLES lfc, MS_DMS_DOCUMENTS  docu
WHERE SYSDATE BETWEEN lfc.START_DATE AND NVL(lfc.END_DATE,SYSDATE+1)  AND stages.lifecycle_id = lfc.lifecycle_id AND
 stages.seq = (SELECT MIN(seq) FROM SI_DMS_LIFECYCLE_STAGES WHERE lifecycle_id=lfc.lifecycle_id) AND
 lc_rl_grp.lifecycle_id = stages.lifecycle_id AND lc_rl_grp.stage_id = stages.stage_id AND
 dms_roles.role_id =lc_rl_grp.role_id  AND dsgrp.role_group_id =lc_rl_grp.role_group_id AND dsgrp.user_id = siu.user_id     
    AND siu.user_name = :2 AND siu.enterprise_id = sie.enterprise_id AND                        
 sie.enterprise_name = :3 AND RTRIM(docu.document_name)=RTRIM(:1)  and    docu.approval_process_id=lfc.lifecycle_id 
 and SUBSTR(dms_roles.access_rights,4,1)=
    case when :4 IN ('Create Document','UpVersion Document') then   '1' 
      when :4  not IN('Obsolete Document' ) then 
        SUBSTR(dms_roles.access_rights,4,1)   end
   AND docu.approval_process_id=lfc.lifecycle_id and obsolete_flag='N'
 UNION
   SELECT DISTINCT lfc.lifecycle_name,  lfc.lifecycle_id FROM
 si_users siu,SI_ENT sie,ms_dms_documents_stage_usr_dtl dsc,
  SI_DMS_LIFECYCLE_STAGES stages, SI_DMS_LIFECYCLES lfc, MS_DMS_DOCUMENTS  docu
WHERE SYSDATE BETWEEN lfc.START_DATE AND NVL(lfc.END_DATE,SYSDATE+1)  AND stages.lifecycle_id = lfc.lifecycle_id AND
 stages.seq = (SELECT MIN(seq) FROM SI_DMS_LIFECYCLE_STAGES WHERE lifecycle_id=lfc.lifecycle_id) AND
  SUBSTR(dsc.access_right,4,1)=
  case when :4 IN ('Create Document','UpVersion Document') then   '1' 
      when :4 not IN('Obsolete Document' ) then  SUBSTR(dsc.access_right,4,1)
       end  AND docu.document_object_id=dsc.document_object_id
       AND siu.user_name = :2 AND siu.enterprise_id = sie.enterprise_id AND  stages.stage_id=dsc.stage_id 
       and lfc.lifecycle_id=dsc.life_cycle_id and   docu.approval_process_id=dsc.life_cycle_id   and                 
        sie.enterprise_name = :3 AND RTRIM(docu.document_name)=RTRIM(:1) AND dsc.stage_seq=1  and
             docu.approval_process_id=lfc.lifecycle_id and siu.user_id=dsc.user_id and obsolete_flag='N'
        UNION ALL 
        select distinct lifecycle_name,  lifecycle_id from si_dms_lifecycles where lifecycle_id =
   ( select case WHEN EXISTS (
    SELECT obs_lfc.lifecycle_id  FROM si_dms_lifecycles obs_lfc
WHERE obs_lfc.lifecycle_id = lfcyc.obsolete_lifecycle_id AND SYSDATE BETWEEN obs_lfc.start_date
AND NVL (obs_lfc.end_date, SYSDATE + 1))
THEN (SELECT DISTINCT obs_lfc.lifecycle_id FROM si_dms_lifecycles obs_lfc  WHERE obs_lfc.lifecycle_id =lfcyc.obsolete_lifecycle_id)
else (SELECT DISTINCT obs_lfc.lifecycle_id FROM si_dms_lifecycles obs_lfc     WHERE obs_lfc.lifecycle_id =lfcyc.lifecycle_id
AND SYSDATE BETWEEN obs_lfc.start_date AND NVL (obs_lfc.end_date,
SYSDATE + 1)) END as lfc_id 
FROM si_dms_lifecycles lfcyc, ms_dms_documents docs WHERE :4 IN ('Obsolete Document') and lfcyc.lifecycle_id=docs.approval_process_id
          and document_name=:1 and obsolete_flag='N'))
