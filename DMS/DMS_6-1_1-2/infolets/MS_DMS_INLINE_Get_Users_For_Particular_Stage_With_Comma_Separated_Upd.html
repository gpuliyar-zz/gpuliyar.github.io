                                                        select rtrim(xmlagg(xmlelement(e,user_id|| ',')).extract('//text()').getclobval(),',') as user_id_list,
rtrim(xmlagg(xmlelement(e,user_name|| ',')).extract('//text()').getclobval(),',') as user_name_list
from (
select user_id,user_name from(
SELECT g.user_id as user_id,b.FULL_NAME as user_name
FROM
ms_dms_lifecycle_all_stage_v a,MS_DMS_LOCAL_USERS_T b, 
SI_DMS_GROUP_DETAILS g,SI_DMS_LIFECYCLE_ROLE_GRPS lg,si_dms_roles rl 
WHERE
g.role_group_id =lg.role_group_id
AND lg.lifecycle_id = a.lifecycle_id
AND lg.stage_id = a.stage_id
and rl.role_id=lg.role_id 
AND LG.STAGE_ID=:4
AND G.USER_ID=B.USER_ID    AND G.USER_ID=(CASE WHEN  B.OUT_OF_OFFICE_USERID>=1 THEN G.USER_ID ELSE 0 END)
union
select user_id,FULL_NAME as user_name from MS_DMS_USERS_PV where user_name=:2 and
  exists( select 1 from si_dms_lifecycle_stages f where add_owner_flag='Y' and f.stage_id=:4 )
  )
   where :1=0
UNION
select to_number(user_id),
       (select full_name from ms_dms_local_users_t where user_id = a.user_id) as user_name from
(SELECT TRIM( SUBSTR ( txt
                     , INSTR (txt, ',', 1, level ) + 1
                     , INSTR (txt, ',', 1, level+1
                     )
               - INSTR (txt, ',', 1, level) -1 ) )
         AS user_id
FROM (SELECT ','||STAGE_USERS ||',' as txt
FROM MS_DMS_INLINE_UPDATE_FORM_V WHERE PROCESS_INSTANCE_ID=:3
and INSTANCE_ID= ( SELECT MAX(INSTANCE_ID) FROM MS_DMS_INLINE_UPDATE_FORM_V WHERE PROCESS_INSTANCE_ID= :3)  and stage_id=:4
AND :1=5
UNION
SELECT ','||STAGE_USERS ||',' as txt
FROM MS_DMS_INLINE_UPDATE_FORM_Vs WHERE PROCESS_INSTANCE_ID=:3
and INSTANCE_ID= ( SELECT MAX(INSTANCE_ID) FROM MS_DMS_INLINE_UPDATE_FORM_Vs WHERE PROCESS_INSTANCE_ID= :3) and stage_id=:4
AND :1=1
)
CONNECT BY level <=
              length(txt)-length(replace(txt,',',''))-1) a
)
