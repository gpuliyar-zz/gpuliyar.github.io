                                                        SELECT Count(1),pid,Doc_Status,Document_Name,Doc_Num,Category,Life_Cycle,Stages,Due_By,MS_APPS_UTILITIES.GET_USER_FULL_NAME(USR_name),Purpose,Ver_Num
FROM (SELECT distinct b.process_instance_id pid,b.document_name,b.document_number doc_num,b.purpose,b.version_number ver_num,
             b.category,b.life_cycle_name life_cycle,ms_apps_utilities.get_user_name(a.user_id) usr_name,
             'Stage-'||( SELECT count(1)+1 cnt 
                   FROM ms_dms_documents_stage_dtl
                   WHERE  process_instance_id= a.process_instance_id
                   AND    stage_seq < a.stage_seq) stages,
             CASE 
                WHEN (to_date(SYSDATE,'DD/MM/YYYY') - to_date(nvl(a.assignment_creation_date,sysdate),'DD/MM/YYYY')) <= 0  THEN '(0 days)' 
                WHEN (to_date(SYSDATE,'DD/MM/YYYY') - to_date(nvl(a.assignment_creation_date,sysdate),'DD/MM/YYYY')) >  0  AND  
                     (to_date(SYSDATE,'DD/MM/YYYY') - to_date(nvl(a.assignment_creation_date,sysdate),'DD/MM/YYYY')) <= 7  THEN '(1-7 days)' 
                WHEN (to_date(SYSDATE,'DD/MM/YYYY') - to_date(nvl(a.assignment_creation_date,sysdate),'DD/MM/YYYY')) >  7  AND 
                     (to_date(SYSDATE,'DD/MM/YYYY') - to_date(nvl(a.assignment_creation_date,sysdate),'DD/MM/YYYY')) <= 21 THEN '(8-21 days)'  
                WHEN (to_date(SYSDATE,'DD/MM/YYYY') - to_date(nvl(a.assignment_creation_date,sysdate),'DD/MM/YYYY')) >  21 THEN '(>21 days)'
             END AS Due_By,
            'In-Process' doc_status 
      FROM  MS_DMS_DOCS_IN_PROCESS_V b,MS_DMS_USER_GRP_CAT_RPT_V  MDGCV,ms_dms_documents_stage_usr_dtl a
               WHERE         b.CATEGORY=MDGCV.CATEGORY
                 AND  NVL(b.SUB_CATEGORY1,'%')=NVL(MDGCV.SUBCATEGORY1,'%')
                AND  NVL(b.SUB_CATEGORY2,'%')=NVL(MDGCV.SUBCATEGORY2,'%')
                 AND NVL(b.SUB_CATEGORY3,'%')=NVL(MDGCV.SUBCATEGORY3,'%')
                AND NVL(b.SUB_CATEGORY4,'%')=NVL(MDGCV.SUBCATEGORY4,'%')
                and a.process_instance_id=b.process_instance_id and current_process_status='Y'
              and  b.usr_name=MDGCV.user_name
             and MDGCV.user_name =:6 
     
)
 WHERE Doc_status  = :1
AND   (Due_By      = :2
OR    Category    = :3
OR    (life_cycle = :4
       AND 
       Stages     = :5))
GROUP BY pid,document_name,doc_num,category,purpose,doc_status,Life_Cycle,ver_num,usr_name,stages,Due_By
