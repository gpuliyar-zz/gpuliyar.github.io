                                                        SELECT month_year,SUBSTR(month_year,5),TO_CHAR(TO_DATE(month_year,'mon-yyyy'),'MM'),
       SUM (published) , SUM (initiated),SUM (approved) 
FROM  (SELECT   month_year,
                DECODE (doc_status,'Published', COUNT (1),0) AS published,
                DECODE (doc_status,'Initiated', COUNT (1),0) AS initiated,
                DECODE (doc_status, 'Approved', COUNT (1),0) AS approved
       FROM    (SELECT DISTINCT aud.document_name,
                       map.doc_status doc_status,
                       TO_CHAR (aud.aud_sys_date,'Mon-YYYY') month_year
                FROM   ms_dms_doc_audit_history_v aud,
                       ms_dms_docs_map_v map,MS_DMS_USER_GRP_CAT_RPT_V MDGCV
                WHERE  aud.document_name = map.doc_name
                AND    TO_CHAR (aud.aud_sys_date, 'Mon-YYYY') IN (SELECT COLUMN_VALUE month_year
                                                                  FROM TABLE (get_age_month_year((SYSDATE-300),SYSDATE))
                                                                  WHERE ROWNUM <= :1)
                   and  aud.CATEGORY_ID=MDGCV.CATEGORY_ID
                AND  NVL(aud.SUB_CATEGORY1_ID,0)=NVL(MDGCV.SUBCATEGORY1_ID,0)
                AND  NVL(aud.SUB_CATEGORY2_ID,0)=NVL(MDGCV.SUBCATEGORY2_ID,0)
                 AND NVL(aud.SUB_CATEGORY3_ID,0)=NVL(MDGCV.SUBCATEGORY3_ID,0)
                AND NVL(aud.SUB_CATEGORY4_ID,0)=NVL(MDGCV.SUBCATEGORY4_ID,0)
                and MDGCV.USER_NAME=:2)
       GROUP BY month_year,doc_status)
GROUP BY SUBSTR(month_year,5),TO_CHAR(TO_DATE(month_year,'mon-yyyy'),'MM'),month_year
ORDER BY 2,3,1
