User ((User_Name = ('@USER@'))OR(User_Name = ('@ALL_DMS@'))) 
                                                        select  a.creation_date,a.current_level_name,a.current_level_number,
        a.category,sub_category1,'WIP Document',a.document_name,
        c.user_name user_name,
        u.user_name approved_by,
        si_common_sv.get_attachment_url(793,a.instance_id,10,'Click Here')"CLICK_HERE"
from    SI_DMS_DOCUMENT_REVIEW_V a,
        MS_DMS_ACCESS_GROUP b,
        MS_DMS_ACCESS_GROUP_USERS c,
        si_users u,
        (select distinct category, access_group from MS_DMS_USER_PROFILES) d
where   NOT EXISTS (SELECT '1'
                    FROM    ms_dms_documents e
                    WHERE   a.process_instance_id = e.process_instance_id)
and     d.category = a.category
and     u.user_id = a.created_by
and     b.group_name = d.access_group
and     b.GROUP_TYPE = 'DRM'
and     b.group_name = c.group_name
and     c.user_name not in (select user_name
                            from   si_users u,
                                   si_profiles p,
                                   si_user_profiles up
                            WHERE  p.profile_name = 'ALL_DMS'
                            AND    p.profile_id = up.profile_id
                            AND    up.profile_value = 'YES'
                            AND    u.user_id = up.user_id)
and    a.instance_id  = (select max(instance_id)
                         FROM   SI_DMS_DOCUMENT_REVIEW_V aa where aa.process_instance_id  = a.process_instance_id)
union
select  a.creation_date,a.current_level_name,a.current_level_number,
        a.category,sub_category1,'WIP Document',a.document_name,
        'YES',
        u.user_name approved_by,
        si_common_sv.get_attachment_url(793,a.instance_id,10,'Click Here')"CLICK_HERE"
from    SI_DMS_DOCUMENT_REVIEW_V a,
        si_users u
where   NOT EXISTS (SELECT '1'
                    FROM    ms_dms_documents e
                    WHERE   a.process_instance_id = e.process_instance_id)
and    a.instance_id  = (select max(instance_id)
                         FROM   SI_DMS_DOCUMENT_REVIEW_V aa where aa.process_instance_id  = a.process_instance_id)
and     u.user_id = a.created_by
