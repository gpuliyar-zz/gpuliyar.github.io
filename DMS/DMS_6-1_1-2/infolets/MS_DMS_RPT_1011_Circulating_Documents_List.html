                                                        SELECT a.purpose,
  a.document_name,
  a.document_number,
  a.version_number,
  ms_dms_mls_utilities.MS_DMS_GET_CATEGORY_MLSVALUE(:1, a.category_id)     AS category,
  ms_dms_mls_utilities.MS_DMS_GET_CATEGORY_MLSVALUE(:1,a.sub_category1_id) AS sub_category1,
  ms_dms_mls_utilities.MS_DMS_GET_CATEGORY_MLSVALUE(:1,a.sub_category2_id) AS sub_category2,
  ms_dms_mls_utilities.MS_DMS_GET_CATEGORY_MLSVALUE(:1,a.sub_category3_id) AS sub_category3,
  ms_dms_mls_utilities.MS_DMS_GET_CATEGORY_MLSVALUE(:1,a.sub_category4_id) AS sub_category4,
  Ms_Dms_Generate_Edit_Url(
  (SELECT metric_id
  FROM SI_METRICS_T
  WHERE metric_name ='MS_DMS_REVIEW_DOCUMENT'
  ),a.process_instance_id,C.USER_ID),
  (SELECT user_name FROM si_users WHERE user_id = c.user_id
  ) AS user_name,
  Ms_Dms_Doc_Url_For_Namereport(a.process_instance_id,d.audit_id,'In Process',a.document_attachment,a.document_name),
  a.document_object_id,
  'I' Flag,
  ms_dms_utilities.ms_dms_get_val_for_locale(:1, 'Attributes') Meta_Data
FROM MS_DMS_DOCUMENTS_HDR a,
  MS_DMS_DOCUMENTS_STAGE_DTL b,
  MS_DMS_DOCUMENTS_STAGE_USR_DTL c,
  MS_DMS_DOC_AUDIT_HISTORY d
WHERE d.process_instance_id  = a.process_instance_id
AND a.process_instance_id    = b.process_instance_id
AND a.transaction_id         = b.transaction_id
AND b.process_instance_id    = c.process_instance_id
AND b.stage_seq              = c.stage_seq
AND in_process_flag          = 'Y'
AND c.current_process_status = 'Y'
AND nvl(a.IS_INLINE,0)             <>'Y'
AND a.current_stage_seq      = b.stage_seq
AND EXISTS
  (SELECT 1
  FROM SI_EVENT_ASSIGNMENTS
  WHERE assignment_id = c.user_assignment_id
  )
AND d.instance_id =
  (SELECT MAX(instance_id)
  FROM MS_DMS_DOC_AUDIT_HISTORY
  WHERE process_instance_id = d.process_instance_id
  GROUP BY process_instance_id
  )
AND ms_apps_utilities.GET_USER_NAME(C.USER_ID) = :1

UNION

SELECT T2.PURPOSE,
  T2.DOCUMENT_NAME,
  T2.DOCUMENT_NUMBER,
  T2.VERSION_NUMBER,
  MS_DMS_MLS_UTILITIES.MS_DMS_GET_CATEGORY_MLSVALUE (:1, T2.category_id)      AS category,
  MS_DMS_MLS_UTILITIES.MS_DMS_GET_CATEGORY_MLSVALUE (:1, T2.sub_category1_id) AS sub_category1,
  MS_DMS_MLS_UTILITIES.MS_DMS_GET_CATEGORY_MLSVALUE (:1, T2.sub_category2_id) AS sub_category2,
  MS_DMS_MLS_UTILITIES.MS_DMS_GET_CATEGORY_MLSVALUE (:1, T2.sub_category3_id) AS sub_category3,
  MS_DMS_MLS_UTILITIES.MS_DMS_GET_CATEGORY_MLSVALUE (:1, T2.sub_category4_id) AS sub_category4,
  Ms_Dms_Generate_Edit_Url(
  (SELECT metric_id
  FROM SI_METRICS_T
  WHERE metric_name ='MS_DMS_INLINE_REVIEW_DOCUMENT'
  ), t2.process_instance_id, t3.user_id,t3.ASSIGNMENT_ID),
  (SELECT user_name FROM si_users WHERE user_id = T3.user_id
  ) AS user_name,
  Ms_Dms_Doc_Url_For_Namereport(t2.process_instance_id,T1.audit_id,'In Process',T2.document_attachment,T2.document_name),
  T2.document_object_id,
  'I' Flag,
  ms_dms_utilities.ms_dms_get_val_for_locale(:1, 'Attributes') Meta_Data
FROM Ms_Dms_Doc_Audit_History T1,
  Ms_Dms_Documents_Hdr T2,
  Si_Event_Assignments T3
WHERE T1.PROCESS_INSTANCE_ID = T2.PROCESS_INSTANCE_ID
AND T1.PROCESS_INSTANCE_ID   = T3.PROCESS_INSTANCE_ID
AND T2.PROCESS_INSTANCE_ID   = T3.PROCESS_INSTANCE_ID
AND T2.IN_PROCESS_FLAG       = 'Y'
AND T2.IS_INLINE             ='Y'
AND T1.INSTANCE_ID          IN
  (SELECT MAX(INSTANCE_ID)
  FROM MS_DMS_DOC_AUDIT_HISTORY
  WHERE PROCESS_INSTANCE_ID = T1.PROCESS_INSTANCE_ID
  GROUP BY PROCESS_INSTANCE_ID
  )
AND MS_APPS_UTILITIES.get_user_name(T3.USER_ID) = :1
