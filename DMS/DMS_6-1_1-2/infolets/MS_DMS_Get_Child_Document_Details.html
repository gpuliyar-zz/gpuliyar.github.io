a.document_name,
a.document_object_id,
a.document_object_type,
a.author_name,
a.owner_name,
a.version_number,
Ms_Dms_Doc_Url_For_Report
(
a.process_instance_id,
a.audit_id,
DECODE(a.document_status_flag,'P','Production','I','In Process','O','Obsolete','T','Terminated'),
a.document_attachment,
a.document_name
),
a.ORIGINAL_CATEGORY_NAME, 
a.ORIGINAL_SUB_CATEGORY1, 
a.ORIGINAL_SUB_CATEGORY2, 
a.ORIGINAL_SUB_CATEGORY3,

a.ORIGINAL_AUTHOR_NAME
FROM MS_DMS_DOC_AUDIT_HISTORY a
WHERE (document_object_id ,parent_audit_id) IN  
(SELECT child_object_id,parent_audit_id  FROM MS_DMS_AUDIT_CHILD_REL b
WHERE parent_audit_id=:1 
START WITH parent_object_id=:2
CONNECT BY PRIOR b.child_object_id=b.parent_object_id)
