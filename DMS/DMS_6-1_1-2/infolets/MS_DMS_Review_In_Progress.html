MS_DMS_Review_In_Progress

Description This report infolet will fetch all the documents details where the documents are in the workflow and at what stage of the workflow they are in. It shall display the In-Process documents for which the current user is the Initiator, Author or Owner.

                                                        with usr as (select user_id from ms_dms_local_users_t where user_name = :9)
select (:1 || :2 ||:3 || :4 || :5 || :6 || :9) Dummy,
    hdr.document_number DOC_NUMBER,
	(select full_name from ms_dms_local_users_t where user_name = hdr.AUTHOR_NAME) DOCUMENT_AUTHOR,
	hdr.document_name DOCUMENT_NAME,
	MS_DMS_MLS_UTILITIES.MS_DMS_GET_LIFECYCLE_MLSVALUE(hdr.LIFE_CYCLE_ID,null,null,:9) LIFECYCLE_NAME,
	MS_DMS_MLS_UTILITIES.MS_DMS_GET_CATEGORY_MLSVALUE(:9,hdr.CATEGORY_ID) CATEGORY_NAME,
	hdr.current_stage_name as CURRENT_STAGE,
	ms_dms_document_review_rep_pkg.get_completed_pending_reviews(hdr.process_instance_id ,hdr.current_stage_id ,hdr.current_stage_seq ,hdr.document_object_id ,'P',hdr.life_cycle_id) PENDING_REVIEWS,
	trim(ms_dms_document_review_rep_pkg.get_myreport_users(hdr.process_instance_id ,hdr.current_stage_id ,hdr.current_stage_seq ,hdr.document_object_id,2,hdr.life_cycle_id,null)) outstanding_users,     
    ms_dms_document_review_rep_pkg.get_assigned_date(hdr.process_instance_id) ASSIGNED_DATE,
    trunc(hdr.EFFECTIVE_FROM ) EFFECTIVE_DATE, 
    ms_dms_document_review_rep_pkg.get_due_date(hdr.process_instance_id) due_date,
    hdr.SUB_CATEGORY1 SUB_CATEGORY1,
    hdr.SUB_CATEGORY2 SUB_CATEGORY2,
    hdr.SUB_CATEGORY3 SUB_CATEGORY3,
    hdr.SUB_CATEGORY4 SUB_CATEGORY4,
    ms_dms_document_review_rep_pkg.get_completed_pending_reviews(hdr.process_instance_id ,hdr.current_stage_id ,hdr.current_stage_seq ,hdr.document_object_id ,'C',hdr.life_cycle_id) COMP_REVIEWS,
    'Attributes' ATTRIBUTES1,
    hdr.DOCUMENT_OBJECT_ID DOCUMENT_OBJECT_ID,
    (SELECT  displayed_value
     FROM ms_qs_lov_org_values_v a,SI_LOCALES b,SI_USERS c
     WHERE LOV_NAME='MS_DMS_INLINE_DOCUMENT_TYPE' and a.locale_id=b.locale_id
     and USER_NAME=:9 and c.locale=b.locale_string and a.stored_value=hdr.DOC_TYPE) DOCUMENT_OBJECT_TYPE,
    1  ASSIGNMENT_ID,
    hdr.process_instance_id PROCESS_INSTANCE_ID,
    hdr.current_stage_id STAGE_ID,
    hdr.current_stage_seq STAGE_SEQ,
    (select full_name from ms_dms_local_users_t where user_name = hdr.Owner_Name) Owner_Name
FROM ms_dms_documents_hdr hdr,ms_dms_documents_stage_dtl STGDTL,usr
WHERE hdr.in_process_flag='Y'
AND hdr.transaction_id =stgdtl.transaction_id
AND hdr.current_stage_id =stgdtl.stage_id
AND stgdtl.current_stage_flag='Y'
AND EXISTS (SELECT 1 FROM si_event_assignments WHERE process_instance_id =hdr.process_instance_id)
AND upper(hdr.document_name) like '%' ||upper(:1) ||'%'
and nvl(hdr.Owner_Name,'x') = nvl(:3,nvl(hdr.Owner_Name,'x'))
and nvl(hdr.author_name,'x') = nvl(:2,nvl(hdr.author_name,'x'))
AND NVL(hdr.CATEGORY_ID,'x') = NVL(:6,nvl(hdr.CATEGORY_ID,'x'))
AND trunc(NVL(hdr.EFFECTIVE_FROM,SYSDATE)) = trunc(NVL(TO_DATE(:4),nvl(hdr.EFFECTIVE_FROM,SYSDATE)))
AND NVL(hdr.LIFE_CYCLE_ID,1) = NVL(:5,nvl(hdr.LIFE_CYCLE_ID,1))
AND  (author_id= usr.user_id or owner_id= usr.user_id  or originator_id= usr.user_id)
AND (CASE 
WHEN to_date(:7) is not null AND to_date(:8) is null AND trunc(ms_dms_document_review_rep_pkg.get_due_date(hdr.process_instance_id)) >= TRUNC(to_date(:7)) THEN 1
WHEN to_date(:8) is not null AND to_date(:7) is null AND trunc(ms_dms_document_review_rep_pkg.get_due_date(hdr.process_instance_id)) <= TRUNC(to_date(:8)) THEN 1
WHEN to_date(:8) is not null and to_date(:7) is not null AND (trunc(ms_dms_document_review_rep_pkg.get_due_date(hdr.process_instance_id)) <= TRUNC(to_date(:8)) 
AND trunc(ms_dms_document_review_rep_pkg.get_due_date(hdr.process_instance_id)) >= TRUNC(to_date(:7))) THEN 1
WHEN to_date(:8) is null and to_date(:7) is null THEN 1 
ELSE 0
END )=1
ORDER BY ms_dms_document_review_rep_pkg.get_assigned_date(hdr.process_instance_id) DESC
