                                                        select DOCUMENT_NAME,Ms_Dms_Doc_Url_For_nameReport
    (
    process_instance_id,
    audit_id,
    'Production',
    DOCUMENT_ATTACHMENT,
    'View'
    ) AS document_link,VERSION_NUMBER,VERSION_NUMBER,EFFECTIVE_DATE,META_DATA,Frequency,Document_object_id from 
(
SELECT B.DOCUMENT_NAME,A.VERSION_NUMBER,B.EFFECTIVE_DATE, MS_DMS_UTILITIES.MS_DMS_GET_VAL_FOR_LOCALE(:3,'Attributes') AS META_DATA,COUNT(1) AS Frequency,
A.DOCUMENT_OBJECT_ID, B.process_instance_id,B.audit_id,B.DOCUMENT_ATTACHMENT FROM 
MS_DMS_POLICY_ACCESS_HISTORY A,MS_DMS_ALL_PRODUCTION_DOCS B WHERE
A.DOCUMENT_OBJECT_ID=B.DOCUMENT_OBJECT_ID
AND A.DOCUMENT_OBJECT_ID=:2
AND A.VERSION_NUMBER=B.REVISION_NUMBER
and decode(:1,'','1',B.EFFECTIVE_DATE)=decode(:1,'','1',:1)
GROUP BY B.DOCUMENT_NAME,A.VERSION_NUMBER,B.EFFECTIVE_DATE,A.DOCUMENT_OBJECT_ID, B.PROCESS_INSTANCE_ID,B.AUDIT_ID,B.DOCUMENT_ATTACHMENT
order by version_number desc)
