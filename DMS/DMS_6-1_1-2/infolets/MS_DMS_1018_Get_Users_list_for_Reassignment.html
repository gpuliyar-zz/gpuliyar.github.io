                                                        SELECT b.user_name,  b.full_name as user_full_name,b.user_id
    FROM MS_DMS_DOCUMENTS_STAGE_USR_DTL a,MS_DMS_LOCAL_USERS_T b 
   WHERE process_instance_id = :1
   AND life_cycle_id=:2
   AND stage_seq=:3
   AND a.ORGINAL_USER_ID=b.user_id  
   and b.out_of_office_userid>=1
   union
   select b.user_name,  b.full_name as user_full_name,b.user_id
    FROM ms_dms_LF_User_list_v a,MS_DMS_LOCAL_USERS_T b 
   WHERE lifecycle_id=:2
   AND seq=:3 and  a.USER_ID=b.user_id
    UNION
SELECT  b.user_name,  b.full_name as user_full_name,b.user_id
 FROM SI_DMS_USER_GROUP_USERS a, MS_DMS_LOCAL_USERS_T b, SI_DMS_USER_GROUPS grp
WHERE a.user_id=b.user_id 
and a.user_group_id = grp.user_group_id and grp.DEFAULT_SELECT_FLAG = 'Y'
AND  a.user_group_id IN
( SELECT  access_group_id   FROM SI_DMS_ACCESS_GRP_CATS
  WHERE   CASE WHEN (NVL(subcategory4_id,'0')>'0') THEN subcategory4_id
                WHEN (NVL(subcategory3_id,'0')>'0') THEN subcategory3_id
                WHEN (NVL(subcategory2_id,'0')>'0') THEN subcategory2_id
                WHEN (NVL(subcategory1_id,'0')>'0') THEN subcategory1_id
                WHEN (NVL(category_id,'0')>'0') THEN category_id END  
 IN (SELECT CASE WHEN (NVL(sub_category4_id,'0')>'0') THEN sub_category4_id
                 WHEN (NVL(sub_category3_id,'0')>'0') THEN sub_category3_id
                WHEN  (NVL(sub_category2_id,'0')>'0') THEN sub_category2_id
                WHEN  (NVL(sub_category1_id,'0')>'0') THEN sub_category1_id
                WHEN       (NVL(category_id,'0')>'0') THEN category_id END AS Cat 
                FROM MS_DMS_DOCUMENTS_HDR b,ms_dms_documents_stage_usr_dtl a
    WHERE a.document_object_id=b.document_object_id
    AND a.transaction_id      =b.transaction_id
    AND a.process_instance_id =:1
    AND a.life_cycle_id       = :2
    AND stage_seq             =:3
    AND EXISTS
      (SELECT 1
      FROM SI_DMS_LIFECYCLES
      WHERE Stage_category_user_flag='Y'
      AND LIFECYCLE_ID              = :2
      AND a.life_cycle_id           =lifecycle_id
      )
    AND stage_seq=:3
   ))
and b.out_of_office_userid>=1
