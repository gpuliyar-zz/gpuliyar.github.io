                                                        Select b.Lifecycle_Name,b.Lifecycle_id 
From
(
SELECT 
       DISTINCT a.lifecycle_name,
       a.lifecycle_id
     FROM 
     ms_dms_list_lf_names_v a     
     WHERE 
     a.user_id = ms_dms_utilities.get_user_id(:1) 
     AND a.enterprise_id = MS_APPS_UTILITIES.get_enterprise_id(:2)
     AND   SUBSTR(a.access_rights,4,1)='1'     
     AND  :3  IN ('Create Document','UpVersion Document','Change Request','RCNUR')      
UNION ALL
(SELECT
     DISTINCT (SELECT obs_lfc.lifecycle_name FROM SI_DMS_LIFECYCLES obs_lfc WHERE obs_lfc.lifecycle_id=lifecycles.obsolete_lifecycle_id),
     lifecycles.obsolete_lifecycle_id   
FROM
    SI_DMS_LIFECYCLES lifecycles,
     MS_DMS_Upd_PROCESS_DOCS_V docs
WHERE
    TRIM(docs.document_name) = TRIM(:4) AND
    lifecycles.lifecycle_id = docs.approval_process_id AND
    :3 NOT IN ('Create Document','UpVersion Document','Change Request','RCNUR'))
UNION
      (SELECT  DISTINCT   a.lifecycle_name, a.lifecycle_id 
    FROM SI_DMS_LIFECYCLES a,SI_DMS_LIFECYCLE_STAGES b 
    WHERE 
          a.lifecycle_id=b.lifecycle_id 
          AND  a.start_date <= SYSDATE AND 
          NVL(a.end_date, SYSDATE+1) > SYSDATE   AND  add_owner_flag='Y'  and seq=1 AND enterprise_id=MS_APPS_UTILITIES.get_enterprise_id(:2)  
AND  :3  IN ('Create Document','UpVersion Document','Change Request','RCNUR'))) a,SI_DMS_LIFECYCLES_V b
where a.lifecycle_id=b.lifecycle_id and b.locale_id=:5
