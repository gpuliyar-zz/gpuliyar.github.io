Name Expression Description
Lifecycle ((Lifecycle_ID = :lifecycle_id)AND(Stage_Seq = :seq)AND(User_Name = :user_name)) 
                                                        select
	   e.lifecycle_id,
       e.lifecycle_name,
       d.STAGE_NAME,
       d.SEQ,
       decode(sum(substr(ACCESS_RIGHTS,7,1)),0,'No','Yes') read_flag,
       decode(sum(substr(ACCESS_RIGHTS,4,1)),0,'No','Yes') write_flag,
       decode(sum(substr(ACCESS_RIGHTS,5,1)),0,'No','Yes') comment_flag,
       decode(sum(substr(ACCESS_RIGHTS,1,1)),0,'No','Yes') delete_flag,
       d.APPROVAL_REQUIREMENT,
       decode(d.EMAIL_FLAG,'Y','Yes','No'),
       e1.user_name,
       e1.user_id
from
		si_dms_GROUP_DETAILS a,
       	SI_DMS_LIFECYCLE_ROLE_GRPS b,
      	si_dms_roles c,
		si_dms_lifecycles E,
        SI_DMS_LIFECYCLE_STAGES D,
        si_users E1
where
E.lifecycle_id = D.lifecycle_id
    and     b.ROLE_GROUP_ID =  a.ROLE_GROUP_ID
    and     b.stage_id = d.stage_id
	and		d.lifecycle_id = e.lifecycle_id
    and     b.role_id = c.role_id
    and     e1.user_id = a.user_id
group by
    e.lifecycle_id,--approval_process_id,
       e.lifecycle_name,--approval_process_name,
       d.STAGE_NAME,
       d.SEQ,
       d.APPROVAL_REQUIREMENT,
       decode(d.EMAIL_FLAG,'Y','Yes','No'),
       e1.user_name,
       e1.user_id
order by
1,2,4
