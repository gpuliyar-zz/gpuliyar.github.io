                                                        SELECT  '<input type=checkbox id="'
         || A.USER_ID
         || '" value="'
         || htf.ESCAPE_SC(a.full_name)
         || '"onClick=checkboxChange(this.id)></input>' AS checkbox, A.USER_ID as User_id, A.USER_NAME as User_Name, a.full_name as Full_Name FROM
MS_DMS_LF_Form_USER_LIST_V  b,MS_DMS_LOCAL_USERS_T a 
WHERE a.user_id=b.user_id AND stage_id=:1
UNION
SELECT  '<input type=checkbox id="'
         || ut.user_id
         || '" value="'
         || htf.ESCAPE_SC(ut.full_name)
         || '"onClick=checkboxChange(this.id)></input>', ut.user_id,ut.user_name,ut.full_name FROM 
(
SELECT (SELECT user_id FROM MS_DMS_LOCAL_USERS_T WHERE user_name=:2) AS USER_ID
                                           FROM SI_DMS_LIFECYCLE_ROLE_GRPS a,
                                                SI_DMS_USER_GROUP_USERS b,
												SI_DMS_LIFECYCLE_STAGES f,SI_DMS_ROLES g
                             WHERE g.role_id=a.role_id AND a.group_id = b.user_group_id(+)
                             and f.lifecycle_id= a.lifecycle_id 
                             AND ADD_OWNER_FLAG='Y' and a.stage_id=f.stage_id
                                 AND f.stage_id = :1
UNION
SELECT  USER_ID FROM SI_DMS_LIFECYCLE_ROLE_GRPS  A, SI_DMS_USER_GROUP_USERS B, SI_DMS_ROLES C
WHERE A.STAGE_ID=:1 AND A.ROLE_ID=C.ROLE_ID AND B.USER_GROUP_ID=A.GROUP_ID 
UNION
(SELECT   B.USER_ID  FROM SI_DMS_ACCESS_GRP_CATS grpcat
   ,SI_DMS_USER_GROUP_USERS B,MS_DMS_LOCAL_USERS_T C   ,SI_DMS_ROLES D, SI_DMS_USER_GROUPS grp
WHERE (grpcat.access_group_id=b.user_group_id  AND B.USER_ID=C.USER_ID 
 and b.user_group_id = grp.user_group_id and grp.DEFAULT_SELECT_FLAG = 'Y'
 AND GRPCAT. ROLE_ID=D.ROLE_ID AND B.USER_ID=C.USER_ID
 AND  DECODE(NVL(subcategory4_id,'0') ,'0',DECODE(NVL(SUBCATEGORY3_ID ,'0') ,'0',DECODE(NVL(SUBCATEGORY2_ID,'0'),'0',
DECODE(NVL( SUBCATEGORY1_ID,'0') ,'0',CATEGORY_ID,SUBCATEGORY1_ID),SUBCATEGORY2_ID),SUBCATEGORY3_ID),SUBCATEGORY4_ID)     =:3
      and EXISTS(SELECT a.lifecycle_id FROM SI_DMS_LIFECYCLES a,SI_DMS_LIFECYCLE_STAGES b
              WHERE a.lifecycle_id=b.lifecycle_id AND stage_id=:1 AND  STAGE_CATEGORY_USER_FLAG='Y'))
         UNION
       (SELECT user_id  FROM SI_DMS_LIFECYCLE_ROLE_GRPS  a,SI_DMS_GROUP_DETAILS b,SI_DMS_ROLES c
WHERE  a.role_group_id=b.role_group_id AND stage_id= :1 AND  a.role_id=c.role_id )
 )) u_tab,
MS_DMS_LOCAL_USERS_T ut WHERE ut.user_id=u_tab.user_id 
and ut.out_of_office_userid >= 1
