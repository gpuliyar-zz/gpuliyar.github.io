                                                        SELECT ut.user_id,ut.user_name,ut.full_name,NULL AS CREATE__A_DOCUMENT, NULL AS VIEW_ANNOTATIONS, NULL AS WRITE_ANNOTATIONS, NULL AS DOWNLOAD, NULL AS WRITE_ALL_USERS_ANNOTATIONS, NULL AS VIEW_A_DOCUMENT, NULL AS UPDATE_A_DOCUMENT, NULL AS PRINT__A_DOCUMENT, NULL AS VIEW_ALL_USERS_DOCUMENTS, NULL AS EDIT_METADATA
FROM 
(
SELECT (SELECT user_id FROM MS_DMS_LOCAL_USERS_T WHERE user_name=:2) AS USER_ID
                                           FROM SI_DMS_LIFECYCLE_ROLE_GRPS a,
                                                SI_DMS_USER_GROUP_USERS b,
												SI_DMS_LIFECYCLE_STAGES f,SI_DMS_ROLES g
                             WHERE g.role_id=a.role_id AND a.group_id = b.user_group_id(+)
                             and f.lifecycle_id= a.lifecycle_id 
                             AND ADD_OWNER_FLAG='Y' and a.stage_id=f.stage_id
                                 AND f.stage_id = :1
UNION
SELECT  USER_ID FROM SI_DMS_LIFECYCLE_ROLE_GRPS  A, SI_DMS_GROUP_DETAILS B, SI_DMS_ROLES C
WHERE A.STAGE_ID=:1 AND A.ROLE_ID=C.ROLE_ID AND B.ROLE_GROUP_ID=A.ROLE_GROUP_ID
UNION
(SELECT   B.USER_ID  FROM SI_DMS_ACCESS_GRP_CATS grpcat
   ,SI_DMS_GROUP_DETAILS B,SI_DMS_LIFECYCLE_ROLE_GRPS C   ,SI_DMS_ROLES D, SI_DMS_USER_GROUPS grp
WHERE (C.role_group_id =b.role_group_id AND c.role_id=D.role_id
 and grp.DEFAULT_SELECT_FLAG = 'Y'
 AND GRPCAT.ROLE_ID =D.ROLE_ID AND C.STAGE_ID =:1
 AND  DECODE(NVL(subcategory4_id,'0') ,'0',DECODE(NVL(SUBCATEGORY3_ID ,'0') ,'0',DECODE(NVL(SUBCATEGORY2_ID,'0'),'0',
DECODE(NVL( SUBCATEGORY1_ID,'0') ,'0',CATEGORY_ID,SUBCATEGORY1_ID),SUBCATEGORY2_ID),SUBCATEGORY3_ID),SUBCATEGORY4_ID)     =:3
    AND  :4 IN(SELECT a.lifecycle_id FROM SI_DMS_LIFECYCLES a,SI_DMS_LIFECYCLE_STAGES b
     WHERE a.lifecycle_id=b.lifecycle_id AND stage_id=:1 AND  STAGE_CATEGORY_USER_FLAG='Y'))
         UNION
       (SELECT user_id  FROM SI_DMS_LIFECYCLE_ROLE_GRPS  a,SI_DMS_GROUP_DETAILS b,SI_DMS_ROLES c
WHERE  a.role_group_id=b.role_group_id AND stage_id= :1 AND  a.role_id=c.role_id  AND :4 IN(SELECT a.lifecycle_id FROM SI_DMS_LIFECYCLES a,SI_DMS_LIFECYCLE_STAGES b
     WHERE a.lifecycle_id=b.lifecycle_id AND stage_id=:1 AND  STAGE_CATEGORY_USER_FLAG='Y') )
 )) u_tab,
MS_DMS_LOCAL_USERS_T ut WHERE ut.user_id=u_tab.user_id 
and ut.out_of_office_userid >= 1
