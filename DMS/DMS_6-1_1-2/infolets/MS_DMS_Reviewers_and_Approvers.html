                                                        SELECT c.lifecycle_id, c.seq, c.stage_name, c.approval_requirement, c.user_list, REPLACE(b.stage_user_list__mlc,',',', '), c.lifecycle_name
                        FROM SI_DMS_CREATE_DOCUMENT_V a,
                             SI_DMS_MODIFY_STAGE_USER_V b,
             	             SI_DMS_STAGE_DEFAULT_USER_LIST c
                        WHERE a.approval_process_id = b.lifecycle_id  AND a.process_instance_id = b.process_instance_id
                          AND a.instance_id = (SELECT MAX(t1.instance_id) FROM SI_DMS_CREATE_DOCUMENT_V t1
                                                WHERE trim(a.document_name) = trim(t1.document_name)
                                                  AND trim(a.document_name) = trim(:1) AND NVL(t1.enable_stage_user_flag, 'No') ='Yes' AND t1.process_instance_id = :2)
						AND  a.approval_process_id = c.lifecycle_id
						  AND b.stage_id = c.stage_id
						  AND b.stage_sequence = c.seq
                          AND NVL(a.enable_stage_user_flag,'No') = 'Yes'
						  AND trim(a.document_name) =  trim(:1)
						  AND a.process_instance_id = :2
						  UNION
						   SELECT c.lifecycle_id, c.seq, c.stage_name, c.approval_requirement,  c.user_list,  DECODE(c. seq, 1, b.user_list, c.USER_LIST), c.lifecycle_name
                            FROM SI_DMS_CREATE_DOCUMENT_V b,
                                 SI_DMS_STAGE_DEFAULT_USER_LIST c
                           WHERE b.approval_process_id = c.lifecycle_id
                             AND trim(b.document_name) = trim(:1)
                             AND b.process_instance_id = :2
                             AND b.instance_id = (SELECT MAX(instance_id) FROM SI_DMS_CREATE_DOCUMENT_V
                                                   WHERE process_instance_id = b.process_instance_id
                                                     AND trim(document_name) = trim(:1) and process_instance_id=:2)
                             AND NVL(b.enable_stage_user_flag, 'No') !='Yes'
