                                                        SELECT lut.user_name, lut.full_name
FROM ms_dms_local_users_t lut
WHERE exists (select 1 from MS_DMS_DISTRIBUTED_ADMIN da, ms_dms_local_users_t lut1 
              where da.user_name = lut1.user_id and lut1.user_name = :1)
UNION
SELECT lut.user_name, lut.full_name
FROM SI_DMS_USER_GROUP_USERS A, SI_DMS_ACCESS_GRP_CATS B, ms_dms_local_users_t lut
WHERE A.USER_GROUP_ID=B.ACCESS_GROUP_ID and lut.user_id = a.user_id
AND exists (select 1 from (SELECT EXTRACT(VALUE(d),'//row/text()').getstringval() AS category
                           FROM (SELECT XMLTYPE('<rows><row>'||REPLACE(CATEGORIES,',','</row><row>')||'</row></rows>') AS xmlval
                                 FROM MS_DMS_DISTRIBUTED_ADMIN da, ms_dms_local_users_t lut1 
                                 WHERE da.user_name = lut1.user_id and lut1.user_name = :1) x,
                           TABLE(XMLSEQUENCE(EXTRACT(X.XMLVAL,'/rows/row'))) d) t_tab 
			where t_tab.category = b.category)
