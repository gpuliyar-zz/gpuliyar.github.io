                                                        SELECT document_name as document_name,
         document_number as document_number,
         revision_number as revision_number,
         MS_DMS_MLS_UTILITIES.MS_DMS_GET_CATEGORY_MLSVALUE (:1, MDD.category_id) AS category,
         MS_DMS_MLS_UTILITIES.MS_DMS_GET_CATEGORY_MLSVALUE (:1, sub_category1_id) AS sub_category1,
         MS_DMS_MLS_UTILITIES.MS_DMS_GET_CATEGORY_MLSVALUE (:1, sub_category2_id) AS sub_category2,
         MS_DMS_MLS_UTILITIES.MS_DMS_GET_CATEGORY_MLSVALUE (:1, sub_category3_id) AS sub_category3,
         MS_DMS_MLS_UTILITIES.MS_DMS_GET_CATEGORY_MLSVALUE (:1, sub_category4_id) AS sub_category4,
         TIMESTAMP as Date_Submitted,
         Ms_Dms_Doc_Url_For_NameReport (process_instance_id,
                                        audit_id,
                                        'Obsolete',
                                        document_attachment,
                                        document_name) as Document_Name_URL,
         Document_Object_id as Document_Object_id,
         'Y' as Flag,
         ms_dms_utilities.ms_dms_get_val_for_locale (:1, 'Attributes') as Attributes,
MS_APPS_UTILITIES.Get_Display_Value(ms_grc_utilities.active_ent_id, 'MS_DMS_INLINE_DOCUMENT_TYPE', mdd.doc_type,1,(select locale_id from si_locales aa , si_users_t bb where aa.locale_string=bb.locale and bb.user_name=:1)) as document_type	    FROM MS_DMS_DOCUMENTS mdd, MS_DMS_USER_GRP_CAT_RPT_V c
   WHERE     obsolete_flag = 'Y'
         AND MDD.CATEGORY = c.CATEGORY
         AND NVL (MDD.SUB_CATEGORY1, '%') = NVL (c.SUBCATEGORY1, '%')
         AND NVL (MDD.SUB_CATEGORY2, '%') = NVL (c.SUBCATEGORY2, '%')
         AND NVL (MDD.SUB_CATEGORY3, '%') = NVL (c.SUBCATEGORY3, '%')
         AND NVL (MDD.SUB_CATEGORY4, '%') = NVL (c.SUBCATEGORY4, '%')
         AND c.user_name = :1
ORDER BY DOCUMENT_NAME
