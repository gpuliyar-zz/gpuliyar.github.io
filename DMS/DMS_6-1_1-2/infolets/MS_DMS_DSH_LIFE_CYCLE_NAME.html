                                                        SELECT count(1) as  cnt,lifecycle_name,doc_status   from (
(
SELECT  distinct process_instance_id, document_name, 
 'In Production' doc_status,MDD.CATEGORY||','|| MDD.SUB_CATEGORY1||','|| MDD.SUB_CATEGORY2||','||MDD.SUB_CATEGORY3||','||MDD.SUB_CATEGORY4  concat,
(select MS_DMS_MLS_UTILITIES.MS_DMS_GET_LIFECYCLE_MLSVALUE(lifecycle_id, 0, NULL, :2) from si_dms_lifecycles where lifecycle_id=mdd.approval_process_id)  lifecycle_name FROM ms_dms_documents MDD,MS_DMS_USER_GRP_CAT_RPT_V MDGCV
 WHERE checked_out_flag IN ('N', 'I')
AND obsolete_flag = 'N'
AND document_object_type = 'N'
and  MDD.CATEGORY_ID=MDGCV.CATEGORY_ID
AND  NVL(MDD.SUB_CATEGORY1_ID,0)=NVL(MDGCV.SUBCATEGORY1_ID,0)
AND  NVL(MDD.SUB_CATEGORY2_ID,0)=NVL(MDGCV.SUBCATEGORY2_ID,0)
AND NVL(MDD.SUB_CATEGORY3_ID,0)=NVL(MDGCV.SUBCATEGORY3_ID,0)
 AND NVL(MDD.SUB_CATEGORY4_ID,0)=NVL(MDGCV.SUBCATEGORY4_ID,0)
 and mdd.approval_process_id is not null
 and
              MDD.CATEGORY||','|| MDD.SUB_CATEGORY1||','|| MDD.SUB_CATEGORY2||','||MDD.SUB_CATEGORY3||','||MDD.SUB_CATEGORY4 in   (SELECT CATEGORY||','|| SUBCATEGORY1||','|| SUBCATEGORY2||','||SUBCATEGORY3||','||SUBCATEGORY4  
                    FROM MS_DMS_USER_GRP_CAT_RPT_V  WHERE USER_NAME=:2 )
             UNION
             (             
             SELECT  distinct mddiv.process_instance_id,mddiv.document_name,
                    'In-Process' doc_status,  MDDIV.CATEGORY||','|| MDDIV.SUB_CATEGORY1||','|| MDDIV.SUB_CATEGORY2||','|| MDDIV.SUB_CATEGORY3||','||MDDIV.SUB_CATEGORY4 cat_concat,
                    MS_DMS_MLS_UTILITIES.MS_DMS_GET_LIFECYCLE_MLSVALUE(mddiv.life_cycle_id, 0, NULL, :2)  AS LIFE_CYCLE_NAME
               FROM ms_dms_docs_in_process_v MDDIV,MS_DMS_USER_GRP_CAT_RPT_V MDGCV 
               WHERE    MDDIV.CATEGORY=MDGCV.CATEGORY
                AND  NVL(MDDIV.SUB_CATEGORY1,'%')=NVL(MDGCV.SUBCATEGORY1,'%')
                AND  NVL(MDDIV.SUB_CATEGORY2,'%')=NVL(MDGCV.SUBCATEGORY2,'%')
                 AND NVL(MDDIV.SUB_CATEGORY3,'%')=NVL(MDGCV.SUBCATEGORY3,'%')
                AND NVL(MDDIV.SUB_CATEGORY4,'%')=NVL(MDGCV.SUBCATEGORY4,'%')
                AND  MDGCV.USER_ID=MS_APPS_UTILITIES.GET_USER_ID(MDDIV.USR_NAME ) 
                and MDDIV.USR_NAME=:2                
                )                
                )             
              )  
              where Doc_status=:1              
group by  doc_status,lifecycle_name
