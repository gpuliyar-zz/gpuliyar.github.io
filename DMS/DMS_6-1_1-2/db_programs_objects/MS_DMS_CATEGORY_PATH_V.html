MS_DMS_CATEGORY_PATH_V 

                                                
  CREATE OR REPLACE FORCE VIEW "PDMS61SP2_DEV"."MS_DMS_CATEGORY_PATH_V" ("LIFECYCLE_ID", "LEAF_CATEGORY_ID", "PARENT_CATEGORY_ID", "CATEGORY_ID_PATH", "CATEGORY_NAME_PATH") AS 
  SELECT lif_cat.LIFECYCLE_ID , cat_det.LEAF_CATEGORY_ID , cat_det.PARENT_CATEGORY_ID , cat_det.category_id_path , cat_det.category_name_path
FROM si_dms_lifecycle_categories lif_cat,
    ( SELECT  CATEGORY_ID as LEAF_CATEGORY_ID 
             ,NULL AS PARENT_CATEGORY_ID 
             ,to_char(CATEGORY_ID) AS category_id_path
             ,CATEGORY_NAME AS category_name_path 
      FROM SI_DMS_CATEGORIES 
      WHERE category_id NOT IN ( SELECT DISTINCT SUBCATEGORY_ID FROM si_dms_CAT_SUBCATEGORIES)
      UNION   
      SELECT  a.SUBCATEGORY_ID AS  LEAF_CATEGORY_ID
            , a.CATEGORY_ID    AS  PARENT_CATEGORY_ID
            , root_category || a.sub_path AS category_id_path 
            , ms_dms_utilities.get_category_names(root_category || a.sub_path) AS category_name_path
      FROM (SELECT  SUBCATEGORY_ID
                  , CATEGORY_ID
                  , LEVEL AS lev 
                  , SYS_CONNECT_BY_PATH(SUBCATEGORY_ID, '/') AS sub_path 
                  , CONNECT_BY_ROOT CATEGORY_ID root_category 
            FROM si_dms_CAT_SUBCATEGORIES
            CONNECT BY PRIOR SUBCATEGORY_ID = CATEGORY_ID
            ) A 
      WHERE  (A.SUBCATEGORY_ID,A.lev) IN (SELECT SUBCATEGORY_ID,MAX(LEVEL) 
                                          FROM  si_dms_CAT_SUBCATEGORIES 
                                          GROUP BY SUBCATEGORY_ID
                                          CONNECT BY PRIOR SUBCATEGORY_ID = CATEGORY_ID 
                                          )
      ) cat_det 
WHERE lif_cat.CATEGORY_ID = cat_det.LEAF_CATEGORY_ID
