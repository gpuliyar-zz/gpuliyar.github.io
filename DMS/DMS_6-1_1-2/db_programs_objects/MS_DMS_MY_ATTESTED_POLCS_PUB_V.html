MS_DMS_MY_ATTESTED_POLCS_PUB_V 

                                                
  CREATE OR REPLACE FORCE VIEW "PDMS61SP2_DEV"."MS_DMS_MY_ATTESTED_POLCS_PUB_V" ("DOCUMENT_NAME", "VERSION_NUMBER", "ACTION_TYPE", "FEEDBACK_DATE", "ACTION_COMMENTS", "ASSIGN_DUE_DATE", "POLICY_END_DATE", "USER_NAME") AS 
  SELECT (''
||B.DOCUMENT_NAME
||'') AS DOCUMENT_NAME
,A.VERSION_NUMBER AS VERSION_NUMBER
,L.DISPLAYED_VALUE AS ACTION_TYPE
,A.FEEDBACK_DATE AS FEEDBACK_DATE
,A.ACTION_COMMENTS AS ACTION_COMMENTS
,P.ASSIGN_DUE_DATE AS ASSIGN_DUE_DATE
,P.POLICY_END_DATE AS POLICY_END_DATE
,A.USER_NAME AS USER_NAME
FROM (SELECT DOC_OBJECTID, DOC_PID, DOC_AUDITID, VERSION_NUMBER, DOCUMENT_NAME, USER_NAME, ACTION_COMMENTS, FEEDBACK_DATE, ACTION_TYPE,
             ROW_NUMBER()OVER(PARTITION BY DOC_OBJECTID, USER_NAME ORDER BY TO_NUMBER(REPLACE(VERSION_NUMBER,'.')) DESC) RN 
      FROM MS_DMS_POLICY_FEEDBACK 
     WHERE ACTION_TYPE IS NOT NULL) A, 
     (SELECT MAX(CASE WHEN UPPER(PNAME.PARAMETER_NAME) = 'MS DMS NATIVE VIEWER ERROR DISPLAY_YN' THEN PVALUE.PARAMETER_VALUE ELSE NULL END) AS ERROR_DISPLAY_YN,
             MAX(CASE WHEN UPPER(PNAME.PARAMETER_NAME) = 'MS DMS NATIVE VIEWER ERROR DISPLAY' THEN PVALUE.PARAMETER_VALUE ELSE NULL END) AS ERROR_DISPLAY,
             MAX(CASE WHEN UPPER(PNAME.PARAMETER_NAME) = 'MS_DMS_CONV_INPROCESS_MSG' THEN PVALUE.PARAMETER_VALUE ELSE NULL END) AS CONV_INPROCESS_MSG ,
             MAX(CASE WHEN UPPER(PNAME.PARAMETER_NAME) = 'MS_DMS_PDF_CONVERSION_RQD_YN' THEN PVALUE.PARAMETER_VALUE ELSE NULL END) AS CONVERSION_RQD_YN 
        FROM MS_QS_PARAMETER_VALUES PVALUE, MS_QS_PARAMETER_TYPES PTYPE, MS_QS_PARAMETER_NAMES PNAME
       WHERE PNAME.PARAMETER_ID      = PVALUE.PARAMETER_ID
         AND PTYPE.PARAMETER_TYPE_ID = PVALUE.PARAMETER_TYPE_ID
         AND PTYPE.PARAMETER_TYPE_ID = PNAME.PARAMETER_TYPE_ID
         AND PVALUE.ORG_ENTITY_ID    = 1
         AND (   (UPPER(PTYPE.PARAMETER_TYPE) = 'MS DMS NATIVE VIEWER ERRORS' AND 
                  UPPER(PNAME.PARAMETER_NAME) IN ('MS DMS NATIVE VIEWER ERROR DISPLAY_YN', 'MS DMS NATIVE VIEWER ERROR DISPLAY', 'MS_DMS_CONV_INPROCESS_MSG'))
              OR (UPPER(PTYPE.PARAMETER_TYPE) = 'MS_DMS_PDF_CONVERSION_RQD_YN' AND UPPER(PNAME.PARAMETER_NAME) = 'MS_DMS_PDF_CONVERSION_RQD_YN'))) PV,
MS_DMS_POLICY P, 
MS_DMS_ALL_PRODUCTION_DOCS B,
MS_PDM_LIST_OF_VALUES_V L,
MS_DMS_LOCAL_USERS_T U
WHERE B.AUDIT_ID = A.DOC_AUDITID
AND A.DOC_AUDITID = P.DOC_AUDITID(+)
AND L.LOCALE_ID = U.LOCALE_ID
AND U.USER_NAME = A.USER_NAME
AND L.STORED_VALUE = A.ACTION_TYPE
AND L.LOV_NAME = 'MS_PDM_ATTESTED_POLICY_ACTIONS'
AND A.RN = 1
