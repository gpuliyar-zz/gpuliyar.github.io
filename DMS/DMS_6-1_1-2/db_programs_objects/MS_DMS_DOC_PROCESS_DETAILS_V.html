MS_DMS_DOC_PROCESS_DETAILS_V 

                                                
  CREATE OR REPLACE FORCE VIEW "PDMS61SP2_DEV"."MS_DMS_DOC_PROCESS_DETAILS_V" ("DOCUMENT_NAME", "DOCUMENT_OBJECT_ID", "DOCUMENT_OBJECT_TYPE", "TRANSACTION_ID", "ENTERPRISE_ID", "PROCESS_INSTANCE_ID", "DOCUMENT_NUMBER", "ACTION_TYPE", "LIFE_CYCLE_ID", "LIFE_CYCLE_NAME", "CURRENT_STAGE_ID", "CURRENT_STAGE_NAME", "CURRENT_STAGE_SEQ", "DOCUMENT_ATTACHMENT", "VERSION_NUMBER", "OWNER_NAME", "AUTHOR_NAME", "ORIGINATOR_NAME", "IN_PROCESS_FLAG") AS 
  SELECT 
   A.DOCUMENT_NAME, 
   A.DOCUMENT_OBJECT_ID, 
   A.DOCUMENT_OBJECT_TYPE, 
   A.TRANSACTION_ID, 
   A.ENTERPRISE_ID, 
   A.PROCESS_INSTANCE_ID, 
   A.DOCUMENT_NUMBER, 
   A.ACTION_TYPE, 
  A.LIFE_CYCLE_ID, 
   A.LIFE_CYCLE_NAME, 
   A.CURRENT_STAGE_ID, 
   A.CURRENT_STAGE_NAME, 
   A.CURRENT_STAGE_SEQ, 
   A.DOCUMENT_ATTACHMENT, 
   A.VERSION_NUMBER, 
   A.OWNER_NAME, 
   A.AUTHOR_NAME, 
   A.ORIGINATOR_NAME, 
   A.IN_PROCESS_FLAG 
   FROM MS_DMS_DOCUMENTS_HDR A 
   WHERE IN_PROCESS_FLAG='Y' 
   UNION 
    SELECT 
   B.DOCUMENT_NAME, 
   B.DOCUMENT_OBJECT_ID, 
   B.DOCUMENT_OBJECT_TYPE, 
   A.TRANSACTION_ID, 
   A.ENTERPRISE_ID, 
   A.PROCESS_INSTANCE_ID, 
   A.DOCUMENT_NUMBER, 
   A.ACTION_TYPE, 
  A.LIFE_CYCLE_ID, 
   A.LIFE_CYCLE_NAME, 
   A.CURRENT_STAGE_ID, 
   A.CURRENT_STAGE_NAME, 
   A.CURRENT_STAGE_SEQ, 
   B.DOCUMENT_ATTACHMENT, 
   B.VERSION_NUMBER, 
   B.OWNER_NAME, 
   B.AUTHOR_NAME, 
   A.ORIGINATOR_NAME, 
   A.IN_PROCESS_FLAG 
   FROM MS_DMS_DOCUMENTS_HDR A,MS_DMS_DOCUMENTS_DTL  B WHERE 
   A.PROCESS_INSTANCE_ID=B.PROCESS_INSTANCE_ID 
   AND A.IN_PROCESS_FLAG='Y' 
   AND A.DOCUMENT_OBJECT_TYPE='V'
