MS_DMS_DOCUMENT_STATUS_V 

                                                
  CREATE OR REPLACE FORCE VIEW "PDMS61SP2_DEV"."MS_DMS_DOCUMENT_STATUS_V" ("CATEGORY_ID", "STATUS", "USER_NAME") AS 
  SELECT B.CATEGORY_ID, B.STATUS, C.USER_NAME
FROM 
( SELECT CATEGORY_ID, SUB_CATEGORY1_ID, SUB_CATEGORY2_ID, SUB_CATEGORY3_ID, SUB_CATEGORY4_ID,
	            CASE WHEN DOCUMENT_STATUS = 'Production' AND TRUNC(EFFECTIVE_DATE) > TRUNC(SYSDATE) AND VN = 1 THEN 'Ready for Production'
                     ELSE DOCUMENT_STATUS END AS STATUS
         FROM (SELECT DOCUMENT_STATUS, EFFECTIVE_DATE, CATEGORY_ID, SUB_CATEGORY1_ID, SUB_CATEGORY2_ID, SUB_CATEGORY3_ID, SUB_CATEGORY4_ID,
                      ROW_NUMBER()OVER(PARTITION BY DOCUMENT_OBJECT_ID ORDER BY PROCESS_INSTANCE_ID DESC) VN,
                      LAG(EFFECTIVE_DATE,1)OVER(PARTITION BY DOCUMENT_OBJECT_ID ORDER BY PROCESS_INSTANCE_ID DESC) MAX_EFFECTIVE_DATE
                 FROM MS_DMS_ALL_PRODUCTION_DOCS)
        WHERE VN = 1 OR (TRUNC(MAX_EFFECTIVE_DATE) > TRUNC(SYSDATE) AND VN = 2 AND TRUNC(EFFECTIVE_DATE) <= TRUNC(SYSDATE))
     UNION ALL 
       SELECT CATEGORY_ID, SUB_CATEGORY1_ID, SUB_CATEGORY2_ID, SUB_CATEGORY3_ID, SUB_CATEGORY4_ID,
		      CASE WHEN H.IN_PROCESS_FLAG = 'Y' THEN 'In-Process' 
                   Else 'Terminated' 
			       END AS STATUS
         FROM MS_DMS_DOCUMENTS_HDR H
        WHERE H.IN_PROCESS_FLAG = 'Y' 
           OR (H.IN_PROCESS_FLAG ='T' AND ACTION_TYPE = 'Create Document')  
     ) B,       MS_DMS_USER_GRP_CAT_RPT_V C
  WHERE B.CATEGORY_ID      = C.CATEGORY_ID
    AND NVL(B.SUB_CATEGORY1_ID,'0') = NVL(C.SUBCATEGORY1_ID,'0')
    AND NVL(B.SUB_CATEGORY2_ID,'0') = NVL(C.SUBCATEGORY2_ID,'0')
    AND NVL(B.SUB_CATEGORY3_ID,'0') = NVL(C.SUBCATEGORY3_ID,'0')
    AND NVL(B.SUB_CATEGORY4_ID,'0') = NVL(C.SUBCATEGORY4_ID,'0')
