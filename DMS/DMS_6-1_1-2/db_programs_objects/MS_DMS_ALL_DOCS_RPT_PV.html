MS_DMS_ALL_DOCS_RPT_PV 

                                                
  CREATE OR REPLACE FORCE VIEW "PDMS61SP2_DEV"."MS_DMS_ALL_DOCS_RPT_PV" ("DOCUMENT_NAME", "DOC_NAME", "DOCUMENT_NUMBER", "REVISION_NUMBER", "FLAG", "DOCUMENT_OBJECT_ID", "DOCUMENT_TYPE", "ATTRIBUTES", "LIFE_CYCLE_NAME1", "DOC_TYPE", "AUTHOR_NAME", "AUTHOR_ID", "AUTHOR_FULL_NAME", "OWNER_NAME", "OWNER_ID", "OWNER_FULL_NAME", "EFFECTIVE_FROM", "AUDIT_ID", "ACTUAL_CATEGORY_ID", "ACTUAL_SUB_CATEGORY1_ID", "ACTUAL_SUB_CATEGORY2_ID", "ACTUAL_SUB_CATEGORY3_ID", "ACTUAL_SUB_CATEGORY4_ID", "CATEGORY", "SUB_CATEGORY1", "SUB_CATEGORY2", "SUB_CATEGORY3", "SUB_CATEGORY4", "LIFE_CYCLE_ID", "LIFE_CYCLE_NAME", "ORIGINATOR_ID", "ORIGINATOR_NAME", "ORIGINATOR_FULL_NAME", "LAST_UPDATED_NAME", "LAST_UPDATED_BY", "LAST_UPDATED_BY1", "LAST_UPDATED_ON", "LOGIN_USER_NAME", "DOC_STATUS_STORED", "STATUS") AS 
  SELECT DOCUMENT_NAME, --Modified on Jul 08 for check In
    DOC_NAME,
    DOCUMENT_NUMBER,
    REVISION_NUMBER,
    flag,
    DOCUMENT_OBJECT_ID,
    (SELECT displayed_value
    FROM ms_dms_local_lov_t v,
      ms_dms_local_users_t t
    WHERE v.lov_name = 'MS_DMS_INLINE_DOCUMENT_TYPE'
    AND v.locale_id  = t.LOCALE_ID
    AND t.user_name  =login_user_name
    AND stored_value = doc_type
    ) AS DOCUMENT_TYPE,
    (SELECT displayed_value
    FROM ms_dms_list_of_values_v v,
      ms_dms_local_users_t t
    WHERE lov_name    ='MS_DMS_STATUS_LOV'
    AND v.locale_id   = t.locale_id
    AND t.user_name   =login_user_name
    AND v.stored_value= 'Attributes'
    ) AS ATTRIBUTES,
    (SELECT LIFECYCLE_NAME
    FROM SI_DMS_LIFECYCLES_V v,
      ms_dms_local_users_t t
    WHERE LIFECYCLE_ID=LIFE_CYCLE_ID
    AND v.LOCALE_ID   = t.LOCALE_ID
    AND t.user_name   =login_user_name
    ) AS LIFE_CYCLE_NAME1,
    doc_type,
    AUTHOR_NAME,
    AUTHOR_ID,
    (SELECT full_name FROM ms_dms_local_users_t WHERE user_id = AUTHOR_ID
    ) AS AUTHOR_FULL_NAME,
    OWNER_NAME,
    OWNER_ID ,
    (SELECT full_name FROM ms_dms_local_users_t WHERE user_id = OWNER_ID
    ) AS OWNER_FULL_NAME,
    EFFECTIVE_FROM,
    AUDIT_ID,
    actual_CATEGORY_ID,
    actual_SUB_CATEGORY1_ID,
    actual_SUB_CATEGORY2_ID,
    actual_SUB_CATEGORY3_ID,
    actual_SUB_CATEGORY4_ID,
    (SELECT cv.category_name
    FROM si_dms_categories_v cv,
      MS_DMS_LOCAL_USERS_T ut
    WHERE cv.locale_id = ut.locale_id
    AND cv.CATEGORY_ID =actual_CATEGORY_ID
    AND ut.user_name   = LOGIN_USER_NAME
    ) AS category,
    (SELECT cv.category_name
    FROM si_dms_categories_v cv,
      MS_DMS_LOCAL_USERS_T ut
    WHERE cv.locale_id = ut.locale_id
    AND cv.CATEGORY_ID =actual_SUB_CATEGORY1_ID
    AND ut.user_name   = LOGIN_USER_NAME
    ) AS SUB_CATEGORY1,
    (SELECT cv.category_name
    FROM si_dms_categories_v cv,
      MS_DMS_LOCAL_USERS_T ut
    WHERE cv.locale_id = ut.locale_id
    AND cv.CATEGORY_ID =actual_SUB_CATEGORY2_ID
    AND ut.user_name   = LOGIN_USER_NAME
    ) AS SUB_CATEGORY2,
    (SELECT cv.category_name
    FROM si_dms_categories_v cv,
      MS_DMS_LOCAL_USERS_T ut
    WHERE cv.locale_id = ut.locale_id
    AND cv.CATEGORY_ID =actual_SUB_CATEGORY3_ID
    AND ut.user_name   = LOGIN_USER_NAME
    ) AS SUB_CATEGORY3,
    (SELECT cv.category_name
    FROM si_dms_categories_v cv,
      MS_DMS_LOCAL_USERS_T ut
    WHERE cv.locale_id = ut.locale_id
    AND cv.CATEGORY_ID =actual_SUB_CATEGORY4_ID
    AND ut.user_name   = LOGIN_USER_NAME
    ) AS SUB_CATEGORY4,
    LIFE_CYCLE_ID,
    LIFE_CYCLE_NAME,
    ORIGINATOR_ID,
    ORIGINATOR_NAME,
    (SELECT full_name FROM ms_dms_local_users_t WHERE user_id = ORIGINATOR_ID
    ) AS ORIGINATOR_FULL_NAME,
    LAST_UPDATED_NAME,
    LAST_UPDATED_BY,
    (SELECT full_name FROM ms_dms_local_users_t WHERE user_id = LAST_UPDATED_BY
    ) AS LAST_UPDATED_BY1,
    LAST_UPDATED_ON ,
    LOGIN_USER_NAME ,
    Doc_status_stored,
    (SELECT displayed_value
    FROM ms_dms_list_of_values_v v,
      si_enabled_locales l,
      si_USERS_T t
    WHERE lov_name      ='MS_DMS_STATUS_LOV'
    AND v.locale_id     = l.locale_id
    AND l.locale_string = t.locale
    AND t.user_name     =login_user_name
    AND v.stored_value  =DECODE(flag,'I','In Process','P','Move To Production','O','Obsolete','T','Terminated','RP','Ready for Production','IN','In-Process')
    ) AS STATUS
  FROM
    (SELECT
      --Ms_Dms_Doc_Url_For_nameReport ( a.process_instance_id, a.audit_id, DECODE (a.Document_Status, 'Production', 'Production', 'Obsoleted', 'Obsolete', NULL),
      --a.document_attachment, a.document_name) AS DOCUMENT_NAME1,
      (
      --Ms_Dms_Doc_Url_For_nameReport ( process_instance_id, audit_id, DECODE (Document_Status, 'Production', 'Production', 'Obsoleted', 'Obsolete', NULL),
      --   document_attachment, document_name)
      (''
      /* || (SELECT NVL(displayed_value,'View')
      FROM MS_DMS_LIST_OF_VALUES_V v,
      si_enabled_locales l,
      si_USERS_T t
      WHERE LOV_NAME='MS_DMS_STATUS_LOV'
      and   l.locale_id = v.locale_id
      and  l.locale_string= t.locale
      and  t.USER_NAME=c.user_name
      and  v.stored_value='View') */
      || a.document_name
      || '') )    AS DOCUMENT_NAME,
      a.DOCUMENT_NAME AS DOC_NAME,
      a.DOCUMENT_NUMBER,
      REVISION_NUMBER,
      a.DOCUMENT_OBJECT_ID,
      --Below decode statement is replaced with the CASE statement to add the RP Status flag for trakr 222712
      --DECODE (a.Document_Status, 'Production', 'P', 'Obsoleted', 'O', NULL) flag,
      CASE
        WHEN TRUNC(a.EFFECTIVE_DATE) > TRUNC(SYSDATE)
        THEN 'RP'
        WHEN a.DOCUMENT_STATUS = 'Production'
        THEN 'P'
        WHEN a.DOCUMENT_STATUS = 'Obsoleted'
        THEN 'O'
        ELSE NULL
      END AS Flag,
      'Attributes',
      a.doc_type,
      b.AUTHOR_NAME,
      b.AUTHOR_ID,
      b.OWNER_NAME,
      b.OWNER_ID ,
      a.EFFECTIVE_DATE AS EFFECTIVE_FROM,
      a.AUDIT_ID,
      a.CATEGORY_ID      AS actual_CATEGORY_ID,
      a.SUB_CATEGORY1_ID AS actual_SUB_CATEGORY1_ID,
      a.SUB_CATEGORY2_ID AS actual_SUB_CATEGORY2_ID,
      a.SUB_CATEGORY3_ID AS actual_SUB_CATEGORY3_ID,
      a.SUB_CATEGORY4_ID AS actual_SUB_CATEGORY4_ID,
      b.life_cycle_id    AS LIFE_CYCLE_ID,
      b.life_cycle_name  AS LIFE_CYCLE_NAME,
      b.ORIGINATOR_ID    AS ORIGINATOR_ID,
      b.ORIGINATOR_NAME  AS ORIGINATOR_NAME,
      a.LAST_USERNAME    AS LAST_UPDATED_NAME,
      a.LAST_USER_ID     AS LAST_UPDATED_BY,
      b.CREATED_ON       AS LAST_UPDATED_ON ,
      c.user_name        AS LOGIN_USER_NAME,
      --MS_DMS_RPT_UTILS.get_doc_status_store_val(DECODE (a.Document_Status, 'Production', 'P', 'Obsoleted', 'O', NULL),a.EFFECTIVE_DATE) AS Doc_status_stored
      CASE
        WHEN TRUNC(a.EFFECTIVE_DATE) > TRUNC(SYSDATE)
        THEN 'Ready for Production'
        WHEN a.Document_Status = 'Production'
        THEN 'Move To Production'
        WHEN a.Document_Status = 'Obsoleted'
        THEN 'Obsolete'
        ELSE NULL
      END AS Doc_status_stored
    FROM MS_DMS_DOCUMENTS a,
      MS_DMS_USER_GRP_CAT_RPT_V c ,
      MS_DMS_DOCUMENTS_HDR b
    WHERE checked_out_flag         = 'N'
    AND a.DOCUMENT_OBJECT_ID       = b.DOCUMENT_OBJECT_ID
    AND a.PROCESS_INSTANCE_ID      = b.PROCESS_INSTANCE_ID
    AND A.CATEGORY                 = C.CATEGORY
	  AND a.PROCESS_INSTANCE_ID      = (SELECT MAX(PROCESS_INSTANCE_ID) FROM MS_DMS_DOCUMENTS_HDR WHERE PROCESS_INSTANCE_ID = b.PROCESS_INSTANCE_ID)
    AND NVL (a.SUB_CATEGORY1, '%') = NVL (c.SUBCATEGORY1, '%')
    AND NVL (a.SUB_CATEGORY2, '%') = NVL (c.SUBCATEGORY2, '%')
    AND NVL (a.SUB_CATEGORY3, '%') = NVL (c.SUBCATEGORY3, '%')
    AND NVL (a.SUB_CATEGORY4, '%') = NVL (c.SUBCATEGORY4, '%')
    UNION ALL
    SELECT
      --(SELECT Ms_Dms_Doc_Url_For_nameReport (b.process_instance_id, a.audit_id, 'In Process', b.document_attachment, b.document_name) FROM DUAL) AS DOCUMENT_NAME1,
      (''
      /* || (  SELECT NVL(displayed_value,'View')
      FROM MS_DMS_LIST_OF_VALUES_V v,
      si_enabled_locales l,
      si_USERS_T t
      WHERE LOV_NAME='MS_DMS_STATUS_LOV'
      and   l.locale_id = v.locale_id
      and  l.locale_string= t.locale
      and  t.USER_NAME= c.user_name
      and  v.stored_value='View') */
      || b.document_name
      || '')      AS DOCUMENT_NAME,
      b.DOCUMENT_NAME AS DOC_NAME,
      b.document_number,
      DECODE(b.in_process_flag,'T',a.version_number,b.version_number) version_number,
      b.document_object_id,
      DECODE (b.in_process_flag, 'Y', 'IN', 'T', 'T', NULL) flag,
      'Attributes' ,
      b.doc_type,
      b.AUTHOR_NAME,
      b.AUTHOR_ID,
      b.OWNER_NAME,
      b.OWNER_ID ,
      b.EFFECTIVE_FROM,
      a.AUDIT_ID,
      b.CATEGORY_ID,
      b.SUB_CATEGORY1_ID,
      b.SUB_CATEGORY2_ID,
      b.SUB_CATEGORY3_ID,
      b.SUB_CATEGORY4_ID,
      b.life_cycle_id                            AS LIFE_CYCLE_ID,
      b.life_cycle_name                          AS LIFE_CYCLE_NAME,
      b.ORIGINATOR_ID                            AS ORIGINATOR_ID,
      b.ORIGINATOR_NAME                          AS ORIGINATOR_NAME,
      a.USER_NAME                                AS LAST_UPDATED_NAME,
      MS_APPS_UTILITIES.get_user_id(a.USER_NAME) AS LAST_UPDATED_BY,
      b.CREATED_ON                               AS LAST_UPDATED_ON ,
      c.user_name                                AS LOGIN_USER_NAME,
      --MS_DMS_RPT_UTILS.get_doc_status_store_val(DECODE (b.in_process_flag,  'Y', 'IN',  'T', 'T',  NULL),b.EFFECTIVE_FROM) as Doc_status_stored
      CASE
          /*Commented by rajeshwar.reddy for trakr 217904 WHEN trunc(b.EFFECTIVE_FROM) > trunc(SYSDATE) THEN 'Ready for Production' */
        WHEN b.in_process_flag = 'Y'
        THEN 'In-Process'
        WHEN b.in_process_flag = 'T'
        THEN 'Terminated'
        ELSE NULL
      END AS Doc_status_stored
    FROM MS_DMS_DOC_AUDIT_HISTORY a,
      MS_DMS_DOCUMENTS_HDR b,
      MS_DMS_USER_GRP_CAT_RPT_V c
    WHERE b.in_process_flag  IN ('Y', 'T')
    AND a.process_instance_id = b.process_instance_id
    AND audit_id              =
      (SELECT MAX (audit_id)
      FROM MS_DMS_DOC_AUDIT_HISTORY
      WHERE process_instance_id = b.process_instance_id
      )
    AND a.CATEGORY                 = c.CATEGORY
    AND NVL (a.SUB_CATEGORY1, '%') = NVL (c.SUBCATEGORY1, '%')
    AND NVL (a.SUB_CATEGORY2, '%') = NVL (c.SUBCATEGORY2, '%')
    AND NVL (a.SUB_CATEGORY3, '%') = NVL (c.SUBCATEGORY3, '%')
    AND NVL (a.SUB_CATEGORY4, '%') = NVL (c.SUBCATEGORY4, '%')
    AND b.CATEGORY                 = c.CATEGORY
    AND NVL (b.SUB_CATEGORY1, '%') = NVL (c.SUBCATEGORY1, '%')
    AND NVL (b.SUB_CATEGORY2, '%') = NVL (c.SUBCATEGORY2, '%')
    AND NVL (b.SUB_CATEGORY3, '%') = NVL (c.SUBCATEGORY3, '%')
    AND NVL (b.SUB_CATEGORY4, '%') = NVL (c.SUBCATEGORY4, '%')
    )
