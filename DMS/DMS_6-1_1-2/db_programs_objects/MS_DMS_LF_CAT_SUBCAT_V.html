MS_DMS_LF_CAT_SUBCAT_V 

                                                
  CREATE OR REPLACE FORCE VIEW "PDMS61SP2_DEV"."MS_DMS_LF_CAT_SUBCAT_V" ("LIFECYCLE_ID", "CATEGORY_SUBCATEGORY_CONCAT", "CATEGORY_ID", "CATEGORY_NAME", "DESCRIPTION") AS 
  SELECT asgv.lifecycle_id, asgv.category_subcategory_concat, asgv.category_id, sdc.category_name, 'Category' description 
FROM SI_DMS_LF_ASSGND_CATEGORIES_V asgv, SI_DMS_CATEGORIES sdc 
 WHERE 
asgv.category_id = sdc.category_id 
AND asgv.category_id = (SELECT category_id FROM SI_DMS_CAT_SUBCATEGORY_ALL_V WHERE 
category_id IS NOT NULL 
AND subcategory1_id IS NULL 
AND subcategory2_id IS NULL 
AND subcategory3_id IS NULL 
AND subcategory4_id IS NULL 
AND category_id = asgv.category_id) 
UNION ALL
