MS_DMS_LF_FIRST_STAGE_USERS_V 

                                                
  CREATE OR REPLACE FORCE VIEW "PDMS61SP2_DEV"."MS_DMS_LF_FIRST_STAGE_USERS_V" ("ENTERPRISE_ID", "LIFECYCLE_ID", "LIFECYCLE_NAME", "DOCUMENT_OBJECT_ID", "DOCUMENT_NAME", "DOCUMENT_OBJECT_TYPE", "USER_ID", "ROLE_GROUP_ID", "GROUP_ID", "ROLE_ID") AS 
  SELECT enterprise_id,lifecycle_id,lifecycle_name,document_object_id,document_name,
document_object_type,user_id,role_group_id,group_id,role_id 
FROM (
select a.enterprise_id,a.lifecycle_id,a.lifecycle_name,b.document_object_id,b.document_name,
b.document_object_type,g.user_id,lg.role_group_id,lg.group_id,lg.role_id  from
ms_dms_lifecycle_first_stage_v a,ms_dms_avail_process_docs_v b,
si_dms_group_details g,si_dms_lifecycle_role_grps lg 
where
b.approval_process_id=a.lifecycle_id
and g.role_group_id =lg.role_group_id
and lg.lifecycle_id = a.lifecycle_id
and lg.stage_id = a.stage_id
 
union
select a.enterprise_id,a.lifecycle_id,a.lifecycle_name,b.document_object_id,b.document_name,
b.document_object_type,g.user_id,lg.role_group_id,lg.group_id,lg.role_id  from
ms_dms_lifecycle_first_stage_v a,ms_dms_avail_process_docs_v b ,
si_dms_user_group_users g,si_dms_lifecycle_role_grps lg
where
b.approval_process_id=a.lifecycle_id
and lg.group_id =  g.user_group_id
and lg.lifecycle_id = a.lifecycle_id
and lg.stage_id = a.stage_id
union
select a.enterprise_id,a.lifecycle_id,a.lifecycle_name,null,null,null,
g.user_id,lg.role_group_id,lg.group_id,lg.role_id  from
ms_dms_lifecycle_first_stage_v a,
si_dms_group_details g,si_dms_lifecycle_role_grps lg 
where
g.role_group_id =lg.role_group_id
and lg.lifecycle_id = a.lifecycle_id
and lg.stage_id = a.stage_id
union
select a.enterprise_id,a.lifecycle_id,a.lifecycle_name,null,null,null,
g.user_id,lg.role_group_id,lg.group_id,lg.role_id  from
ms_dms_lifecycle_first_stage_v a,
si_dms_user_group_users g,si_dms_lifecycle_role_grps lg 
where
lg.group_id =  g.user_group_id
and lg.lifecycle_id = a.lifecycle_id
and lg.stage_id = a.stage_id
) a
where a.user_id= (case when  ms_dms_chk_outofoffice_user(a.user_id)>=1 then a.user_id
else 0 end)
