MS_DMS_PRINTLAYOUT_V 

                                                
  CREATE OR REPLACE FORCE VIEW "PDMS61SP2_DEV"."MS_DMS_PRINTLAYOUT_V" ("MAX_COPIES", "DOC_INFO", "MANIFESTATION", "MANIFESTATION_PAGES", "WATERMARK", "WATERMARK_TEXT", "USER_TEXT", "DOCUMENT_TYPE", "CATEGORY", "SUB_CATEGORY1", "SUB_CATEGORY2", "SUB_CATEGORY3", "SUB_CATEGORY4", "CATEGORY_ID", "SUB_CATEGORY1_ID", "SUB_CATEGORY2_ID", "SUB_CATEGORY3_ID", "SUB_CATEGORY4_ID", "EFFECTIVE_DATE", "DOCUMENT_NUMBER", "VERSION_NUMBER", "DOCUMENT_NAME", "HEADER_TEXT", "FOOTER_TEXT", "HEADER_STYLE", "FOOTER_STYLE", "AUDIT_ID", "STAGE_ID", "PROCESS_INSTANCE_ID", "DOCUMENT_ATTACHMENT", "DOCUMENT_OBJECT_ID") AS 
  SELECT NVL (p.max_copies, -1),
          p.doc_info,
          p.manifestation,
          p.manifestation_pages,
          p.watermark,
          p.watermark_text,
          p.user_text,
          T.document_type,
          T.CATEGORY,
          T.SUB_CATEGORY1,
          T.SUB_CATEGORY2,
          T.SUB_CATEGORY3,
          T.SUB_CATEGORY4,
          T.CATEGORY_ID,
          T.SUB_CATEGORY1_ID,
          T.SUB_CATEGORY2_ID,
          T.SUB_CATEGORY3_ID,
          T.SUB_CATEGORY4_ID,
          T.EFFECTIVE_FROM,
          T.document_number,
          T.version_number,
          T.document_name,
          p.header_text,
          p.footer_text,
          p.header_style,
          p.footer_style,
          T.audit_ID,
          NVL (p.stage_id, 0),
          t.process_instance_id,
          t.document_attachment,
          DOCUMENT_OBJECT_ID
     FROM SI_DMS_PRINTLAYOUT p,
          SI_DMS_LIFECYCLE_STAGES s,
          MS_DMS_DOC_AUDIT_HISTORY T
    WHERE p.LIFECYCLE_id = s.lifecycle_id
          AND p.lifecycle_id =
                 CASE
                    WHEN parent_audit_id IS NOT NULL
                    THEN
                       (SELECT DISTINCT a.life_cycle_id
                          FROM ms_dms_doc_audit_history a
                         WHERE a.audit_id = parent_audit_id
                               AND a.stage_id = t.stage_id)
                    ELSE
                       t.life_cycle_id
                 END
          AND p.stage_id IS NOT NULL
          AND (p.stage_id = s.stage_id(+))
   UNION
   SELECT NVL (p.max_copies, -1),
          p.doc_info,
          p.manifestation,
          p.manifestation_pages,
          p.watermark,
          p.watermark_text,
          p.user_text,
          T.document_type,
          T.CATEGORY,
          T.SUB_CATEGORY1,
          T.SUB_CATEGORY2,
          T.SUB_CATEGORY3,
          T.SUB_CATEGORY4,
          T.CATEGORY_ID,
          T.SUB_CATEGORY1_ID,
          T.SUB_CATEGORY2_ID,
          T.SUB_CATEGORY3_ID,
          T.SUB_CATEGORY4_ID,
          T.EFFECTIVE_FROM,
          T.document_number,
          T.version_number,
          T.document_name,
          p.header_text,
          p.footer_text,
          p.header_style,
          p.footer_style,
          T.audit_ID,
          NVL (p.stage_id, 0),
          t.process_instance_id,
          t.document_attachment,
          DOCUMENT_OBJECT_ID
     FROM SI_DMS_PRINTLAYOUT p, MS_DMS_DOC_AUDIT_HISTORY T
      WHERE DECODE(T.LIFE_CYCLE_Name, 'N/A', 0,T.LIFE_CYCLE_id) = p.lifecycle_id 
    AND (p.stage_id IS NULL or p.stage_id = 0)
   UNION
   SELECT 0 max_copies,
          '',
          '',
          '',
          '',
          '',
          '',
          T.document_type,
          T.CATEGORY,
          T.SUB_CATEGORY1,
          T.SUB_CATEGORY2,
          T.SUB_CATEGORY3,
          T.SUB_CATEGORY4,
          T.CATEGORY_ID,
          T.SUB_CATEGORY1_ID,
          T.SUB_CATEGORY2_ID,
          T.SUB_CATEGORY3_ID,
          T.SUB_CATEGORY4_ID,
          T.EFFECTIVE_FROM,
          T.document_number,
          T.version_number,
          T.document_name,
          '',
          '',
          '',
          '',
          T.audit_ID,
          NVL (t.stage_id, 0),
          t.process_instance_id,
          t.document_attachment,
          DOCUMENT_OBJECT_ID
     FROM MS_DMS_DOC_AUDIT_HISTORY T, SI_DMS_LIFECYCLE_STAGES p
    WHERE     T.LIFE_CYCLE_id = p.lifecycle_id
          AND p.stage_id = t.stage_id
          AND 0 = (SELECT COUNT (1)
                     FROM SI_DMS_PRINTLAYOUT
                    WHERE STAGE_ID = t.stage_id)
     UNION
        SELECT NVL (p.max_copies, -1),
          p.doc_info,
          p.manifestation,
          p.manifestation_pages,
          p.watermark,
          p.watermark_text,
          p.user_text,
          T.document_type,
          T.CATEGORY,
          T.SUB_CATEGORY1,
          T.SUB_CATEGORY2,
          T.SUB_CATEGORY3,
          T.SUB_CATEGORY4,
          T.CATEGORY_ID,
          T.SUB_CATEGORY1_ID,
          T.SUB_CATEGORY2_ID,
          T.SUB_CATEGORY3_ID,
          T.SUB_CATEGORY4_ID,
          T.EFFECTIVE_FROM,
          T.document_number,
          T.version_number,
          T.document_name,
          p.header_text,
          p.footer_text,
          p.header_style,
          p.footer_style,
          T.audit_ID,
          NVL (p.stage_id, 0),
          t.process_instance_id,
          t.document_attachment,
          DOCUMENT_OBJECT_ID
     FROM SI_DMS_PRINTLAYOUT p, MS_DMS_DOC_AUDIT_HISTORY T
      WHERE DECODE(T.LIFE_CYCLE_Name, 'N/A', 0,T.LIFE_CYCLE_id) = nvl(p.lifecycle_id,0) 
    and t.document_attachment is not null
