MS_DMS_VIRTUAL_DOCS_V 

                                                
  CREATE OR REPLACE FORCE VIEW "PDMS61SP2_DEV"."MS_DMS_VIRTUAL_DOCS_V" ("DOCUMENT_AUTHOR", "DOCUMENT_OWNER", "DOCUMENT_ATTACHMENT", "DOCUMENT_OBJECT_ID", "DOCUMENT_OBJECT_TYPE", "DOCUMENT_NAME", "REVISION_NUMBER", "PARENT_DOCUMENT_ID", "DOCUMENT_SEQ_NUM", "DOCUMENT_NUMBER", "EXCLUSIVE_FLAG", "INDEPENDENT_UPVERSION_FLAG", "VIRTUAL_VERSION_NUMBER", "DOCUMENT_STATUS", "CATEGORY", "SUB_CATEGORY1", "SUB_CATEGORY2", "SUB_CATEGORY3", "SUB_CATEGORY4") AS 
  SELECT  DOCUMENT_AUTHOR, DOCUMENT_OWNER,  document_attachment,DOCUMENT_OBJECT_ID, DOCUMENT_OBJECT_TYPE, document_name,
revision_number,b.parent_object_id as parent_document_id,document_seq_num,document_number,
b.exclusive_flag,b.independent_upversion_flag,b.version_number as virtual_version_number,a.document_status,category,sub_category1
,sub_category2,sub_category3,sub_category4 FROM
ms_dms_doc_meta_v a,MS_DMS_DOC_RELATIONSHIP b
WHERE
a.document_object_id=b.child_object_id(+)
AND a.document_object_type='N'
UNION
SELECT  DOCUMENT_AUTHOR, DOCUMENT_OWNER,  document_attachment,DOCUMENT_OBJECT_ID, DOCUMENT_OBJECT_TYPE, document_name,
revision_number,b.parent_object_id AS parent_document_id,DOCUMENT_SEQ_NUM,DOCUMENT_NUMBER,
b.exclusive_flag,b.independent_upversion_flag,b.version_number as virtual_version_number,a.document_status,category,sub_category1
,sub_category2,sub_category3,sub_category4 FROM
ms_dms_doc_meta_v a,MS_DMS_DOC_RELATIONSHIP b
WHERE
a.document_object_id=b.child_object_id
AND a.document_object_type='V'
UNION
SELECT  DOCUMENT_AUTHOR, DOCUMENT_OWNER,  document_attachment,DOCUMENT_OBJECT_ID, DOCUMENT_OBJECT_TYPE, document_name,
revision_number,NULL AS parent_document_id,DOCUMENT_SEQ_NUM,DOCUMENT_NUMBER,
b.exclusive_flag,b.independent_upversion_flag,b.version_number as virtual_version_number,a.document_status,category,sub_category1
,sub_category2,sub_category3,sub_category4
FROM
ms_dms_doc_meta_v a,MS_DMS_DOC_RELATIONSHIP b
WHERE
a.document_object_id=b.parent_object_id
AND a.document_object_type='V'
