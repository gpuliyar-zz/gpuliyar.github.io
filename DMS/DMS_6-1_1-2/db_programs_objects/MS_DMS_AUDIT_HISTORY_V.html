MS_DMS_AUDIT_HISTORY_V 

                                                
  CREATE OR REPLACE FORCE VIEW "PDMS61SP2_DEV"."MS_DMS_AUDIT_HISTORY_V" ("DOCUMENT_OBJECT_ID", "DOCUMENT_OBJECT_TYPE", "DOCUMENT_NAME", "PROCESS_INSTANCE_ID", "ATTACHMENT", "REVISION_NUMBER", "LAST_UPDATED_BY", "CREATION_DATE", "DOCUMENT_STATUS", "DOCUMENT_AUTHOR", "RIGHTS", "INSTANCE_ID", "STAGE_SEQ", "AUDIT_ID", "USER_ID", "FORM_CODE", "USER_ACTION_TYPE", "AUTHOR_NAME", "AUD_SYS_DATE", "DOCUMENT_ATTACHMENT", "KEYWORDS", "DESCRIPTION", "PURPOSE", "DOCUMENT_NUMBER", "CATEGORY", "SUB_CATEGORY1", "SUB_CATEGORY2", "SUB_CATEGORY3", "SUB_CATEGORY4", "CATEGORY_ID", "SUB_CATEGORY1_ID", "SUB_CATEGORY2_ID", "SUB_CATEGORY3_ID", "SUB_CATEGORY4_ID", "SUPPORTING_DOCUMENTS") AS 
  SELECT a.document_object_id,
    a.document_object_type,
    a.document_name,
    a.process_instance_id,
    a.document_attachment attachment,
    a.version_number version_number,
    f.user_name last_updated_by,
    a.aud_sys_date creation_date,
    Ms_Dms_Process_Sv.ms_dms_get_doc_status(DOCUMENT_STATUS_FLAG,effective_from) document_status,
    a.author_name document_author,
    '0' rights,
    a.instance_id,
    stage_seq,
    a.audit_id,
    f.user_id,
    a.form_code,
    user_action_type,
    author_name,
    aud_sys_date,
    document_attachment,
    keywords,
    description,
    purpose,
    document_number ,
    CATEGORY,
    sub_category1,
    sub_category2,
    sub_category3,
    SUB_CATEGORY4,
    a.category_id,
    a.sub_category1_id,
    a.sub_category2_id,
    a.sub_category3_id,
    a.sub_category4_id,
    a.supporting_documents
  FROM MS_DMS_DOC_AUDIT_HISTORY a,
    SI_USERS_T f
  WHERE form_code IN (
  'MS DMS Document Review','MS DMS Create Document','MS_DMS_CREATE_DOC','MS_DMS_REVIEW_DOCUMENT', 'MS_DMS_UPDATE_DOCUMENT'
   ,'MS_DMS_INLINE_REVIEW_DOCUMENT','MS_DMS_INLINE_CREATE_DOCMUNENT','MS_DMS_INLINE_UPDATE_FORM'
   )
  AND NVL(user_action_type,'0') NOT IN('Continue Working','0')
  and F.USER_ID                      =a.USER_ID
  ORDER BY a.instance_id DESC
