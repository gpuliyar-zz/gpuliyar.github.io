MS_DMS_INPROCESS_DOCS_V 

                                                
  CREATE OR REPLACE FORCE VIEW "PDMS61SP2_DEV"."MS_DMS_INPROCESS_DOCS_V" ("METRIC_ID", "PROCESS_INSTANCE_ID", "INSTANCE_ID", "ATTACHMENT", "USER_NAME", "CREATION_DATE", "CREATED_BY", "CREATION_DATE_SQL", "CREATION_DATE_CMP", "DOCUMENT_TYPE", "DOCUMENT_NUMBER", "VERSION_NUMBER", "ACTION", "DOCUMENT_NAME", "CATEGORY", "SUB_CATEGORY1", "SUB_CATEGORY2", "SUB_CATEGORY3", "SUB_CATEGORY4", "CURRENT_LEVEL_NAME", "NEXT_LEVEL_NAME", "DESCRIPTION", "PURPOSE", "REVIEW_SCHEDULE_DAYS", "KEYWORDS", "DOCUMENT_AUTHOR", "USER_LIST", "CURRENT_USER", "NEXT_STAGE", "DOCUMENT_STATUS", "LIFECYCLE_NAME", "USER_ID", "ENTERPRISE_ID") AS 
  (SELECT
     t.metric_id,
     t.process_instance_id,
     t.instance_id,
     t.Document attachment,
     p.user_name,
     TO_CHAR(t.creation_date, 'mm/dd/yyyy HH:MI AM') creation_date,
     t.created_by,
     t.creation_date creation_date_sql,
     TO_CHAR(t.creation_date, 'mm dd yyyy HH24:MI:SS') creation_date_cmp,
     t.document_type,
     t.document_number,
     t.version_number,
     t.action_type,
     t.document_name,
     t.CATEGORY,
     t.sub_category1,
     t.sub_category2,
     t.sub_category3,
     t.sub_category4,
     t.current_level_name,
     t.next_level_name,
     t.description,
     t.purpose,
     t.review_schedule_days,
     t.keywords,
     t.document_creator document_author,
     --ms_dms_document_author(t.document_name) document_author,
     t.user_list,
     t.CURRENT_USER,
     t.next_stage,
     'In Process' document_status,
     a.lifecycle_name,
     p.user_id,
     p.enterprise_id
FROM
     si_dms_lifecycles a,  -- to get lifecycle name
     SI_DMS_LIFECYCLE_STAGES d,  -- stages for current document
     si_users p,  -- User table to get author's name
     SI_DMS_DOCUMENT_REVIEW_V t,  -- Doc Approval Table
     si_event_assignments ea  -- Open event assignments table
WHERE
     ea.infolet_id = (SELECT metric_id
		                       FROM si_metrics_t
		                       WHERE metric_name LIKE 'MS DMS Document Review') AND
     t.process_instance_id = ea.process_instance_id AND
     p.user_id = t.created_by AND
     NOT EXISTS
     (
       SELECT 1
       FROM   ms_dms_documents k
       WHERE  k.process_instance_id = t.process_instance_id
     ) AND
	d.lifecycle_id = t.approval_process_id AND
	a.lifecycle_id = t.approval_process_id AND
	d.seq = NVL(t.next_stage,t.current_level_number) AND
	NOT EXISTS
	(
     	SELECT	'Later Record'
        FROM 	SI_DMS_DOCUMENT_REVIEW_V t2
        WHERE 	t2.process_instance_id = t.process_instance_id AND
        	t2.instance_id > t.instance_id
        )
UNION
SELECT
     t.metric_id,
     t.process_instance_id,
     t.instance_id,
     'N/A',
     p.user_name,
     TO_CHAR(t.creation_date, 'mm/dd/yyyy HH:MI AM') creation_date,
     t.created_by,
     t.creation_date creation_date_sql,
     TO_CHAR(t.creation_date, 'mm dd yyyy HH24:MI:SS') creation_date_cmp,
     t.document_type,
     t.document_number,
     t.version_number,
     t.action_type,
     t.document_name,
     t.CATEGORY,
     t.sub_category1,
     t.sub_category2,
     t.sub_category3,
     t.sub_category4,
     t.current_level_name,
     MS_DMS_GET_NEXT_STAGE(d.seq,t.approval_process_id) next_stage,
     t.description,
     t.purpose,
     t.review_schedule_days,
     t.keywords,
     t.originator document_author,
     --ms_dms_document_author(t.document_name) document_author,
    to_char( t.user_list),
     p.user_name,
     d.seq,
     'In Process' document_status,
     a.lifecycle_name,
     p.user_id,
     p.enterprise_id
FROM
     si_dms_lifecycles a,  -- to get lifecycle name
     SI_DMS_LIFECYCLE_STAGES d,  -- stages for current document
     si_users p,  -- User table to get author's name
     SI_DMS_CREATE_DOCUMENT_V t,  -- Doc Approval Table
     si_event_assignments ea  -- Open event assignments table
WHERE
     ea.infolet_id = (SELECT metric_id
		        FROM si_metrics_t
		       WHERE metric_name LIKE 'MS DMS Document Review') AND
     p.user_id = t.created_by AND
     t.process_instance_id = ea.process_instance_id AND
     NOT EXISTS
     (
       SELECT 1
       FROM   ms_dms_documents mdd
       WHERE  mdd.process_instance_id = t.process_instance_id
     ) AND
	NOT EXISTS
	(
	  SELECT 1
          FROM 	 SI_DMS_DOCUMENT_REVIEW_V
          WHERE	 process_instance_id = t.process_instance_id
	) AND
        d.lifecycle_id = t.approval_process_id AND
        a.lifecycle_id = t.approval_process_id AND
        d.stage_name = t.current_level_name AND
        NOT EXISTS
	(
     	  SELECT 'Later Instance ID'
          FROM   SI_DMS_CREATE_DOCUMENT_V t2
          WHERE  t2.process_instance_id = t.process_instance_id AND
                 t2.instance_id > t.instance_id
        )
)
