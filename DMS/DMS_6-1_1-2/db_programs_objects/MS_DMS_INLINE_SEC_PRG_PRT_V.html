MS_DMS_INLINE_SEC_PRG_PRT_V 

                                                
  CREATE OR REPLACE FORCE VIEW "PDMS61SP2_DEV"."MS_DMS_INLINE_SEC_PRG_PRT_V" ("DOCUMENT_NAME", "VERSION", "SECTION_NUMBER", "SECTION_TITLE", "AUTHOR", "APPROVE_LEVEL", "APPROVER", "APPROVAL_TYPE", "DUE_DATE", "STATUS", "SECTION_APPROVED_BY_ALL_USERS", "IS_DOC_READY_FOR_PROD", "DOCUMENT_TYPE", "INLINE_REPORT_ID", "REVIEW_RESPONSE", "ACTIVE") AS 
  SELECT DOCUMENT_NAME,
    VERSION,
    SECTION_NUMBER,
    SECTION_TITLE,
    AUTHOR,
    DECODE(APPROVE_LEVEL, 'STAGE AUTHOR', 'AUTHOR', 'REVIEW AUTHOR', 'AUTHOR', 'AUTHOR', 'AUTHOR', APPROVE_LEVEL) APPROVE_LEVEL,
    APPROVER,
    APPROVAL_TYPE,
    DUE_DATE,
    DECODE(STATUS,'Author Approved','Submitted',STATUS) STATUS,
    SECTION_APPROVED_BY_ALL_USERS,
    IS_DOC_READY_FOR_PROD,
    DOCUMENT_TYPE,
    INLINE_REPORT_ID,
    REVIEW_RESPONSE,
    ACTIVE
  FROM ms_dms_inline_sec_prog_report ispr
  WHERE ((approve_level IN ('INITIATOR', 'STAGE AUTHOR','AUTHOR')
  AND NOT (status        = 'No Response'
  AND review_response   IS NULL)
  AND NOT (STATUS        = 'Author Approved'
  AND REVIEW_RESPONSE   IS NULL))
  OR (approve_level NOT IN ('INITIATOR', 'STAGE AUTHOR','AUTHOR')
  AND (review_response  IS NOT NULL
  OR (review_response   IS NULL
  AND document_type      = 'Policy Type'))))
  AND EXISTS
    (SELECT 1
    FROM MS_DMS_GET_INLINE_PROG_DOC_V
    WHERE document_name = ispr.document_name
    )
  AND ACTIVE                         ='Y'
  AND (status                       <> 'Rejected and sent to Author'
  OR (status                         = 'Rejected and sent to Author'
  AND SECTION_APPROVED_BY_ALL_USERS IN ('000 -100', '100 -100', '110 -100')))
