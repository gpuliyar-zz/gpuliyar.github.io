MS_DMS_ALL_INPROGRESS_DOCS_V 

                                                
  CREATE OR REPLACE FORCE VIEW "PDMS61SP2_DEV"."MS_DMS_ALL_INPROGRESS_DOCS_V" ("PURPOSE", "CATEGORY_ID", "USER_NAME", "DOCUMENT_NAME", "DOCUMENT_TYPE", "VERSION_NUMBER", "DOCUMENT_NUMBER", "AUTHOR_NAME", "OWNER_NAME", "LIFE_CYCLE_NAME", "STAGE_NAME", "DUE_DATE", "ATTRIBUTES", "LAST_UPDATED_ON", "CATEGORY", "SUB_CATEGORY1", "SUB_CATEGORY2", "SUB_CATEGORY3", "SUB_CATEGORY4", "DOCUMENT_OBJECT_ID") AS 
  WITH TEMP_TAB AS
(SELECT distinct a.lifecycle_id as lifecycle_id,b.user_name as user_name
FROM ms_dms_lifecycle_all_stage_v a,SI_USERS_T b, 
SI_DMS_GROUP_DETAILS g,SI_DMS_LIFECYCLE_ROLE_GRPS lg,si_dms_roles rl 
WHERE g.role_group_id =lg.role_group_id AND lg.lifecycle_id = a.lifecycle_id
AND lg.stage_id = a.stage_id AND RL.ROLE_ID=LG.ROLE_ID AND G.USER_ID=B.USER_ID )
SELECT b.purpose as purpose,
		b.CATEGORY_ID as CATEGORY_ID,
		c.USER_NAME as USER_NAME,
--Ms_Dms_Doc_Url_For_nameReport (b.process_instance_id, a.audit_id, 'In Process', b.document_attachment, b.document_name) AS DOCUMENT_NAME, 
(''
      || (  SELECT NVL(displayed_value,'View')
          FROM MS_DMS_LIST_OF_VALUES_V v,
                si_enabled_locales l,
                si_USERS_T t
          WHERE LOV_NAME='MS_DMS_STATUS_LOV' 
          and   l.locale_id = v.locale_id
          and  l.locale_string= t.locale
          and  t.USER_NAME= c.user_name
          and  v.stored_value='View')
      || b.DOCUMENT_NAME||'') as 
       DOCUMENT_NAME, 
 MS_APPS_UTILITIES.Get_Display_Value(ms_grc_utilities.active_ent_id, 'MS_DMS_INLINE_DOCUMENT_TYPE', b.doc_type,1,(select locale_id from si_locales aa , si_users_t bb where aa.locale_string=bb.locale and bb.user_name=c.user_name)) as document_type,
 a.version_number as version_number,
 b.document_number AS document_number,
 --MS_APPS_UTILITIES.get_user_full_name(b.AUTHOR_NAME) as Author_Name,
 (select full_name from MS_DMS_LOCAL_USERS_T where user_name = b.AUTHOR_NAME) as Author_Name,
 --MS_APPS_UTILITIES.get_user_full_name(b.Owner_NAME) as Owner_NAME,
 (select full_name from MS_DMS_LOCAL_USERS_T where user_name = b.Owner_NAME) as Owner_NAME,
decode(b.is_inline,'Y',MS_DMS_RPT_UTILS.get_inline_rpt_approvals
(b.process_instance_id,MS_DMS_MLS_UTILITIES.MS_DMS_GET_LIFECYCLE_MLSVALUE(b.LIFE_CYCLE_ID,null,null,c.user_name)), MS_DMS_MLS_UTILITIES.MS_DMS_GET_LIFECYCLE_MLSVALUE(b.LIFE_CYCLE_ID,null,null,c.user_name)) AS LIFE_CYCLE_NAME,
 decode(b.is_inline,'Y',MS_DMS_RPT_UTILS.get_inline_rpt_approvals( b.process_instance_id,a.Stage_name), b.current_stage_name) as Stage_name,
 ms_dms_document_review_rep_pkg.get_due_date(b.process_instance_id) as due_date,
 ms_dms_utilities.ms_dms_get_val_for_locale (c.user_name, 'Attributes')
 as Attributes,
 b.created_on as last_updated_on,
 --MS_DMS_MLS_UTILITIES.MS_DMS_GET_CATEGORY_MLSVALUE (:7, b.category_ID) AS category,
 --MS_DMS_MLS_UTILITIES.MS_DMS_GET_CATEGORY_MLSVALUE (:7, b.sub_category1_ID) AS sub_category1,
 --MS_DMS_MLS_UTILITIES.MS_DMS_GET_CATEGORY_MLSVALUE (:7, b.sub_category2_ID) AS sub_category2,
 --MS_DMS_MLS_UTILITIES.MS_DMS_GET_CATEGORY_MLSVALUE (:7, b.sub_category3_ID) AS sub_category3,
 --MS_DMS_MLS_UTILITIES.MS_DMS_GET_CATEGORY_MLSVALUE (:7, b.sub_category4_ID) AS sub_category4,
        (select cv.category_name
    from
    si_dms_categories_v cv,
    MS_DMS_LOCAL_USERS_T ut
    where cv.LOCALE_ID = ut.locale_id
    and cv.CATEGORY_ID=b.category_id
    and ut.user_name = c.user_name) as category,
       (select cv.category_name
    from
    si_dms_categories_v cv,
    MS_DMS_LOCAL_USERS_T ut
    where cv.LOCALE_ID = ut.locale_id
    and cv.CATEGORY_ID=b.sub_category1_ID
    and ut.user_name = c.user_name) as sub_category1,
       (select cv.category_name
    from
    si_dms_categories_v cv,
    MS_DMS_LOCAL_USERS_T ut
    where cv.LOCALE_ID = ut.locale_id
    and cv.CATEGORY_ID=b.sub_category2_ID
    and ut.user_name = c.user_name) as sub_category2,
              (select cv.category_name
    from
    si_dms_categories_v cv,
    MS_DMS_LOCAL_USERS_T ut
    where cv.LOCALE_ID = ut.locale_id
    and cv.CATEGORY_ID=b.sub_category3_ID
    and ut.user_name = c.user_name) as sub_category3,
       (select cv.category_name
    from
    si_dms_categories_v cv,
    MS_DMS_LOCAL_USERS_T ut
    where cv.LOCALE_ID = ut.locale_id
    and cv.CATEGORY_ID=b.sub_category4_ID
    and ut.user_name = c.user_name) as sub_category4,	
 b.document_object_id as document_object_id 
FROM MS_DMS_DOC_AUDIT_HISTORY a,MS_DMS_DOCUMENTS_HDR b,TEMP_TAB c
WHERE A.LIFE_CYCLE_ID = C.LIFECYCLE_ID 
AND A.LIFE_CYCLE_ID = B.LIFE_CYCLE_ID
AND b.in_process_flag = 'Y' 
AND a.process_instance_id = b.process_instance_id
AND audit_id = (SELECT MAX (audit_id) FROM MS_DMS_DOC_AUDIT_HISTORY where process_instance_id = b.process_instance_id)
