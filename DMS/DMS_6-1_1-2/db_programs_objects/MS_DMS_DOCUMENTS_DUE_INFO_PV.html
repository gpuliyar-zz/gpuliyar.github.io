MS_DMS_DOCUMENTS_DUE_INFO_PV 

                                                
  CREATE OR REPLACE FORCE VIEW "PDMS61SP2_DEV"."MS_DMS_DOCUMENTS_DUE_INFO_PV" ("DOCUMENT_NAME", "CATEGORY_ID", "SUB_CATEGORY1", "SUB_CATEGORY2", "SUB_CATEGORY3", "SUB_CATEGORY4", "LOCALE_ID", "LAST_UPDATED_BY", "DUE_DATE", "DUE_DATE_COLOR", "AUTHOR_FULL_NAME", "AUTHOR_NAME", "OWNER_FULL_NAME", "OWNER_NAME", "EFFECTIVE_FROM", "DOCUMENT_OBJECT_ID", "DOCUMENT_ID", "INITIATOR_FULL_NAME", "INITIATOR_NAME", "CREATED_ON", "DOC_TYPE_STORED", "DOCUMENT_TYPE", "VERSION", "NEXT_SCHEDULE_DATE", "REASSIGN_URL", "PROCESS_INSTANCE_ID", "USER_NAME", "STAGE_ID", "LAST_UPDATED_ON", "STAGE_NAME") AS 
  select "DOCUMENT_NAME","CATEGORY_ID","SUB_CATEGORY1","SUB_CATEGORY2","SUB_CATEGORY3","SUB_CATEGORY4","LOCALE_ID","LAST_UPDATED_BY","DUE_DATE","DUE_DATE_COLOR","AUTHOR_FULL_NAME","AUTHOR_NAME","OWNER_FULL_NAME","OWNER_NAME","EFFECTIVE_FROM","DOCUMENT_OBJECT_ID","DOCUMENT_ID","INITIATOR_FULL_NAME","INITIATOR_NAME","CREATED_ON","DOC_TYPE_STORED","DOCUMENT_TYPE","VERSION","NEXT_SCHEDULE_DATE","REASSIGN_URL","PROCESS_INSTANCE_ID","USER_NAME","STAGE_ID","LAST_UPDATED_ON","STAGE_NAME" from (
select 
       --MS_DMS_RPT_UTILS.get_form_url(B.process_instance_id,b.infolet_id, a.document_name,b.ASSIGNMENT_ID )as document_name,
       ('' || SUBSTR('' || htf.ESCAPE_SC(a.document_name) ||'',1,3800) || '') as document_name,
       A.CATEGORY_ID as category_id,
       A.SUB_CATEGORY1_ID as  SUB_CATEGORY1,
       A.SUB_CATEGORY2_ID AS  SUB_CATEGORY2,
       A.SUB_CATEGORY3_ID AS  SUB_CATEGORY3,
       A.SUB_CATEGORY4_ID AS  SUB_CATEGORY4,
       d.locale_id        as  locale_id,
      -- ms_apps_utilities.get_user_full_name(b.last_updated_by) as last_updated_by,
       (select full_name from ms_dms_local_users_t where user_id = b.last_updated_by) as last_updated_by,
       B.due_date as due_date,
       CASE WHEN (TRUNC(B.due_date)-TRUNC(SYSDATE)) IS NULL THEN 'NONE'
       WHEN (TRUNC(B.due_date)-TRUNC(SYSDATE)) < 0 THEN 'RED'
       WHEN (TRUNC(B.due_date)-TRUNC(SYSDATE)) >= 0 THEN 'GREEN'
       end as due_date_color,
      -- ms_apps_utilities.get_user_full_name(a.author_id) as author_full_name,
      (select full_name from ms_dms_local_users_t where user_id = a.author_id) as author_full_name,
       a.author_name as author_name,
      -- ms_apps_utilities.get_user_full_name(a.owner_id) as owner_full_name,
      (select full_name from ms_dms_local_users_t where user_id = a.owner_id) as owner_full_name,
       a.owner_name  as owner_name,
       nvl(a.effective_from,a.created_on) as effective_from,
       a.document_object_id  as document_object_id,
       a.document_number as document_id,
      -- ms_apps_utilities.get_user_full_name(originator_id) as initiator_full_name,
      (select full_name from ms_dms_local_users_t where user_id = originator_id) as initiator_full_name,

       a.created_on as created_on,

       ( SELECT displayed_value
             FROM ms_qs_list_of_values lovvalue, ms_qs_lov_names lovname
            WHERE lovvalue.lov_id = lovname.lov_id
            AND lovvalue.org_entity_id = 1
            AND lovvalue.enterprise_id = (SELECT enterprise_id
                                                FROM si_metrics
                                               WHERE metric_id = b.infolet_id)
            AND lovvalue.stored_value = a.doc_type
            AND lovname.enterprise_id = (SELECT enterprise_id
                                                FROM si_metrics
                                               WHERE metric_id = b.infolet_id)
            and lovname.lov_name = 'MS_DMS_INLINE_DOCUMENT_TYPE'
            AND lovvalue.locale_id = d.locale_id) as document_type,
       a.version_number as version,
       (CASE WHEN SCHEDULE_PERIOD_TYPE=1 THEN (nvl(a.effective_from,a.created_on) + REVIEW_SCHEDULE_DAYS)
       when schedule_period_type=2 then (add_months(nvl(a.effective_from,a.created_on),review_schedule_days))
       WHEN SCHEDULE_PERIOD_TYPE=3 THEN (ADD_MONTHS(nvl(a.effective_from,a.created_on),(REVIEW_SCHEDULE_DAYS*12)))
       else null
       end) as next_schedule_date,
     --  MS_DMS_RPT_UTILS.get_reassign_url(B.process_instance_id,b.infolet_id,b.assignment_id) as reassign_url,
     ('') as reassign_url,
       b.process_instance_id as process_instance_id,
       d.user_name as user_name,
	   to_char(a.current_stage_id) as stage_id,
	   nvl(b.last_update_date,b.creation_date) as last_updated_on,
	  -- MS_DMS_MLS_UTILITIES.MS_DMS_GET_STAGE_MLSVALUE(a.CURRENT_STAGE_ID,d.locale_id) as stage_name
      (SELECT STAGE_NAME
          FROM SI_DMS_LIFECYCLE_STAGES_V
          where stage_id=a.current_stage_id
          AND LOCALE_ID = d.locale_id) as stage_name
 from
 MS_DMS_DOCUMENTS_HDR_ALL_V a , si_event_assignments b,
 (select aa.user_id,aa.user_name,aa.locale,bb.locale_id from si_users_t aa , si_locales bb where aa.locale = bb.locale_string) d
where a.process_instance_id = b.process_instance_id
and b.user_id = d.user_id
and b.infolet_id IN
                   (select metric_id
                     from si_metrics_t
           where metric_name in ('MS_DMS_CREATE_DOC','MS_DMS_REVIEW_DOCUMENT','MS_DMS_UPDATE_DOCUMENT'
           ,'MS_DMS_INLINE_CREATE_DOCMUNENT','MS_DMS_INLINE_REVIEW_DOCUMENT','MS_DMS_INLINE_UPDATE_FORM')
            )

        (select full_name from ms_dms_local_users_t where user_id = t1.LAST_UPDATED_BY) as last_updated_by,
        t1.DUE_DATE as due_date,
        CASE
         WHEN (TRUNC(t1.due_date) - TRUNC(SYSDATE)) IS NULL THEN
          'NONE'
         WHEN (TRUNC(t1.due_date) - TRUNC(SYSDATE)) < 0 THEN
          'RED'
         when (TRUNC(t1.DUE_DATE) - TRUNC(sysdate)) >= 0 then
          'GREEN'
       end as due_date_color,
       --MS_APPS_UTILITIES.GET_USER_FULL_NAME(ms_apps_utilities.get_user_id(t.AUTHOR_NAME)) as author_full_name,
       (select full_name from ms_dms_local_users_t where user_name = t.AUTHOR_NAME) as author_full_name,
       t.author_name as author_name,
       --(ms_apps_utilities.get_user_full_name(ms_apps_utilities.get_user_id(t.owner_name))) as owner_full_name,
       (select full_name from ms_dms_local_users_t where user_name = t.owner_name) as owner_full_name,
       t.owner_name as owner_name,
       NVL(T.EFFECTIVE_FROM_DATE, T.CREATION_DATE) as effective_from,

       null as document_id,
       --(MS_APPS_UTILITIES.GET_USER_FULL_NAME(INITIATOR_NAME)) as initiator_full_name,

       t.creation_date as created_on,
       t.document_type as doc_type_stored, 
       /*MS_APPS_UTILITIES.Get_Display_Value(MS_APPS_UTILITIES.get_enterprise_id(t1.infolet_id),
                                           'MS_DMS_INLINE_DOCUMENT_TYPE',
                                           t.document_type,
                                           1,
                                           t2.locale_id) as document_type, */
          ( SELECT displayed_value
                       FROM ms_qs_list_of_values lovvalue, ms_qs_lov_names lovname
                      WHERE lovvalue.lov_id = lovname.lov_id
                      AND lovvalue.org_entity_id = 1
                      AND lovvalue.enterprise_id = (SELECT enterprise_id
                                                          FROM si_metrics
                                                         WHERE metric_id = t1.infolet_id)
                      AND lovvalue.stored_value = t.document_type
                      AND lovname.enterprise_id = (SELECT enterprise_id
                                                          FROM si_metrics
                                                         WHERE metric_id = t1.infolet_id)
                      and lovname.lov_name = 'MS_DMS_INLINE_DOCUMENT_TYPE'
                      AND lovvalue.locale_id = t2.locale_id) as document_type,
       NULL as version,
       NULL AS NEXT_SCHEDULE_DATE,
       NULL AS reassign_url,
       t1.process_instance_id as process_instance_id,
       t2.user_name as user_name,
	  -- null as stage_id
	   t.CURRENT_STAGE as stage_id,
	   nvl(t1.last_update_date,t1.creation_date) as last_updated_on,
	   --MS_DMS_MLS_UTILITIES.MS_DMS_GET_STAGE_MLSVALUE(to_number(t.CURRENT_STAGE),t2.locale_id) as stage_name
     (SELECT STAGE_NAME
          FROM SI_DMS_LIFECYCLE_STAGES_V
          where stage_id=to_number(t.current_stage)
          AND LOCALE_ID = t2.locale_id) as stage_name 
	   from MS_DMS_COMMON_VS t, SI_EVENT_ASSIGNMENTS t1,
       (select aa.user_id,aa.user_name,aa.locale,bb.locale_id from si_users_t aa , si_locales bb where aa.locale = bb.locale_string) t2
       where t.process_instance_id = t1.process_instance_id
       and t.instance_id in
                     (select max(instance_id)
                        from MS_DMS_COMMON_VS
                       where process_instance_id = t1.process_instance_id)
        and t.instance_rec_num = 1
        and t1.user_id = t2.user_id
        and t1.infolet_id IN
             (select metric_id
                from si_metrics_t
               where metric_name in
                     ('MS_DMS_CREATE_DOC', 'MS_DMS_REVIEW_DOCUMENT',
                      'MS_DMS_UPDATE_DOCUMENT', 'MS_DMS_INLINE_CREATE_DOCMUNENT',
                      'MS_DMS_INLINE_REVIEW_DOCUMENT', 'MS_DMS_INLINE_UPDATE_FORM'))
       and not exists ( select 1 from ms_dms_documents_hdr_all_v x where x.process_instance_id = t.process_instance_id)
union ALL
select  
        --MS_DMS_RPT_UTILS.GET_FORM_URL(t1.PROCESS_INSTANCE_ID,t1.INFOLET_ID,t.object_name,t1.assignment_id) as document_name,
        ('' || SUBSTR('' || htf.ESCAPE_SC(t.object_name) ||'',1,3800) || '') as document_name,
        t.HDN_SELECTED_CATEGORY_ID as category_id,
        T.HDN_SUB_CATEGORY1 as SUB_CATEGORY1,
        T.HDN_SUB_CATEGORY2 as SUB_CATEGORY2,
        T.HDN_SUB_CATEGORY3 as SUB_CATEGORY3,
        t.HDN_SUB_CATEGORY4 as SUB_CATEGORY4,
        t2.locale_id        as locale_id,
        --MS_APPS_UTILITIES.GET_USER_FULL_NAME(t1.LAST_UPDATED_BY) as last_updated_by,
        (select full_name from ms_dms_local_users_t where user_id = t1.LAST_UPDATED_BY) as last_updated_by,
        t1.DUE_DATE as due_date,
        CASE
         WHEN (TRUNC(t1.due_date) - TRUNC(SYSDATE)) IS NULL THEN
          'NONE'
         WHEN (TRUNC(t1.due_date) - TRUNC(SYSDATE)) < 0 THEN
          'RED'
         when (TRUNC(t1.DUE_DATE) - TRUNC(sysdate)) >= 0 then
          'GREEN'
       end as due_date_color,
       --ms_apps_utilities.get_user_full_name(ms_apps_utilities.get_user_id(t.author)) as author_full_name,
       (select full_name from ms_dms_local_users_t where user_name = t.author) as author_full_name,
       t.author as author_name,
       --(ms_apps_utilities.get_user_full_name(ms_apps_utilities.get_user_id(t.owner_name))) as owner_full_name,
       (select full_name from ms_dms_local_users_t where user_name = t.owner_name) as owner_full_name,
       t.owner_name as owner_name,
       NVL(T.EFFECTIVE_FROM, T.CREATION_DATE) as effective_from,
       null as document_object_id,
       null as document_id,
       --(MS_APPS_UTILITIES.GET_USER_FULL_NAME(INITIATOR)) as initiator_full_name,
       (select full_name from ms_dms_local_users_t where user_name = INITIATOR) as initiator_full_name,
       INITIATOR as initiator_name,
       t.creation_date as created_on,
       t.document_type as doc_type_stored,
      /* MS_APPS_UTILITIES.Get_Display_Value(MS_APPS_UTILITIES.get_enterprise_id(t1.infolet_id),
                                           'MS_DMS_INLINE_DOCUMENT_TYPE',
                                           t.document_type,
                                           1,
                                           t2.locale_id) as document_type, */
       ( SELECT displayed_value
                       FROM ms_qs_list_of_values lovvalue, ms_qs_lov_names lovname
                      WHERE lovvalue.lov_id = lovname.lov_id
                      AND lovvalue.org_entity_id = 1
                      AND lovvalue.enterprise_id = (SELECT enterprise_id
                                                          FROM si_metrics
                                                         WHERE metric_id = t1.infolet_id)
                      AND lovvalue.stored_value = t.document_type
                      AND lovname.enterprise_id = (SELECT enterprise_id
                                                          FROM si_metrics
                                                         WHERE metric_id = t1.infolet_id)
                      and lovname.lov_name = 'MS_DMS_INLINE_DOCUMENT_TYPE'
                      AND lovvalue.locale_id = t2.locale_id) as document_type,
       NULL as version,
       NULL AS NEXT_SCHEDULE_DATE,
       NULL AS reassign_url,
       t1.process_instance_id as process_instance_id,
       t2.user_name as user_name,
	   --null as stage_id
	   t.CURRENT_STAGE as stage_id,
	   nvl(t1.last_update_date,t1.creation_date) as last_updated_on,
	   --MS_DMS_MLS_UTILITIES.MS_DMS_GET_STAGE_MLSVALUE(to_number(t.CURRENT_STAGE),t2.locale_id) as stage_name
     (SELECT STAGE_NAME
          FROM SI_DMS_LIFECYCLE_STAGES_V
          where stage_id=to_number(t.current_stage)
          AND LOCALE_ID = t2.locale_id) as stage_name

        and t.instance_rec_num = 1
        and t1.user_id = t2.user_id
        and t1.infolet_id IN
       (select metric_id
          from si_metrics_t

               ('MS_DMS_CREATE_DOC', 'MS_DMS_REVIEW_DOCUMENT',

                'MS_DMS_INLINE_REVIEW_DOCUMENT', 'MS_DMS_INLINE_UPDATE_FORM'))
