MS_DMS_DOC_PRODUCTION_V 

                                                
  CREATE OR REPLACE FORCE VIEW "PDMS61SP2_DEV"."MS_DMS_DOC_PRODUCTION_V" ("CATEGORY", "SUB_CATEGORY1", "SUB_CATEGORY2", "SUB_CATEGORY3", "SUB_CATEGORY4", "CATEGORY_ID", "SUB_CATEGORY1_ID", "SUB_CATEGORY2_ID", "SUB_CATEGORY3_ID", "SUB_CATEGORY4_ID", "DOCUMENT_NAME", "DOCUMENT_NUMBER", "PROCESS_INSTANCE_ID", "DOCUMENT_ATTACHMENT", "REVISION_NUMBER", "LAST_UPDATED_BY", "TIMESTAMP", "DOCUMENT_AUTHOR", "INSTANCE_ID", "DOCUMENT_OBJECT_ID", "DOCUMENT_OBJECT_TYPE", "AUDIT_ID", "KEYWORDS", "LAST_REVIEW_DATE", "PURPOSE", "DESCRIPTION", "EFFECTIVE_DATE", "SUPPORTING_DOCUMENTS", "IS_INLINE", "DOCUMENT_ORIGINATOR", "NEXT_REVIEW_DATE", "DOC_TYPE", "DOCUMENT_OWNER", "ORATXT_IDX") AS 
  SELECT A.CATEGORY, A.sub_category1, A.sub_category2, A.sub_category3, A.sub_category4, A.CATEGORY_id, A.sub_category1_id, A.sub_category2_id, A.sub_category3_id, A.sub_category4_id, a.document_name, a.document_number, a.process_instance_id, a.document_attachment, a.revision_number, A.LAST_USERNAME, a.TIMESTAMP, a.document_author , a.instance_id, a.document_object_id, a.document_object_type, a.audit_id , a.key_words, a.LAST_REVIEW_DATE, a.purpose, a.description, a.effective_date, a.supporting_documents, a.is_inline, A.DOCUMENT_ORIGINATOR, A.NEXT_REVIEW_DATE,A.DOC_TYPE,A.DOCUMENT_OWNER,A.ORATXT_IDX
  FROM MS_DMS_DOCUMENTS a
  WHERE (a.effective_date IS NULL OR TRUNC(A.effective_date) <= TRUNC(SYSDATE)) AND obsolete_flag ='N'
  UNION ALL
  SELECT B.CATEGORY, B.sub_category1, B.sub_category2, B.sub_category3, B.sub_category4, B.CATEGORY_id, B.sub_category1_id, B.sub_category2_id, B.sub_category3_id, B.sub_category4_id, a.document_name, a.document_number, a.process_instance_id, a.document_attachment, a.revision_number, B.LAST_USERNAME, a.TIMESTAMP, a.document_author , a.instance_id, a.document_object_id, a.document_object_type, a.audit_id , a.key_words, b.LAST_REVIEW_DATE, a.purpose, a.description, a.effective_date, a.supporting_documents, a.is_inline, B.DOCUMENT_ORIGINATOR,B.NEXT_REVIEW_DATE,B.DOC_TYPE,B.DOCUMENT_OWNER,A.ORATXT_IDX
  FROM MS_DMS_ALL_PRODUCTION_DOCS a , MS_DMS_DOCUMENTS b
  WHERE EXISTS
    (SELECT document_object_id
    FROM MS_DMS_DOCUMENTS MDD
    WHERE effective_date      > SYSDATE AND b.document_object_id=mdd.document_object_Id
    ) AND a.document_object_id=b.document_object_id AND a.audit_id =
    (SELECT MAX(audit_id)
    FROM MS_DMS_ALL_PRODUCTION_DOCS
    WHERE document_object_id=a.document_object_id AND audit_id <>b.audit_id AND effective_date<=sysdate
    )
  ORDER BY 13 DESC
