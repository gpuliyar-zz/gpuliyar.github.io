MS_DMS_PRODUCTION_DOCS_V 

                                                
  CREATE OR REPLACE FORCE VIEW "PDMS61SP2_DEV"."MS_DMS_PRODUCTION_DOCS_V" ("RIGHTS", "METRIC_ID", "PROCESS_INSTANCE_ID", "INSTANCE_ID", "ATTACHMENT", "COLSEQ", "USER_NAME", "CREATION_DATE", "ATTCOLNAME", "CREATED_BY", "CREATION_DATE_SQL", "CREATION_DATE_CMP", "DOCUMENT_TYPE", "DOCUMENT_NUMBER", "VERSION_NUMBER", "ACTION", "DOCUMENT_NAME", "CATEGORY", "SUB_CATEGORY1", "SUB_CATEGORY2", "SUB_CATEGORY3", "SUB_CATEGORY4", "CURRENT_LEVEL_NAME", "NEXT_LEVEL_NAME", "DESCRIPTION", "PURPOSE", "REVIEW_SCHEDULE_DAYS", "KEYWORDS", "DOCUMENT_CREATOR", "USER_LIST", "CURRENT_USER", "NEXT_STAGE", "DOCUMENT_STATUS", "LIFECYCLE_NAME", "USER_ID", "DAYS_TO_REVIEW", "NEXT_REVIEW_DATE", "RELATIONSHIP_DETAILS", "EFFECTIVE_DATE") AS 
  ( 
SELECT 
	DECODE(SUM(SUBSTR(g.access_rights,1,1)),0,0,1)||DECODE(SUM(SUBSTR(g.access_rights,2,1)),0,0,1)|| 
	DECODE(SUM(SUBSTR(g.access_rights,3,1)),0,0,1)||DECODE(SUM(SUBSTR(g.access_rights,4,1)),0,0,1)|| 
	DECODE(SUM(SUBSTR(g.access_rights,5,1)),0,0,1)||DECODE(SUM(SUBSTR(g.access_rights,6,1)),0,0,1)|| 
	DECODE(SUM(SUBSTR(g.access_rights,7,1)),0,0,1)||DECODE(SUM(SUBSTR(g.access_rights,8,1)),0,0,1)|| 
	DECODE(SUM(SUBSTR(g.access_rights,9,1)),0,0,1)||DECODE(SUM(SUBSTR(g.access_rights,10,1)),0,0,1)|| 
	DECODE(SUM(SUBSTR(g.access_rights,11,1)),0,0,1)||DECODE(SUM(SUBSTR(g.access_rights,12,1)),0,0,1)|| 
	DECODE(SUM(SUBSTR(g.access_rights,13,1)),0,0,1)||DECODE(SUM(SUBSTR(g.access_rights,14,1)),0,0,1)|| 
	DECODE(SUM(SUBSTR(g.access_rights,15,1)),0,0,1)||DECODE(SUM(SUBSTR(g.access_rights,16,1)),0,0,1)|| 
	DECODE(SUM(SUBSTR(g.access_rights,17,1)),0,0,1)||DECODE(SUM(SUBSTR(g.access_rights,18,1)),0,0,1)|| 
	DECODE(SUM(SUBSTR(g.access_rights,19,1)),0,0,1)||DECODE(SUM(SUBSTR(g.access_rights,20,1)),0,0,1)|| 
	DECODE(SUM(SUBSTR(g.access_rights,21,1)),0,0,1)||DECODE(SUM(SUBSTR(g.access_rights,22,1)),0,0,1)|| 
	DECODE(SUM(SUBSTR(g.access_rights,23,1)),0,0,1)||DECODE(SUM(SUBSTR(g.access_rights,24,1)),0,0,1)|| 
	DECODE(SUM(SUBSTR(g.access_rights,25,1)),0,0,1)||DECODE(SUM(SUBSTR(g.access_rights,26,1)),0,0,1)|| 
	DECODE(SUM(SUBSTR(g.access_rights,27,1)),0,0,1)||DECODE(SUM(SUBSTR(g.access_rights,28,1)),0,0,1)|| 
	DECODE(SUM(SUBSTR(g.access_rights,29,1)),0,0,1)||DECODE(SUM(SUBSTR(g.access_rights,30,1)),0,0,1)|| 
	DECODE(SUM(SUBSTR(g.access_rights,31,1)),0,0,1) access_rights, 
	(select metric_id from si_metrics_t where metric_name like 'MS DMS Document Review' and end_Date is null), -- metric_id 
	a.process_instance_id, 
	a.instance_id, 
	a.document_attachment attachment, 
	10 colseq, 
	a.last_username, 
	TO_CHAR(a.TIMESTAMP, 'mm/dd/yyyy HH:MI AM') creation_date, 
	'Document' attcolname, 
	a.last_username, 
	a.TIMESTAMP, 
	NULL, --creation_date_cmp 
	NULL, --a.document_type 
	a.document_number, 
	a.revision_number, 
	NULL, --action_type 
	a.document_name, 
	a.CATEGORY, 
	a.sub_category1, 
	a.sub_category2, 
	a.sub_category3, 
	a.sub_category4, 
	NULL, --t.current_level_name, 
	NULL, --t.next_level_name, 
	a.description, 
	a.purpose, 
	a.review_schedule_days, 
	a.key_words, 
	a.document_owner, 
	--ms_dms_document_author(a.document_name), 
	NULL, --t.user_list, 
	NULL, --t.current_user, 
	NULL, --t.next_stage, 
       'Production' document_status, 
	(SELECT lifecycle_name FROM SI_DMS_LIFECYCLES WHERE lifecycle_id = a.approval_process_id) lifecycle_name, 
	e.user_id, 
	TRUNC((NVL(a.last_review_date, a.TIMESTAMP) + a.review_schedule_days)) - TRUNC(SYSDATE), 
	TO_CHAR((NVL(a.last_review_date, a.TIMESTAMP)+a.review_schedule_days), 'mm/dd/yyyy HH:MI AM'), 
	Ms_Dms_Get_Assoc_Docs_List(a.document_name) asso_docs, 
	TO_CHAR(NVL(a.effective_date, a.TIMESTAMP), 'mm/dd/yyyy HH:MI AM') eff_Dat 
FROM 
     MS_DMS_DOCUMENTS a, 
     SI_DMS_ROLES g, 
     si_users siu, 
     SI_DMS_ACCESS_GRP_CATS f, 
	SI_DMS_USER_GROUP_USERS e 
WHERE 
f.access_group_id = e.user_group_id AND 
siu.user_id = e.user_id AND 
g.enterprise_id = siu.enterprise_id AND 
g.role_id = f.role_id 
AND A.CATEGORY = f.CATEGORY 
AND ((f.subcategory1 IS NULL)OR (A.sub_category1 = f.subcategory1)) 
AND ((f.subcategory2 IS NULL)OR (A.sub_category2 = f.subcategory2)) 
AND ((f.subcategory3 IS NULL)OR (A.sub_category3 = f.subcategory3)) 
AND ((f.subcategory4 IS NULL)OR (A.sub_category4 = f.subcategory4)) 
AND A.frozen_flag = 'Y' 
AND ( NVL(A.obsolete_flag,'N') = 'N' OR 
      (A.obsolete_flag = 'Y' AND a.effective_date > SYSDATE) ) 
/* Added by Peter Chow */ 
AND a.process_instance_id = 
 (SELECT MAX(process_instance_id) FROM MS_DMS_DOCUMENTS mdd 
   WHERE mdd.document_name = a.document_name 
   AND   DECODE(TRUNC(SYSDATE),TRUNC(NVL(a.effective_date,SYSDATE)), 'Production', 
	     DECODE(((TRUNC(SYSDATE)-TRUNC(NVL(a.effective_date,SYSDATE)))+ 
	  	         TRUNC(SYSDATE)-TRUNC(NVL(a.effective_date,SYSDATE))),0, 
			  	  'Ready For Production','Production')) = 'Production') 
GROUP BY 
a.process_instance_id, a.instance_id,a.document_attachment,a.last_username, a.TIMESTAMP, 
TO_CHAR(a.TIMESTAMP, 'mm/dd/yyyy HH:MI AM') , 
a.document_author, 
a.document_number,a.revision_number,a.document_name, 
a.CATEGORY, a.sub_category1,a.sub_category2,a.sub_category3,a.sub_category4, 
a.description,a.purpose,a.review_schedule_days, 
a.key_words, a.document_owner, 'Production', 
a.approval_process_id, e.user_id, a.last_review_date, ABS(TRUNC((NVL(a.last_review_date, a.TIMESTAMP) + a.review_schedule_days)) - TRUNC(SYSDATE)), 
TO_CHAR((NVL(a.last_review_date, a.TIMESTAMP)+a.review_schedule_days), 'mm/dd/yyyy HH:MI AM'), 
Ms_Dms_Get_Assoc_Docs_List(a.document_name), TO_CHAR(NVL(a.effective_date, a.TIMESTAMP), 'mm/dd/yyyy HH:MI AM') 
)
