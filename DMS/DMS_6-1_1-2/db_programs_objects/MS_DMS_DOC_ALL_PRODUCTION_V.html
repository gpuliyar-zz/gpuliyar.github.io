MS_DMS_DOC_ALL_PRODUCTION_V 

                                                
  CREATE OR REPLACE FORCE VIEW "PDMS61SP2_DEV"."MS_DMS_DOC_ALL_PRODUCTION_V" ("CATEGORY", "SUB_CATEGORY1", "SUB_CATEGORY2", "SUB_CATEGORY3", "SUB_CATEGORY4", "DOCUMENT_NAME", "PROCESS_INSTANCE_ID", "DOCUMENT_ATTACHMENT", "REVISION_NUMBER", "LAST_UPDATED_BY", "LAST_USERNAME", "TIMESTAMP", "DOCUMENT_AUTHOR", "INSTANCE_ID", "DOCUMENT_OBJECT_ID", "DOCUMENT_OBJECT_TYPE", "AUDIT_ID", "KEY_WORDS", "PURPOSE", "DESCRIPTION", "DOCUMENT_NUMBER", "CATEGORY_ID", "SUB_CATEGORY1_ID", "SUB_CATEGORY2_ID", "SUB_CATEGORY3_ID", "SUB_CATEGORY4_ID", "LAST_REVIEW_DATE", "EFF_DATE", "EFFECTIVE_DATE", "SUPPORTING_DOCUMENTS", "IS_INLINE", "DOCUMENT_ORIGINATOR", "DOC_TYPE", "NEXT_REVIEW_DATE", "ORATXT_IDX", "DOCUMENT_OWNER") AS 
  SELECT A.CATEGORY,A.SUB_CATEGORY1, A.SUB_CATEGORY2, A.SUB_CATEGORY3, A.SUB_CATEGORY4, A.DOCUMENT_NAME, A.PROCESS_INSTANCE_ID, A.DOCUMENT_ATTACHMENT, A.REVISION_NUMBER,A.LAST_USERNAME, A.LAST_USERNAME, A.TIMESTAMP, A.DOCUMENT_AUTHOR, A.INSTANCE_ID,DOCUMENT_OBJECT_ID,DOCUMENT_OBJECT_TYPE,AUDIT_ID,KEY_WORDS,PURPOSE,DESCRIPTION,DOCUMENT_NUMBER, A.CATEGORY_ID,A.SUB_CATEGORY1_ID, A.SUB_CATEGORY2_ID, A.SUB_CATEGORY3_ID, A.SUB_CATEGORY4_ID, A.LAST_REVIEW_DATE, A.EFFECTIVE_DATE, A.EFFECTIVE_DATE, A.SUPPORTING_DOCUMENTS,NVL(A.IS_INLINE,'N') IS_INLINE ,A.DOCUMENT_ORIGINATOR,A.DOC_TYPE,A.NEXT_REVIEW_DATE,A.ORATXT_IDX,A.DOCUMENT_OWNER
FROM MS_DMS_ALL_PRODUCTION_DOCS A
WHERE NVL( A.EFFECTIVE_DATE,SYSDATE) <= SYSDATE AND OBSOLETE_FLAG!='Y' AND DOCUMENT_OBJECT_ID NOT IN
  (SELECT DOCUMENT_OBJECT_ID FROM MS_DMS_DOCUMENTS WHERE OBSOLETE_FLAG='Y'
  )
UNION ALL
SELECT A.CATEGORY,A.SUB_CATEGORY1, A.SUB_CATEGORY2, A.SUB_CATEGORY3, A.SUB_CATEGORY4, A.DOCUMENT_NAME, A.PROCESS_INSTANCE_ID, A.DOCUMENT_ATTACHMENT, A.REVISION_NUMBER, A.LAST_USERNAME,A.LAST_USERNAME, A.TIMESTAMP, A.DOCUMENT_AUTHOR, A.INSTANCE_ID ,A.DOCUMENT_OBJECT_ID,A.DOCUMENT_OBJECT_TYPE,A.AUDIT_ID,A.KEY_WORDS,A.PURPOSE,A.DESCRIPTION,A.DOCUMENT_NUMBER, A.CATEGORY_ID,A.SUB_CATEGORY1_ID, A.SUB_CATEGORY2_ID, A.SUB_CATEGORY3_ID, A.SUB_CATEGORY4_ID, A.LAST_REVIEW_DATE, A.EFFECTIVE_DATE, A.EFFECTIVE_DATE, A.SUPPORTING_DOCUMENTS,NVL(A.IS_INLINE,'N') IS_INLINE ,A.DOCUMENT_ORIGINATOR,A.DOC_TYPE,A.NEXT_REVIEW_DATE,A.ORATXT_IDX,A.DOCUMENT_OWNER
FROM MS_DMS_ALL_PRODUCTION_DOCS A , MS_DMS_DOCUMENTS B
WHERE EXISTS
  (SELECT DOCUMENT_NAME
  FROM MS_DMS_DOCUMENTS MDD
  WHERE OBSOLETE_FLAG       ='Y' AND EFFECTIVE_DATE > SYSDATE AND B.DOCUMENT_NAME =MDD.DOCUMENT_NAME
  ) AND A.DOCUMENT_OBJECT_ID=B.DOCUMENT_OBJECT_ID AND A.PROCESS_INSTANCE_ID=
  (SELECT MAX(PROCESS_INSTANCE_ID)
  FROM MS_DMS_ALL_PRODUCTION_DOCS
  WHERE DOCUMENT_OBJECT_ID=A.DOCUMENT_OBJECT_ID AND PROCESS_INSTANCE_ID<>B.PROCESS_INSTANCE_ID
  )
ORDER BY 13 DESC
