MS_DMS_LF_USER_LIST_V 

                                                
  CREATE OR REPLACE FORCE VIEW "PDMS61SP2_DEV"."MS_DMS_LF_USER_LIST_V" ("LIFECYCLE_ID", "SEQ", "STAGE_ID", "USER_ID", "USER_NAME", "CREATE_DOCUMENT", "VIEW_ANNOTATION", "WRITE_ANOTATION", "DOWNLOAD", "WRITE_ALL_ANOT", "VIEW_DOCUMENT", "UPDATE_DOCUMENT", "PRINT_DOCUMENT", "VIEW_ALL_ANOTATION") AS 
  SELECT lifecycle_id,seq,stage_id,b.user_id,b.user_name, 
DECODE(SUM(SUBSTR(ACCESS_RIGHTS,1,1)),0,'No','Yes')  create_document, 
DECODE(SUM(SUBSTR(ACCESS_RIGHTS,3,1)),0,'No','Yes') view_annotation, 
DECODE(SUM(SUBSTR(ACCESS_RIGHTS,5,1)),0,'No','Yes') write_anotation, 
DECODE(SUM(SUBSTR(ACCESS_RIGHTS,7,1)),0,'No','Yes') download, 
DECODE(SUM(SUBSTR(ACCESS_RIGHTS,9,1)),0,'No','Yes') write_all_anot, 
DECODE(SUM(SUBSTR(ACCESS_RIGHTS,2,1)),0,'No','Yes')  AS View_document, 
DECODE(SUM(SUBSTR(ACCESS_RIGHTS,4,1)),0,'No','Yes')  AS update_document, 
DECODE(SUM(SUBSTR(ACCESS_RIGHTS,6,1)),0,'No','Yes') print_document, 
DECODE(SUM(SUBSTR(ACCESS_RIGHTS,8,1)),0,'No','Yes') view_all_anotation 
FROM 
ms_dms_lf_user_role_details_v a,SI_USERS_T b 
WHERE 
a.user_id=b.user_id    and a.user_id=(case when  ms_dms_chk_outofoffice_user(a.user_id)>=1 then a.user_id
else 0 end)
GROUP BY lifecycle_id,seq,stage_id,b.user_id,b.user_name order by b.user_name
