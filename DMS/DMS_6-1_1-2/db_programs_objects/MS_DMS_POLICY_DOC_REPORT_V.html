MS_DMS_POLICY_DOC_REPORT_V 

                                                
  CREATE OR REPLACE FORCE VIEW "PDMS61SP2_DEV"."MS_DMS_POLICY_DOC_REPORT_V" ("DOC_OBJECTID", "DOCUMENT_NAME", "VERSION_NUMBER", "CATEGORY", "CATEGORY1", "CATEGORY2", "CATEGORY3", "CATEGORY4", "CATEGORY_ID", "SUB_CATEGORY1_ID", "SUB_CATEGORY2_ID", "SUB_CATEGORY3_ID", "SUB_CATEGORY4_ID", "DOCUMENT_STATUS", "EFFECTIVE_DATE", "COL1", "COL2", "COL3", "COL4", "CHECKED_OUT_FLAG", "DOCUMENT_OBJECT_ID", "DOC_NAME") AS 
  SELECT (a.doc_objectid),
    (SELECT ''
      || a.document_name
      || ''
    FROM DUAL
    ) AS document_name,
    a.version_number,
    (SELECT c.category_name
    FROM si_dms_categories c
    WHERE c.category_id = a.category_id
    ) CATEGORY,
    (SELECT c.category_name
    FROM si_dms_categories c
    WHERE c.category_id = a.sub_category1_id
    ) category1,
    (SELECT c.category_name
    FROM si_dms_categories c
    WHERE c.category_id = a.sub_category2_id
    ) category2,
    (SELECT c.category_name
    FROM si_dms_categories c
    WHERE c.category_id = a.sub_category3_id
    ) category3,
    (SELECT c.category_name
    FROM si_dms_categories c
    WHERE c.category_id = a.sub_category4_id
    ) category4,
    a.category_id      AS CATEGORY_ID,
    a.sub_category1_id AS SUB_CATEGORY1_ID,
    a.sub_category2_id AS SUB_CATEGORY2_ID,
    a.sub_category3_id AS SUB_CATEGORY3_ID,
    a.sub_category4_id AS SUB_CATEGORY4_ID,
    b.document_status,
    a.effective_date,
    (
    CASE
      WHEN (NVL (a.assign_due_date, SYSDATE + 1) > SYSDATE)
      THEN
        (SELECT DECODE (COUNT (c.feedback_pid), COUNT (user_name), 'Completed', 'In Progress')
        FROM ms_dms_policy_feedback c
        WHERE c.pol_object_id = TO_CHAR (a.doc_objectid)
        AND c.user_name      IS NOT NULL
        )
      ELSE 'Completed'
    END) col1,
    'Attributes' AS col2,
    (SELECT MAX (c.seq_no)
    FROM ms_dms_policy_feedback c
    WHERE c.doc_objectid = a.doc_objectid
    AND c.user_name     IS NOT NULL
    )    AS COL3,
    NULL AS COL4,
    B.CHECKED_OUT_FLAG,
    B.DOCUMENT_OBJECT_ID,
    b.document_name as doc_name
  FROM ms_dms_manage_policy_v a,
    ms_dms_documents b
  WHERE a.doc_objectid      = b.document_object_id
  AND a.process_instance_id =
    (SELECT MAX (process_instance_id)
    FROM ms_dms_manage_policy_v
    WHERE doc_objectid = a.doc_objectid
    )
  AND a.instance_id =
    (SELECT MAX (INSTANCE_ID)
    FROM ms_dms_manage_policy_v
    WHERE process_instance_id = a.process_instance_id
    )
  AND A.HDN_FIELD3 = 'DUMMY'
  AND A.OBJECT_ID IS NOT NULL
