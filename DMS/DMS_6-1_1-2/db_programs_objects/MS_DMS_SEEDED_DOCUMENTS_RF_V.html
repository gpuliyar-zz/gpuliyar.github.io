MS_DMS_SEEDED_DOCUMENTS_RF_V 

                                                
  CREATE OR REPLACE FORCE VIEW "PDMS61SP2_DEV"."MS_DMS_SEEDED_DOCUMENTS_RF_V" ("BATCH_NUMBER", "SEED_VERSION", "PREVIOUS_DOC_NAME", "RECORD_NUMBER", "CURRENT_VERSION", "BASE_VERSION", "DOCUMENT_OBJECT_ID", "DOC_ENTERPRISE_NAME", "DOCUMENT_NAME", "LIFECYCLE_NAME", "CATEGORY", "SUB_CATEGORY1", "SUB_CATEGORY2", "SUB_CATEGORY3", "SUB_CATEGORY4", "PURPOSE", "DESCRIPTION", "REVIEW_SCHEDULE_DAYS", "SCHEDULE_PERIOD_TYPE", "KEYWORDS", "EFFECTIVE_DATE", "USER_NAME", "FILE_NAME", "FILE_PATH", "SEEDED_FLAG", "PROCESS_TYPE", "AUTHOR", "OWNER", "NO_OF_UPVERSIONS", "OBSOLUTE_SEQ", "SUPPORTING_DOCS") AS 
  select DISTINCT af.BATCH_NUMBER      ,
af.SEED_VERSION      ,
af.PREVIOUS_DOC_NAME ,
af.RECORD_NUMBER     ,
af.CURRENT_VERSION   ,
af.BASE_VERSION      ,
af.DOCUMENT_OBJECT_ID,
af.DOC_ENTERPRISE_NAME  ,
af.DOCUMENT_NAME     ,
af.LIFECYCLE_NAME    ,
cat.category_id   ,
DECODE(cat.SUBcategory1_id,'0',NULL,cat.SUBcategory1_id)     ,
DECODE(cat.SUBcategory2_id,'0',NULL,cat.SUBcategory2_id)    ,
DECODE(cat.SUBcategory3_id,'0',NULL,cat.SUBcategory3_id)     ,
DECODE(cat.SUBcategory4_id,'0',NULL,cat.SUBcategory4_id)   ,
af.PURPOSE           ,
af.DESCRIPTION       ,
af.REVIEW_SCHEDULE_DAYS ,
af.SCHEDULE_PERIOD_TYPE ,
af.KEYWORDS          ,
af.EFFECTIVE_DATE    ,
af.USER_NAME         ,
af.FILE_NAME         ,
af.FILE_PATH         ,
af.SEEDED_FLAG       ,
af.PROCESS_TYPE      ,
af.AUTHOR            ,
af.OWNER             ,
af.NO_OF_UPVERSIONS  ,
af.OBSOLUTE_SEQ      ,
af.SUPPORTING_DOCS  from  (SELECT BATCH_NUMBER      ,
SEED_VERSION      ,
PREVIOUS_DOC_NAME ,
RECORD_NUMBER     ,
CURRENT_VERSION   ,
BASE_VERSION      ,
DOCUMENT_OBJECT_ID,
DOC_ENTERPRISE_NAME  ,
DOCUMENT_NAME     ,
LIFECYCLE_NAME    ,
CATEGORY          ,
SUB_CATEGORY1     ,
SUB_CATEGORY2     ,
SUB_CATEGORY3     ,
SUB_CATEGORY4     ,
PURPOSE           ,
DESCRIPTION       ,
REVIEW_SCHEDULE_DAYS ,
SCHEDULE_PERIOD_TYPE ,
KEYWORDS          ,
EFFECTIVE_DATE    ,
USER_NAME         ,
FILE_NAME         ,
FILE_PATH         ,
SEEDED_FLAG       ,
PROCESS_TYPE      ,
AUTHOR            ,
OWNER             ,
NO_OF_UPVERSIONS  ,
OBSOLUTE_SEQ      ,
SUPPORTING_DOCS ,
TRIM(CATEGORY
    ||DECODE(NVL(SUB_CATEGORY1,0),'0',NULL,'-'
    ||SUB_CATEGORY1)
    ||DECODE(NVL(SUB_CATEGORY2,0),'0',NULL,'-'
    ||SUB_CATEGORY2)
    ||DECODE(NVL(SUB_CATEGORY3,0),'0',NULL,'-'
    ||SUB_CATEGORY3)
    ||DECODE(NVL(SUB_CATEGORY4,0),'0',NULL,'-'
    ||SUB_CATEGORY4)) CATEGORIES_NAME_CONCAT
from 
   MS_DMS_SEEDED_DOCUMENTS_RF ) af, MS_DMS_USER_GRP_CAT_RPT_V CAT
   WHERE AF.CATEGORIES_NAME_CONCAT=cat.categories_concat
