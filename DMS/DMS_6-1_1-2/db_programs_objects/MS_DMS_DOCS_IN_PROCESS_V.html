MS_DMS_DOCS_IN_PROCESS_V 

                                                
  CREATE OR REPLACE FORCE VIEW "PDMS61SP2_DEV"."MS_DMS_DOCS_IN_PROCESS_V" ("PROCESS_INSTANCE_ID", "DOCUMENT_NAME", "DOCUMENT_NUMBER", "PURPOSE", "CATEGORY", "SUB_CATEGORY1", "SUB_CATEGORY2", "SUB_CATEGORY3", "SUB_CATEGORY4", "VERSION_NUMBER", "USR_NAME", "LIFE_CYCLE_NAME", "LIFE_CYCLE_ID", "DOCUMENT_OBJECT_TYPE", "ACTION_TYPE") AS 
  SELECT process_instance_id,
          document_name,
          document_number,
          purpose,
          category,
          sub_category1,
          sub_category2,
          sub_category3,
          sub_category4,
          version_number,
          usr_name,
          life_cycle_name,
          life_cycle_id,
          document_object_type,
          action_type
     FROM (SELECT a.life_cycle_id,
                  a.document_object_id,
                  a.document_object_type,
                  a.document_name,
                  purpose,
                  a.category,
                  a.sub_category1,
                  a.sub_category2,
                  a.sub_category3,
                  sub_category4,
                  action_type,
                  a.life_cycle_name,
                  a.process_instance_id,
                  a.document_number,
                  a.version_number,
                  ms_apps_utilities.get_user_name (USER_ID) usr_name
             FROM MS_DMS_DOCUMENTS_HDR a, ms_dms_documents_stage_usr_dtl b
            WHERE     in_process_flag = 'Y'
                  AND a.document_object_type = 'N'
                  AND a.transaction_id = b.transaction_id
                  AND current_process_status = 'Y'
           UNION
           SELECT a.life_cycle_id,
                  b.document_object_id,
                  b.document_object_type,
                  b.document_name,
                  B.purpose,
                  B.category_NAME,
                  a.sub_category1,
                  a.sub_category2,
                  a.sub_category3,
                  a.sub_category4,
                  a.action_type,
                  a.life_cycle_name,
                  a.process_instance_id,
                  a.document_number,
                  a.version_number,
                  ms_apps_utilities.get_user_name (d.user_ID) usr_name
             FROM MS_DMS_DOCUMENTS_HDR a,
                  MS_DMS_DOCUMENTS_DTL b,
                  MS_DMS_DOCUMENTS C,
                  MS_DMS_DOCUMENTS_STAGE_USR_dTL d
            WHERE     a.in_process_flag = 'Y'
                  AND a.document_object_type = 'V'
                  AND c.obsolete_flag = 'N'
                  AND b.document_object_type = 'N'
                  AND d.current_process_status = 'Y'
                  AND b.document_object_id = c.document_object_id
                  AND d.transaction_id = a.transaction_id
                  AND a.transaction_id = b.transaction_id
                  AND b.transaction_id = d.transaction_id)
