MS_DMS_DOCUMENTS_STAGE_DTL_V 

                                                
  CREATE OR REPLACE FORCE VIEW "PDMS61SP2_DEV"."MS_DMS_DOCUMENTS_STAGE_DTL_V" ("APPROVAL_REQUIREMENT", "ELECTRONIC_SIGNATURE_REQ", "EMAIL_FLAG", "NEXT_STAGE_SEQ", "PREV_STAGE_SEQ", "VERSION_NUMBER_MODEL", "DOC_DEFAULT_USERS1", "MODIFY_STAGE_USERS", "STAGE_CATEGORY_USER_FLAG", "ACTUAL_APP_REQ", "DOC_DEFAULT_USERS", "DOCUMENT_OBJECT_ID", "DOCUMENT_OBJECT_TYPE", "DOCUMENT_SEQ_NUM", "STAGE_USER_REFRESH", "TRANSACTION_ID", "ENTERPRISE_ID", "PROCESS_INSTANCE_ID", "DOCUMENT_NAME", "LIFE_CYCLE_ID", "LIFE_CYCLE_NAME", "STAGE_ID", "STAGE_NAME", "STAGE_SEQ", "CURRENT_STAGE_FLAG", "LOCALE_ID") AS 
  SELECT T.APPROVAL_REQUIREMENT,
  T.ELECTRONIC_SIGNATURE_REQ,
  T.EMAIL_FLAG,
  T.NEXT_STAGE_SEQ,
  T.PREV_STAGE_SEQ,
  T.VERSION_NUMBER_MODEL,
  T.DOC_DEFAULT_USERS1,
  T.MODIFY_STAGE_USERS,
  T.STAGE_CATEGORY_USER_FLAG,
  T.ACTUAL_APP_REQ,
  T.DOC_DEFAULT_USERS,
  T.DOCUMENT_OBJECT_ID,
  T.DOCUMENT_OBJECT_TYPE,
  T.DOCUMENT_SEQ_NUM,
  T.STAGE_USER_REFRESH,
  T.TRANSACTION_ID,
  T.ENTERPRISE_ID,
  T.PROCESS_INSTANCE_ID,
  T.DOCUMENT_NAME,
  T.LIFE_CYCLE_ID,
  NVL(TL.LIFE_CYCLE_NAME,T.LIFE_CYCLE_NAME),
  T.STAGE_ID,
  NVL(TL.STAGE_NAME, T.STAGE_NAME),
  T.STAGE_SEQ,
  T.CURRENT_STAGE_FLAG,
  nvl(TL.LOCALE_ID,T.LOCALE_ID)
FROM (SELECT S.APPROVAL_REQUIREMENT,
  S.ELECTRONIC_SIGNATURE_REQ,
  S.EMAIL_FLAG,
  S.NEXT_STAGE_SEQ,
  S.PREV_STAGE_SEQ,
  S.VERSION_NUMBER_MODEL,
  S.DOC_DEFAULT_USERS1,
  S.MODIFY_STAGE_USERS,
  S.STAGE_CATEGORY_USER_FLAG,
  S.ACTUAL_APP_REQ,
  S.DOC_DEFAULT_USERS,
  S.DOCUMENT_OBJECT_ID,
  S.DOCUMENT_OBJECT_TYPE,
  S.DOCUMENT_SEQ_NUM,
  S.STAGE_USER_REFRESH,
  S.TRANSACTION_ID,
  S.ENTERPRISE_ID,
  S.PROCESS_INSTANCE_ID,
  S.DOCUMENT_NAME,
  S.LIFE_CYCLE_ID,
  S.LIFE_CYCLE_NAME,
  S.STAGE_ID,
  S.STAGE_NAME,
  S.STAGE_SEQ,
  S.CURRENT_STAGE_FLAG,
  V.LOCALE_ID
FROM MS_DMS_DOCUMENTS_STAGE_DTL S,
  SI_ENABLED_LOCALES V WHERE V.LOCALE_ID=1009) T,MS_DMS_DOCUMENTS_STAGE_DTL_TL TL
WHERE T.LIFE_CYCLE_ID = TL.LIFE_CYCLE_ID(+) 
  AND T.STAGE_ID = TL.STAGE_ID(+)
  AND T.PROCESS_INSTANCE_ID = TL.PROCESS_INSTANCE_ID(+) 
  AND T.DOCUMENT_OBJECT_ID = TL.DOCUMENT_OBJECT_ID(+)
