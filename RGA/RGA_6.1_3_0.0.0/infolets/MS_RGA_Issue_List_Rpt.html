CHANNEL_NAME_GRCI,
ALERT_TITLE_GRCI,
(select L2.DISPLAYED_VALUE from MS_QS_LOV_DISPLAY_VALUES_V l1, MS_QS_LOV_DISPLAY_VALUES_V l2 where l1.lov_name=l2.lov_name and L1.STORED_VALUE = L2.STORED_VALUE and L1.LOV_NAME='MS_RGA_Issue_Status' and L1.DISPLAYED_VALUE = ISSUE_STATUS and L2.LOCALE_ID = LOCALES.LOCALE_ID) as ISSUE_STATUS,
ISSUE_OWNER_GRCI,
ms_apps_utilities.get_display_value(users.enterprise_id, 'MS RGA List of Issue Rating',ISSUE_RATING_GRCI,1,LOCALES.LOCALE_ID) AS ISSUE_RATING_GRCI,

ms_apps_utilities.get_display_value(users.enterprise_id, 'MS RGA List of Issue Types', ISSUE_TYPE_GRCI,1,LOCALES.LOCALE_ID) AS ISSUE_TYPE_GRCI,
ISSUE_DETAILS_GRCI,
RECOMMENDATION_GRCI,
OWNER_ORGANIZATION,
APPROVER_ORGANIZATION,
ISSUE_APPROVER_GRCI,
''''    || ATTACHMENTS    || ''', 10' AS  ATTACHMENTS,
BODY,
Ms_Rga_Utilities.GET_NOTIFIED_USERS(T.ISSUE_CREATED_BY,T.FEED_DATA_ID) ACTIONS,
ATTACHMENT,
ISSUE_DUE_DATE_GRCI  as ISSUE_DUE_DATE
 FROM MS_RGA_ISSUE_REPORT T
,SI_USERS_T USERS, SI_LOCALES LOCALES
WHERE  T.ISSUE_CREATED_BY = :1
AND T.ISSUE_CREATED_BY=USERS.USER_NAME
AND USERS.LOCALE=LOCALES.LOCALE_STRING
