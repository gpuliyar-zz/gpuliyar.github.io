                                                        SELECT cnlmaster.channel_id as channel_id,
       (case
         when length(cnlmaster.channel_name) > 100 then
          substr(cnlmaster.channel_name, 1, 100) || '...'
         when length(cnlmaster.channel_name) <= 100 then
          cnlmaster.channel_name
       end ) as channel_name,
       (SELECT ms_apps_utilities.get_display_value(users.enterprise_id,
                                                   'MS RGA Channel Type',
                                                   cnlmaster.channel_type,1,LOCALES.LOCALE_ID)
          FROM DUAL) AS channel_type_disp,
       details.source_id as source_id,
       (SELECT ms_apps_utilities.get_display_value(users.enterprise_id,
                                                   'MS RGA Channel Status',
                                                   details.active,1,LOCALES.LOCALE_ID)
          FROM DUAL) AS source_status,
       (SELECT ms_apps_utilities.get_display_value(users.enterprise_id,
                                                   'MS_RGA_Source_Type',
                                                   details.source_type,1,LOCALES.LOCALE_ID)
          FROM DUAL) AS source_type_disp,
       DECODE(cnlmaster.channel_type, '1', 'N/A', cnlmaster.infolet_name) AS infolet_name,
       ms_rga_utilities.fn_noty_user(msrcc.subscription_id, user_id) AS notify_users,
       (SELECT ms_apps_utilities.get_display_value(users.enterprise_id,
                                                   'MS RGA Channel Status',
                                                   cnlmaster.active,1,LOCALES.LOCALE_ID)
          FROM DUAL) AS channel_active_disp,
       ms_apps_utilities.get_edit_url(ms_apps_utilities.get_infolet_id('MS_RGA_CHANNEL_DTLS'),
                                      cnlmaster.pid) AS edit_url,
       cnlmaster.channel_description as channel_description,
       --details.server_add_url as server_add_url,
       ms_rga_utilities.get_rss_url(details.server_add_url) as server_add_url,
       details.scr_user_name as scr_user_name,
       details.email_from as email_from,
       details.keywords as keywords,
       details.source_description as source_description,
       details.server_parameters as server_parameters,
       user_id,
       pid,
       (SELECT ms_rga_utilities.get_user_name(user_id) FROM DUAL) AS user_name,
       ms_rga_utilities.fn_subscription_status(MSRCC.Status, user_id)  AS cust_col2,
       NULL AS cust_col3,
       NULL AS cust_col4,
       NULL AS cust_col5
  FROM ms_rga_channel_master   cnlmaster,
       ms_rga_channel_details  details,
       si_users_t              users,
       ms_rga_subs_channel_chn msrcc
,
       SI_LOCALES  locales
 WHERE cnlmaster.channel_id = details.channel_id
   AND cnlmaster.Src_Pk=details.source_id
   AND cnlmaster.active = details.active
    AND cnlmaster.channel_id IN
       (Select a.Channel_Name
          From ms_rga_subs_channel_chn a
         where a.CHN_CREATED_BY = :1)
   AND cnlmaster.channel_id = msrcc.channel_name
   and details.channel_id = msrcc.channel_name
   and users.user_name = msrcc.chn_created_by
   and users.user_name = :1
  and USERS.LOCALE = LOCALES.LOCALE_STRING
 order by SOURCE_ID desc
