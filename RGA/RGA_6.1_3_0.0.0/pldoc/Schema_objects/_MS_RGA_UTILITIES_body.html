IF (i_hyper_text IS NULL) THEN x_url := '<a target="_self" href="javascript:callreportdetails(''Pushinfolet?id=' || i_metric_id || CHR(38) || 'proc=' || i_process_instance_id || CHR(38) || 'flag=1' || CHR(38) || 'emd=1' || CHR(38) || 'wrapper=no'')"><img border="0" src="/si_image/new_pencil.gif"></a>'; ELSE x_url := '<a target="_self" href="javascript:callreportdetails(''Pushinfolet?id=' || i_metric_id || CHR(38) || 'proc=' || i_process_instance_id || CHR(38) || 'flag=1' || CHR(38) || 'emd=1' || CHR(38) || 'wrapper=no'')">' || HTF.escape_sc(i_hyper_text) || '</a>'; END IF; RETURN x_url; EXCEPTION WHEN OTHERS THEN RETURN NULL; END get_edit_url;
