MS_RGA_MANAGE_CHANNEL_V 

                                                
  CREATE OR REPLACE FORCE VIEW "GRCIDEV"."MS_RGA_MANAGE_CHANNEL_V" ("CHANNEL_ID", "CHANNEL_NAME", "CHANNEL_TYPE_DISP", "SOURCE_ID", "SOURCE_STATUS", "SOURCE_TYPE_DISP", "INFOLET_NAME", "NOTIFY_USERS", "CHANNEL_ACTIVE_DISP", "EDIT_URL", "CHANNEL_DESCRIPTION", "SERVER_ADD_URL", "SCR_USER_NAME", "EMAIL_FROM", "KEYWORDS", "SOURCE_DESCRIPTION", "SERVER_PARAMETERS", "CUST_COL1", "CUST_COL2", "CUST_COL3", "CUST_COL4", "CUST_COL5") AS 
  SELECT Cnlmaster.Channel_Id,
    (
    CASE
      WHEN LENGTH(channel_name) > 100
      THEN SUBSTR(channel_name, 1, 100)
        || '...'
      WHEN LENGTH(channel_name) <= 100
      THEN channel_name
    END ) channel_name,
    /*   (SELECT Ms_Apps_Utilities.Get_Display_Value(Users.Enterprise_Id, 'MS RGA Channel Type', Cnlmaster.Channel_Type)
    FROM Dual
    ) AS Channel_Type_Disp,*/
    Cnlmaster.Channel_Type AS Channel_Type_Disp,
    Details.Source_Id,
  (SELECT Ms_Apps_Utilities.Get_Display_Value(Users.Enterprise_Id, 'MS RGA Channel Status', Details.Active)
    FROM Dual
    ) AS Source_Status,
    Details.Source_Type AS Source_Type_Disp,
    DECODE(Cnlmaster.Channel_Type, '1', 'N/A', Cnlmaster.Infolet_Name) AS Infolet_Name,
    DECODE(Cnlmaster.Channel_Type, '1', 'N/A',
    (SELECT Ms_Concat(Ms_Apps_Utilities.Get_User_Full_Name(a.User_Id))
    FROM Ms_Rga_Channel_Notify_Users a
    WHERE Channel_Id = Cnlmaster.Channel_Id
    )) AS Notify_Users,
   Cnlmaster.Active AS Channel_Active_Disp,
    Ms_Apps_Utilities.Get_Edit_Url(Ms_Apps_Utilities.Get_Infolet_Id('MS_RGA_CHANNEL_DTLS'), Cnlmaster.Pid) Edit_Url,
    Cnlmaster.Channel_Description,
    Details.Server_Add_Url,
    Details.Scr_User_Name,
    Details.Email_From,
    Details.Keywords,
    Details.Source_Description,
    Details.Server_Parameters,
    NULL AS Cust_Col1,
    NULL AS Cust_Col2,
    NULL AS Cust_Col3,
    pid  AS Cust_Col4,
    NULL AS Cust_Col5
  FROM Ms_Rga_Channel_Master Cnlmaster,
    Ms_Rga_Channel_Details Details,
    Si_Users_t Users
  WHERE Cnlmaster.Channel_Id = Details.Channel_Id
  AND Cnlmaster.Created_By   = Users.User_Name
  AND Cnlmaster.Active       = Details.Active
  AND Cnlmaster.src_pk=Details.source_id
  ORDER BY Cnlmaster.channel_id DESC ,
    Details.source_id DESC
