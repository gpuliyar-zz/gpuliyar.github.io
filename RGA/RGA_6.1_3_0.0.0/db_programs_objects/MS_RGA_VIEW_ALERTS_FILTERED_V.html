MS_RGA_VIEW_ALERTS_FILTERED_V 

                                                
  CREATE OR REPLACE FORCE VIEW "GRCIDEV"."MS_RGA_VIEW_ALERTS_FILTERED_V" ("RESPONSE_ID", "CHANNEL_ID", "CHANNEL_NAME", "RESPONSE_DATE", "SOURCE_ID", "TITLE_SUBJECT", "RESPONSE_BODY", "ATTACHMENTS", "ACTIVE", "PROCESSED", "CHANNEL_RESPONSE_ID", "PID", "BODY", "FEED_DATA_ID", "PROCESSED_FLAG", "ATTACHMENT", "EMAIL_BODY", "RECIPIENT", "SENDER", "SERVER", "CREATION_DATE", "REVIEW_UNCHECKED", "FOLLOW_UNCHECKED", "REVIEW_CHECKED", "FOLLOW_CHECKED") AS 
  select 
		Ch_Resp_Dtls.Response_Id,
		Ch_Resp_Dtls.Channel_Id,
		--MS_RGA_UTILITIES.GET_CHANNEL_NAME(Ch_Resp_Dtls.CHANNEL_ID) AS CHANNEL_NAME,
    chn_dtl.channel_name AS CHANNEL_NAME,
		Ch_Resp_Dtls.Response_Date as RESPONSE_DATE,
		Ch_Resp_Dtls.Source_Id,
		Ch_Resp_Dtls.Title_Subject,
		--Ch_Resp_Dtls.Response_Body,
    null Response_Body,
		Ch_Resp_Dtls.Attachments,
		Ch_Resp_Dtls.Active,
		CH_RESP_DTLS.PROCESSED,
		Ch_Resp_Dtls.response_id channel_response_id,
		Ch_Resp_Dtls.FEED_DATA_ID PID,
		FEEDS.body ,
		Feeds.Feed_Data_Id,
		Feeds.Processed_Flag,
		 ''''    || Feeds.Attachment    || ''', 10' AS Attachment,
    --null Attachment,
		Feeds.Body email_body,
		Feeds.Recipient,
		Feeds.Sender,
		FEEDS.SERVER,
		FEEDS.CREATION_DATE,
    '' REVIEW_UNCHECKED,
    
    '' FOLLOW_UNCHECKED,
               
    '' REVIEW_CHECKED,
               
    '' FOLLOW_CHECKED
	From 
     Ms_Rga_Channel_Response_Dtls Ch_Resp_Dtls,
     MS_RGA_FEED_AGGREGATOR_V FEEDS,
     MS_RGA_CHANNEL_DTLS chn_dtl
        where 
        Ch_Resp_Dtls.feed_data_id       =    FEEDS.FEED_DATA_ID  
       and FEED_SENT_DATE >= trunc(sysdate) - ms_rga_utilities.get_parameter_value(1,'MS_RGA_Configuration','RGA_Day_Alerts_Filter')
       and Ch_Resp_Dtls.PROCESSED = 'P'
       and chn_dtl.object_id=Ch_Resp_Dtls.CHANNEL_ID
