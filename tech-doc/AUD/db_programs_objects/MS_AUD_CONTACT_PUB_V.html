MS_AUD_CONTACT_PUB_V 

  CREATE OR REPLACE FORCE VIEW "MS_AUD_CONTACT_PUB_V" ("AUDIT_ID", "CONTACT_ID", "USER_IN_SYSTEM_STO", "USER_IN_SYSTEM_DISP", "USER_NAME", "USER_FULL_NAME", "SEND_OR_NOTIFY_STO", "SEND_OR_NOTIFY_DISP", "TITLE", "AUDITEE_CONTACT_STO", "AUDITEE_CONTACT_DISP", "COMMENTS", "NAME", "EMAIL", "PHONE") AS 
  SELECT a.audit_id Audit_ID, b.co_contact_id Contact_ID,
          b.co_user_in_system User_In_System_Sto,
          ms_Apps_utilities.get_display_Value(100000, 'MS AUD Yes No' , b.co_user_in_system, 1, NULL, NULL)
                                                     User_In_System_Disp,
          b.hdn_mng_select_user User_Name,
          ms_apps_utilities.get_user_full_name (b.co_user)
                                                          User_Full_Name,
          ms_aud_utilities.mlov_values
                          (b.co_contact_id,
                           'co_contact_id',
                           'co_notification_options',
                           'ms_aud_audit_san'
                          ) Send_or_Notify_Sto,
          ms_Apps_utilities.get_display_Value(100000, 'MS AUD Notify Send' , b.co_contact_id, 1, NULL, NULL)
                                                     Send_or_Notify_Disp,
          co_title Title, b.co_auditee_contact Auditee_Contact_Sto,
           ms_Apps_utilities.get_display_Value(100000, 'MS AUD Yes No' , b.co_auditee_contact, 1, NULL, NULL)
                                                    Auditee_Contact_Disp,
          b.co_comments Comments, b.co_name Name,
          b.co_email Email, co_phone Phone
     FROM ms_aud_audit a, ms_aud_audit_con b
    WHERE a.audit_id = b.audit_id
