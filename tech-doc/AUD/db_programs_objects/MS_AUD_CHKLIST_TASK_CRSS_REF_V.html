MS_AUD_CHKLIST_TASK_CRSS_REF_V 

  CREATE OR REPLACE FORCE VIEW "MS_AUD_CHKLIST_TASK_CRSS_REF_V" ("TASK_ID", "AUDIT_TITLE_GROUP", "TASK_NAME", "TASK_STATUS", "AUDIT_STATUS", "CHECKLIST", "CHECKLIST_TYPE", "CHECKLIST_SCORE", "TASK_OWNER", "APPROVER", "AUDIT_ID", "AUDIT_TITLE_URL", "AUDIT_TITLE_DISP", "TASK_NAME_DISP", "DD_OBJECT_TYPE", "RESTRICT_ACCESS_TO", "CONDUCTED_BY_ORG", "AUDIT_TITLE_GROUP_JSON", "TASK_NAME_JSON", "CHECKLIST_JSON", "AUDIT_TITLE_URL_JSON") AS 
  select  a.task_id as task_id,
  ms_aud_execution_rpt_utils.fn_concat_title_status(b.Audit_title,b.audit_status,b.audit_id,b.dd_process_instance_id,'CT') AS Audit_title_group,
  a.task_name  As Task_Name,
nvl (ms_aud_utilities.fn_get_status_display_name (a.status), 'None' ) as task_status, ms_aud_utilities.fn_get_status_display_name(b.audit_status)  as audit_status,
c.checklist_name as checklist,
(SELECT displayed_value
          FROM ms_qs_list_of_values lovvalue
                ,ms_qs_lov_names lovname, si_metrics_t m
          WHERE lovvalue.lov_id = lovname.lov_id
          AND lovvalue.org_entity_id = 1--i_org_entity_id
          AND lovvalue.enterprise_id = m.enterprise_id
          AND lovvalue.stored_value = c.checklist_type
      and m.metric_name = b.dd_object_type
          AND lovname.lov_name = 'MS CHK Checklist Type'
          AND lovvalue.locale_id = nvl(cast(si_db_locale_sv.getvalue as number),1009)
      AND (lovname.end_date IS NULL OR lovname.end_date > SYSDATE)
          AND lovname.start_date <= SYSDATE
      AND (lovvalue.end_date IS NULL OR lovvalue.end_date > SYSDATE)) as checklist_type,
C.Score As Checklist_Score,
Ms_Apps_Utilities.Get_User_Full_Name(A.Task_Owner) As Task_owner,
ms_apps_utilities.get_user_full_name (A.approver) AS approver,
b.audit_id,
ms_aud_execution_rpt_utils.fn_concat_title_status(b.Audit_title,b.audit_status,b.audit_id,b.dd_process_instance_id,' ') AS audit_title_url,
b.audit_title AS audit_title_disp,
a.task_name as task_name_disp,b.dd_object_type as dd_object_type,b.restrict_access_to as restrict_access_to,b.conducted_by_org as conducted_by_org,
ms_aud_execution_rpt_utils.fn_concat_title_status_rpt_jsn(b.Audit_title,b.audit_status,b.audit_id,b.dd_process_instance_id,'CT') AS Audit_title_group_Json,
ms_aud_execution_rpt_utils.fn_get_task_url_cross_ref(a.dd_process_instance_id,a.task_type,a.audit_id,a.task_id,a.task_name)  As Task_Name_Json,
ms_aud_execution_rpt_utils.get_checklist_url(c.checklist_id,c.checklist_name) as checklist_Json,
ms_aud_execution_rpt_utils.fn_concat_title_status_rpt_jsn(b.Audit_title,b.audit_status,b.audit_id,b.dd_process_instance_id,' ') AS audit_title_url_Json
FROM ms_aud_audit_task A, ms_aud_audit b , ms_aud_checklist_task c WHERE A.audit_id = b.audit_id  AND A.task_id = c.task_id
and a.task_type in ('3')
AND ((A.status IN ('11','18','2','15','9','49','50','51','53','54') AND ms_aud_utilities.get_parameter_value('Reference In-progress Tasks')='Y')
     or (A.status IN ('15','53') AND ms_aud_utilities.get_parameter_value('Reference In-progress Tasks')='N'))
AND b.audit_status<>'4'
