MS_AUD_AUDIT_LIST_REPORT_V 

  CREATE OR REPLACE FORCE VIEW "MS_AUD_AUDIT_LIST_REPORT_V" ("AUDIT_ID", "AUDIT_TITLE", "AUDIT_TITLE_LINK", "PLAN_TITLE", "AUDIT_TYPE", "AUDIT_CATEGORY", "AUDIT_STATUS", "CORE_OBJECTS", "ORGANIZATIONS", "RELATED_RISKS", "CREATED_BY", "AUDIT_START_DATE", "AUDIT_END_DATE", "VIEW_TASKS", "RESTRICT_ACCESS_TO", "TEAM_MEMBER", "APPROVE_FLAG", "STATUS", "CONDUCTED_BY_ORG", "DD_OBJECT_TYPE", "AUDIT_MANAGER", "LEAD_AUDITOR", "AUDIT_CLASSIFICATION", "AUDIT_CLOSURE_DATE", "DD_PROCESS_INSTANCE_ID", "DD_INSTANCE_ID") AS 
  SELECT a.audit_id AS audit_id,
    a.audit_title   AS audit_title,
    a.audit_title AS Audit_Title_Link,
    COALESCE(
    CASE
      WHEN a.plan_title='AD-ID'
      THEN 'Adhoc'
      ELSE
        (SELECT plan_title
        FROM ms_aud_audit_plan
        WHERE internal_plan_id = a.plan_title
        AND rownum             =1
        )
    END,'Adhoc')AS Plan_Title,
    (SELECT code_description
    FROM ms_aud_lov_config_tl_v
    WHERE locale_id         =NVL(CAST(si_db_locale_sv.getvalue AS NUMBER),1009)
    AND code_group          = 'MS AUD Audit Type'
    AND code_code           = a.audit_type
    AND COALESCE(status,'Y')='Y'
    ) AS Audit_Type,
    (SELECT code_description
    FROM ms_aud_lov_config_tl_v
    WHERE locale_id         =NVL(CAST(si_db_locale_sv.getvalue AS NUMBER),1009)
    AND code_group          = 'MS AUD Classification'
    AND code_code           = a.audit_classification
    AND COALESCE(status,'Y')='Y'
    )AS Audit_Category,
    (SELECT status_name
    FROM ms_aud_status_tl_v
    WHERE locale_id=NVL(CAST(si_db_locale_sv.getvalue AS NUMBER),1009)
    AND status_id  =a.audit_status
    ) AS Audit_Status ,
    ms_aud_report_utils.fn_get_audit_core_objects(a.audit_id,a.SCOPE_BASED_ON) AS core_objects ,
    ms_Aud_report_utils.fn_get_audit_organizations(a.audit_id,a.SCOPE_BASED_ON) AS organizations,
    ms_Aud_report_utils.fn_get_audit_related_risks(a.audit_id,a.SCOPE_BASED_ON) AS related_risks,
    (SELECT full_name FROM SI_USERS_T WHERE user_name = a.dd_created_by
    )                  AS Created_By,
    a.audit_start_date AS Audit_Start_Date,
    a.audit_end_date   AS audit_end_date,
    (SELECT to_number(DECODE(COUNT(t.task_id),0,NULL,COUNT(t.task_id)))
    FROM ms_aud_audit_task t
    WHERE t.audit_id = a.audit_id
    AND t.TASK_TYPE IN ('2','3','4')
    )                        AS view_tasks,
    a.restrict_access_to     AS restrict_access_to,
    NULL                     AS Team_Member,
    a.approve_flag           AS approve_flag,
    a.status                 AS status,
    a.conducted_by_org       AS conducted_by_org,
    a.dd_object_type         AS dd_object_type,
    a.audit_manager          AS audit_manager,
    a.lead_auditor           AS lead_auditor,
    a.audit_classification   AS audit_classification,
    a.audit_closure_date     AS audit_closure_date,
    a.dd_process_instance_id AS dd_process_instance_id,
    a.dd_instance_id         as dd_instance_id
  FROM ms_aud_audit a
