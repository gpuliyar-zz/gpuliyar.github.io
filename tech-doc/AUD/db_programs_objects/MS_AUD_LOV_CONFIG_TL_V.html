          t.last_update_date, t.code_group, t.code_code,
          NVL (tl.code_description, t.code_description), t.code_link,
          t.status, t.VALUE, t.data_type, t.code_order, t.locale_id
     FROM (SELECT a.created_by, a.creation_date, a.last_updated_by,
                  a.last_update_date, a.code_group, a.code_code,
                  a.code_description, a.code_link, a.status, a.VALUE,
                  a.data_type, a.code_order, s.locale_id
             FROM ms_aud_lov_config a, si_enabled_locales s) t,
          ms_aud_lov_config_tl tl
    WHERE t.code_code = tl.code_code(+) AND t.locale_id = tl.locale_id(+)
