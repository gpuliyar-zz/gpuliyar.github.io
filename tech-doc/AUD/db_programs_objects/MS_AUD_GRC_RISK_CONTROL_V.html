MS_AUD_GRC_RISK_CONTROL_V 

  CREATE OR REPLACE FORCE VIEW "MS_AUD_GRC_RISK_CONTROL_V" ("CONTROL_ID", "CONTROL_NAME", "CONTROL_DESCRIPTION", "CONTROL_DESCRIPTION_HTM", "RISK_ID", "RISK_NAME", "VALID_FROM", "VALID_UNTIL", "DD_PROCESS_INSTANCE_ID", "DD_INSTANCE_ID", "SEQ_NO", "KEY_COMPLIANCE") AS 
  SELECT "CONTROL_ID","CONTROL_NAME","CONTROL_DESCRIPTION","CONTROL_DESCRIPTION_HTM","RISK_ID","RISK_NAME","VALID_FROM","VALID_UNTIL","DD_PROCESS_INSTANCE_ID","DD_INSTANCE_ID","SEQ_NO","KEY_COMPLIANCE" FROM
(SELECT  con.object_id AS control_id, con.object_name AS control_name,
          con.description AS control_description,
          con.description_htm AS control_description_htm,
          ris.object_id AS risk_id, ris.object_name AS risk_name,
          con.valid_from as valid_from, con.valid_until as valid_until,
          con.DD_PROCESS_INSTANCE_ID as DD_PROCESS_INSTANCE_ID,
          con.DD_INSTANCE_ID as DD_INSTANCE_ID,
          row_number() over (partition by con.object_id , con.object_name ,
          ris.object_id, ris.object_name,
          con.valid_from, con.valid_until,
          con.DD_PROCESS_INSTANCE_ID,
          con.DD_INSTANCE_ID order by con.object_id) as seq_no,
          con.key_compliance as key_compliance
     FROM ms_grc_control_pub_v con,
          ms_grc_risk_pub_v ris,
          ms_grc_orb_relations_pub_v orb
    WHERE UPPER (src_obj_type) = 'MS_GRC_RISK'
      AND UPPER (tgt_obj_type) = 'MS_GRC_CONTROL'
      AND orb.tgt_obj_id = con.object_id
      AND orb.src_obj_id = ris.object_id) where seq_no=1
