MS_AUD_TASKS_EDS_V 

  CREATE OR REPLACE FORCE VIEW "MS_AUD_TASKS_EDS_V" ("ID", "TASK_ID", "TASK_NAME", "START_DATE", "STATUS", "OWNER", "REVIEWER", "APPROVER", "BUDGETED_DAYS", "INSTRUCTIONS", "EXECUTION_RESULT", "TIME_SPENT_DAYS", "WORK_DONE", "RESULTS", "COMPLETION_DATE", "DOCUMENT_ID", "DOCUMENT_NAME", "DOCUMENT_TYPE", "DOCUMENT_CATEGORY", "DOCUMENT_AUDITABLE_ENTITIES", "DOCUMENT_RELATED_RISKS", "DOCUMENT_COMMENTS", "FINDING_ID", "FINDING_TITLE", "FINDING_DETAILS", "ISSUE_RATING", "FINDING_TYPE", "ISSUE_RESOLUTION", "OWNER_ORG", "ISSUE_OWNER", "ISSUE_DUE_DATE", "ISSUE_APPROVER_ORG", "ISSUE_APPROVER", "TASK_AUDITABLE_ENTITIES", "RELATED_RISKS", "RELATED_ORGANIZATIONS", "ISSUE_RECOMMENDATION", "PRIOR_FINDING", "ROOT_CAUSE", "CONTRIBUTING_FACTORS", "TASK_ACTION_ID", "TASK_ACTION_TITLE", "ACTION_DESCRIPTION", "PRIORITY", "ACTION_DUE_BY", "ACTION_DUE_DATE", "OWNER_ORGANIZATION", "ACTION_OWNER", "CREATED_BY", "CREATED_ON", "MODIFIED_BY", "MODIFIED_ON") AS 
  SELECT /*@public*/
--Task Details
A.AUDIT_ID AS ID,
A.TASK_ID AS TASK_ID,
A.TASK_NAME AS TASK_NAME,
A.TASK_START_DATE AS START_DATE,
(SELECT Z.STATUS_NAME FROM MS_AUD_STATUS_TL_V Z WHERE Z.STATUS_ID=A.STATUS) AS STATUS,
(SELECT Z.USER_FULL_NAME FROM MS_GRC_USER_FULL_NAME_V Z WHERE Z.USER_NAME = A.TASK_OWNER) AS OWNER,
(SELECT Z.USER_FULL_NAME FROM MS_GRC_USER_FULL_NAME_V Z WHERE Z.USER_NAME = A.TASK_REVIEWER) AS REVIEWER,
(SELECT Z.USER_FULL_NAME FROM MS_GRC_USER_FULL_NAME_V Z WHERE Z.USER_NAME = A.APPROVER) AS APPROVER,
A.BUDGETED_HOURS AS BUDGETED_DAYS,
--TO BE FILLED WITH CHECKLIST & OTHER DETAILS
TO_CHAR(Regexp_Replace(A.INSTRUCTIONS,'<[^>]+>',NULL)) AS INSTRUCTIONS,
(SELECT Z.DISPLAYED_VALUE FROM MS_QS_LOV_ORG_VALUES_V Z WHERE LOV_NAME = 'MS AUD Task Execution Result' AND Z.STORED_VALUE=A.TASK_EXECUTION_RESULT
AND z.locale_id=ms_grc_common_utils.fn_get_default_locale_id) AS EXECUTION_RESULT,
A.TIME_SPENT AS TIME_SPENT_DAYS,
TO_CHAR(Regexp_Replace(A.WORK_DONE,'<[^>]+>',NULL)) AS WORK_DONE,
TO_CHAR(Regexp_Replace(A.RESULTS,'<[^>]+>',NULL)) AS RESULTS,
A.TASK_COMPLETION_DATE AS COMPLETION_DATE,

--Document Details
K.DO_ID AS DOCUMENT_ID,
K.DO_NAME AS DOCUMENT_NAME,
(SELECT Z.DISPLAYED_VALUE FROM MS_QS_LOV_ORG_VALUES_V Z WHERE LOV_NAME = 'MS AUD Document Types' AND Z.STORED_VALUE=K.DO_TYPE
AND z.locale_id=ms_grc_common_utils.fn_get_default_locale_id)  AS DOCUMENT_TYPE,
(SELECT Z.DISPLAYED_VALUE FROM MS_QS_LOV_ORG_VALUES_V Z WHERE LOV_NAME = 'MS AUD Document Categories' AND Z.STORED_VALUE=K.DO_CATEGORY
AND z.locale_id=ms_grc_common_utils.fn_get_default_locale_id) AS DOCUMENT_CATEGORY,
D.DO_AUDITABLE_ENTITIES AS DOCUMENT_AUDITABLE_ENTITIES,
L.DO_RELATED_RISKS AS DOCUMENT_RELATED_RISKS,
K.DO_COMMENTS AS DOCUMENT_COMMENTS,

--Finding Details
M.FN_ID AS FINDING_ID,
M.FN_FINDING_TITLE AS FINDING_TITLE,
TO_CHAR(Regexp_Replace(M.FN_FINDING_DETAILS,'<[^>]+>',NULL)) AS FINDING_DETAILS,
(SELECT Z.DISPLAYED_VALUE FROM MS_QS_LOV_ORG_VALUES_V Z WHERE LOV_NAME = 'MS AUD Audit Task Issue Rating' AND Z.STORED_VALUE=M.FN_ISSUE_RATING
AND z.locale_id=ms_grc_common_utils.fn_get_default_locale_id) AS ISSUE_RATING,
(SELECT Z.DISPLAYED_VALUE FROM MS_QS_LOV_ORG_VALUES_V Z WHERE LOV_NAME = 'MS ISM Issue Type' AND Z.STORED_VALUE=M.FN_FINDING_TYPE
AND z.locale_id=ms_grc_common_utils.fn_get_default_locale_id) AS FINDING_TYPE,
--M.FN_DISPOSITION AS ISSUE_DISPOSITION,
(SELECT Z.DISPLAYED_VALUE FROM MS_QS_LOV_ORG_VALUES_V Z WHERE LOV_NAME = 'MS AUD Audit Task Issue Resol' AND Z.STORED_VALUE=M.FN_ISSUE_RESOLUTION
AND z.locale_id=ms_grc_common_utils.fn_get_default_locale_id) AS ISSUE_RESOLUTION,

(SELECT Z.ORG_ENTITY_NAME FROM SI_ORG_ENTITIES_T Z WHERE Z.ORG_ENTITY_ID=M.FN_OWNER_ORGANIZATION) AS OWNER_ORG,
(SELECT Z.USER_FULL_NAME FROM MS_GRC_USER_FULL_NAME_V Z WHERE Z.USER_NAME = M.FN_ISSUE_OWNER) AS ISSUE_OWNER,
M.FN_ISSUE_DUE_DATE AS ISSUE_DUE_DATE,
(SELECT Z.ORG_ENTITY_NAME FROM SI_ORG_ENTITIES_T Z WHERE Z.ORG_ENTITY_ID=M.FN_APPROVER_ORG) AS ISSUE_APPROVER_ORG,
(SELECT Z.USER_FULL_NAME FROM MS_GRC_USER_FULL_NAME_V Z WHERE Z.USER_NAME = M.FN_ISSUE_APPROVER) AS ISSUE_APPROVER,
(SELECT Z.OBJECT_NAME FROM MS_GRC_CORE_OBJECT Z WHERE Z.OBJECT_ID=F.FN_CORE_OBJECTS) AS TASK_AUDITABLE_ENTITIES,
(SELECT Z.OBJECT_NAME FROM MS_GRC_RISK Z WHERE Z.OBJECT_ID=T.FN_RELATED_RISKS) AS RELATED_RISKS,
P.FN_ORGANIZATIONS AS RELATED_ORGANIZATIONS,
TO_CHAR(Regexp_Replace(M.FN_RECOMMENDATION,'<[^>]+>',NULL)) AS ISSUE_RECOMMENDATION,
M.PRIOR_FINDING AS PRIOR_FINDING,
(SELECT Z.DISPLAYED_VALUE FROM MS_QS_LOV_ORG_VALUES_V Z WHERE LOV_NAME = 'MS AUD Root Cause' AND Z.STORED_VALUE=U.FN_ROOT_CAUSE
AND z.locale_id=ms_grc_common_utils.fn_get_default_locale_id) AS ROOT_CAUSE,
(SELECT Z.DISPLAYED_VALUE FROM MS_QS_LOV_ORG_VALUES_V Z WHERE LOV_NAME = 'MS AUD Contributing Factors' AND Z.STORED_VALUE=I.FN_CONTRIBUTING_FACTORS
AND z.locale_id=ms_grc_common_utils.fn_get_default_locale_id) AS CONTRIBUTING_FACTORS,
C.AC_ID AS TASK_ACTION_ID,
C.AC_ACTION_TITLE AS TASK_ACTION_TITLE,
TO_CHAR(Regexp_Replace(C.AC_ACTION_DESCRIPTION,'<[^>]+>',NULL)) AS ACTION_DESCRIPTION,
(SELECT Z.DISPLAYED_VALUE FROM MS_QS_LOV_ORG_VALUES_V Z WHERE LOV_NAME = 'MS ISM Priority' AND Z.STORED_VALUE=C.AC_PRIORITY
AND z.locale_id=ms_grc_common_utils.fn_get_default_locale_id) AS PRIORITY,
C.AC_DUE_BY AS ACTION_DUE_BY,
C.AC_DUE_DATE AS ACTION_DUE_DATE,
(SELECT Z.ORG_ENTITY_NAME FROM SI_ORG_ENTITIES_T Z WHERE Z.ORG_ENTITY_ID=C.AC_OWNER_ORGANIZATION) AS OWNER_ORGANIZATION,
(SELECT Z.USER_FULL_NAME FROM MS_GRC_USER_FULL_NAME_V Z WHERE Z.USER_NAME = C.AC_ACTION_OWNER) AS ACTION_OWNER,
A.DD_CREATED_BY AS CREATED_BY,
A.DD_CREATED_ON AS CREATED_ON,
A.DD_MODIFIED_BY AS MODIFIED_BY,
A.DD_MODIFIED_ON AS MODIFIED_ON

FROM MS_AUD_AUDIT_TASK A
LEFT JOIN MS_AUD_AUDIT_TASK_FND M ON 
A.AUDIT_ID=M.AUDIT_ID AND
A.TASK_ID=M.TASK_ID
LEFT JOIN MS_AUD_AUDIT_TASK_AUE F ON 
A.AUDIT_ID=F.AUDIT_ID AND
A.TASK_ID=F.TASK_ID AND
F.FN_ID=M.FN_ID
LEFT JOIN MS_AUD_AUDIT_TASK_COF I ON 
A.AUDIT_ID=I.AUDIT_ID AND
A.TASK_ID=I.TASK_ID  AND
M.FN_ID=I.FN_ID
LEFT JOIN MS_AUD_AUDIT_TASK_FOO N ON 
A.AUDIT_ID=N.AUDIT_ID AND
A.TASK_ID=N.TASK_ID  AND
M.FN_ID=N.FN_ID
LEFT JOIN MS_AUD_AUDIT_TASK_ORG P ON 
A.AUDIT_ID=P.AUDIT_ID AND
A.TASK_ID=P.TASK_ID  AND
M.FN_ID=P.FN_ID
LEFT JOIN MS_AUD_AUDIT_TASK_RSK T ON 
A.AUDIT_ID=T.AUDIT_ID AND
A.TASK_ID=T.TASK_ID  AND
M.FN_ID=T.FN_ID
LEFT JOIN MS_AUD_AUDIT_TASK_RUC U ON 
A.AUDIT_ID=U.AUDIT_ID AND
A.TASK_ID=U.TASK_ID  AND
M.FN_ID=U.FN_ID
LEFT JOIN MS_AUD_AUDIT_TASK_ACT C ON 
A.AUDIT_ID=C.AUDIT_ID AND
A.TASK_ID=C.TASK_ID AND
M.FN_ID=C.AC_FINDING_ID
LEFT JOIN MS_AUD_AUDIT_TASK_DOC K ON 
A.AUDIT_ID=K.AUDIT_ID AND
A.TASK_ID=K.TASK_ID
LEFT JOIN MS_AUD_AUDIT_TASK_ADE D ON 
A.AUDIT_ID=D.AUDIT_ID AND
A.TASK_ID=D.TASK_ID AND
D.DO_ID=K.DO_ID
LEFT JOIN MS_AUD_AUDIT_TASK_DRR L ON 
A.AUDIT_ID=L.AUDIT_ID AND
a.task_id=l.task_id  and
K.DO_ID=L.DO_ID
