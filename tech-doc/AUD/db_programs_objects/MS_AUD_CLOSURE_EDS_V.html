MS_AUD_CLOSURE_EDS_V 

  CREATE OR REPLACE FORCE VIEW "MS_AUD_CLOSURE_EDS_V" ("ID", "COMMENTS", "AUDITOR", "FOLLOW_UP_DATE", "FOLLOW_UP_STATUS", "REAUD_LEAD_AUDITOR", "REAUD_START_DATE", "ORG_CLOSE_COMMENTS", "ORGANIZATION_ID", "ORG_RISK_RATING", "AE_CLOSE_COMMENTS", "AE_ID", "AE_RISK_RATING", "RISK_CLOSE_COMMENTS", "RISK_ID", "RS_RISK_RATING", "CREATED_BY", "CREATED_ON", "MODIFIED_BY", "MODIFIED_ON") AS 
  SELECT /*@public*/
A.AUDIT_ID AS ID,
TO_CHAR(Regexp_Replace(A.CLOSING_COMMENTS,'<[^>]+>',NULL)) AS COMMENTS,
A.FOLLOW_UP_AUDITOR AS AUDITOR,
A.FOLLOW_UP_DATE AS FOLLOW_UP_DATE,
A.FOLLOW_UP_STATUS AS FOLLOW_UP_STATUS,
A.RE_AUD_LEAD_AUDITOR AS REAUD_LEAD_AUDITOR,
A.RE_AUDIT_START_DATE AS REAUD_START_DATE,
TO_CHAR(Regexp_Replace(F.ORG_CLOSING_COMMENTS,'<[^>]+>',NULL)) AS ORG_CLOSE_COMMENTS,
F.ORG_ID AS ORGANIZATION_ID,
(SELECT Z.DISPLAYED_VALUE FROM MS_QS_LOV_ORG_VALUES_V Z WHERE LOV_NAME = 'MS AUD Audit Rating' 
AND Z.STORED_VALUE=F.ORG_RISK_RATING
AND z.locale_id=ms_grc_common_utils.fn_get_default_locale_id)  AS ORG_RISK_RATING,
TO_CHAR(Regexp_Replace(D.AE_CLOSING_COMMENTS,'<[^>]+>',NULL)) AS AE_CLOSE_COMMENTS,
D.AE_ID AS AE_ID,
(SELECT Z.DISPLAYED_VALUE FROM MS_QS_LOV_ORG_VALUES_V Z WHERE LOV_NAME = 'MS AUD Audit Rating' 
AND Z.STORED_VALUE=D.AE_RISK_RATING
AND z.locale_id=ms_grc_common_utils.fn_get_default_locale_id) AS AE_RISK_RATING,
TO_CHAR(Regexp_Replace(H.RS_CLOSING_COMMENTS,'<[^>]+>',NULL)) AS RISK_CLOSE_COMMENTS,
H.RS_ID AS RISK_ID,
(SELECT Z.DISPLAYED_VALUE FROM MS_QS_LOV_ORG_VALUES_V Z WHERE LOV_NAME = 'MS AUD Audit Rating' 
AND Z.STORED_VALUE=H.RS_RISK_RATING 
AND z.locale_id=ms_grc_common_utils.fn_get_default_locale_id) AS RS_RISK_RATING,
A.DD_CREATED_BY AS CREATED_BY,
A.DD_CREATED_ON AS CREATED_ON,
A.DD_MODIFIED_BY AS MODIFIED_BY,
A.DD_MODIFIED_ON AS MODIFIED_ON
FROM MS_AUD_AUDIT_CLOSURE A
LEFT JOIN MS_AUD_AUDIT_CLOSURE_ORG F ON 
A.AUDIT_ID=F.AUDIT_ID AND
A.FORM_TYPE=F.FORM_TYPE
LEFT JOIN MS_AUD_AUDIT_CLOSURE_AUE D ON 
A.AUDIT_ID=D.AUDIT_ID AND
A.FORM_TYPE=D.FORM_TYPE
LEFT JOIN MS_AUD_AUDIT_CLOSURE_RSK H ON 
a.audit_id=h.audit_id and
A.FORM_TYPE=H.FORM_TYPE
