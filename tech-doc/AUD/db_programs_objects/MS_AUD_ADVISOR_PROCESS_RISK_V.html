MS_AUD_ADVISOR_PROCESS_RISK_V 

  CREATE OR REPLACE FORCE VIEW "MS_AUD_ADVISOR_PROCESS_RISK_V" ("OBJ_REL_ID", "SRC_OBJ_TYPE", "SRC_OBJ_ID", "TGT_OBJ_TYPE", "TGT_OBJ_ID") AS 
  SELECT rel_orb.REL_INST_ID      AS OBJ_REL_ID,
       decode(defn.SRC_OBJ_TYPE, 'MS_GRC_RISK', defn.dest_OBJ_TYPE, defn.SRC_OBJ_TYPE) AS src_obj_type,
       decode(defn.SRC_OBJ_TYPE, 'MS_GRC_RISK', rel_orb.tgt_obj_id, rel.REL_SOURCE_OBJECT_ID) AS src_obj_id,
       decode(defn.dest_OBJ_TYPE, 'MS_GRC_RISK', defn.dest_obj_type, defn.SRC_OBJ_TYPE) AS tgt_obj_type,
       decode(defn.dest_OBJ_TYPE, 'MS_GRC_RISK', rel_orb.tgt_obj_id, rel.REL_SOURCE_OBJECT_ID) AS tgt_obj_id
  FROM (select SRC_OBJ_TYPE, dest_obj_type, rel_id
          from ms_grc_rel_defn
        where ((src_obj_type in (SELECT object_type FROM ms_aud_ae_lib_conf where upper(active) = 'Y') and dest_obj_type = 'MS_GRC_RISK')
                or
                (src_obj_type = 'MS_GRC_RISK' and dest_obj_type in (SELECT object_type FROM ms_aud_ae_lib_conf where upper(active) = 'Y')))
              union all
        select SRC_OBJ_TYPE, dest_obj_type, rel_id
          from ms_aud_ae_rel_defn
         where ((src_obj_type in (SELECT object_type FROM ms_aud_ae_lib_conf where upper(active) = 'Y') and dest_obj_type = 'MS_GRC_RISK')
                 or
                 (src_obj_type = 'MS_GRC_RISK' and dest_obj_type in (SELECT object_type FROM ms_aud_ae_lib_conf where upper(active) = 'Y')))) defn,
       ms_Grc_orb_relations rel,
       ms_Grc_orb_relations_orb rel_orb
 WHERE rel.rel_source_id = rel_orb.rel_source_id
   AND defn.rel_id = rel_orb.rel_config_id
