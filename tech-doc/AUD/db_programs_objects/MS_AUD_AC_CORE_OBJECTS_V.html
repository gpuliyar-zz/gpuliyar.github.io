MS_AUD_AC_CORE_OBJECTS_V 

  CREATE OR REPLACE FORCE VIEW "MS_AUD_AC_CORE_OBJECTS_V" ("PROCESS_NAME", "PROCESS_ID", "CORE_TYPE", "RISK_NAME", "RISK_ID", "ORG_ENTITY_NAME", "ORG_ENTITY_ID", "PROCESS_LEVEL", "VERSION_NO", "OWNER_ORGANIZATIONS", "RESTRICT_ACCESS_TO", "DD_OBJECT_TYPE", "OBJECT_ID", "ATO_ORGANIZATIONS", "LOCALE_ID") AS 
  SELECT DISTINCT po.process_name as process_name,
                        po.process_id as process_id,
                        ms_aud_utilities.fn_get_ae_obj_name(pub.DD_OBJECT_TYPE) AS core_type,
                        NULL AS risk_name,
                        NULL AS risk_id,
                        NULL AS org_entity_name,
                        NULL AS org_entity_id,
                        OBJECT_LEVEL as process_level,
                        (select ae_version
                           from ms_aud_auditable_entit
                          where object_id = po.process_id) as version_no,
                        PO.owner_organizations as owner_organizations, restrict_access_to, pub.dd_object_type as dd_object_type, 
                        pub.object_id as object_id, ato_organizations,
                        loc.locale_id as locale_id
          FROM ms_aud_grc_process_v         po,
               ms_aud_grc_core_object_pub_v pub,
               si_enabled_locales           loc
         WHERE po.process_id = pub.object_id
           AND trunc(nvl(pub.valid_from, sysdate)) <= trunc(sysdate)
           AND NVL(TRUNC(pub.valid_until), TRUNC(SYSDATE)) >= TRUNC(SYSDATE)
