MS_AUD_AUDPLAN_EDS_V 

  CREATE OR REPLACE FORCE VIEW "MS_AUD_AUDPLAN_EDS_V" ("ID", "TITLE", "TYPE", "START_DATE", "END_DATE", "CONDUCTED_BY_ORG", "USE_RISK_ASSESSMENT_FROM", "STATUS", "VERSION", "OVERVIEW", "TOTAL_BUDGET_DAYS", "BUDGETED_EFFORT", "BUDGETED_COST", "SCOPE", "APPROACH", "BUSINESS_OVERVIEW", "PROCESS_OVERVIEW", "SIGNIFICANT_CHALLENGES", "ACCOMPLISHMENTS", "ATTACH_FILES", "CREATED_BY", "CREATED_ON", "MODIFIED_BY", "MODIFIED_ON") AS 
  SELECT /*@public*/
A.PLAN_ID AS ID,
A.PLAN_TITLE AS TITLE,
(SELECT Z.DISPLAYED_VALUE FROM MS_QS_LOV_ORG_VALUES_V Z WHERE Z.LOV_NAME='MS AUD Plan Types' AND Z.STORED_VALUE=A.PLAN_TYPE
AND z.locale_id=ms_grc_common_utils.fn_get_default_locale_id) AS TYPE,
A.START_DATE AS START_DATE,
A.END_DATE AS END_DATE,
(SELECT Z.ORG_ENTITY_NAME FROM SI_ORG_ENTITIES_T Z WHERE Z.ORG_ENTITY_ID=A.HDR_ORGANIZATION) AS CONDUCTED_BY_ORG,
(SELECT Z.OBJECT_NAME FROM MS_RSK_SCENARIOS_SCN Z WHERE Z.SCENARIO_ID=A.USE_ASSESSMENT_FORM) AS USE_RISK_ASSESSMENT_FROM,
(SELECT Z.STATUS_NAME FROM MS_AUD_STATUS_TL_V Z WHERE Z.STATUS_ID=A.STATUS) AS STATUS,
A.VERSION_NO AS VERSION,
TO_CHAR(Regexp_Replace(A.OVERVIEW,'<[^>]+>',NULL)) AS OVERVIEW,
A.TOTAL_BUDGETED_EFFORT AS TOTAL_BUDGET_DAYS,
A.BUDGETED_EFFORT_FROM_AC AS BUDGETED_EFFORT,
A.BUDGETED_COST AS BUDGETED_COST,
TO_CHAR(Regexp_Replace(A.ADD_SCOPE,'<[^>]+>',NULL)) AS SCOPE,
TO_CHAR(Regexp_Replace(A.APPROACH,'<[^>]+>',NULL)) AS APPROACH,
TO_CHAR(Regexp_Replace(A.BUSINESS_OVERVIEW,'<[^>]+>',NULL)) AS BUSINESS_OVERVIEW,
TO_CHAR(Regexp_Replace(A.PROCESS_OVERVIEW,'<[^>]+>',NULL)) AS PROCESS_OVERVIEW,
TO_CHAR(Regexp_Replace(A.SIGNIFICANT_CHALLENGES,'<[^>]+>',NULL)) AS SIGNIFICANT_CHALLENGES,
TO_CHAR(Regexp_Replace(A.ACCOMPLISHMENTS,'<[^>]+>',NULL)) AS ACCOMPLISHMENTS,
TO_CHAR(SUBSTR(A.ATTACHMENTS,1,INSTR(A.ATTACHMENTS,'#',1,1)-1)) AS ATTACH_FILES,
A.DD_CREATED_BY AS CREATED_BY,
A.DD_CREATED_ON AS CREATED_ON,
A.DD_MODIFIED_BY AS MODIFIED_BY,
a.dd_modified_on as modified_on
FROM MS_AUD_AUDIT_PLAN A
