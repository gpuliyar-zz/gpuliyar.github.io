MS_AUD_CONTROLTSK_WRKPAPER_V 

  CREATE OR REPLACE FORCE VIEW "MS_AUD_CONTROLTSK_WRKPAPER_V" ("AUDIT_ID", "TASK_ID", "AUDITABLE_ENTITY", "CONTROL", "NO_OF_TESTS", "ASSIGNED_TO", "KEY_CONTROL", "START_DATE", "END_DATE", "ISSUES", "APPROVER", "TASK_TYPE", "CO_AUDITORS", "FINAL_APPROVER", "ASSIGNED_TO_STORED", "APPROVER_STORED", "FINAL_APPROVER_STORED") AS 
  SELECT a.audit_id                                                                                                                                     AS audit_id,
       a.task_id                                                                                                                                      AS task_id,
       ( SELECT  gr.object_name
         FROM   MS_AUD_GRC_CORE_OBJECT_PUB_V gr,
                MS_AUD_AUDIT_TASK_COB c
         WHERE  gr.object_id = c.core_objects AND
                c.task_id = a.task_id )                                                                                                               AS auditable_entity,
        ( SELECT MS_JOIN_CLOB_FOR_DISPLAY( ct_name )
         FROM   MS_AUD_CTRL_TEST_TASK_CTL c
         WHERE  c.task_id = a.task_id )                                                                                                                                               control,
       ( SELECT Decode( Count( m.ts_id ), 0, NULL,
                                          Count( m.ts_id ) )
         FROM   MS_AUD_CTRL_TEST_TASK_TST m,
                MS_AUD_CTRL_TEST_TASK_CTL n
         WHERE  m.ts_ct_id = n.ct_id AND
                n.testing_req = 1 AND
                m.task_id = a.task_id )                                                                                                               AS no_of_tests,
       ( SELECT user_full_name
         FROM   MS_AUD_USERS_V u
         WHERE  u.user_name = a.task_owner )                                                                                                      AS assigned_to,
       ( SELECT Ms_concat( Decode( ct_key_control, 1, 'Yes',
                                                   2, 'No' ) )
         FROM   MS_AUD_CTRL_TEST_TASK_CTL c
         WHERE  c.task_id = a.task_id )                                                                                                               AS key_control,
       a.task_start_date                                                                                                                              AS start_date,
       a.task_completion_date                                                                                                                         AS end_date,
       ( SELECT Decode( Count( fn_finding_title ), 0, NULL,
                                                   Count( fn_finding_title ) )
         FROM   MS_AUD_AUDIT_TASK_FND
         WHERE  task_id = a.task_id )                                                                                                                 AS issues,
       ( SELECT user_full_name
         FROM   MS_AUD_USERS_V u
         WHERE  u.user_name = a.approver )                                                                                                            AS approver,
       a.task_type                                                                                                                                    task_type,
       ms_aud_report_utils.Fn_get_mlov_lead_aud_disp( ms_aud_utilities.Mlov_values( a.task_id, 'task_id', 'co_auditors', 'ms_aud_ctrl_test_task_car' ) ) AS co_auditors,
       ( SELECT user_full_name
         FROM   MS_AUD_USERS_V u
         WHERE  u.user_name = a.final_approver )AS final_approver,
          a.task_owner as assigned_to_stored,
          a.approver as approver_stored,
           a.final_approver as final_approver_stored
FROM   MS_AUD_AUDIT_TASK a
WHERE  a.task_type = '4'
