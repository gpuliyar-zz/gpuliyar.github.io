MS_AUD_USERS_V 

  CREATE OR REPLACE FORCE VIEW "MS_AUD_USERS_V" ("USER_NAME", "USER_ID", "USER_FULL_NAME", "ENDDATE") AS 
  SELECT
                                   users.user_name        user_name
                    ,users.user_id          user_id
        ,(CASE pvalue.parameter_value
          WHEN '2'
          THEN users.last_name || ' ' || users.first_name
          WHEN '3'
          THEN users.last_name || ',' || ' ' || users.first_name || ' ' || '(' || users.user_name || ')'
          WHEN '1'
          THEN users.first_name|| ' ' || users.last_name
          WHEN '4'
          THEN users.first_name|| ',' || ' ' || users.last_name || ' ' || '(' || users.user_name || ')'
          ELSE users.last_name || ' ' || users.first_name  END ) AS user_full_name,
         end_Date AS enddate
FROM si_users_t users,
     ms_qs_parameter_values pvalue,
     ms_qs_parameter_types ptype,
     ms_qs_parameter_names pname
WHERE
  UPPER (ptype.parameter_type)   = UPPER ('MS APPS USER NAME DISPLAY FORMAT')
  AND pname.parameter_id             = pvalue.parameter_id
  AND ptype.parameter_type_id        = pvalue.parameter_type_id
  AND ptype.parameter_type_id        = pname.parameter_type_id
  AND UPPER (pname.parameter_name)   = UPPER ('DISPLAY_FORMAT')
  AND pvalue.org_entity_id           = 1
