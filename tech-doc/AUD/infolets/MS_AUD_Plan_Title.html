SELECT 'AD-ID' as plan_id,  
       'Adhoc Audit' AS plan_title, 
       'NA' as start_date, 
       'NA' as end_date, 
       'NA' as organization 
  FROM DUAL 
UNION ALL 
SELECT plan_id, plan_title, 
                 to_char(start_date,'dd/mm/yyyy') as start_date, 
                 to_char(end_date,'dd/mm/yyyy') as end_date, 
                 to_char(organization) as organization 
  FROM (SELECT   plan_id AS plan_id,  
                 plan_title,  
                 VERSION_NO, 
                 MS_GRC_COMMON_UTILS.Fn_Get_Org_Display_Names (hdr_organization) as organization, 
                 start_date, 
                 end_date 
  FROM ms_aud_audit_plan  
           WHERE STATUS = '3' 
)
