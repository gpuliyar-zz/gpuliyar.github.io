SELECT a.fn_id as fnId, a.fn_finding_title as findingTitle,
       MS_AUD_REPORT_UTILS.fn_get_status_display_name
          (ms_aud_locking_utilities.fn_get_status_from_trans (a.audit_id,
                                                              a.fn_id
                                                             )
          ) AS issue_status,
       DECODE
          (b.fn_ref_task_id,
           NULL, NULL,
           ms_apps_utilities.get_user_full_name
              (ms_aud_utilities.fn_get_task_owner_by_task_id (b.fn_ref_task_id)
              )
          ) AS task_owner,
       NVL (ms_aud_f009_helper.fn_get_task_name (b.fn_ref_task_id),
            'Draft Issue'
           ) AS task_name,
       NULL AS ex_col_1, NULL AS ex_col_2,c.locale_id AS localeId,
                (SELECT COUNT (DISTINCT (ac_id))
           FROM ms_aud_audit_task_act act
          WHERE act.ac_finding_id = a.fn_id) AS actions_count
  FROM ms_aud_draft_issues_fnd b, ms_aud_audit_task_fnd a, si_enabled_locales c
 WHERE a.fn_issue_resolution IN ('1','2')
   AND b.audit_id = a.audit_id
   AND b.task_id = a.task_id
   AND b.fn_id = a.fn_id
   AND a.audit_id = :x_audit_id
   AND (case when (nvl(:current_stage,'80')='80' and nvl(:previous_stage,'0') ='0') then   
            ms_aud_locking_utilities.fn_get_issue_status_ar (a.audit_id, a.fn_id)
       else '1' end ='1')
   AND a.fn_id IS NOT NULL
   order by b.fnd_orderkey
