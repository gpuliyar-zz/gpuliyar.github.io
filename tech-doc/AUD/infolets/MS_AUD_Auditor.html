SELECT ms_apps_utilities.get_user_full_name(a.auditor_user_name) AS Auditor_Name,
          (SELECT lovvalue.displayed_value AS displayed_value
			  FROM ms_qs_list_of_values lovvalue,
				ms_qs_lov_names lovname
			  WHERE lovvalue.lov_id      = lovname.lov_id
			  AND lovvalue.org_entity_id = 1
			  AND lovname.lov_name       = 'MS AUD Auditor Profile Title'
			  AND lovvalue.locale_id     = NVL(CAST(si_db_locale_sv.getvalue AS NUMBER),1009)
			  AND stored_value           =a.AUDITOR_TITLE
			  AND lovvalue.enterprise_id =(SELECT enterprise_id FROM si_ent)
			  AND (lovname.end_date     IS NULL
			  OR lovname.end_date        > SYSDATE)
			  AND lovname.start_date    <= sysdate
			  AND (lovvalue.end_date    IS NULL
			  OR lovvalue.end_date       > SYSDATE)
			  ) as Title,
          EMAIL_ADDRESS,
          ms_aud_setup_rpt_utils.fn_get_auditor_group(ms_aud_utilities.mlov_values(a.auditor_user_name,'AUDITOR_USER_NAME','AUDITOR_GROUP','ms_aud_auditor_profile_aug'),'MS AUD Auditor Group') as Auditor_Groups,
          ms_aud_setup_rpt_utils.fn_get_auditor_group(ms_aud_utilities.mlov_values(a.auditor_user_name,'AUDITOR_USER_NAME','PROFESSIONAL_QUAL','ms_aud_auditor_profile_pro'),'MS AUD Auditor Qualifications') as Professional_Qualifications,
          ms_aud_setup_rpt_utils.fn_get_auditor_group(ms_aud_utilities.mlov_values(a.auditor_user_name,'AUDITOR_USER_NAME','TRAINING','ms_aud_auditor_profile_trn'),'MS AUD Auditor Profile Training') as Training,
          (SELECT lovvalue.displayed_value AS displayed_value
			  FROM ms_qs_list_of_values lovvalue,
				ms_qs_lov_names lovname
			  WHERE lovvalue.lov_id      = lovname.lov_id
			  AND lovvalue.org_entity_id = 1
			  AND lovname.lov_name       = 'MS AUD Auditor Proficiency'
			  AND lovvalue.locale_id     = NVL(CAST(si_db_locale_sv.getvalue AS NUMBER),1009)
			  AND stored_value           =a.PROFIENCY_LEVEL
			  AND lovvalue.enterprise_id =(SELECT enterprise_id FROM si_ent)
			  AND (lovname.end_date     IS NULL
			  OR lovname.end_date        > SYSDATE)
			  AND lovname.start_date    <= sysdate
			  AND (lovvalue.end_date    IS NULL
			  OR lovvalue.end_date       > SYSDATE)
			  ) as Proficiency,
          ms_aud_setup_rpt_utils.fn_get_auditor_group(ms_aud_utilities.mlov_values(a.auditor_user_name,'AUDITOR_USER_NAME','SUBJECT_MATTER_EXPERT','ms_aud_auditor_profile_sme'),'MS AUD Auditor Subject Matter Expert') as Subject_Matter_Expert,
          ms_aud_setup_rpt_utils.fn_get_auditor_group(ms_aud_utilities.mlov_values(a.auditor_user_name,'AUDITOR_USER_NAME','LANGUAGE_SKILLS','ms_aud_auditor_profile_lns'),'MS AUD Auditor Language Skills') as Language_Skills,
          a.COST_PER_HOUR,
         ms_aud_report_utils.fn_get_report_json(
  (SELECT metric_id FROM si_metrics_t WHERE metric_name='MS_AUD_AUDITOR_PROFILE'
  ),
  a.dd_process_instance_id,
  null,
  'flag=1'
  ||CHR(38)
  ||'emd=1'
  ||CHR(38)
  ||'x_edit_mode=Yes'
  ||CHR(38)
  ||'Audit_user_name='
  ||a.auditor_user_name
  ||CHR (38)
  ||'Auditor='
  ||ms_apps_utilities.get_user_full_name(a.auditor_user_name)) AS report_json, a.auditor_user_name AS auditor_user_name 
  FROM ms_aud_auditor_profile a,si_users_t 
  where USER_NAME=a.auditor_user_name
