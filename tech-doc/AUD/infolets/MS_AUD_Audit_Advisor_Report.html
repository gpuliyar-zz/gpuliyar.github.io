SELECT   g.object_name AS object_name,
         ms_aud_advisor_utilities.fn_get_rating (MS_AUD_ADVISOR_UTILITIES.FN_GET_PROCESS_SCORE (g.object_id,:Perspective,'1'),g.object_id,:Perspective,'C','1') as score,
         to_date(a.start_date,'MM/DD/RRRR') as start_date,to_date(a.end_date,'MM/DD/RRRR') as end_date, to_date(a.next_audit_date,'MM/DD/RRRR') as next_audit_date,
         ms_aud_advisor_utilities.fn_get_suggested_year(:Perspective, to_date(a.start_date,'MM/DD/RRRR') , to_date(a.next_audit_date,'MM/DD/RRRR'), ms_aud_advisor_utilities.fn_get_rating(ms_aud_advisor_utilities.fn_get_process_score (g.object_id, :Perspective,'1'),g.object_id,:Perspective,'C','1'),g.object_id) AS SuggestedYears,
         MS_AUD_PLANNING_RPT_UTILS.get_plan_years (g.object_id) AS plan_year,
         ms_aud_advisor_utilities.fn_get_open_issue(g.object_id,'Y') AS open_issues,
 MS_AUD_PLANNING_RPT_UTILS.fn_highlight_ae(ms_aud_advisor_utilities.fn_get_suggested_year(:Perspective, to_date(a.start_date,'MM/DD/RRRR'),  to_date(a.next_audit_date,'MM/DD/RRRR'), ms_aud_advisor_utilities.fn_get_rating(ms_aud_advisor_utilities.fn_get_process_score (g.object_id, :Perspective,'1'),g.object_id,:Perspective,'C','1'),g.object_id)) as Highlight,
         MS_AUD_PLANNING_RPT_UTILS.fn_get_actitle (g.object_id) AS planned_candidates,
         g.object_id AS ID, ms_aud_advisor_utilities.fn_get_no_of_audits(g.object_id, 'AE') as no_of_audits,a.audit_rating as aud_rating
    ,MS_AUD_REPORT_UTILS.fn_get_audits_detail(g.object_id, 'AE') AS audit_ids,
ms_aud_report_utils.fn_get_report_json(
            (SELECT metric_id FROM si_metrics_t WHERE metric_name = g.dd_object_type),
              g.dd_process_instance_id,
              null,
              'edit_flag=N' || CHR (38)|| 'flag=2'|| CHR (38)|| 'emd=1') AS report_json,
   DECODE(g.dd_object_type,'MS_AUD_AUDITABLE_ENTITY','MS_AUD_AUDITABLE_ENTIT',g.dd_object_type) AS CORE_OBJ_TYPE
    FROM ms_aud_ae_advisor_data_v a, ms_aud_grc_core_object_pub_v g
   WHERE a.object_id(+)=g.object_id
      AND g.object_id IN (
 	SELECT object_id
  	FROM ms_aud_grc_core_object_pub_v
 	WHERE NVL (TRUNC (valid_until), TRUNC (SYSDATE)) >= TRUNC (SYSDATE)
    and obj_status not in ('EXP','INACT')
   	AND (restrict_access_to IN ('N') OR (MS_GRC_STR_COMMON_UTILITIES.Is_String_exists_in_list(owner_organizations,MS_AUD_REPORT_UTILS.fn_get_org_entity_id(:LOGIN_ID)) = 1))
	AND ((object_id IN (SELECT MAX(b.OBJECT_ID) FROM ms_aud_grc_core_object_pub_v a, ms_aud_auditable_entit b WHERE a.object_id = b.object_id 
AND NVL(TRUNC(a.valid_until),TRUNC(SYSDATE)) >= TRUNC(SYSDATE) and a.obj_status not in ('EXP','INACT') GROUP BY b.internal_object_id) 
	AND dd_object_type='MS_AUD_AUDITABLE_ENTITY') OR dd_object_type<>'MS_AUD_AUDITABLE_ENTITY'))
ORDER BY to_number(MS_AUD_ADVISOR_UTILITIES.FN_GET_PROCESS_SCORE (g.object_id, :Perspective,'1')) DESC NULLS LAST
