select ae_type,ae_type_stored_val,SCOPE_IN,SCOPE_OUT,ae_type_count,ae_total_count,locale_id,:Year as YEAR 
 from (select lovv.displayed_value as ae_type,
       lovv.stored_value as ae_type_stored_val,
       'SI' AS SCOPE_IN,
       'SO' AS SCOPE_OUT,
       count(co.core_type) as ae_type_count,
       (SELECT COUNT(1)
          FROM ms_aud_grc_core_object_pub_v a, MS_AUD_AE_LIB_CONF_PUB_V c
         where ms_apps_security_pkg.check_object_access(a.dd_object_type,
                                                        a.restrict_access_to,
                                                        a.owner_organizations,
                                                        :LOGIN_ID,
                                                        'VIEW') = 'ACCESS'
           and nvl(a.valid_until, sysdate) >= trunc(sysdate)
           and c.object_type = a.dd_object_type
           and c.active = 'Y'
           and a.core_type is not null
           AND lovv.stored_value = a.core_type) as ae_total_count,
       lovv.locale_id as locale_id
  from (select DISTINCT a.dd_object_type , a.core_type, A.OBJECT_ID
    from ms_aud_grc_core_object_pub_v a,
         MS_AUD_AE_LIB_CONF_PUB_V     c,
         MS_AUD_AUDIT_AUE.html">MS_AUD_AUDIT_AUE             D,
         MS_AUD_AUDIT                 E
   where ms_apps_security_pkg.check_object_access(a.dd_object_type,
                                                  a.restrict_access_to,
                                                  a.owner_organizations,
                                                  :LOGIN_ID,
                                                  'VIEW') = 'ACCESS'
     and nvl(a.valid_until, sysdate) >= trunc(sysdate)
     and c.object_type = a.dd_object_type
     AND e.audit_id(+) = d.audit_id
     and c.active = 'Y'
     and a.core_type is not null
     And Extract(Year From Audit_Start_Date) =Nvl(:Year,Ms_Aud_Constants.Fn_Year)
     AND D.CORE_OBJECTS = A.OBJECT_ID(+)
     AND trunc(e.audit_start_date) >
         ADD_MONTHS(trunc(sysdate, 'MON'), - (10))) co
 right outer join ms_qs_list_of_values lovv
    on lovv.stored_value = co.core_type
  left outer join si_metrics_t m
 on m.metric_name = co.dd_object_type , ms_qs_lov_names lovn
 where lovv.lov_id = lovn.lov_id
   and lovv.org_entity_id = 1
   and lovn.lov_name in ('MS_AUD_AE_Process_Type','MS GRC Process Type')
   and lovv.locale_id = nvl(cast(si_db_locale_sv.getvalue as number), 1009)
 group by lovv.displayed_value, lovv.stored_value, lovv.locale_id) aa
 where aa.ae_total_count > 0
