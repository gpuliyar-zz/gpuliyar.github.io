SELECT   a.audit_id as Audit_ID, a.audit_title as Audit_Title,
         NVL(MS_AUD_PLANNING_RPT_UTILS.fn_get_plan_title(a.plan_title),'Adhoc') AS plan_title,
         MS_AUD_EXECUTION_RPT_UTILS.get_code_description
                                          ('MS AUD Audit Type',
                                           a.audit_type
                                          ) AS audit_type,
         MS_AUD_EXECUTION_RPT_UTILS.get_code_description
                                  ('MS AUD Classification',
                                   a.audit_classification
                                  ) AS audit_category,
         MS_AUD_REPORT_UTILS.fn_get_status_display_name
                                                (a.audit_status)
                                                              AS audit_status,
         null AS core_objects,
         null  AS risks,
          null  AS organizations,
            MS_GRC_COMMON_UTILS.Fn_Get_Org_Display_Names(conducted_by_org) as conducted_org,
         COALESCE(ms_apps_utilities.get_user_full_name (a.lead_auditor),MS_AUD_REPORT_UTILS.fn_get_mlov_lead_aud_disp(MS_GRC_COMMON_UTILS.Fn_Get_Val_By_Id(a.audit_id,'ms_aud_audit_alm','lead_auditor_mlov','audit_id'))) AS 
        lead_auditor,COALESCE(ms_apps_utilities.get_user_full_name (a.audit_manager),MS_AUD_REPORT_UTILS.fn_get_mlov_aud_man_disp(MS_GRC_COMMON_UTILS.Fn_Get_Val_By_Id(a.audit_id,'ms_aud_audit_amm','audit_manager_mlov','audit_id'))) AS 
         audit_manager,
         ms_apps_utilities.get_user_full_name (a.dd_created_by) AS created_by,
         TO_DATE(a.audit_start_date) AS audit_start_date,
         MS_AUD_EXECUTION_RPT_UTILS.fn_get_task_count (a.audit_id, :LOGIN_ID) AS view_tasks,
         lead_auditor as lead_auditor_stored,
         audit_manager as audit_manager_stored,a.audit_end_date as audit_end_date,ms_aud_report_utils.fn_get_report_json((SELECT metric_id
                                                FROM si_metrics_t
                                               WHERE metric_name =
                                                     'MS_AUD_MANAGE_AUDIT'),
                                              a.dd_process_instance_id,
                                              NULL,
                                              'flag=1' || CHR(38) || 'emd=2' ||
                                              CHR(38) || 'x_edit_mode=No' ||
                                              CHR(38) || 'audit_id=' ||
                                              a.audit_id || CHR(38) ||
                                              'mode=view') as report_json,
NULL as Issues,
  case
    WHEN a.AUDIT_STATUS = '38'
    THEN
      (SELECT URL_LINK
      FROM
        (SELECT URL_LINK,
          AUDIT_ID,
          dense_rank() over(partition BY AUDIT_ID order by PUBLISHED_DATE DESC) AS RNK
        FROM MS_AUD_AUDIT_PUBLISHED_REPORTS
        ) C
      WHERE C.AUDIT_ID=a.AUDIT_ID
      AND RNK         =1
      )
  END AS DOWNLOAD
 FROM ms_aud_audit a 
 WHERE a.status not in (4,8,31)
