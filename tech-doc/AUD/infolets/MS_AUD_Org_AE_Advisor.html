select org_object_id,object_name,
 score,
start_date,
end_date,
next_audit_date,
suggested_years,
planned_years,
open_issue,
planned_candidates,
Fk,
null as object_url,no_of_audits,aud_rating,audit_ids,org_score
from (
select a.object_id as org_object_id,
        A.OBJECT_NAME AS OBJECT_NAME,null as ae_process_object_id,null as ae_process_name,null as risk_object_id, null risk_object_name,null as Temp_fld,
         ms_aud_advisor_utilities.fn_get_rating((select ms_aud_advisor_utilities.Fn_Get_Org_Score_4_scope3 (a.object_id,:risk_perspective) from dual),a.object_id,:risk_perspective,'O','2') AS score,
         to_date(a.start_date,'MM/DD/RRRR') AS start_date,
         to_date(a.end_date,'MM/DD/RRRR') AS end_date,
         to_date(a.next_audit_date,'MM/DD/RRRR') as next_audit_date,
         ms_aud_advisor_utilities.fn_get_suggested_year
            (:risk_perspective,
            to_date(a.start_date,'MM/DD/RRRR'),
             to_date(a.next_audit_date,'MM/DD/RRRR'),
             MS_AUD_ADVISOR_UTILITIES.FN_GET_RATING( MS_AUD_ADVISOR_UTILITIES.Fn_Get_Org_Score_4_scope3 (a.object_id, :risk_perspective),a.object_id, :risk_perspective,'O','2')
,a.object_id
            )
          as suggested_years,
         MS_AUD_PLANNING_RPT_UTILS.GET_PLAN_YEARS_FOR_ORG(object_id) as planned_years,                         
          ms_aud_advisor_utilities.fn_get_org_open_issue (a.object_id) AS open_issue,
                           MS_AUD_PLANNING_RPT_UTILS.fn_get_AcTitle_for_org (object_id) AS planned_candidates,
                           a.fk as fk,
                            ms_aud_advisor_utilities.fn_get_no_of_audits(b.org_entity_id, 'ORG') as no_of_audits,a.audit_rating as aud_rating,MS_AUD_REPORT_UTILS.fn_get_audits_detail(a.object_id, 'ORG') AS audit_ids,MS_AUD_ADVISOR_UTILITIES.Fn_Get_Org_Score_4_scope3(a.object_id,:risk_perspective) as org_score
FROM ms_aud_org_advisor_data_v a, si_org_entities_t b
WHERE a.object_id = b.org_entity_id
AND (   b.end_date IS NULL
      or trunc (b.end_date) > trunc (sysdate)
     )                 
 )A  Where (:organization_id Is Null Or Org_Object_Id In (Select Column_Value From Table ( Cast (Ms_Grc_Str_Common_Utilities.Split_String_Stored(:organization_id) As Ms_Varchar2_Array))))
                             order by org_score desc nulls last
