select process_name,
       process_id,
       core_type,
       risk_name,
       risk_id,
       org_entity_name,
       org_entity_id,
       process_level,
       version_no
  from (SELECT process_name,
               process_id,
               core_type,
               risk_name,
               risk_id,
               org_entity_name,
               org_entity_id,
               process_level,
               version_no
          FROM ms_aud_tmp_core_objects_v
         WHERE (:scope = 5 OR :scope = 1)
           AND ((object_id IN
               (SELECT max(b.object_id)
                    FROM ms_aud_grc_core_object_pub_v a,
                         ms_aud_auditable_entit       b
                   WHERE a.object_id = b.object_id
                     AND NVL(TRUNC(a.valid_from), TRUNC(SYSDATE)) <=
                         TRUNC(SYSDATE)
                     AND NVL(TRUNC(a.valid_until), TRUNC(SYSDATE)) >=
                         TRUNC(SYSDATE)
                   group by b.internal_object_id) AND
               dd_object_type = 'MS_AUD_AUDITABLE_ENTITY') OR
               dd_object_type <> 'MS_AUD_AUDITABLE_ENTITY' OR
               object_id IN
               (SELECT tmp_core_objects
                   FROM ms_aud_audit_plan_cro
                  where plan_id IN
                        (SELECT PLAN_ID
                           FROM ms_aud_audit_plan
                          where internal_plan_id = :plan_id)))
           AND ms_apps_security_pkg.check_object_access(dd_object_type,
                                                        restrict_access_to,
                                                        owner_organizations,
                                                        :user_name,
                                                        'VIEW') = 'ACCESS'
        union
        SELECT null as process_name,
               process_id||ms_aud_constants.scope_delimiter()||ato_organizations as process_id,
               core_type,
               risk_name,
               risk_id,
               org_entity_name,
               org_entity_id,
               process_level,
               version_no
          FROM ms_aud_ac_core_objects_v
         WHERE ato_organizations IN
               (SELECT COLUMN_VALUE
                  FROM TABLE(ms_grc_str_common_utilities.Split_String_Stored(NVL(:org, 1))))
           AND :scope = 3
           AND ((object_id IN
               (SELECT max(b.object_id)
                    from ms_aud_grc_core_object_pub_v a,
                         ms_aud_auditable_entit       b
                   WHERE a.object_id = b.object_id
                     AND NVL(TRUNC(a.valid_from), TRUNC(SYSDATE)) <=
                         TRUNC(SYSDATE)
                     AND NVL(TRUNC(a.valid_until), TRUNC(SYSDATE)) >=
                         TRUNC(SYSDATE)
                   group by b.internal_object_id) AND
               dd_object_type = 'MS_AUD_AUDITABLE_ENTITY') OR
               dd_object_type <> 'MS_AUD_AUDITABLE_ENTITY' OR
               object_id IN
               (SELECT TMP_CORE_OBJECTS
                   FROM ms_aud_audit_plan_cro
                  where plan_id in
                        (select plan_id
                           from ms_aud_audit_plan
                          where internal_plan_id = :plan_id)))
           AND ms_apps_security_pkg.check_object_access(dd_object_type,
                                                        restrict_access_to,
                                                        owner_organizations,
                                                        :user_name,
                                                        'VIEW') = 'ACCESS')
 ORDER BY process_name
