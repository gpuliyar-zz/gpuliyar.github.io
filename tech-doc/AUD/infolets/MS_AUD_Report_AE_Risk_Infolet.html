SELECT a.object_id as object_id, 
   (SELECT displayed_value
       FROM ms_qs_list_of_values lovvalue,ms_qs_lov_names lovname,si_metrics_t m
       WHERE lovvalue.lov_id = lovname.lov_id
       AND lovvalue.org_entity_id = 1
       AND lovname.lov_name in('MS_AUD_AE_Process_Type','MS GRC Process Type')
       AND lovvalue.locale_id = nvl(cast(si_db_locale_sv.getvalue as number),1009) 
       and stored_value=a.core_type
       and lovvalue.enterprise_id=m.enterprise_id
       and m.metric_name = a.dd_object_type) object_type,
              a.object_name  AS object_name ,
               trim(substr(ms_aud_advisor_utilities.fn_get_rating(MS_AUD_PLANNING_RPT_UTILS.get_risk_score (a.object_id,:Scenario),a.object_id,:Scenario,'C',1),1,
              instr(ms_aud_advisor_utilities.fn_get_rating(MS_AUD_PLANNING_RPT_UTILS.get_risk_score (a.object_id,:Scenario),a.object_id,:Scenario,'C',1),' '))) AS risk_rating,
              (SELECT max(audit_start_date) keep (dense_rank first order by to_number(audit_status)) FROM ms_aud_audit WHERE audit_id IN (
               SELECT b.audit_id FROM ms_aud_audit_aue b,ms_aud_audit ad
				WHERE b.core_objects = a.object_id
				and ad.audit_id=b.audit_id
				and ad.audit_status not in (4)) AND audit_start_date is not null) AS last_audited_on,
                (SELECT TRUNC (MIN (audit_start_date)) FROM ms_aud_audit WHERE audit_id IN (SELECT audit_id FROM ms_aud_audit_aue b
                            WHERE b.core_objects = a.object_id)
                    AND audit_status = '33') AS next_audit_date,
                MS_AUD_PLANNING_RPT_UTILS.get_plan_years (a.object_id) AS plan_years,
                NVL((SELECT COUNT(b.audit_id) FROM ms_aud_audit_aue b,ms_aud_audit aud
                     WHERE b.core_objects = a.object_id and aud.status not in (4,8) and aud.audit_id=b.audit_id),0) AS no_of_audits_till_date,
                 ms_aud_advisor_utilities.fn_get_open_issue(a.object_id,'Y') AS open_issues,
                (SELECT displayed_value
          FROM ms_qs_list_of_values lovvalue,ms_qs_lov_names lovname ,si_metrics_t m
          WHERE lovvalue.lov_id = lovname.lov_id
              AND lovvalue.org_entity_id = 1
              AND lovname.lov_name in('MS GRC Core Category','MS_AUD_AE_Category_Type')
              AND lovvalue.locale_id = nvl(cast(si_db_locale_sv.getvalue as number),1009) 
              and stored_value=a.core_category
              and lovvalue.enterprise_id=m.enterprise_id
              and m.metric_name = a.dd_object_type) AS core_category ,
                ms_apps_utilities.get_user_full_name(d.obj_created_by) AS created_by,TRUNC (d.dd_created_on) AS creation_date,
            ms_aud_report_utils.fn_get_report_json(
            (SELECT metric_id FROM si_metrics_t WHERE metric_name = a.dd_object_type),
              a.dd_process_instance_id,a.dd_instance_id,
              'edit_flag=N' || CHR (38)|| 'flag=2'|| CHR (38)|| 'emd=1','Y') AS report_json,d.obj_created_by  AS created_by_id, DECODE(d.dd_object_type,'MS_AUD_AUDITABLE_ENTITY','MS_AUD_AUDITABLE_ENTIT',d.dd_object_type) as core_obj_type
             FROM ms_aud_grc_core_object_pub_v a, MS_GRC_CORE_OBJECT_PUB_V d
          WHERE a.object_id = d.object_id
          AND ((d.object_id IN (
                SELECT MAX(a.OBJECT_ID) FROM ms_aud_auditable_entit a,ms_grc_core_object_pub_v b 
                where a.object_id=b.object_id
                and b.obj_status not in ('INACT','EXP')
                GROUP BY internal_object_id) 
          AND d.dd_object_type='MS_AUD_AUDITABLE_ENTITY') OR d.dd_object_type<>'MS_AUD_AUDITABLE_ENTITY') 
         AND ms_apps_security_pkg.check_object_access(a.dd_object_type, a.restrict_access_to, a.owner_organizations, :LOGIN_ID, 'VIEW') = 'ACCESS'
         AND  TRUNC(NVL(a.VALID_UNTIL,SYSDATE)) >= TRUNC(SYSDATE)   
         ORDER BY MS_AUD_PLANNING_RPT_UTILS.get_risk_score (a.object_id,:Scenario) desc NULLS LAST
