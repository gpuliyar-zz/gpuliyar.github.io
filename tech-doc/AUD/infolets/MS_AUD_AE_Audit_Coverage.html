SELECT 
  audit_id,
  audit_title,
  audit_title_link,
  audit_status,
  audit_rating,
  plan_scope,
  Audit_Type,
  Audit_Classification,
  actual_effort ,
  audit_closure_date,
  budgeted_effort,
  audit_report_id,
  audit_start_date,
  open_issues,
  version_no,
  report_json_audit_title,
  report_json_plan_scope
FROM
  (SELECT audit_id,
    audit_title ,
    a.audit_title AS audit_title_link ,
    (SELECT status_name FROM ms_aud_status WHERE status_id = a.audit_status
    ) AS audit_status ,
    (SELECT ms_apps_utilities.get_display_value(ms_apps_utilities.get_enterprise_id (ms_apps_utilities.get_infolet_id('MS_AUD_MANAGE_AUDIT')), 'MS AUD Audit Rating',audit_rating,'1',nvl(cast(si_db_locale_sv.getvalue as number),1009)) AS audit_rating
    FROM
      (SELECT audit_rating,
        row_number() over (partition by audit_id order by task_id DESC) AS rnum,
        audit_id
      FROM ms_aud_audit_report
      )
    WHERE rnum  =1 
    AND audit_id=a.audit_id
    )  AS Audit_Rating ,
    (SELECT s.status_name
  FROM ms_aud_plan_scope ps,
    ms_aud_status_tl_v s
  WHERE ps.audit_id= a.audit_id
  AND ps.form_type ='P'
  AND s.locale_id  =
    (SELECT b.locale_id
    FROM si_users_t a,
      si_locales b
    WHERE a.locale  = b.locale_string
    AND a.user_name = :LOGIN_ID
    )
  AND PS.STATUS = S.STATUS_ID
  ) AS plan_scope ,
    a.audit_start_date AS Audit_Start_Date ,
    (select code_description from ms_aud_lov_config_tl_v where locale_id=nvl(cast(si_db_locale_sv.getvalue as number),1009) AND code_group = 'MS AUD Audit Type' and code_code = a.audit_type) AS Audit_Type,
    (select code_description from ms_aud_lov_config_tl_v where locale_id=nvl(cast(si_db_locale_sv.getvalue as number),1009) and code_group = 'MS AUD Classification' and code_code = a.audit_classification) AS Audit_Classification ,
    actual_effort ,
    planned_audit_length AS budgeted_effort ,
    audit_closure_date ,
    (SELECT audit_report_id
    FROM
      (SELECT audit_report_id,
        row_number() over (partition by audit_id order by task_id DESC) AS rnum,
        audit_id
      FROM ms_aud_audit_report
      )
    WHERE rnum  =1
    AND audit_id=a.audit_id
    )  AS audit_report_id ,
     ms_aud_f014_helper.fn_get_open_issue_for_ae(:auditable_entity_id,a.audit_id,:internal_object_id) AS open_issues,
     (select ae_version from ms_aud_auditable_entit where object_id = (SELECT core_objects from ms_aud_audit_aue 
      where audit_id=a.audit_id and core_objects IN  (SELECT object_id from ms_aud_auditable_entit 
      where internal_object_id=:internal_object_id
      and object_id<=:auditable_entity_id
      ) and rownum=1)) as version_no,
       ms_aud_F011_helper.fn_get_current_audit_urls_mod('MS_AUD_MANAGE_AUDIT',:LOGIN_ID,a.audit_id,a.audit_title,a.dd_process_instance_id) as report_json_audit_title,
        ms_aud_F011_helper.fn_get_current_audit_urls_mod('MS_AUD_PLAN_SCOPE',:LOGIN_ID,a.audit_id,'x',a.dd_process_instance_id) as report_json_plan_scope
  FROM ms_aud_audit a
  WHERE ms_aud_audit_security.check_view_access (a.AUDIT_ID,a.DD_OBJECT_TYPE,a.RESTRICT_ACCESS_TO,a.CONDUCTED_BY_ORG,:LOGIN_ID,'VIEW')='ACCESS'
  AND a.status NOT IN ('2','31','44','51','59','4')
  AND a.approve_flag  = '1'
  AND EXISTS
    (SELECT 1 FROM ms_aud_audit_aue WHERE audit_id=a.audit_id and core_objects IN (SELECT object_id from ms_aud_auditable_entit 
      where internal_object_id=:internal_object_id
      and object_id<=:auditable_entity_id
      )
    )
  ORDER BY audit_start_date DESC
  )
WHERE rownum<=5
