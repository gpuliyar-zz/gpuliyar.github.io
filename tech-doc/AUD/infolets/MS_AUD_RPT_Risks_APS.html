SELECT risk_id,
  risk_name,
  object_level,
  object_parent
FROM
  (SELECT object_name AS risk_name,
    object_id         AS risk_id,
    object_level,
    ms_aud_notification_utilities.fn_get_risk_display_values (object_parent) AS object_parent
  FROM ms_grc_risk_pub_v
    START WITH object_id IN
    (SELECT a.object_id
    FROM ms_grc_risk_pub_v a,
      ms_aud_grc_risk_process_v b
    WHERE a.object_id = b.tgt_obj_id
    AND b.src_obj_id IN
      (SELECT COLUMN_VALUE
      FROM TABLE (ms_grc_str_common_utilities.Split_String_Stored (:parent_id) )
      )

    AND :scope                                                                                                                                   = 1
    AND ms_aud_utilities.get_parameter_value_for_id (:x_audit_id, 'IS_AE_Risks_Relation_Required' )                                             <> 'N'
    AND NVL (TRUNC (a.valid_until), TRUNC (SYSDATE))                                                                                            >= TRUNC (SYSDATE)
    AND ms_apps_security_pkg.check_object_access (a.dd_object_type, a.restrict_access_to, a.owner_organizations, :dd_current_user_name, 'VIEW' ) = 'ACCESS'
    )
    CONNECT BY PRIOR object_id = object_parent
  UNION
  SELECT object_name AS risk_name,
    object_id        AS risk_id,
    object_level,
    ms_aud_notification_utilities.fn_get_risk_display_values (object_parent) AS object_parent
  FROM ms_grc_risk_pub_v
    START WITH object_id IN
    (SELECT a.object_id
    FROM ms_grc_risk_pub_v a

    WHERE :scope                                                                                                                                 = 1
    AND ms_aud_utilities.get_parameter_value_for_id (:x_audit_id, 'IS_AE_Risks_Relation_Required' )                                              = 'N'
    AND NVL (TRUNC (a.valid_until), TRUNC (SYSDATE))                                                                                            >= TRUNC (SYSDATE)
    AND ms_apps_security_pkg.check_object_access (a.dd_object_type, a.restrict_access_to, a.owner_organizations, :dd_current_user_name, 'VIEW' ) = 'ACCESS'
    )
    CONNECT BY PRIOR object_id = object_parent
  UNION
  SELECT DISTINCT object_name AS risk_name,
    object_id                 AS risk_id,
    object_level,
    ms_aud_notification_utilities.fn_get_risk_display_values (object_parent) AS object_parent
  FROM ms_grc_risk_pub_v
  WHERE object_id IN
    (SELECT ro.risk_id AS object_id
    FROM ms_aud_grc_risk_v ro
    WHERE ato_organizations IN
      (SELECT COLUMN_VALUE
      FROM TABLE (ms_grc_str_common_utilities.Split_String_Stored (:parent_id) )
      )

    AND :scope                                         = 2
    AND NVL (TRUNC (ro.valid_until), TRUNC (SYSDATE)) >= TRUNC (SYSDATE)
    )
  UNION
  SELECT object_name AS risk_name,
    object_id        AS risk_id,
    object_level,
    ms_aud_notification_utilities.fn_get_risk_display_values (object_parent) AS object_parent
  FROM ms_grc_risk_pub_v
  WHERE object_id IN
    (SELECT object_id
    FROM ms_grc_risk_pub_v
      START WITH object_id IN
      (SELECT tgt_obj_id AS object_id
      FROM ms_aud_grc_process_risk_v
      WHERE src_obj_id IN
        (SELECT COLUMN_VALUE
        FROM TABLE (ms_grc_str_common_utilities.Split_String_Stored (:parent_id) )
        )

      AND :scope = 3
      )
      CONNECT BY PRIOR object_id = object_parent
    )
  AND NVL (TRUNC (valid_until), TRUNC (SYSDATE)) >= TRUNC (SYSDATE)
  )
ORDER BY ROWNUM ASC
