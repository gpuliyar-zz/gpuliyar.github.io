SELECT   a.audit_id, a.audit_title,
         a.audit_title AS audit_title_link,
         NVL(MS_AUD_PLANNING_RPT_UTILS.fn_get_plan_title(a.plan_title),'Adhoc') AS plan_title,
         MS_AUD_EXECUTION_RPT_UTILS.get_code_description
                                          ('MS AUD Audit Type',
                                           a.audit_type
                                          ) AS audit_type,
         MS_AUD_EXECUTION_RPT_UTILS.get_code_description
                                  ('MS AUD Classification',
                                   a.audit_classification
                                  ) AS audit_category,
         MS_AUD_REPORT_UTILS.fn_get_status_display_name
                                                (a.audit_status)
                                                              AS audit_status,
          ms_aud_report_utils.fn_get_audit_core_objects(a.audit_id,a.SCOPE_BASED_ON)  AS core_objects,
         ms_Aud_report_utils.fn_get_audit_related_risks(a.audit_id,a.SCOPE_BASED_ON) AS risks,
         ms_Aud_report_utils.fn_get_audit_organizations(a.audit_id,a.SCOPE_BASED_ON)  AS organizations,
            MS_GRC_COMMON_UTILS.Fn_Get_Org_Display_Names(conducted_by_org) as conducted_org,
         NVL(ms_apps_utilities.get_user_full_name (a.lead_auditor),MS_AUD_REPORT_UTILS.fn_get_mlov_lead_aud_disp(MS_GRC_COMMON_UTILS.Fn_Get_Val_By_Id(a.audit_id,'ms_aud_audit_alm','lead_auditor_mlov','AUDIT_ID'))) AS 
lead_auditor,NVL(ms_apps_utilities.get_user_full_name (a.audit_manager),MS_AUD_REPORT_UTILS.fn_get_mlov_aud_man_disp(MS_GRC_COMMON_UTILS.Fn_Get_Val_By_Id(a.audit_id,'ms_aud_audit_amm','audit_manager_mlov','audit_id'))) AS 
audit_manager,
         ms_apps_utilities.get_user_full_name (a.dd_created_by) AS created_by,
         TO_CHAR (a.audit_start_date, 'MM/DD/YYYY') AS audit_start_date,
         MS_AUD_EXECUTION_RPT_UTILS.fn_get_task_count (a.audit_id, :LOGIN_ID) AS view_tasks, a.restrict_access_to,
         MS_AUD_REPORT_UTILS.mlov_values (a.audit_id,
                                       'audit_id',
                                       'team_member',
                                       'ms_aud_audit_tmm'
                                      ) AS team_member , decode(status,3,'START',null) as starts,decode(status,20,'SCH',null) as  SCH,
                                      ms_aud_report_utils.fn_get_report_json(
  (SELECT metric_id FROM si_metrics_t WHERE metric_name='MS_AUD_MANAGE_AUDIT'
  ),
  a.dd_process_instance_id,
  NULL,
  'flag=1'
  ||CHR(38)
  ||'emd=2'
  ||CHR(38)
  ||'x_edit_mode=No'
  || CHR (38)
  || 'audit_id='
  || a.audit_id) AS report_json,
  NVL(a.lead_auditor,MS_GRC_COMMON_UTILS.Fn_Get_Val_By_Id(a.audit_id,'ms_aud_audit_alm','lead_auditor_mlov','audit_id')) AS 
lead_auditor_stored,
NVL(a.audit_manager,MS_GRC_COMMON_UTILS.Fn_Get_Val_By_Id(a.audit_id,'ms_aud_audit_amm','audit_manager_mlov','audit_id')) AS 
audit_manager_stored,
a.dd_created_by as created_by_stored,a.audit_end_date as audit_end_date
     FROM ms_aud_audit a
 where
    ms_aud_audit_security.check_view_access(a.audit_id,a.dd_object_type,a.restrict_access_to,a.conducted_by_org,:LOGIN_ID,'VIEW') = 'ACCESS'
       and  audit_start_date between to_date(:audit_start_after)+1 and to_date(:audit_after_before)
  order by a.audit_title
