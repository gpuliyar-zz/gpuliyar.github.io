SELECT a.audit_title,
       a.audit_manager,
       (select user_full_name from ms_aud_users_v u where u.user_name = a.audit_manager) audit_manager_disp,
       a.lead_auditor,
       (select user_full_name from ms_aud_users_v u where u.user_name = a.lead_auditor)  lead_auditor_disp,
       a.conducted_by_org,
       (select org_entity_name from si_org_entities_t where org_entity_id=a.conducted_by_org and rownum=1) conducted_by_org_disp,
       TRUNC(a.audit_start_date) AS audit_date,
       DECODE(CROSS_FUNCTIONAL,1,(select REPLACE(ms_join(audit_manager_mlov),',','~') from ms_aud_audit_amm amm where amm.audit_id=a.audit_id),null) audit_manager_mlov_sto,
       DECODE(CROSS_FUNCTIONAL,1, (select MS_JOIN_FOR_DISPLAY(u.first_name ||' '|| u.last_name)
            from si_users_t u
                ,ms_aud_audit_amm amm
            where amm.audit_id = a.audit_id
               and amm.audit_manager_mlov = u.user_name
          ),null) audit_manager_mlov_disp,
        DECODE(CROSS_FUNCTIONAL,1,(select REPLACE(ms_join(lead_auditor_mlov),',','~') from ms_aud_audit_alm alm where alm.audit_id=a.audit_id),null) lead_auditor_mlov_sto,
       DECODE(CROSS_FUNCTIONAL,1, (select MS_JOIN_FOR_DISPLAY(u.first_name ||' '|| u.last_name)
            from si_users_t u
                ,ms_aud_audit_alm alm
            where alm.audit_id = a.audit_id
               and alm.lead_auditor_mlov = u.user_name
       ),null) lead_auditor_mlov_disp,
       COALESCE(a.audit_classification,'01') AS audit_classification,
  (SELECT status_name FROM ms_aud_status b WHERE b.status_id = a.STATUS) as status,
 a.STATUS as status_stored_value,
TRUNC(a.audit_end_date) AS end_date,
(select code_description from ms_aud_lov_config where code_group='MS AUD Classification' and code_code = a.audit_classification) as audit_class_display,
(select code_description from ms_aud_lov_config where code_group='MS AUD Audit Type' and code_code = a.audit_type) as audit_type,
a.audit_id,
NVL(a.CROSS_FUNCTIONAL,'2') AS CROSS_FUNCTIONAL
  FROM ms_aud_audit a
 WHERE a.audit_id =:i_audit_id
