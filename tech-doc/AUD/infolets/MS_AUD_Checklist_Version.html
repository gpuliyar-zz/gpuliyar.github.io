SELECT cm.checklist_name,
       cm.version_no,       ms_apps_utilities.get_display_value(ms_apps_utilities.get_enterprise_id(ms_apps_utilities.get_infolet_id('MS_CHK_CHECKLIST_MASTER')),
                                            'MS CHK Checklist Type',
                                            CM.CHECKLIST_TYPE,
                                            1,
                                            NULL,                                            'ms_apps_utilities.get_enterprise_name(ms_apps_utilities.get_enterprise_id 
(ms_apps_utilities.get_infolet_id (''MS_CHK_CHECKLIST_MASTER'') ))') AS checklist_type,
       (select ms_concat(tgt_obj_name)
          from ms_chk_chklist_master_orb
         where tgt_obj_type = 'Auditable Entities'
           and checklist_id = cm.checklist_id) as auditable_entities,
       (select ms_concat(tgt_obj_name)
          from ms_chk_chklist_master_orb
         where tgt_obj_type = 'Risk(s)'
           and checklist_id = cm.checklist_id) as risks,
       (select ms_concat(tgt_obj_name)
          from ms_chk_chklist_master_orb
         where tgt_obj_type = 'Control(s)'
           and checklist_id = cm.checklist_id) as controls,
       null as Test_procedures,
       (select ms_concat(tgt_obj_name)
          from ms_chk_chklist_master_orb
         where tgt_obj_type = 'Applies to Organization(s)'
           and checklist_id = cm.checklist_id) as related_to_organizations,
       (select ms_concat(tgt_obj_name)
          from ms_chk_chklist_master_orb
         where tgt_obj_type = 'Process(es)'
           and checklist_id = cm.checklist_id) as Processes,
           (select ms_concat(tgt_obj_id)
          from ms_chk_chklist_master_orb
         where tgt_obj_type = 'Auditable Entities'
           and checklist_id = cm.checklist_id) as auditable_entities_id,
           (select ms_concat(tgt_obj_id)
          from ms_chk_chklist_master_orb
         where tgt_obj_type = 'Risk(s)'
           and checklist_id = cm.checklist_id) as risks_id,
       (select ms_concat(tgt_obj_id)
          from ms_chk_chklist_master_orb
         where tgt_obj_type = 'Control(s)'
           and checklist_id = cm.checklist_id) as controls_id,       
       (select ms_concat(tgt_obj_id)
          from ms_chk_chklist_master_orb
         where tgt_obj_type = 'Applies to Organization(s)'
           and checklist_id = cm.checklist_id) as related_to_organizations_id,
       (select ms_concat(tgt_obj_id)
          from ms_chk_chklist_master_orb
         where tgt_obj_type = 'Process(es)'
           and checklist_id = cm.checklist_id) as Processes_id,
 cm.checklist_type as checklist_type_Stored,
cm.checklist_id as Checklist_ID
  from ms_chk_chklist_master cm
 where (cm.restrict_access_to = 'N' OR
       (cm.restrict_access_to = 'O' AND
       :x_owner_org IN (SELECT b.owner_organizations
                      FROM ms_chk_chklist_master_org b
                     WHERE b.checklist_id = cm.checklist_id)))
   and (CASE
         WHEN VALID_FROM IS NULL THEN
          2
         WHEN VALID_FROM >= SYSDATE THEN
          2
         WHEN VALID_UNTILL IS NULL THEN
          1
         WHEN VALID_UNTILL >= SYSDATE THEN
          1
         ELSE
          2
       END) = 1
       and cm.checklist_name=:checklist_name ORDER BY VERSION_NO
