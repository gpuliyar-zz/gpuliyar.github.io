AE ID trim(upper(OBJECT_ID)) like upper(:Auditable Entity ID) Auditable Entity ID filter
AERep AE Name trim(upper(OBJECT_NAME)) like upper(:Auditable Entity Name) Auditable Entity Name
AERep AE Status trim(upper(STATUS)) like upper(:Status)
AERep AE Type trim(upper(PROCESS_TYPE)) like upper(:Type) Auditable Entity Type
AERep AE ATO EXISTS (SELECT column_value FROM TABLE(ms_apps_utilities.parse_string(APPLIES_TO_ORGANIZATIONS_ID,',')) WHERE column_value IN :Applies to Organizations ) Applies to Organization
AERep AE Risk EXISTS (SELECT column_value FROM TABLE(ms_apps_utilities.parse_string(RELATED_RISKS_ID,',')) WHERE column_value IN :Related Risks )
AERep AE Req EXISTS (SELECT column_value FROM TABLE(ms_apps_utilities.parse_string(RELATED_REQUIREMENTS_ID,',')) WHERE column_value IN :Related Requirements )
AERep AE Fra Ref EXISTS (SELECT column_value FROM TABLE(ms_apps_utilities.parse_string(RELATED_FRAMEWORK_REF_ID,',')) WHERE column_value IN :Related Framework References )
AERep AE Obj EXISTS (SELECT column_value FROM TABLE(ms_apps_utilities.parse_string(RELATED_OBJECTIVES_ID,',')) WHERE column_value IN :Related Objectives )
AERep AE Rel Ref EXISTS (SELECT column_value FROM TABLE(ms_apps_utilities.parse_string(RELATED_REFERENCES_ID,',')) WHERE column_value IN :Related References )
AERep AE Func EXISTS (SELECT column_value FROM TABLE(ms_apps_utilities.parse_string(RELATED_FUNCTIONS_ID,',')) WHERE column_value IN :Related Functions )
AERep AE Exp Item (trim(EXPIRED_FLAG) <= :Include Expired Items) Expired Item
AERep AE Fin Acc EXISTS (SELECT column_value FROM TABLE(ms_apps_utilities.parse_string(RELATED_FINANCIAL_ACC_ID,',')) WHERE column_value IN :Related Financial Accounts )
AERep Info AOC EXISTS (SELECT column_value FROM TABLE(ms_apps_utilities.parse_string(RELATED_AREAS_OF_CMPLNCE_ID,',')) WHERE column_value IN :Related Areas of Compliance )
AEBusinessEntity EXISTS (SELECT column_value FROM TABLE(ms_apps_utilities.parse_string(GD_BUSINESS_ENTITIES,',')) WHERE column_value IN :Business Entities) OR EXISTS(SELECT column_value FROM TABLE(ms_apps_utilities.parse_string(GE_BUSINESS_ENTITIES,',')) WHERE column_value IN :Business Entities) Business Entities Filter
AELegalEntites EXISTS (SELECT column_value FROM TABLE(ms_apps_utilities.parse_string(GD_LEGAL_ENTITIES,',')) WHERE column_value IN :Legal Entities) OR EXISTS(SELECT column_value FROM TABLE(ms_apps_utilities.parse_string(GE_LEGAL_ENTITIES,',')) WHERE column_value IN :Legal Entities) Legal Entities
