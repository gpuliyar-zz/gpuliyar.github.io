SELECT plan_id,plan_title FROM
(SELECT 'AD-ID' as plan_id, 
       'Adhoc Audit' AS plan_title,
       locale_id as locale_id
  FROM si_enabled_locales
UNION ALL
SELECT plan_id, plan_title,l.locale_id
  FROM (SELECT   internal_plan_id AS plan_id, 
                 plan_title
  FROM ms_aud_audit_plan 
           WHERE STATUS = '3'
        GROUP BY INTERNAL_PLAN_ID, PLAN_TITLE),SI_ENABLED_LOCALES L)
  WHERE PLAN_ID=:PLAN_ID
