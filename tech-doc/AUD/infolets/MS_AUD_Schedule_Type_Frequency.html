SELECT frequencey,YEAR,NO 
  FROM (SELECT DISTINCT :plan_type as plan_type, :end_date as end_date, 'Quarter '||TO_CHAR((LAST_DAY(TO_DATE(td.end_date+1-ROWNUM))),'Q')||' '||TO_CHAR((LAST_DAY(TO_DATE(td.end_date+1-ROWNUM))),'YYYY') AS frequencey,
                        TO_CHAR((LAST_DAY(TO_DATE(td.end_date+1-ROWNUM))),'YYYY') AS YEAR,
                        TO_CHAR((LAST_DAY(TO_DATE(td.end_date +1-ROWNUM))),'Q') AS NO
                   FROM all_objects,
                        (SELECT :start_date start_date,
                                TO_DATE('31-12-'||TO_CHAR(TO_DATE(ADD_MONTHS(TRUNC(:start_date),12)),'yyyy'),'dd-mm-yyyy') end_date
                           FROM DUAL) td
                  WHERE TRUNC(td.end_date+1-ROWNUM,'MM')>=TRUNC(td.start_date,'MM'))
   WHERE :sch_type='Quarter'
UNION
SELECT frequencey,YEAR,NO
    FROM (SELECT DISTINCT :plan_type as plan_type, :end_date as end_date, TO_CHAR((LAST_DAY(TO_DATE(td.end_date+1-ROWNUM))),'Mon')||' '||TO_CHAR((LAST_DAY(TO_DATE(td.end_date+1-ROWNUM))),'YYYY') AS frequencey,
                          TO_CHAR((LAST_DAY(TO_DATE(td.end_date+1-ROWNUM))),'YYYY') AS YEAR,
                          TO_CHAR((LAST_DAY(TO_DATE(td.end_date+1-ROWNUM))),'MM') AS NO
                     FROM all_objects,
                          (SELECT :start_date start_date,
                                  TO_DATE ('31-12-'||TO_CHAR(TO_DATE(ADD_MONTHS(TRUNC(:start_date),12)),'yyyy'),'dd-mm-yyyy') end_date
                             FROM DUAL) td
                    WHERE TRUNC(td.end_date+1-ROWNUM,'MM')>=TRUNC(td.start_date,'MM'))
   WHERE :sch_type='Month'
UNION
SELECT 'Time Window '||ms_aud_planning_rpt_utils.fn_get_time_window(q)||' '||TO_CHAR(TO_DATE(e),'yyyy') AS frequencey,TO_CHAR(TO_DATE (e),'yyyy') AS YEAR,TO_CHAR(ms_aud_planning_rpt_utils.fn_get_time_window(q)) AS NO
  FROM (SELECT :plan_type as plan_type, :end_date as end_date, ROWNUM q,TO_CHAR((td.start_date+ms_aud_planning_rpt_utils.fn_get_time_window_days(ROWNUM - 1)),'dd-mon-yyyy') AS s,TO_CHAR((td.start_date+ms_aud_planning_rpt_utils.fn_get_time_window_days(ROWNUM))-1,'dd-mon-yyyy') AS e
          FROM all_objects,
               (SELECT TO_DATE('1-1-'||TO_CHAR(:start_date,'yyyy'),'dd-mm-yyyy') AS start_date,
                       TO_DATE('31-12-'||TO_CHAR(TO_DATE(ADD_MONTHS(TRUNC(:start_date),12)),'yyyy'),'dd-mm-yyyy') end_date
                    FROM DUAL) td
         WHERE TRUNC (td.end_date+1-ROWNUM,'MM')>=TRUNC(td.start_date,'MM'))
   WHERE TO_DATE (e,'dd-mon-yyyy')<=TRUNC(TO_DATE('31-12-'||TO_CHAR(TO_DATE(ADD_MONTHS(TRUNC(:start_date),12)),'yyyy'),'dd-mm-yyyy'))
     AND ((TO_DATE(s,'dd-mon-yyyy')>TO_DATE(TRUNC(:start_date))) OR TO_DATE(TRUNC(:start_date)) BETWEEN TO_DATE(s,'dd-mon-yyyy') AND TO_DATE(e,'dd-mon-yyyy'))
     AND :sch_type = 'TW'
ORDER BY 2, 3
