SELECT DISTINCT b.checklist_id, b.checklist_name, b.version_no,
                ms_apps_utilities.get_display_value
                   (ms_apps_utilities.get_enterprise_id
                       (ms_apps_utilities.get_infolet_id
                                                    ('MS_CHK_CHECKLIST_MASTER')
                       ),
                    'MS CHK Checklist Type',
                    b.checklist_type,
                    1,
                    NULL,
                    'ms_apps_utilities.get_enterprise_name(ms_apps_utilities.get_enterprise_id 
(ms_apps_utilities.get_infolet_id (''MS_CHK_CHECKLIST_MASTER'') ))'
                   ) AS checklist_type,
                ms_apps_utilities.get_display_value
                   (ms_apps_utilities.get_enterprise_id
                       (ms_apps_utilities.get_infolet_id
                                                    ('MS_CHK_CHECKLIST_MASTER')
                       ),
                    'MS CHK Scoring Method',
                    b.scoring_method,
                    1,
                    NULL,
                    'ms_apps_utilities.get_enterprise_name(ms_apps_utilities.get_enterprise_id 
(ms_apps_utilities.get_infolet_id (''MS_CHK_CHECKLIST_MASTER'') ))'
                   ) AS scoring_method,
                MS_GRC_COMMON_UTILS.Fn_Get_Org_Display_Names
                   (MS_GRC_COMMON_UTILS.Fn_Get_Val_By_Id
                                                  (c.checklist_id,
                                                   'ms_chk_chklist_master_org','owner_organizations','CHECKLIST_ID'
                                                  )
                   ) AS owner_organizations,
                ms_chk_utilities.fn_get_user_disp_val
                   (MS_GRC_COMMON_UTILS.Fn_Get_Val_By_Id
                                                  (d.checklist_id,
                                                  'ms_chk_chklist_master_own',
                                                   'owners',
                                                   'CHECKLIST_ID'
                                                  )
                   ) AS owners,
                ms_apps_utilities.get_user_full_name
                                               (b.dd_created_by)
                                                                AS created_by,
               'Edit' AS edit,
                ms_chk_utilities.include_expired
                                       (TRUNC (b.valid_from),
                                        TRUNC (b.valid_untill)
                                       ) AS expired_flag,
           ms_aud_report_utils.fn_get_report_json(
  (SELECT metric_id FROM si_metrics_t WHERE metric_name='MS_CHK_CHECKLIST_MASTER'
  ),
  b.dd_process_instance_id,
  b.dd_instance_id,
  'flag=1'
  ||CHR(38)
  ||'emd=1'
  ||CHR(38)
  ||'edit_flag=No'
  ) AS report_json,
                MS_GRC_COMMON_UTILS.Fn_Get_Val_By_Id(d.checklist_id,'ms_chk_chklist_master_own' ,'owners','CHECKLIST_ID') AS owners_stored,
                b.dd_created_by AS created_by_stored
           FROM ms_chk_chklist_master b,
                ms_chk_chklist_master_org c,
                ms_chk_chklist_master_own d,
                ms_aud_chk_question_qst e
            where
                e.question_id=:question_id 
            AND e.qst_checklist_id=b.checklist_id AND e.qst_checklist_id=c.checklist_id
            AND e.qst_checklist_id=d.checklist_id  AND b.checklist_id = c.checklist_id
            AND b.checklist_id = d.checklist_id AND d.checklist_id = c.checklist_id
AND ms_apps_security_pkg.check_object_access (b.dd_object_type, b.restrict_access_to,
MS_GRC_COMMON_UTILS.Fn_Get_Val_By_Id (c.checklist_id,'ms_chk_chklist_master_org','owner_organizations','CHECKLIST_ID'),:LOGIN_ID,'VIEW')='ACCESS'
