SELECT 
    v.issue_popup_link AS ISSUE_LINK,
    V.ISSUE_TITLE       AS ISSUE_TITLE,
    T.TGT_OBJ_ID              AS OBJECT_ID,
    V.ISSUE_STATUS            AS ISSUE_STATUS,
    V. issue_owner_full_name    as ISSUE_OWNER,
    V.ISSUE_RATING            AS ISSUE_RATING,
    V.PRIORITY                AS ISSUE_PRIORITY,
    V.ISSUE_SOURCE_TYPE       AS ISSUE_SOURCE_TYPE,
    V.ISSUE_DUE_DATE          AS ISSUE_DUE_DATE,
    V.created_by_full_name              AS CREATED_BY,
    V.ISSUE_IDENTIFIED_ON     AS ISSUE_IDENTIFIED_ON,
    V.ISSUE_ID                AS ISSUE_ID,
    V.NO_OF_ACTIONS_TRIGGERED AS NO_OPEN_ACTION,
    V.ISSUE_OWNERORG          AS ISSUE_OWNER_ORG,
    V.INITIAL_APPROVER_ORG    AS INTITIAL_APPROVER_ORG,
    V.PROGRESS_STATUS         AS PROGRESS_STATUS,
    V.RELATED_AREA_OF_COMP AS RELATED_AREA_OF_COMP ,
    V.RELATED_RISKS as RELATED_RISKS,
    V.EXCEPTION_TYPE AS EXCEPTION_TYPE,
    V.RELATED_CONTROLS as RELATED_CONTROLS,
    V.RELATED_CORE_PROCESSES as RELATED_CORE_PROCESSES,
    V.RELATED_ORGS as RELATED_ORGS,
    V.RELATED_STANDARD as RELATED_STANDARD,
    V. RELATED_ASSET as RELATED_ASSET,
    V.RELATED_ASSET_CLASS as RELATED_ASSET_CLASS,
    V.RELATED_FINANCIAL_ACOUNT as RELATED_FINANCIAL_ACOUNT,
    V.RELATED_FUNCTION as RELATED_FUNCTION ,
    V.RELATED_FRAMEWORK_REF as RELATED_FRAMEWORK_REF,
    V.RELATED_OBJECTIVE as RELATED_OBJECTIVE,
    V.RELATED_PROCESS as RELATED_PROCESS,
    V.RELATED_REFERENCE as RELATED_REFERENCE,
    V.RELATED_REGULATORY_BODY as RELATED_REGULATORY_BODY,
    V.RELATED_REQUIREMENT as RELATED_REQUIREMENT,
    V.RELATED_QUESTION_PROCEDURE as RELATED_QUESTION_PROCEDURE,
    V.RELATED_SUPPLIER as RELATED_SUPPLIER,
    V.RELATED_AUDITABLE_ENTITY as RELATED_AUDITABLE_ENTITY,
    V.RELATED_PRODUCT as RELATED_PRODUCT,
    V.CREATED_ON as CREATED_ON,
    V.ISSUE_SRC_TYPE_STORED  AS ISSUE_SRC_TYPE_STORED,
    V.ISSUE_TYPE as ISSUE_TYPE,
   (SELECT ae_version from ms_aud_auditable_entit where object_id= T.TGT_OBJ_ID)  as version_no 
  FROM MS_ISM_INTG_ISSUE_PUB_V V,
    MS_ISM_ISSUE_ORB_REL_PUB_V T
  WHERE V.ISSUE_ID = T.ISSUE_ID
AND V.REACHED_TO_MANAGE = 'Y'
AND (1=ms_ism_security_pkg.FN_ISM_SECURITIES(V.ISSUE_ID,:LOGIN_ID))
AND T.TGT_OBJ_ID IN (SELECT object_id from ms_aud_auditable_entit 
      where internal_object_id=:INTERNAL_OBJECT_ID
      and object_id<=:OBJECT_ID)
and v.Issue_Src_Type_Stored='1'
AND ms_apps_utilities.get_parameter_value
                                                 ('',
                                                  'MS AUD Integrated Modules',
                                                  'Is_ISM_Integrated'
                                                 )
              = 'Y'
