SELECT a.object_id AS object_id,a.object_name AS object_name ,
               trim(substr(ms_aud_advisor_utilities.fn_get_rating(MS_AUD_ADVISOR_UTILITIES.FN_GET_PROCESS_SCORE (a.object_id,:Scenario),a.object_id,:Scenario,'C',1),1,
              instr(ms_aud_advisor_utilities.fn_get_rating(MS_AUD_ADVISOR_UTILITIES.FN_GET_PROCESS_SCORE (a.object_id,:Scenario),a.object_id,:Scenario,'C',1),' '))) AS risk_rating,
              (SELECT max(audit_start_date) keep (dense_rank first order by to_number(audit_status))
                   FROM ms_aud_audit WHERE audit_id IN (
                           SELECT b.audit_id FROM ms_aud_audit_aue b
,ms_aud_audit ad 
                            WHERE b.core_objects = a.object_id and ad.audit_id=b.audit_id and ad.audit_status not in (4))
                    AND audit_start_date is not null) AS last_audited_on,
                NVL ((SELECT COUNT (b.audit_id) FROM ms_aud_audit_aue b,ms_aud_audit ad
                       WHERE b.core_objects = a.object_id
                       and ad.audit_id=b.audit_id
                       and ad.audit_status not in (4,8)),0) AS no_of_audits_till_date,
                ms_aud_advisor_utilities.fn_get_open_issue(a.object_id,'Y') AS open_issues,
               NULL AS FREE1,NULL AS FREE2,NULL AS FREE3,to_date(MS_AUD_PLANNING_RPT_UTILS.fn_get_last_assessment_date(:Scenario,a.object_id)) AS FREE4, ms_aud_report_utils.fn_get_report_json((SELECT metric_id FROM si_metrics_t WHERE metric_name = a.dd_object_type),a.dd_process_instance_id,a.dd_instance_id,'edit_flag=N'||CHR(38)||'flag=2'||CHR(38)||'emd=1') AS Report_Json FROM ms_aud_grc_core_object_pub_v a, MS_GRC_CORE_OBJECT_PUB_V d
          WHERE a.object_id = d.object_id
         AND (a.restrict_access_to IN ('N') OR (MS_GRC_STR_COMMON_UTILITIES.Is_String_exists_in_list
                                     (a.owner_organizations,MS_AUD_REPORT_UTILS.fn_get_org_entity_id(:LOGIN_ID)) = 1))
         AND  TRUNC(NVL(a.VALID_UNTIL,SYSDATE)) >= TRUNC(SYSDATE)
     AND ((d.object_id IN (
                SELECT MAX(a.OBJECT_ID) FROM ms_aud_auditable_entit a,ms_grc_core_object_pub_v b 
                where a.object_id=b.object_id
                and b.obj_status not in ('INACT','EXP')
                GROUP BY internal_object_id      
      ) 
          AND d.dd_object_type='MS_AUD_AUDITABLE_ENTITY') OR d.dd_object_type<>'MS_AUD_AUDITABLE_ENTITY')     
         ORDER BY MS_AUD_PLANNING_RPT_UTILS.get_risk_score (a.object_id,:Scenario) desc NULLS LAST
