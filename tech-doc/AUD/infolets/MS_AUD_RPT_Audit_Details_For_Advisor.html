MS_AUD_RPT_Audit_Details_For_Advisor Report SYSTEMI 
SELECT a.audit_id as audit_id,a.audit_title as audit_title,a.Audit_Title_Link as Audit_Title_Link,a.Plan_Title as Plan_Title,a.Audit_Type as Audit_Type,
a.Audit_Category as Audit_Category,a.Audit_Status as Audit_Status ,a.Core_Objects as Core_Objects , related_risks as risks,
a.organizations as organizations,case when a.lead_auditor is null then la.la_disp_name else (select user_full_name from ms_aud_users_v where user_name=a.lead_auditor) end as lead_auditor,case when a.audit_manager is null then ma.ma_disp_name else (select user_full_name from ms_aud_users_v where user_name=a.audit_manager) end audit_manager 
,a.Created_By as Created_By,
a.Audit_Start_Date as Audit_Start_Date,a.view_tasks as view_tasks,a.restrict_access_to as restrict_access_to,a.Team_Member as Team_Member,ms_aud_report_utils.fn_get_report_json((SELECT metric_id FROM si_metrics_t WHERE metric_name = 'MS_AUD_MANAGE_AUDIT'),a.dd_process_instance_id,a.dd_instance_id,'flag=1'||CHR(38)||'emd=2'||CHR(38)||'x_edit_mode=No'||CHR(38)||'audit_id='||a.audit_id||CHR(38)) as report_json,
a.lead_auditor as lead_auditor_ID,case when a.audit_status ='Final Report' then (select url_link from (select url_link,audit_id,dense_rank() over(partition by audit_id order by published_date desc) as rnk From MS_AUD_AUDIT_PUBLISHED_REPORTS) c  where c.audit_id=a.audit_id and rnk =1) end as download,case when a.lead_auditor is null then la.LEAD_AUDITOR_MLOV else (select user_name from ms_aud_users_v where user_name = a.lead_auditor) end as lead_auditor_stored,case when a.audit_manager is null then ma.AUDIT_MANAGER_MLOV else (select user_name from ms_aud_users_v where user_name = a.audit_manager) end audit_manager_stored,s.issues,a.audit_end_date as audit_end_date
from (SELECT Created_By,Audit_Start_Date,view_tasks,team_member,dd_process_instance_id,dd_instance_id,audit_end_date,DD_OBJECT_TYPE,restrict_access_to,conducted_by_org,status,approve_flag,audit_id,audit_title,Audit_Title_Link,Plan_Title,Audit_Type,Audit_Category,Audit_Status,Core_Objects,related_risks,organizations,lead_auditor,audit_manager FROM ms_aud_audit_list_report_v WHERE audit_id IN ((SELECT COLUMN_VALUE AS audit_id
            FROM TABLE (ms_grc_str_common_utilities.Split_String_Stored (:audit_id))))) a LEFT OUTER JOIN (SELECT AUDIT_ID,MS_JOIN_FOR_DISPLAY(LEAD_AUDITOR_MLOV) LEAD_AUDITOR_MLOV,MS_JOIN_FOR_DISPLAY(user_full_name) la_disp_name 
FROM MS_AUD_AUDIT_ALM alm,ms_aud_users_v u where u.user_name = alm.LEAD_AUDITOR_MLOV group by AUDIT_ID)LA ON (A.AUDIT_ID = LA.AUDIT_ID) LEFT OUTER JOIN  
(SELECT AUDIT_ID,MS_JOIN_FOR_DISPLAY(AUDIT_MANAGER_MLOV) AUDIT_MANAGER_MLOV,MS_JOIN_FOR_DISPLAY(user_full_name) ma_disp_name FROM MS_AUD_AUDIT_AMM amm,ms_aud_users_v u where u.user_name = amm.AUDIT_MANAGER_MLOV GROUP BY AUDIT_ID)MA ON (A.AUDIT_ID = MA.AUDIT_ID)
LEFT OUTER JOIN (SELECT audit_id,count(1) issues from TABLE (MS_AUD_REPORTING_RPT_UTILS.get_issue_data)
group by audit_id)s on(a.audit_id=s.audit_id)
WHERE ms_aud_audit_security.check_view_access (a.AUDIT_ID,a.DD_OBJECT_TYPE,a.RESTRICT_ACCESS_TO,a.CONDUCTED_BY_ORG,:LOGIN_ID,'VIEW')='ACCESS' AND a.status NOT IN ('2','31','44','51','59') and a.approve_flag = '1' order by a.audit_title
