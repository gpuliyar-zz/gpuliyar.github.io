MS_AUD_RPT_Issues_Under_Internal_Review Report SYSTEMI 
SELECT DISTINCT   B.AUDIT_TITLE AS AUDIT_TITLE_DISP,
MS_AUD_REPORT_UTILS.fn_get_mlov_aud_man_disp
  (NVL (b.lead_auditor,
   (MS_AUD_REPORT_UTILS.mlov_values (b.audit_id,
   'audit_id',
   'LEAD_AUDITOR_MLOV',
   'ms_aud_audit_alm'
   ))) ) AS lead_auditor,
  MS_AUD_REPORT_UTILS.fn_get_mlov_aud_man_disp
   (NVL (b.audit_manager,
   (MS_AUD_REPORT_UTILS.mlov_values (a.audit_id,
   'audit_id',
   'AUDIT_MANAGER_MLOV',
   'ms_aud_audit_amm')))) AS audit_manager,
   MS_AUD_REPORT_UTILS.fn_get_mlov_aud_man_disp(issue_reviewer) AS reviewer,
    (select status_name from ms_aud_status_tl_v where locale_id = (select b.locale_id from si_users_t a, si_locales b 
      where a.locale = b.locale_string and a.user_name = :LOGIN_ID) AND status_id = a.status) AS current_status,
ms_aud_report_utils.fn_get_report_json(
        (select metric_id from si_metrics_t where metric_name ='MS_AUD_AUDIT_ISSUE_REV_TASK'),
        a.process_instance_id,
        null,
        decode(MS_AUD_REPORTING_RPT_UTILS.FN_SEND_ISSUES_REV_EDIT(:LOGIN_ID,A.PROCESS_INSTANCE_ID),'Y','flag=1'|| CHR (38)|| 'emd=4'|| CHR (38) || 'x_edit_mode=Yes'|| CHR (38) || 'audit_id='	|| a.audit_id,
                'flag=1'|| CHR (38)	|| 'emd=2'|| CHR (38)|| 'x_edit_mode=No'|| CHR (38)	|| 'audit_id=' || a.audit_id)
      ) AS report_json,
	  NVL(b.lead_auditor,(MS_AUD_REPORT_UTILS.mlov_values(b.audit_id,
           'audit_id',
           'LEAD_AUDITOR_MLOV',
           'ms_aud_audit_alm')))  as lead_auditor_stored,
  NVL(b.audit_manager,
      (MS_AUD_REPORT_UTILS.mlov_values(a.audit_id,
           'audit_id',
           'AUDIT_MANAGER_MLOV',
           'ms_aud_audit_amm')))  as  audit_manager_stored,
issue_reviewer AS REVIEWER_STORED
           FROM ms_aud_audit_issue_rev_task_v a, ms_aud_audit b
 WHERE instance_id = (SELECT MAX (instance_id)
                        FROM ms_aud_audit_issue_rev_task_v
                       WHERE audit_id = a.audit_id and process_instance_id=a.process_instance_id)
                       and A.AUDIT_ID=B.AUDIT_ID
   AND a.instance_rec_num = 1
   AND review_type = 1
   and a.status in (24,29)
   and b.status not in (4)
  and (Ms_Aud_Audit_Security.Check_Team_Member(A.Audit_Id,:LOGIN_ID) > 0 or (SELECT issue_reviewer FROM ms_aud_audit_issue_rev_task_v
 WHERE audit_id = A.AUDIT_ID AND process_instance_id = A.PROCESS_INSTANCE_ID
   AND instance_id = (SELECT MAX (instance_id) FROM ms_aud_audit_issue_rev_task_v
            WHERE audit_id = A.AUDIT_ID and process_instance_id = A.PROCESS_INSTANCE_ID
              AND instance_id <>(SELECT MAX (instance_id) FROM ms_aud_audit_issue_rev_task_v
                       where AUDIT_ID =a.AUDIT_ID  and PROCESS_INSTANCE_ID = a.PROCESS_INSTANCE_ID))
   AND instance_rec_num = 1)=:LOGIN_ID)
