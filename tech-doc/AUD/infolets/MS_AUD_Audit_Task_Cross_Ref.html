SELECT  a.task_id AS task_id,
         ms_aud_execution_rpt_utils.fn_concat_title_status(b.Audit_title,b.audit_status,b.audit_id,b.dd_process_instance_id,'CT') AS Audit_title_group,
           a.task_name  AS Task_Name,  
           Nvl (Ms_Aud_Utilities.Fn_Get_Status_Display_Name (A.Status), 'None' ) AS Task_Status, 
           Ms_Aud_Utilities.Fn_Get_Status_Display_Name(b.audit_status)AS audit_status,
           ms_aud_execution_rpt_utils.Fn_Get_Risk_Url( Ms_Aud_Utilities.Mlov_Values (A.task_id, 'task_id','CORE_OBJECTS', 'MS_AUD_AUDIT_TASK_COB' ),'AUE') AS auditable_entity, 
           ms_aud_execution_rpt_utils.fn_get_risk_form_url(A.task_id,'task_id','RELATED_RISKS','MS_AUD_AUDIT_TASK_RRS',b.audit_id,b.scope_based_on) AS risk, 
		       null as dummyrow,
           Ms_Apps_Utilities.Get_User_Full_Name(a.Task_Owner) AS Task_owner,
           ms_apps_utilities.get_user_full_name (A.approver) AS approver,
           a.audit_id,
           a.task_name AS task_name_disp,
           ms_aud_execution_rpt_utils.fn_concat_title_status(b.Audit_title,b.audit_status,b.audit_id,b.dd_process_instance_id,' ') AS audit_title_url,
           b.audit_title as audit_title_disp, ms_aud_execution_rpt_utils.fn_concat_title_status_rpt_jsn(b.Audit_title,b.audit_status,b.audit_id,b.dd_process_instance_id,'CT') AS Audit_title_group_Json,  ms_aud_execution_rpt_utils.fn_get_task_url_cross_ref(a.dd_process_instance_id,a.task_type,a.audit_id,a.task_id,a.task_name)  AS Task_Name_Json,  ms_aud_execution_rpt_utils.fn_get_risk_url_rpt_json( Ms_Aud_Utilities.Mlov_Values (A.task_id, 'task_id','CORE_OBJECTS', 'MS_AUD_AUDIT_TASK_COB' ),'AUE') AS auditable_entity_Json,  ms_aud_execution_rpt_utils.fn_get_risk_form_url_rpt_json(A.task_id,'task_id','RELATED_RISKS','MS_AUD_AUDIT_TASK_RRS',b.audit_id) AS risk_Json,  ms_aud_execution_rpt_utils.fn_concat_title_status_rpt_jsn(b.Audit_title,b.audit_status,b.audit_id,b.dd_process_instance_id,' ') AS audit_title_url_Json, 'MS_AUD_AUDIT_TASK' AS form_name
           FROM ms_aud_audit_task A, ms_aud_audit b 
           WHERE A.audit_id = b.audit_id
           AND b.audit_status<>'4'
           AND A.task_type IN ('2')
           AND 
           ms_aud_audit_security.check_view_access (b.audit_id,b.dd_object_type,b.restrict_access_to,b.conducted_by_org,:LOGIN_ID,'VIEW') =  'ACCESS'
           AND ( (ms_aud_utilities.get_parameter_value ('Reference In-progress Tasks')='N' AND A.status IN ('15','53')) OR (ms_aud_utilities.get_parameter_value ('Reference In-progress Tasks')='Y' AND A.status IN ('11','18','2','15','9','49','50','51','53','54')))
