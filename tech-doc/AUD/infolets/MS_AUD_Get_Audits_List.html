MS_AUD_Rpt_Get_Audits_List Report SYSTEMI 
SELECT  a.audit_id                                                             AS audit_id ,
  a.audit_title                                                          AS audit_title ,
  (SELECT status_name
  FROM ms_aud_status_tl_v
  WHERE locale_id =
    (SELECT b.locale_id
    FROM si_users_t a,
      si_locales b
    WHERE a.locale  = b.locale_string
    AND a.user_name = :LOGIN_ID
    )
  AND status_id = a.audit_status
  )                                                                                                                                                                                            AS audit_status ,
  NVL(ms_apps_utilities.get_user_full_name (a.lead_auditor),MS_AUD_REPORT_UTILS.fn_get_mlov_lead_aud_disp(MS_GRC_COMMON_UTILS.Fn_Get_Val_By_Id(audit_id,'ms_aud_audit_alm','lead_auditor_mlov','audit_id')))  AS lead_auditor ,
  NVL(ms_apps_utilities.get_user_full_name (a.audit_manager),MS_AUD_REPORT_UTILS.fn_get_mlov_aud_man_disp(MS_GRC_COMMON_UTILS.Fn_Get_Val_By_Id(audit_id,'ms_aud_audit_amm','audit_manager_mlov','audit_id'))) AS audit_manager ,
  MS_AUD_EXECUTION_RPT_UTILS.fn_get_task_count(a.audit_id,:LOGIN_ID)                                                                                                                               AS view_tasks ,
  audit_start_date                                                                                                                                                                             AS audit_start_date ,
  (SELECT s.status_name
  FROM ms_aud_plan_scope ps ,
    ms_aud_status_tl_v s
  WHERE ps.audit_id= a.audit_id
  AND ps.form_type ='P'
  AND s.locale_id  =
    (SELECT b.locale_id
    FROM si_users_t a,
      si_locales b
    WHERE a.locale  = b.locale_string
    AND a.user_name = :LOGIN_ID
    )
  AND ps.status = s.status_id
  )                                                                                                                                                                           AS plan_scope ,
  MS_AUD_REPORTING_RPT_UTILS.fn_get_draft_issue_count(a.audit_id)                                                                                                                       AS draft_issues ,
  ms_aud_f011_helper.fn_get_attachment_count(a.audit_id)                                                                                                                      AS attachment_no,
  ms_aud_F011_helper.fn_get_current_audit_urls_mod('MS_AUD_MANAGE_AUDIT',:LOGIN_ID,a.audit_id,a.audit_title,a.dd_process_instance_id)                                         AS report_json_audit_title,
  ms_aud_F011_helper.fn_get_current_audit_urls_mod('MS_AUD_PLAN_SCOPE',:LOGIN_ID,a.audit_id,'x',a.dd_process_instance_id)                                                     AS report_json_audit_plan,
  ms_aud_F011_helper.fn_get_current_audit_urls_mod('MS_AUD_DRAFT_ISSUES',:LOGIN_ID,a.audit_id,MS_AUD_REPORTING_RPT_UTILS.fn_get_draft_issue_count(a.audit_id),a.dd_process_instance_id) AS report_json_draft_issue,
  ms_aud_f011_helper.fn_get_attachment_details(a.audit_id)  as attachments,a.audit_end_date as AUDIT_END_DATE
FROM ms_aud_audit a
WHERE a.status                     IN ('3','8')
AND TO_CHAR(audit_start_date,'RRRR')= NVL(:i_year,TO_CHAR(audit_start_date,'RRRR'))
AND audit_id                        = NVL(:i_audit_id, audit_id)
AND conducted_by_org                = NVL(:i_org_id,conducted_by_org)
AND (NVL(lead_auditor,'1')          = NVL(:i_lead_auditor,NVL(lead_auditor,'1'))
OR :i_lead_auditor                 IN
  ( SELECT lead_auditor_mlov FROM ms_aud_audit_alm WHERE audit_id = a.audit_id
  ))
AND (NVL(audit_manager,'1') = NVL(:i_audit_manager,NVL(audit_manager,'1'))
OR :i_audit_manager        IN
  (SELECT audit_manager_mlov FROM ms_aud_audit_amm WHERE audit_id = a.audit_id
  ))
AND ms_aud_audit_security.check_view_access (audit_id,dd_object_type,restrict_access_to,conducted_by_org,:LOGIN_ID,'VIEW') = 'ACCESS'
ORDER BY a.dd_process_instance_id DESC
