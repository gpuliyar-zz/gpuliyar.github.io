 select OrgClass,StartCount,SchCount,StartAuditId,SchAuditId,Total from
(select OrgClass,StartCount,SchCount,StartAuditId,SchAuditId,(StartCount+SchCount) as Total
from
(select distinct 
       MS_GRC_COMMON_UTILS.Fn_Get_Org_Display_Names(conducted_by_org)||'-  '||ms_aud_utilities.get_code_description('MS AUD Classification',audit_classification) 
       as OrgClass,
       NVL(ms_aud_utilities.fn_status_count_aud_class1(audit_classification,3,conducted_by_org,:LOGIN_ID,:Audit_starting_after,:Audit_starting_before),0) StartCount,
       NVL(ms_aud_utilities.fn_status_count_aud_class1(audit_classification,20,conducted_by_org,:LOGIN_ID,:Audit_starting_after,:Audit_starting_before),0) SchCount,
       ms_aud_utilities.fn_get_auditid_classification1(audit_classification,:LOGIN_ID,conducted_by_org,:Audit_starting_after,:Audit_starting_before) as StartAuditId,
       ms_aud_utilities.fn_get_auditid_classification2(audit_classification,:LOGIN_ID,conducted_by_org,:Audit_starting_after,:Audit_starting_before) as SchAuditId
 from ms_aud_audit where audit_classification is not null
 and ms_aud_audit_security.check_view_access 
    (audit_id,dd_object_type,restrict_access_to,conducted_by_org,:LOGIN_ID,'VIEW') = 
    'ACCESS' and
    trunc(audit_start_date) between trunc(:Audit_starting_after)+1 and trunc(:Audit_starting_before)-1)
order by Total desc)      
where rownum<=5 
and Total<>0
