SELECT 
 control_id as ctrl_id, control_name as ctrl_name, test_id as tst_id, test_name as tst_name, 'View' as view_tst_form, view_json,attachments as attached_files,response_type,question_desc
  from (select a.control_id, a.control_name, a.test_id, a.test_name, 
        MS_AUD_REPORT_UTILS.fn_get_report_json((SELECT metric_id FROM si_metrics_t WHERE UPPER (metric_name) = UPPER(b.dd_object_type)),
                                                b.dd_process_instance_id,
                                                b.dd_instance_id,
                                                chr (38)|| 'edit_flag=N'|| CHR (38)|| 'flag=1'|| CHR (38)|| 'emd=3'
                                                || CHR (38)|| 'x_edit_mode=3'|| CHR (38)|| 'object_id='|| b.object_id
                                                || CHR (38)|| 'wrapper=no') as view_json,
b.attachments,b.response_type,b.question as question_desc,row_number() over (partition by test_id order by test_id) as seq_no
        FROM ms_aud_risk_control_test_v a, ms_Grc_question_proc b
        where a.test_id = b.object_id
		and a.control_id = :i_control_id and nvl(trunc(a.valid_until),trunc(sysdate)) >= trunc(sysdate) and nvl(trunc(a.valid_from),trunc(sysdate)) <= trunc(sysdate) 
		and (:i_test_id is null or (:i_test_id is not null and a.test_id not in (select column_value
                                FROM TABLE(ms_grc_str_common_utilities.Split_String_Stored(rtrim(:i_test_id,','))))))) where seq_no=1
