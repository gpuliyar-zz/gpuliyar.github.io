 SELECT a.audit_id as audit_id ,a.audit_title as audit_title, a.issue_title as issue_title, a.issue_owner as issue_owner,a.issue_rating as issue_rating,
 a.priority as priority,a.issue_due_date as issue_due_date, a.issue_created_by as issue_created_by, a.issue_approver as issue_approver,
  a.issue_disposition as issue_disposition,a.issue_status as issue_status, a.issue_progress as issue_progress,a.no_of_actions as no_of_actions,
 a.auditable_entities as auditable_entities,a.risks as risks,a.AREA_compliance as AREA_compliance,a.related_to_org as related_to_org,
 a.COMPLIANCE_CNTRL as COMPLIANCE_CNTRL,a.FN_ORGANIZATION as FN_ORGANIZATION,a.AUDIT_CLASSIFICATION as AUDIT_CLASSIFICATION,
 a.AUDIT_TYPE as AUDIT_TYPE,a.AUDIT_MANAGER as AUDIT_MANAGER,a.LEAD_AUDITOR as LEAD_AUDITOR,a.issue_id as issue_id,locale_id, report_json as Report_Json,
 a.Task_name as Task_Name,  issue_owner_stored, created_by_stored, issue_approver_stored
FROM TABLE (MS_AUD_REPORTING_RPT_UTILS.get_issue_data) a,ms_aud_audit b,si_enabled_locales  
  WHERE a.audit_id = b.audit_id
   AND ms_aud_audit_security.check_view_access(b.audit_id,b.dd_object_type,b.restrict_access_to,b.conducted_by_org,:LOGIN_ID,'VIEW') = 'ACCESS'
   AND b.audit_id = :AUDIT_ID
