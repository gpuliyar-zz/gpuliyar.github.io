SELECT auditid AS audit_id,
  taskid       AS task_id,
  AUDITTITLE
  ||' ('
  ||AUDITID
  ||')'           AS audit_title,
  taskname        AS task_name,
  tsk_status      AS status,
  auditableentity AS auditable_entity,
  tsk_risk        AS risk,
  assignedto      AS assigned_to,
  startdate       AS start_date,
  enddate         AS end_date,
  tsk_issues      AS issues,
  tsK_approver    AS approver,
  task_type1      AS task_type,
  CO_AUDITORS     AS coauditors,
  report_json      AS  report_json_task,
  task_owner_id AS task_owner_id,
  approver_id AS approver_id
FROM
  (SELECT a.task_type AS task_type1,a.task_owner AS task_owner_id,
a.audit_id AS auditid,
a.task_id AS taskid,
a.task_name AS taskname,
COALESCE((SELECT status_name FROM ms_aud_status_tl_v WHERE locale_id = (SELECT b.locale_id FROM si_users_t a,si_locales b WHERE a.locale  = b.locale_string AND a.user_name = :LOGIN_ID)
AND status_id = a.status), 'None')  AS tsk_status,
    MS_AUD_EXECUTION_RPT_UTILS.fn_get_risk_url( MS_AUD_REPORT_UTILS.mlov_values (a.task_id, 'task_id','CORE_OBJECTS', 'MS_AUD_AUDIT_TASK_COB' ),'AUE') AS auditableentity,
    MS_AUD_EXECUTION_RPT_UTILS.fn_get_risk_form_url(a.task_id,'task_id','RELATED_RISKS','MS_AUD_AUDIT_TASK_RRS',b.audit_id,b.scope_based_on) AS tsk_risk,
    ms_apps_utilities.get_user_full_name (a.task_owner)AS assignedto,
    a.task_start_date AS startdate,
    a.task_completion_date AS enddate,
    MS_AUD_EXECUTION_RPT_UTILS.fn_get_ctrl_issues (a.task_id) AS tsk_issues,
    ms_apps_utilities.get_user_full_name (a.approver)AS tsk_approver,a.approver AS approver_id,
    ms_apps_utilities.get_user_full_name (a.final_approver)AS tsk_final_approver,
    ms_apps_utilities.get_display_value (ms_apps_utilities.get_enterprise_id(
    (SELECT metric_id FROM si_metrics WHERE metric_name = 'MS_AUD_MANAGE_AUDIT'
    )),'MS AUD Task Type', a.task_type ) AS task_type,
    b.audit_title AS audittitle,
    MS_AUD_REPORT_UTILS.fn_get_mlov_lead_aud_disp(MS_AUD_REPORT_UTILS.mlov_values(a.task_id,'task_id','co_auditors_at','ms_aud_audit_task_cad')) AS CO_AUDITORS,
    CASE
      WHEN (SELECT DECODE(COUNT(1),0,'No','Yes')
        FROM ms_aud_audit_task
        WHERE (task_owner=:LOGIN_ID
        OR 1 =
          (SELECT 1
          FROM ms_aud_audit_task_cad bb
          WHERE bb.task_id = a.task_id
          AND :LOGIN_ID   IN
           (SELECT COLUMN_VALUE
           FROM TABLE (ms_grc_str_common_utilities.Split_String_Stored(bb.co_auditors_at))
          )))
        AND status IN (31,11,49)
        AND task_id =A.TASK_ID
        AND NVL(MS_AUD_EXECUTION_RPT_UTILS.fn_get_offline_status(a.dd_process_instance_id),'N')='N')='Yes'
      THEN  
      ms_aud_report_utils.fn_get_report_json((SELECT metric_id FROM si_metrics_t WHERE metric_name='MS_AUD_AUDIT_TASK'),
      a.dd_process_instance_id,null,'flag=1'|| CHR (38) || 'emd=4' || CHR (38) || 'x_edit_mode=Yes' || CHR (38) 
      || 'Task_id=' || a.task_id)
      ELSE  ms_aud_report_utils.fn_get_report_json((SELECT metric_id FROM si_metrics_t WHERE metric_name='MS_AUD_AUDIT_TASK'
      ),a.dd_process_instance_id,null,'flag=1'|| CHR (38) || 'emd=4' || CHR (38) || 'x_edit_mode=No'  || CHR (38) || 'Task_id='
      || a.task_id) END AS report_json FROM ms_aud_audit_task a, ms_aud_audit b
      WHERE a.audit_id = b.audit_id
      AND a.task_type IN ('2')
      AND ms_aud_audit_security.check_team_member(a.audit_id,:LOGIN_ID) > 0
      AND b.status NOT IN ('8','4')
      ORDER BY a.dd_process_instance_id)
