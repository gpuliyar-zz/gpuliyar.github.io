MS_AUD_Top_10_Risk_Cat_Dd_Rpt Report SYSTEMI 
SELECT object_id,
		object_name,
		risk_rating,
		last_audited_on,
		next_audit_date,
		plan_years,
		no_of_audits_till_date,
		open_issues,
		creation_date,
		report_json,
		aud_con from
		(SELECT distinct a.object_id as object_id, 
              a.object_name  AS object_name ,
               trim(substr(ms_aud_advisor_utilities.fn_get_rating(MS_AUD_PLANNING_RPT_UTILS.get_risk_score (a.object_id,:i_scenario_id),a.object_id,:i_scenario_id,'R',3),1,
              instr(ms_aud_advisor_utilities.fn_get_rating(MS_AUD_PLANNING_RPT_UTILS.get_risk_score (a.object_id,:i_scenario_id),a.object_id,:i_scenario_id,'R',3),' '))) AS risk_rating,
              (SELECT max(audit_start_date) keep (dense_rank first order by to_number(audit_status))
                   FROM ms_aud_audit WHERE audit_id IN (SELECT b.audit_id FROM ms_aud_audit_ris b,ms_aud_audit ad
                            WHERE substr(b.related_risks,1,instr(b.related_risks,ms_aud_constants.scope_delimiter)-1) = a.object_id and ad.audit_id=b.audit_id and ad.audit_status not in (4))
                    AND audit_start_date is not null) AS last_audited_on,
                (SELECT TRUNC (MIN (audit_start_date))
                   FROM ms_aud_audit WHERE audit_id IN (
                           SELECT audit_id FROM ms_aud_audit_ris b
                            WHERE substr(b.related_risks,1,instr(b.related_risks,ms_aud_constants.scope_delimiter)-1) = a.object_id)
                    AND audit_status = '33') AS next_audit_date,
                MS_AUD_PLANNING_RPT_UTILS.get_plan_years (a.object_id) AS plan_years,
                NVL((SELECT COUNT(b.audit_id) FROM ms_aud_audit_ris b,ms_aud_audit aud
                     WHERE substr(b.related_risks,1,instr(b.related_risks,ms_aud_constants.scope_delimiter)-1) = a.object_id and aud.status not in (4,8) and aud.audit_id=b.audit_id 
                     AND trunc(aud.audit_start_date) > ADD_MONTHS(trunc(sysdate,'MON'),-(3))
					AND Extract(YEAR From aud.Audit_Start_Date) = Nvl(:Year,Ms_Aud_Constants.Fn_Year)),0) AS no_of_audits_till_date,
                 ms_aud_advisor_utilities.fn_get_open_issue(a.object_id,'Y') AS open_issues,               
                TRUNC (a.CREATED_ON) AS creation_date,
ms_aud_report_utils.fn_get_report_json(
            (SELECT metric_id FROM si_metrics_t WHERE metric_name = a.dd_object_type),
              a.dd_process_instance_id,
              null,
              'edit_flag=N' || CHR (38)|| 'flag=2'|| CHR (38)|| 'emd=1' ||CHR(38)) AS report_json,(SELECT ms_join_clob_for_display(b.audit_id)  FROM ms_aud_audit_ris b,ms_aud_audit aud
                     where substr(b.related_risks,1,instr(b.related_risks,ms_aud_constants.scope_delimiter)-1) = a.OBJECT_ID and AUD.STATUS not in (4,8) and AUD.AUDIT_ID=B.AUDIT_ID 
                     AND trunc(aud.audit_start_date) > ADD_MONTHS(trunc(sysdate,'MON'),-(3)) AND Extract(Year From aud.Audit_Start_Date) =Nvl(:Year,Ms_Aud_Constants.Fn_Year)) AS aud_con
             FROM MS_GRC_RISK_PUB_V a
          WHERE ms_apps_security_pkg.check_object_access(a.dd_object_type, a.restrict_access_to, (SELECT MS_JOIN_FOR_DISPLAY(DISTINCT(OWNER_ORGANIZATIONS))
    FROM MS_GRC_RISK_PUB_V), :LOGIN_ID, 'VIEW') = 'ACCESS'
         AND instr(','||a.category_risk||',',','||:i_risk_cat||',')>0         
         and TRUNC(NVL(a.VALID_UNTIL,SYSDATE)) >= TRUNC(SYSDATE)   
         ORDER BY MS_AUD_PLANNING_RPT_UTILS.get_risk_score (a.object_id,:i_scenario_id) desc NULLS LAST)
