SELECT DISTINCT a1.audit_id, a1.audit_title ||' [' ||a1.audit_id ||']' AS aud_title, CASE a1.status WHEN '3' THEN 'In Process' WHEN '4' THEN 'Canceled' WHEN '8' THEN 'Closed' WHEN '20' THEN 'Future Audit' ELSE 'Status not Found' END KM_STATUS, trim(c1.km_milestones) AS KM_MILESTONES, NVL (TO_CHAR (TRUNC (c1.km_proposed_date), 'MM/DD/YYYY'), '-') AS proposed_date, 'Planned' AS proposed, CASE WHEN TRUNC (c1.km_proposed_date)<TRUNC(c1.km_actual_date) THEN 'Delayed' WHEN TRUNC (c1.km_proposed_date)>=TRUNC(c1.km_actual_date) THEN 'On-Time' ELSE 'Not-Applicable' END AS KM_CATEGORY, C1.KM_EVENT_IN_SYSTEM, NVL((TRUNC(c1.km_actual_date))-TRUNC((c1.km_proposed_date)),0) date_color FROM ms_aud_audit a1, ms_aud_audit_kms c1 WHERE a1.audit_id =c1.audit_id AND NVL(TRIM(a1.plan_title),'1') =NVL(TRIM(:PLAN_TITLE),NVL(TRIM(a1.plan_title),'1')) AND NVL(TRIM(a1.conducted_by_org),'0') =NVL(TRIM (:ORGANIZATION),NVL(TRIM(a1.conducted_by_org),'0')) AND NVL (TRIM (a1.audit_type), '0') = NVL (TRIM (:AUDIT_TYPE), NVL (TRIM (a1.audit_type), '0')) AND NVL (TRIM (a1.audit_classification), '0') = NVL (TRIM (:AUDIT_CLASSIFICATION), NVL (TRIM (a1.audit_classification), '0')) AND TRUNC (a1.audit_start_date) <= TRUNC (SYSDATE) AND TRUNC (a1.audit_start_date) > trunc(TO_DATE (NVL(:STARTED_AFTER,(ADD_MONTHS(:STARTED_BEFORE,-12))))) AND TRUNC (a1.audit_start_date) < trunc(TO_DATE (:STARTED_BEFORE)) AND ( :AUDIT_STATUS IS NULL OR CASE a1.status WHEN '3' THEN 'In Process' WHEN '4' THEN 'Canceled' WHEN '8' THEN 'Closed' WHEN '20' THEN 'Future Audit' ELSE 'Status not Found' END = :AUDIT_STATUS) AND ms_aud_audit_security.check_view_access(a1.audit_id,a1.dd_object_type,a1.restrict_access_to, a1.conducted_by_org,:LOGIN_ID,'VIEW') = 'ACCESS' AND ( (:SELECT_TEMPLATE IS NULL AND a1.milestone_template = '100') OR ( :SELECT_TEMPLATE IS NOT NULL AND a1.milestone_template = :SELECT_TEMPLATE) ) AND a1.status NOT IN ('2','31','51') AND a1.approve_flag = '1' UNION SELECT DISTINCT a1.audit_id, a1.audit_title ||' [' ||a1.audit_id ||']' AS aud_title, CASE a1.status WHEN '3' THEN 'In Process' WHEN '4' THEN 'Canceled' WHEN '8' THEN 'Closed' WHEN '20' THEN 'Future Audit' ELSE 'Status not Found' END KM_STATUS, trim(c1.km_milestones), NVL (TO_CHAR (TRUNC (c1.km_actual_date), 'MM/DD/YYYY'), '-') AS proposed_date , 'Actual' AS proposed, CASE WHEN TRUNC (c1.km_proposed_date) < TRUNC (c1.km_actual_date) THEN 'Delayed' WHEN TRUNC (c1.km_proposed_date) >= TRUNC (c1.km_actual_date) THEN 'On-Time' ELSE 'Not-Applicable' END AS KM_CATEGORY , C1.KM_EVENT_IN_SYSTEM , NVL((TRUNC(c1.km_actual_date))-TRUNC((c1.km_proposed_date)),0) date_color FROM ms_aud_audit a1, ms_aud_audit_kms c1 WHERE a1.audit_id = c1.audit_id AND NVL (TRIM (a1.plan_title), '1') = NVL (TRIM (:PLAN_TITLE), NVL (TRIM (a1.plan_title), '1')) AND NVL (TRIM (a1.conducted_by_org), '0') = NVL (TRIM (:ORGANIZATION), NVL (TRIM (a1.conducted_by_org), '0')) AND NVL (TRIM (a1.audit_type), '0') = NVL (TRIM (:AUDIT_TYPE), NVL (TRIM (a1.audit_type), '0')) AND NVL (TRIM (a1.audit_classification), '0') = NVL (TRIM (:AUDIT_CLASSIFICATION), NVL (TRIM (a1.audit_classification), '0')) AND TRUNC (a1.audit_start_date) <= TRUNC (SYSDATE) AND TRUNC (a1.audit_start_date) > trunc(TO_DATE (NVL(:STARTED_AFTER,(ADD_MONTHS(:STARTED_BEFORE,-12))))) AND TRUNC (a1.audit_start_date) < trunc(TO_DATE (:STARTED_BEFORE)) AND ( :AUDIT_STATUS IS NULL OR CASE a1.status WHEN '3' THEN 'In Process' WHEN '4' THEN 'Canceled' WHEN '8' THEN 'Closed' WHEN '20' THEN 'Future Audit' ELSE 'Status not Found' END = :AUDIT_STATUS) AND ms_aud_audit_security.check_view_access(a1.audit_id,a1.dd_object_type,a1.restrict_access_to, a1.conducted_by_org,:LOGIN_ID,'VIEW') = 'ACCESS' AND ( (:SELECT_TEMPLATE IS NULL AND a1.milestone_template = '100') OR ( :SELECT_TEMPLATE IS NOT NULL AND a1.milestone_template = :SELECT_TEMPLATE) ) AND a1.status NOT IN ('2','31','51') AND a1.approve_flag = '1' ORDER BY aud_title, KM_EVENT_IN_SYSTEM, proposed desc
