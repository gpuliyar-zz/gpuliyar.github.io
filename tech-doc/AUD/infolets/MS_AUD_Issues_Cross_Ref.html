SELECT  a.FIND_ID,
  ms_aud_execution_rpt_utils.fn_concat_title_status(b.Audit_title,b.audit_status,b.audit_id,b.dd_process_instance_id,'CT') AS Audit_title_group, 
  a.task_name,
  a.task_status,
  a.issue_title,
  a.issue_status,
  A.issue_owner,
  a.issue_due_date,
  a.owner_org,
  a.task_owner,
  a.audit_id,
  ms_aud_execution_rpt_utils.fn_concat_title_status(b.Audit_title,b.audit_status,b.audit_id,b.dd_process_instance_id,' ') AS Audit_title_url,
  A.audit_title,
  A.issue_title_disp,
  A.task_id,
  a.draft_task_id, ms_aud_execution_rpt_utils.fn_concat_title_status_rpt_jsn(b.Audit_title,b.audit_status,b.audit_id,b.dd_process_instance_id,'CT') AS Audit_title_group_Json, ms_aud_execution_rpt_utils.fn_concat_title_status_rpt_jsn(b.Audit_title,b.audit_status,b.audit_id,b.dd_process_instance_id,'CT') AS Audit_title_url_Json, a.task_name_json as task_name_jsn
FROM TABLE (ms_aud_execution_rpt_utils.get_issue_data_cross_ref) A, ms_aud_audit b
WHERE A.audit_id = b.audit_id  AND 
((a.draft_task_id != :Task_ID) OR (a.draft_task_id IS NULL ))
AND ((a.task_id != :Task_ID) OR (a.task_id IS null )) AND
ms_aud_audit_security.check_view_access (b.audit_id,b.dd_object_type,b.restrict_access_to,b.conducted_by_org,:LOGIN_ID,'VIEW') =  'ACCESS'
