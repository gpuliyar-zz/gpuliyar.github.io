SELECT DISTINCT
ms_apps_utilities.get_user_full_name(b1.user_name) AS Auditor_Name, 
b1.org_entity_name AS Organizations,
ms_apps_utilities.get_display_value(ms_apps_utilities.get_enterprise_id(ms_apps_utilities.get_infolet_id ('MS_AUD_AUDITOR_PROFILE')),'MS AUD Auditor Profile Title',auditor_title,1,NULL,NULL) AS Title,
ms_apps_utilities.get_display_value(ms_apps_utilities.get_enterprise_id(ms_apps_utilities.get_infolet_id ('MS_AUD_AUDITOR_PROFILE')),'MS AUD Auditor Proficiency',profiency_level,1,NULL,NULL) AS Proficiency_Level,
ms_apps_utilities.get_display_values(ms_apps_utilities.get_enterprise_id(ms_apps_utilities.get_infolet_id ('MS_AUD_AUDITOR_PROFILE')),'MS AUD Auditor Group',
    (ms_aud_utilities.mlov_values(a1.auditor_user_name,'auditor_user_name','auditor_group','ms_aud_auditor_profile_aug')),',',1,NULL,NULL) AS Auditor_Groups,        
ms_apps_utilities.get_display_values(ms_apps_utilities.get_enterprise_id(ms_apps_utilities.get_infolet_id ('MS_AUD_AUDITOR_PROFILE')),'MS AUD Auditor Qualifications',
    (ms_aud_utilities.mlov_values(a1.auditor_user_name,'auditor_user_name','professional_qual','ms_aud_auditor_profile_pro')),',',1,NULL,NULL) AS Professional_Qualification,
ms_apps_utilities.get_display_values(ms_apps_utilities.get_enterprise_id(ms_apps_utilities.get_infolet_id ('MS_AUD_AUDITOR_PROFILE')),'MS AUD Auditor Language Skills',
    (ms_aud_utilities.mlov_values(a1.auditor_user_name,'auditor_user_name','language_skills','ms_aud_auditor_profile_lns')),',',1,NULL,NULL) AS Language_Skills,                    
ms_apps_utilities.get_display_values(ms_apps_utilities.get_enterprise_id(ms_apps_utilities.get_infolet_id ('MS_AUD_AUDITOR_PROFILE')),'MS AUD Auditor Profile Training',
    (ms_aud_utilities.mlov_values(a1.auditor_user_name,'auditor_user_name','training','ms_aud_auditor_profile_trn')),',',1,NULL,NULL) AS Training,
ms_apps_utilities.get_display_values(ms_apps_utilities.get_enterprise_id(ms_apps_utilities.get_infolet_id ('MS_AUD_AUDITOR_PROFILE')),'MS AUD Auditor Subject Matter Expert',
    (ms_aud_utilities.mlov_values(a1.auditor_user_name,'auditor_user_name','subject_matter_expert','ms_aud_auditor_profile_sme')),',',1,NULL,NULL) AS Subject_Matter_Expertise,
ms_aud_setup_rpt_utils.get_auditor_availability(:Task_Start_Date,:Task_End_Date,b1.user_name) AS Available
FROM ms_aud_auditor_profile a1,
        (SELECT DISTINCT user_name, org_entity_id, org_entity_name
            FROM ms_aud_org_flow_v
            WHERE activity_name = 'AUD - Work on Workpapers') b1, ms_aud_auditor_profile_aug c1
WHERE
--(:Auditor_Group IS NULL OR (:Auditor_Group IS NOT NULL AND (c1.auditor_group LIKE '%' || :Auditor_Group || '%')))  
(:Auditor_Group IS NULL OR (:Auditor_Group IS NOT NULL AND UPPER (TRIM (:Auditor_Group)) LIKE
                       '%'
                    || (SELECT COLUMN_VALUE
                          FROM TABLE
                                  (ms_grc_str_common_utilities.Split_String_Stored
                                          ((SELECT UPPER (TRIM (c1.auditor_group))
                                              FROM DUAL)
                                          )
                                  ))
                    || '%') and c1.auditor_group IS NOT NULL)
AND b1.user_name = a1.auditor_user_name(+) 
AND a1.auditor_user_name = c1.auditor_user_name(+)
AND b1.org_entity_id = :Organization
