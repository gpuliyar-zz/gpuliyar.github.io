select 
case when rownum =1 then X.PLAN_TITLE else null end as PLAN_TITLE,
case when rownum =1 then X.plan_start_date else null end as plan_start_date,
case when rownum =1 then X.plan_end_date else null end as plan_end_date,
case when rownum =1 then X.organization else null end as organization,
case when rownum =1 then X.PLANTYPE else null end as PLANTYPE,
y.status as status, 
y.date_time as date_time, 
y.user_name as user_name,y.comments as comments,Y.ELECTRONIC_SIGNATURES as ELECTRONIC_SIGNATURES, Y.PLAN_ID as PLAN_ID, Y.PROCESS_INSTANCE_ID as PROCESS_INSTANCE_ID, Y.REPORT_JSON as Report_Json,y.user_id as user_id
from
(select distinct PLAN_TITLE as PLAN_TITLE,
         TRUNC (start_date) AS plan_start_date,
         TRUNC (end_date) AS plan_end_date,
         MS_GRC_COMMON_UTILS.Fn_Get_Org_Display_Names (hdr_organization) as organization,
         (SELECT lovvalue.displayed_value AS displayed_value
		  FROM ms_qs_list_of_values lovvalue,
			ms_qs_lov_names lovname
		  WHERE lovvalue.lov_id      = lovname.lov_id
		  AND lovvalue.org_entity_id = 1
		  AND lovname.lov_name       = 'MS AUD Plan Types'
		  AND lovvalue.locale_id     = NVL(CAST(si_db_locale_sv.getvalue AS NUMBER),1009)
		  AND stored_value           =plan_type
		  AND lovvalue.enterprise_id =(SELECT enterprise_id FROM si_ent)
		  AND (lovname.end_date     IS NULL
		  OR lovname.end_date        > SYSDATE)
		  AND lovname.start_date    <= sysdate
		  AND (lovvalue.end_date    IS NULL
		  OR lovvalue.end_date       > SYSDATE)
		  )as PLANTYPE
from MS_AUD_PLAN_AUDIT_V 
where  INTERNAL_PLAN_ID = :Plan_Title and instance_id=(select max(instance_id) from MS_AUD_PLAN_AUDIT_V where INTERNAL_PLAN_ID = :Plan_Title)) x,
( select  MS_AUD_REPORT_UTILS.fn_get_status_display_name (a.status) AS status,
         TO_CHAR (a.creation_date,'MM/DD/YYYY hh24:mi:ss') AS Date_Time, MS_AUD_REPORT_UTILS.fn_get_user_display (a.assignee) AS user_name,a.assignee user_id,
         NVL (a.plan_comments, MS_AUD_PLANNING_RPT_UTILS.fn_get_comments (a.plan_id, a.instance_id, a.plan_action)) AS comments,
                                                                                                                   --Function added for the bug  51328
         a.ELECTRONICALLY_SIGNED as ELECTRONIC_SIGNATURES, a.PLAN_ID, a.PROCESS_INSTANCE_ID, ms_aud_report_utils.fn_get_report_json((SELECT metric_id FROM si_metrics_t WHERE metric_name = 'MS_AUD_PLAN_AUDIT'),a.process_instance_id,a.instance_id,'flag=1'||CHR(38)||'emd=2'||CHR(38)||'x_edit_mode=No'||CHR(38)||'Plan_id='||a.plan_id||CHR(38)||'mode=view') AS Report_Json
         FROM ms_aud_plan_audit_v a
   WHERE a.internal_plan_id = :Plan_Title
     and a.INSTANCE_REC_NUM = 1
     AND a.internal_plan_id is not null
     and a.PLAN_ID is not null   
     and (CREATED_BY = (select user_id from si_users_t where user_name = 'SYSTEMI'))
ORDER BY a.instance_id) y
