Select Stored_Value,
       Displayed_Value,
       lov_name,
       Locale_Id
From   ms_qs_lov_org_values_v
where (end_date IS NULL OR end_date > SYSDATE)
         AND start_date <= SYSDATE and lov_name in (
'MS AUD Contributing Factors',
'MS AUD Root Cause',
'MS_AUD_Finding_Type',
'MS AUD Audit Task Issue Rating', 
'MS AUD Priority',
'MS AUD Document Categories',
'MS AUD Audit Task Issue Resol',
'MS AUD Document Types'
)
order by display_order
