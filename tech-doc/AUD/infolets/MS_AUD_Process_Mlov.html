SELECT    process_id as ProcessID,
          process_name as ProcessName,Process_level as PrLevel ,core_type,
          version_no
          FROM (SELECT object_name AS process_name, object_id AS process_id,
                 ms_aud_utilities.fn_get_ae_obj_name(DD_OBJECT_TYPE) AS core_type,
                  object_level as Process_level,
                 (select ae_version from ms_aud_auditable_entit where object_id=ms_aud_grc_core_object_pub_v.object_id) as version_no
             FROM ms_aud_grc_core_object_pub_v
           WHERE (:scope = 5 OR :scope = 1) 
            AND (( ms_aud_grc_core_object_pub_v.object_id IN (SELECT max(b.object_id) from ms_aud_grc_core_object_pub_v a, 
                ms_aud_auditable_entit b WHERE a.object_id = b.object_id 
                AND NVL(TRUNC(a.valid_from),TRUNC(SYSDATE)) <= TRUNC(SYSDATE) 
                AND NVL(TRUNC(a.valid_until),TRUNC(SYSDATE)) >= TRUNC(SYSDATE) 
                group by b.internal_object_id)
                  AND dd_object_type='MS_AUD_AUDITABLE_ENTITY') OR dd_object_type <> 'MS_AUD_AUDITABLE_ENTITY'
                  OR ms_aud_grc_core_object_pub_v.object_id IN (SELECT CORE_OBJECTS FROM ms_aud_audit_aue where audit_id=:ae_ids and :ae_ids<>'0'))
             AND  nvl(TRUNC(valid_from),TRUNC(sysdate)) <= trunc(sysdate) AND NVL(TRUNC(valid_until),TRUNC(SYSDATE)) >= TRUNC(SYSDATE)
             AND ms_apps_security_pkg.check_object_access
                                                         (dd_object_type,
                                                          restrict_access_to,
                                                          owner_organizations,
                                                          :user_name,
                                                          'VIEW'
                                                         ) = 'ACCESS'
             UNION
            SELECT DISTINCT po.process_name as process_name,
                        po.process_id||ms_aud_constants.scope_delimiter()|| po.ato_organizations  as process_id,
                         ms_aud_utilities.fn_get_ae_obj_name(pub.DD_OBJECT_TYPE) AS core_type,
                         object_level as Process_level,
                        (select ae_version from ms_aud_auditable_entit where object_id=po.process_id) as version_no
          FROM ms_aud_grc_process_v po, ms_aud_grc_core_object_pub_v pub
         WHERE po.ato_organizations IN (SELECT COLUMN_VALUE FROM TABLE(ms_grc_str_common_utilities.Split_String_Stored(NVL(:org_ids, 1))))
           AND :scope = 3
           and po.process_id = pub.object_id
           and (( pub.object_id IN (SELECT max(b.object_id) from ms_aud_grc_core_object_pub_v a, ms_aud_auditable_entit b 
          WHERE a.object_id = b.object_id 
          AND NVL(TRUNC(a.valid_from),TRUNC(SYSDATE)) <= TRUNC(SYSDATE) 
          AND NVL(TRUNC(a.valid_until),TRUNC(SYSDATE)) >= TRUNC(SYSDATE) 
          group by b.internal_object_id)
                    AND pub.dd_object_type='MS_AUD_AUDITABLE_ENTITY') OR pub.dd_object_type <> 'MS_AUD_AUDITABLE_ENTITY'
                    OR pub.object_id IN (SELECT CORE_OBJECTS FROM ms_aud_audit_aue where audit_id=:ae_ids and :ae_ids<>'0'))
                     AND nvl(trunc(pub.valid_from), trunc(sysdate)) <= trunc(sysdate)
                     AND NVL(TRUNC(pub.valid_until), TRUNC(SYSDATE)) >= TRUNC(SYSDATE)
                     AND ms_apps_security_pkg.check_object_access(dd_object_type,
                                                                  restrict_access_to,
                                                                  pub.owner_organizations,
                                                                  :user_name,
                                                                  'VIEW') = 'ACCESS'
             )
