MS_AUD_Search_Auditors_Profile Report SYSTEMI 
SELECT Auditor_Name, Title, EMAIL_ADDRESS, Auditor_Groups, Professional_Qualifications, Training, Proficiency, Subject_Matter_Expert, 
Language_Skills, a.COST_PER_HOUR,a.report_json,EMAIL_ADDRESS_URL,Auditor_user_name
FROM ms_aud_search_auditors_prof_v a
WHERE (:Auditor_User_Name IS NULL
OR (:Auditor_User_Name    IS NOT NULL
AND (lower(ms_apps_utilities.get_user_full_name(a.auditor_user_name)) LIKE '%'
  ||lower(:Auditor_User_Name)
  ||'%')))
AND (:First_Name                  IS NULL
OR (:First_Name                   IS NOT NULL
AND (a.auditor_user_name IN
  (SELECT user_name
  FROM si_users_t
  WHERE lower(first_name) LIKE '%'
    ||lower(:First_Name)
    ||'%'
  ))))
AND (:Last_Name                  IS NULL
OR (:Last_Name                   IS NOT NULL
AND (a.auditor_user_name IN
  (SELECT user_name
  FROM si_users_t
  WHERE lower(last_name) LIKE '%'
    ||lower(:Last_Name)
    ||'%'
  ))))
AND (:Email_Address IS NULL
OR (:Email_Address  IS NOT NULL
AND (lower(email_address) LIKE '%'
  ||lower(:Email_Address)
  ||'%') ))
AND (:Phone_Number IS NULL
OR (:Phone_Number  IS NOT NULL
AND (lower(phone_number) LIKE '%'
  ||lower(:Phone_Number)
  ||'%')))
AND a.auditor_title=NVL(:Title,a.auditor_title)
AND (:Training           IS NULL
OR (:Training            IS NOT NULL
AND (MS_AUD_REPORT_UTILS.mlov_values(a.auditor_user_name,'AUDITOR_USER_NAME','TRAINING','ms_aud_auditor_profile_trn') LIKE '%'
  ||:Training
  ||'%') ))
AND (:Proficiency_Level               IS NULL
OR (:Proficiency_Level                IS NOT NULL
AND (a.PROFIENCY_LEVEL =:Proficiency_Level)))
AND (:Professional_Qualification               IS NULL
OR (:Professional_Qualification                IS NOT NULL
AND (MS_AUD_REPORT_UTILS.mlov_values(a.auditor_user_name,'AUDITOR_USER_NAME','PROFESSIONAL_QUAL','ms_aud_auditor_profile_pro') LIKE '%'
  ||:Professional_Qualification
  ||'%')))
AND (:Auditor_User_Name0 IS NULL
OR (:Auditor_User_Name0  IS NOT NULL
AND (MS_AUD_REPORT_UTILS.mlov_values(a.auditor_user_name,'AUDITOR_USER_NAME','LANGUAGE_SKILLS','ms_aud_auditor_profile_lns') LIKE '%'
  ||:Auditor_User_Name0
  ||'%')))
AND (:Auditor_User_Name1 IS NULL
OR (:Auditor_User_Name1  IS NOT NULL
AND (MS_AUD_REPORT_UTILS.mlov_values(a.auditor_user_name,'AUDITOR_USER_NAME','SUBJECT_MATTER_EXPERT','ms_aud_auditor_profile_sme') LIKE '%'
  ||:Auditor_User_Name1
  ||'%')))
