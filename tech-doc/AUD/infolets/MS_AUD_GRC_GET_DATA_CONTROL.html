SELECT DISTINCT object_id, object_name, valid_from, valid_until,
                restrict_access_to, dd_process_instance_id, dd_instance_id,
                owner_organizations, display_name, dd_object_type object_type,
                ato_organization, risks, over_ride_action, expired_flag,
                edit_action_parameters
           FROM (SELECT object_id, object_name as object_name,
                        SUBSTR (object_name, 1, 30) display_name,
                        dd_object_type object_type, 'Y' AS over_ride_action,
                        MS_GRC_COMMON_UTILS.FN_INCLUDE_EXPIRED
                                                 (
                                                  valid_until
                                                 ) AS expired_flag,
                        valid_from, valid_until, restrict_access_to,
                        dd_object_type, dd_process_instance_id,
                           'proc='
                        || TO_CHAR (dd_process_instance_id)
                        || '|instid='
                        || TO_CHAR (dd_instance_id)
                        || '|edit_flag='
                        || NVL (NULL, 'Y') AS edit_action_parameters,
                        dd_instance_id,
                        a.OWNER_ORGANIZATIONS   AS owner_organizations,
                        NULL risks,
                        NULL ato_organization
                   FROM ms_grc_compliance_obje_pub_v a) t1
       ORDER BY object_name ASC
