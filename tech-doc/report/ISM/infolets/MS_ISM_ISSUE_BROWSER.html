SELECT 
  REPLACE(REPLACE(a.issue_title,'<', '&'||'lt;'), '>', '&'||'gt;') as display_name, 
  a.issue_id object_id, 
  UPPER (ms_apps_utilities.get_user_full_name (a.issue_owner)) as user_full_name, 
  NULL AS dd_instance_id, 
  a.dd_object_type as object_type, 
  a.issue_ownerorg as owner_org, 
  a.issue_owner as issue_owner, 
  'Y' AS over_ride_action, 
   'proc=' 
  || TO_CHAR (a.dd_process_instance_id) 
  || '|instid=' 
  || (a.dd_instance_id) 
  || '|emd=' 
  || '4' 
  || '|edit_flag=' 
  || NVL (a.dd_edit_flag, 'Y') AS view_action_parameters, 
  ms_apps_utilities.get_display_value (100000, 'MS ISM Source Type', a.issue_source_type, 1, NULL, NULL) AS source_type, 
  ms_apps_utilities.get_display_value (100000, 'MS ISM Risk Rating', a.issue_risk_rating, 1, NULL, NULL) AS issue_rating, 
  ms_apps_utilities.get_display_value (100000, 'MS ISM Priority', a.issue_risk_priority, 1, NULL, NULL)  AS priority, 
 ms_ism_helper.get_status_display_name(a.ISSUE_STATUS)          AS issue_status, 
 ms_ism_helper.get_coma_seprtd_orb_value ('Organizations', a.issue_id)            AS org, 
  ms_ism_helper.get_coma_seprtd_orb_value ('Areas of Compliance', a.issue_id)            AS area_of_compliance, 
  ms_ism_helper.get_coma_seprtd_orb_value ('Controls', a.issue_id)            AS controls, 
  ms_ism_helper.get_coma_seprtd_orb_value ('Risks', a.issue_id)            AS risks, 
  ms_ism_helper.get_coma_seprtd_orb_value ('Core Objects', a.issue_id)            AS core_objects
FROM ms_ism_issue a 
WHERE a.reached_manage        ='Y' 
AND a.issue_status NOT  IN ('1', '5', '6') 
AND ((1 = ms_ism_security_pkg.fn_ism_view_issue(:1) 
      AND 1 = ms_ism_security_pkg.fn_ism_view_issue_org(:1,a.issue_ownerorg,a.issue_approverorg,a.act_plan_approverorg, a.final_iss_approverorg)) 
           OR 1 = ms_ism_security_pkg.fn_ism_view_issue_all(:1)
           OR 1 = decode((SELECT MS_JOIN(tgt_obj_id) as tgt_obj_id
                                       FROM ms_ism_orb_relation_orb
                                      WHERE rel_source_id = a.issue_id
                                        AND tgt_obj_type = 'Organizations'
                                       GROUP BY rel_source_id),NULL,-1,ms_ism_security_pkg.fn_ism_view_issue_orb(:1,(SELECT MS_JOIN(tgt_obj_id) as tgt_obj_id
                                       FROM ms_ism_orb_relation_orb
                                      WHERE rel_source_id = a.issue_id
                                        AND tgt_obj_type = 'Organizations'
                                       GROUP BY rel_source_id))))  
ORDER BY display_name
