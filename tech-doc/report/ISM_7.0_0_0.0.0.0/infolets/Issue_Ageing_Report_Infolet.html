Select Issue_Source_Type,lvl,Period, Sum(Decode(No_Of_Issue,'Y',1,'N',0,0)) As No_Of_Issue,total,NVL(:2,'') AS Owner_Org
from (
SELECT 
ms_ism_rpt_utils.fn_get_lov_dis_val('MS ISM Source Type',source_type_stored,:1) as issue_source_type,lvl,
(select replace(period,'X',ms_apps_utilities.get_parameter_value (1, 'MS_ISM_Configuration', 'Enable_Ageing_Due_In_Months')) 
from Ms_Ism_Ageing_Table_Tl_V
where locale_id=(select locale_id from si_users_t where user_name=:1)
and period_stored=lvl)  AS period,
       DECODE(lvl, 1, OVERDUE_THIS_MONTH,
                           2, OD_PAST_THREE_MONTH,
                           3, OD_BEYOND_THREE_MONTH,
                           4, DUE_THIS_MONTH,
                           5, DUE_WITHIN_X_MONTH,
                           6, Due_Beyond_X_Month,
                           'N/A') As No_Of_Issue,
      null as total
FROM   ms_ism_issue_reporting_v,
       (Select Level As Lvl From Dual Connect By Level <= 6)
       where (1 = ms_ism_security_pkg.fn_ism_view_issue(:1)
       AND 1 = ms_ism_security_pkg.fn_ism_view_issue_org(:1,ISSUE_OWNER_ORG,ISSUE_APPROVERORG,
       ACT_PLAN_APPROVERORG,FINAL_ISS_APPROVERORG)
       OR 1 = ms_ism_security_pkg.fn_ism_view_issue_all(:1)
       OR 1 = decode(tgt_obj_id,NULL,-1,ms_ism_security_pkg.fn_ism_view_issue_orb(:1,tgt_obj_id))
       )
       and (:2 Is Null Or Issue_Owner_Org =:2)
       )
group by issue_source_type,period,total,lvl
