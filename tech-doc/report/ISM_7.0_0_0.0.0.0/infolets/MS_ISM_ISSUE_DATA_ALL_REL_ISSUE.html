SELECT issue_title,
   issue_id,
   ms_apps_utilities.get_user_full_name(issue_owner) issue_owner,
   issue_reviewer,
   issue_approverorg,
   issue_ownerorg
FROM ms_ism_issue
WHERE issue_title    != :2
AND issue_status NOT IN ('1','5','6')
AND issue_action NOT IN ('1', '2', '3')
AND ((1 = ms_ism_security_pkg.fn_ism_view_issue(:1) 
           AND 1 = ms_ism_security_pkg.fn_ism_view_issue_org(:1,issue_ownerorg,issue_approverorg,act_plan_approverorg, final_iss_approverorg)) 
           OR 1 = ms_ism_security_pkg.fn_ism_view_issue_all(:1)
           OR 1 = decode((SELECT MS_JOIN(tgt_obj_id) as tgt_obj_id
                                       FROM ms_ism_orb_relation_orb
                                      WHERE rel_source_id = issue_id
                                        AND tgt_obj_type = 'Organizations'
                                       GROUP BY rel_source_id),NULL,-1,ms_ism_security_pkg.fn_ism_view_issue_orb(:1,(SELECT MS_JOIN(tgt_obj_id) as tgt_obj_id
                                       FROM ms_ism_orb_relation_orb
                                      WHERE rel_source_id = issue_id
                                        AND tgt_obj_type = 'Organizations'
                                       GROUP BY rel_source_id))))
