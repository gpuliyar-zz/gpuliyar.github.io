  CREATE OR REPLACE FORCE VIEW "METRICSTREAM"."MS_ISM_USER_SECURITY_CHECK_V" ("ACTIVITY_ID", "ACTIVITY_NAME", "ACTIVITY_SHORT_NAME", "ORG_ENTITY_ID", "USER_NAME", "USER_ID", "PARENT_ORG_ENTITY_ID", "FLOW_DOWN") AS 
  SELECT  a.activity_id,
          a.activity_name,
          a.activity_short_name,
              d.Org_Entity_Id,
              F.User_Name,
              F.user_id,
              d.parent_org_entity_id, b.Flow_Down
      FROM  si_activities_t a,
            si_role_activities b,
            si_org_roles c,
            si_org_entities_t d,
            si_user_org_roles e,
            si_users_t f
      Where A.Activity_Name in ('ISM - Reopen Closed Actions','ISM - Update Closed Actions','ISM - Update Closed Issues',
                'ISM - Approver Comments','ISM - Reopen Closed Issues','ISM - View All Issues','ISM - View Issue',
                'ISM - View Issue By Related To Org','ISM - View Issue By  Issue Owner Org','ISM - View Issue By Issue Approver Org',
                 'ISM - Approve Action','ISM - Implement Action', 'ISM - Review Action')
            and a.activity_id = b.activity_id
            AND b.role_id       = c.role_id
            AND C.ORG_ENTITY_ID = D.ORG_ENTITY_ID
            AND C.ORG_ROLE_ID   = E.ORG_ROLE_ID
            And E.User_Id       = F.User_Id 
            And Sysdate Between (Nvl (F.Start_Date, Sysdate - 1)) And Nvl (F.End_Date, Sysdate + 1)
