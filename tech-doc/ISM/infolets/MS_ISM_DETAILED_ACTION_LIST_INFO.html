SELECT ACTION_TITLE,
  ACTION_STATUS AS ACTION_STATUS,
   ACTION_OWNER,
   Priority as PRIORITY,
   ACTION_DUE_DATE,
   ACTION_COMPLETED_ON,
   ISSUE_TITLE,
   ISSUE_OWNER,
   ACTION_TYPE,
   PERCENTAGE_COMPLETED AS PERCENTAGE_COMPLETED,
   PID,
   INSTANCEID,
   OWNER_ORGANIZATION,
   ISSUE_ID,
   APPROVER_ORG,
   ACTION_APPROVER,
   FINAL_APPROVER,
   ACTION_START_DATE,
   ACTION_DETAILS,
   ACTION_CLOSED_ON,
   ACTION_WORK_DONE,
   ACTION_RESULTS,
   NO_OF_EXTENSIONS,
   REOPEN_ACTION,
   OWNER_ORG_ID,
   APPROVER_ORG_ID,
   (	MS_ISM_RPT_UTILS.fn_get_report_json
                               ((select metric_id from si_metrics_t where metric_name='MS_ISM_ACTION'),
                                 PID,
                                 INSTANCEID,
                                '''flag=1'
                                ||CHR(38)
                                ||'emd=1'
                                ||CHR(38)
                                ||'x_edit_mode=Yes'
                                || CHR (38)  
                                || 'wrapper= No'
	          || CHR (38)  
                                || 'mode=view''') )As report_json,
ACTION_TITLE as action_title_adt,
action_id,
ACTION_OWNER_stored,
ISSUE_OWNER_stored,
ACTION_APPROVER_stored,
FINAL_APPROVER_stored,

ISSUE_OWNER as ISSUE_OWNER_ADT,
ACTION_APPROVER as ACTION_APPROVER_ADT,

OTHER_IMPACT

ORDER BY ORG_ACTION_TITLE
