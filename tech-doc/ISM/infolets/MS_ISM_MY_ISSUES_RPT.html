SELECT a.issue_title,
         ms_grc_comm_lov_cache_pkg.get_display_value_lov ('MS_ISM','MS_ISM_FORM_STATUS',a.issue_status, NULL,f.locale) AS issue_status,
         a.no_of_extensions AS no_of_extensions,
         a.issue_id AS issue_id,
         null AS issue_owner_full_name,
         NVL (a.no_of_action, 0) AS actions,
         a.issue_due_date AS issue_due_date,
         GREATEST (TRUNC (SYSDATE) - TRUNC (issue_due_date), 0) AS days_past_due,
         ms_grc_comm_lov_cache_pkg.get_display_value_lov ('MS_ISM','MS ISM Source Type',issue_source_type, NULL, f.locale) AS issue_source_type,
         CASE WHEN a.upstream_source_type = '0' THEN a.adhoc_source ELSE ms_ism_helper.get_issue_source_title (a.issue_source_type, a.issue_source_type_adt, a.issue_source) END AS issue_source,
         ms_grc_comm_lov_cache_pkg.get_display_value_lov ('MS_ISM','MS ISM Risk Rating',a.issue_risk_rating, NULL,f.locale) AS issue_risk_rating,
         (MS_ISM_RPT_UTILS.fn_get_report_json
                               (b.metric_id,
                                 a.dd_process_instance_id,
                                 a.dd_instance_id,
                                '''flag=1'
                                ||CHR(38)
                                ||'emd=2'
                                ||CHR(38)
                                ||'x_edit_mode=Yes'
                                || CHR (38)  
                                || 'wrapper= No''') ) AS report_json,
         a.issue_title AS issue_title_adt,
         d.user_full_name AS issue_owne_adt,
         a.issue_owner AS issue_owner_stored,
         a.issue_delegatee AS issue_delegatee_stored,
         e.user_full_name AS issue_delegatee_adt,
         a.issue_source AS issue_source_stored,
         a.issue_source_type AS issue_source_type_stored,
         issue_source_type_adt,
         issue_risk_rating AS rating_stored
    FROM ms_ism_issue a,
         ms_ism_user_detls_v d,
         ms_ism_user_detls_v e,
         (SELECT metric_id FROM si_metrics_t WHERE metric_name = 'MS_ISM_ISSUE') b,
         (SELECT x.locale FROM si_users_t x WHERE x.user_name = :LOGIN_ID) f
   WHERE (a.issue_owner = :LOGIN_ID OR a.issue_delegatee = :LOGIN_ID)
     AND a.issue_status NOT IN ('7', '10')
     AND a.issue_owner = d.user_name(+)
     AND a.issue_delegatee = e.user_name(+)
ORDER BY a.dd_created_on DESC
