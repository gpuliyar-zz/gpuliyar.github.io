MS_ISM_ISSUE_REPORTING_V 

  CREATE OR REPLACE FORCE VIEW "MS_ISM_ISSUE_REPORTING_V" ("SOURCE_TYPE_STORED", "ISSUE_ID", "ISSUE_OWNER_ORG", "ISSUE_SOURCE_TYPE", "UPSTREAM_SOURCE_TYPE", "ISSUE_DUE_DATE", "OVERDUE_THIS_MONTH", "OD_PAST_THREE_MONTH", "OD_BEYOND_THREE_MONTH", "DUE_THIS_MONTH", "DUE_WITHIN_X_MONTH", "DUE_BEYOND_X_MONTH", "ISSUE_APPROVERORG", "ACT_PLAN_APPROVERORG", "FINAL_ISS_APPROVERORG", "TGT_OBJ_ID", "ISSUE_SOURCE_TYPE_ADT") AS 
  SELECT source_type_stored, 
       issue_id,
       issue_owner_org,  
       issue_source_type,  
       UPSTREAM_SOURCE_TYPE,
       issue_due_date, 
  CASE
           WHEN TRUNC (issue_due_date) >= TRUNC (SYSDATE, 'MONTH')
            AND TRUNC (issue_due_date) < TRUNC (SYSDATE)
           THEN
               'Y'
           ELSE
               'N'
       END
           overdue_this_month,
       CASE
           WHEN TRUNC (issue_due_date) >= ADD_MONTHS (TRUNC (SYSDATE, 'MONTH'), -3)
            AND TRUNC (issue_due_date) < TRUNC (SYSDATE, 'MONTH')
           THEN
               'Y'
           ELSE
               'N'
       END
           od_past_three_month,
       CASE WHEN TRUNC (issue_due_date) < ADD_MONTHS (TRUNC (SYSDATE, 'MONTH'), -3) THEN 'Y' ELSE 'N' END od_beyond_three_month,
       CASE
           WHEN TRUNC (issue_due_date) >= TRUNC (SYSDATE)
            AND TRUNC (issue_due_date) <= LAST_DAY (TRUNC (SYSDATE))
           THEN
               'Y'
           ELSE
               'N'
       END
           due_this_month,
       CASE
           WHEN TRUNC (issue_due_date) > LAST_DAY (TRUNC (SYSDATE))
            AND TRUNC (issue_due_date) <= LAST_DAY (ADD_MONTHS (TRUNC (SYSDATE, 'MONTH'), config.aging_due_in_months))
           THEN
               'Y'
           ELSE
               'N'
  END due_within_x_month ,
  (CASE WHEN TRUNC (issue_due_date) > LAST_DAY (TRUNC (ADD_MONTHS (TRUNC (SYSDATE, 'MONTH'), config.aging_due_in_months))) THEN 'Y' ELSE 'N' END)
    due_beyond_x_month,
 issue_approverorg,
 act_plan_approverorg,
 final_iss_approverorg, 
 orb_tgt_obj_id,
 ISSUE_SOURCE_TYPE_ADT
FROM ms_ism_issue_reporting a,
(SELECT ms_apps_utilities.get_parameter_value (1, 'MS_ISM_Configuration', 'Enable_Ageing_Due_In_Months') aging_due_in_months FROM DUAL) config
