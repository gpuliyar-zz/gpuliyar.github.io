MS_ISM_ACTION_EDS_V 

  CREATE OR REPLACE FORCE VIEW "MS_ISM_ACTION_EDS_V" ("ISSUE_ID", "ACTION_ID", "TITLE", "STATUS", "SOURCE", "TYPE", "DESCRIPTION", "OWNER_ID", "OWNER_NAME", "APPROVER_ID", "APPROVER_NAME", "FINAL_APPROVER_ID", "INAL_APPROVER_NAME", "START_DATE", "DUE_DATE", "ORIGINAL_DUE_DATE", "PRIORITY", "WORK_DONE", "RESULTS", "ATTACH_FILES", "CLOSED_ON", "COMMENTS", "DELEGATEE_ID", "DELEGATEE_NAME", "PROGRESS_STATUS", "REVIEWER_ID", "REVIEWER_NAME", "DUE_DATE_EXTENSTION", "REOPEN_ACTION", "REQUESTED_DUE_DATE", "ACTION_UPDATE", "DOCUMENT_ID", "DOCUMENT_NAME", "FILE_DESCRIPTION", "DOC_ATTACHMENTS", "CREATED_BY", "CREATED_ON", "MODIFIED_BY", "MODIFIED_ON") AS 
  SELECT /*@public*/
A.ISSUE_ID AS ISSUE_ID,
A.ACTION_ID AS ACTION_ID,
A.ACTION_TITLE AS TITLE,
(SELECT Z.STATUS_NAME FROM MS_ISM_STATUS_TL_V Z WHERE Z.STATUS_ID=A.ACTION_STATUS) AS STATUS,
(SELECT Z.DISPLAYED_VALUE FROM MS_QS_LOV_ORG_VALUES_V Z WHERE Z.LOV_NAME = 'MS ISM Action Source' AND Z.STORED_VALUE=A.ACTION_SOURCE
AND z.locale_id=ms_grc_common_utils.fn_get_default_locale_id) AS SOURCE,
(SELECT Z.DISPLAYED_VALUE FROM MS_QS_LOV_ORG_VALUES_V Z WHERE Z.LOV_NAME = 'MS ISM Action Source Type' AND Z.STORED_VALUE=A.ACTION_SOURCE_TYPE
AND z.locale_id=ms_grc_common_utils.fn_get_default_locale_id) AS TYPE,
TO_CHAR(Regexp_Replace(A.ACTION_DETAILS,'<[^>]+>',NULL)) AS DESCRIPTION,
A.ACTION_OWNER AS OWNER_ID,
(SELECT Z.USER_FULL_NAME FROM MS_GRC_USER_FULL_NAME_V Z WHERE Z.USER_NAME=A.ACTION_OWNER) AS OWNER_NAME,
A.ACTION_APPROVER1 AS APPROVER_ID,
(SELECT Z.USER_FULL_NAME FROM MS_GRC_USER_FULL_NAME_V Z WHERE Z.USER_NAME=A.ACTION_APPROVER1) AS APPROVER_NAME,
A.ACTION_APPROVER2 AS FINAL_APPROVER_ID,
(SELECT Z.USER_FULL_NAME FROM MS_GRC_USER_FULL_NAME_V Z WHERE Z.USER_NAME=A.ACTION_APPROVER2) AS INAL_APPROVER_NAME,
A.ACTION_START_DATE AS START_DATE,
A.ACTION_DUE_DATE AS DUE_DATE,
A.ORIG_ACTION_DUE_DT AS ORIGINAL_DUE_DATE,
(SELECT Z.DISPLAYED_VALUE FROM MS_QS_LOV_ORG_VALUES_V Z WHERE Z.LOV_NAME = 'MS ISM Priority' AND Z.STORED_VALUE=A.ACTION_PRIORITY
AND z.locale_id=ms_grc_common_utils.fn_get_default_locale_id) AS PRIORITY,
TO_CHAR(Regexp_Replace(A.ACTION_WORK_DONE,'<[^>]+>',NULL)) AS WORK_DONE,
TO_CHAR(Regexp_Replace(A.ACTION_RESULTS,'<[^>]+>',NULL)) AS RESULTS,
SUBSTR(A.ACTION_ATTACHMENT,1,INSTR(A.ACTION_ATTACHMENT,'#',1,1)-1) AS  ATTACH_FILES,
A.ACTION_CLOSED_ON AS CLOSED_ON,
A.ACTION_COMMENTS AS COMMENTS,
A.ACTION_DELEGATEE AS DELEGATEE_ID,
(SELECT Z.USER_FULL_NAME FROM MS_GRC_USER_FULL_NAME_V Z WHERE Z.USER_NAME=A.ACTION_DELEGATEE) AS DELEGATEE_NAME,
(SELECT Z.DISPLAYED_VALUE FROM MS_QS_LOV_ORG_VALUES_V Z WHERE Z.LOV_NAME = 'MS ISM Update Status' AND TO_CHAR(Z.STORED_VALUE)=TO_CHAR(A.ACTION_PER_COMPLTE)
AND z.locale_id=ms_grc_common_utils.fn_get_default_locale_id) AS PROGRESS_STATUS,
A.ACTION_REVIEWER AS REVIEWER_ID,
(SELECT Z.USER_FULL_NAME FROM MS_GRC_USER_FULL_NAME_V Z WHERE Z.USER_NAME=A.ACTION_REVIEWER) AS REVIEWER_NAME,
A.NO_OF_EXTENSIONS AS DUE_DATE_EXTENSTION,
A.REOPEN_ACTION AS REOPEN_ACTION,
A.REQUESTED_DUE_DATE AS REQUESTED_DUE_DATE,
A.UPDATE_ACTION AS ACTION_UPDATE,
B.DOC_ID AS DOCUMENT_ID,
B.DOCUMENT_NAME AS DOCUMENT_NAME,
B.FILE_DESCRIPTION AS FILE_DESCRIPTION,
B.DOC_ATTACHMENTS AS DOC_ATTACHMENTS,
A.DD_CREATED_BY AS CREATED_BY,
A.DD_CREATED_ON AS CREATED_ON,
A.DD_MODIFIED_BY AS MODIFIED_BY,
A.DD_MODIFIED_ON AS MODIFIED_ON
FROM MS_ISM_ACTION A
LEFT JOIN MS_ISM_ACTION_DOC B ON
A.ACTION_ID=B.ACTION_ID
