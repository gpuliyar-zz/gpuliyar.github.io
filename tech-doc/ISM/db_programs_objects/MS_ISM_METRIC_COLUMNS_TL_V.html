MS_ISM_METRIC_COLUMNS_TL_V 

  CREATE OR REPLACE FORCE VIEW "MS_ISM_METRIC_COLUMNS_TL_V" ("METRIC_COLUMN_NAME", "METRIC_RESULT_COLUMN_NAME", "METRIC_ID", "LOCALE_ID", "LOCALE_STRING") AS 
  (SELECT NVL(TL.VALUE,T.COLUMN_NAME),NVL(TL.KEY,T.RESULT_COLUMN_NAME),nvl(t.metric_id,tl.page_id),T.LOCALE_ID,T.LOCALE_STRING 
  FROM (select a.column_name,a.metric_id,a.result_column_name,s.locale_id,s.locale_string from si_metric_columns a, 
    si_enabled_locales s )t, 
    si_labels_tl tl
     WHERE t.RESULT_COLUMN_NAME = tl.KEY(+) AND t.locale_id = tl.locale_id(+) 
    AND T.METRIC_ID=TL.PAGE_ID(+)
     and t.metric_id in (select metric_id from si_metrics_t where metric_name in ('MS_ISM_ISSUE','MS_ISM_ACTION')) )
