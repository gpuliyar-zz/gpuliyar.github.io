MS_ISM_ISSUE_EDS_V 

  CREATE OR REPLACE FORCE VIEW "MS_ISM_ISSUE_EDS_V" ("ISSUE_ID", "ISSUE_TITLE", "STATUS", "OWNER_ORG_ID", "OWNER_ORG_NAME", "OWNER_ID", "OWNER_NAME", "DESCRIPTION", "IDENTIFIED_ON", "TYPE", "EXCEPTION_TYPE", "RELATED_ISSUES", "REPEAT_ISSUE", "SOURCE_TYPE", "SOURCE", "ADHOC_SOURCE", "SOURCE_DETAILS", "REOPEN_ISSUE", "PRIORITY", "DUE_DATE", "ORIGINAL_DUE_DATE", "RATING", "IMPACT", "LIKELIHOOD", "LOSS_GAIN", "LOSS_GAIN_AMT", "SAME_APPROVER_FLAG", "ACTION_PLAN_APPROVER_ORG_ID", "ACTION_PLAN_APPROVER_ORG_NAME", "ACTION_PLAN_APPROVER_ID", "ACTION_PLAN_APPROVER_NAME", "FINAL_APPROVER_ORG_ID", "FINAL_APPROVER_ORG_NAME", "FINAL_APPROVER_ID", "FINAL_APPROVER_NAME", "APPROVER_ORG_ID", "APPROVER_ORG_NAME", "APPROVER_ID", "APPROVER_NAME", "ACTION_APPROVAL_BY", "ATTACH_FILES", "CLOSED_ON", "RESOLUTION_SUMMARY", "REVIEWER_ID", "REVIEWER_NAME", "CREATED_BY", "CREATED_ON", "MODIFIED_BY", "MODIFIED_ON", "RECOMMENDATION") AS 
  SELECT /*@public*/
A.ISSUE_ID AS ISSUE_ID,
A.ISSUE_TITLE AS ISSUE_TITLE,
(SELECT Z.STATUS_NAME FROM MS_ISM_STATUS_TL_V Z WHERE Z.STATUS_ID=A.ISSUE_STATUS) AS STATUS,
A.ISSUE_OWNERORG AS OWNER_ORG_ID,
(SELECT Z.ORG_ENTITY_NAME FROM SI_ORG_ENTITIES_T Z WHERE Z.ORG_ENTITY_ID=A.ISSUE_OWNERORG) AS OWNER_ORG_NAME,
A.ISSUE_OWNER AS OWNER_ID,
(SELECT Z.USER_FULL_NAME FROM MS_GRC_USER_FULL_NAME_V Z WHERE Z.USER_NAME=A.ISSUE_OWNER) AS OWNER_NAME,
TO_CHAR(Regexp_Replace(A.ISSUE_DETAILS,'<[^>]+>',NULL)) AS DESCRIPTION,
A.ISSUE_IDENTIFIED_ON AS IDENTIFIED_ON,
(SELECT Z.DISPLAYED_VALUE FROM MS_QS_LOV_ORG_VALUES_V Z WHERE Z.LOV_NAME = 'MS ISM Issue Type' AND Z.STORED_VALUE=B.ISSUE_TYPE
AND z.locale_id=ms_grc_common_utils.fn_get_default_locale_id) AS TYPE,
(SELECT Z.DISPLAYED_VALUE FROM MS_QS_LOV_ORG_VALUES_V Z WHERE Z.LOV_NAME = 'MS_CMP_ISM_EXCEPTION_TYPE' AND Z.STORED_VALUE=A.TYPE_OF_EXECUTION
AND z.locale_id=ms_grc_common_utils.fn_get_default_locale_id)
 AS EXCEPTION_TYPE,
C.ISSUE_RELATED_ISSUES AS RELATED_ISSUES,
(SELECT Z.DISPLAYED_VALUE FROM MS_QS_LOV_ORG_VALUES_V Z WHERE Z.LOV_NAME = 'MS_ISM_YES_NO' AND Z.STORED_VALUE=A.REPEAT_ISSUE
AND z.locale_id=ms_grc_common_utils.fn_get_default_locale_id) AS REPEAT_ISSUE,
(SELECT Z.DISPLAYED_VALUE FROM MS_QS_LOV_ORG_VALUES_V Z WHERE Z.LOV_NAME = 'MS ISM Source Type' AND Z.STORED_VALUE=A.ISSUE_SOURCE_TYPE
AND z.locale_id=ms_grc_common_utils.fn_get_default_locale_id) AS SOURCE_TYPE,
A.ISSUE_SOURCE AS SOURCE,
A.ADHOC_SOURCE AS ADHOC_SOURCE,
TO_CHAR(Regexp_Replace(A.ADHOC_SRC_DETAILS,'<[^>]+>',NULL)) AS SOURCE_DETAILS,
A.REOPEN_ISSUE AS REOPEN_ISSUE,
(SELECT Z.DISPLAYED_VALUE FROM MS_QS_LOV_ORG_VALUES_V Z WHERE Z.LOV_NAME = 'MS ISM Priority' AND Z.STORED_VALUE=A.ISSUE_RISK_PRIORITY
AND z.locale_id=ms_grc_common_utils.fn_get_default_locale_id) AS PRIORITY,
A.ISSUE_DUE_DATE AS DUE_DATE,
A.ORIG_ISSUE_DUE_DT AS ORIGINAL_DUE_DATE,
(SELECT Z.DISPLAYED_VALUE FROM MS_QS_LOV_ORG_VALUES_V Z WHERE Z.LOV_NAME = 'MS ISM Risk Rating' AND Z.STORED_VALUE=A.ISSUE_RISK_RATING
AND z.locale_id=ms_grc_common_utils.fn_get_default_locale_id) AS RATING,
(SELECT Z.DISPLAYED_VALUE FROM MS_QS_LOV_ORG_VALUES_V Z WHERE Z.LOV_NAME = 'MS ISM Impact' AND Z.STORED_VALUE=A.ISSUE_RISK_IMPACT
AND z.locale_id=ms_grc_common_utils.fn_get_default_locale_id) AS IMPACT,
(SELECT Z.DISPLAYED_VALUE FROM MS_QS_LOV_ORG_VALUES_V Z WHERE Z.LOV_NAME = 'MS ISM Likelihood' AND Z.STORED_VALUE=A.ISSUE_RISK_LIKLIHOOD
AND z.locale_id=ms_grc_common_utils.fn_get_default_locale_id) 
AS LIKELIHOOD,
(SELECT Z.DISPLAYED_VALUE FROM MS_QS_LOV_ORG_VALUES_V Z WHERE Z.LOV_NAME = 'MS ISM Loss Gain' AND Z.STORED_VALUE=A.ISSUE_RISK_LOSS_GAIN
AND z.locale_id=ms_grc_common_utils.fn_get_default_locale_id) 
AS LOSS_GAIN,
A.ISSUE_RISK_LOSS_GAIN_AMT AS LOSS_GAIN_AMT,
A.SAME_APPROVER_FLAG AS SAME_APPROVER_FLAG,
A.ACT_PLAN_APPROVERORG AS ACTION_PLAN_APPROVER_ORG_ID,
(SELECT Z.ORG_ENTITY_NAME FROM SI_ORG_ENTITIES_T Z WHERE Z.ORG_ENTITY_ID=A.ACT_PLAN_APPROVERORG) AS ACTION_PLAN_APPROVER_ORG_NAME,
A.ACTION_PLAN_APPROVER AS ACTION_PLAN_APPROVER_ID,
(SELECT Z.USER_FULL_NAME FROM MS_GRC_USER_FULL_NAME_V Z WHERE Z.USER_NAME=A.ACTION_PLAN_APPROVER) AS ACTION_PLAN_APPROVER_NAME,
A.FINAL_ISS_APPROVERORG AS FINAL_APPROVER_ORG_ID,
(SELECT Z.ORG_ENTITY_NAME FROM SI_ORG_ENTITIES_T Z WHERE Z.ORG_ENTITY_ID=A.FINAL_ISS_APPROVERORG) AS FINAL_APPROVER_ORG_NAME,
A.FINAL_ISS_APPROVER AS FINAL_APPROVER_ID,
(SELECT Z.USER_FULL_NAME FROM MS_GRC_USER_FULL_NAME_V Z WHERE Z.USER_NAME=A.FINAL_ISS_APPROVER) AS FINAL_APPROVER_NAME,
A.DUMMY_APPROVER_ORG AS Approver_Org_ID,
(SELECT Z.ORG_ENTITY_NAME FROM SI_ORG_ENTITIES_T Z WHERE Z.ORG_ENTITY_ID=A.DUMMY_APPROVER_ORG) AS APPROVER_ORG_NAME,
A.DUMMY_APPROVER AS APPROVER_ID,
(SELECT Z.USER_FULL_NAME FROM MS_GRC_USER_FULL_NAME_V Z WHERE Z.USER_NAME=A.DUMMY_APPROVER) AS APPROVER_NAME,
(SELECT Z.DISPLAYED_VALUE FROM MS_QS_ISM_LOV_ORG_VALUES_V Z WHERE Z.LOV_NAME='MS ISM Final Action Approval' AND Z.STORED_VALUE=A.FINAL_ACTION_APPROVAL
AND z.locale_id=ms_grc_common_utils.fn_get_default_locale_id) 
AS ACTION_APPROVAL_BY,
SUBSTR(A.ISSUE_ATTACH_FILES,1,INSTR(A.ISSUE_ATTACH_FILES,'#',1,1)-1) AS ATTACH_FILES,
A.ISSUE_CLOSED_ON AS CLOSED_ON,
TO_CHAR(Regexp_Replace(A.ISSUE_RESOLUTION_SUMMARY,'<[^>]+>',NULL))AS RESOLUTION_SUMMARY,
A.ISSUE_REVIEWER AS REVIEWER_ID,
(SELECT Z.USER_FULL_NAME FROM MS_GRC_USER_FULL_NAME_V Z WHERE Z.USER_NAME=A.ISSUE_REVIEWER) AS REVIEWER_NAME,
A.DD_CREATED_BY AS CREATED_BY,
A.DD_CREATED_ON AS CREATED_ON,
A.DD_MODIFIED_BY AS MODIFIED_BY,
A.DD_MODIFIED_ON AS MODIFIED_ON,
TO_CHAR(Regexp_Replace(A.ISSUE_RECOMMENDATION,'<[^>]+>',NULL))AS RECOMMENDATION
FROM MS_ISM_ISSUE A
LEFT JOIN MS_ISM_ISSUE_IST B ON
A.ISSUE_ID=B.ISSUE_ID
LEFT JOIN MS_ISM_ISSUE_RIS C ON
A.ISSUE_ID=C.ISSUE_ID
