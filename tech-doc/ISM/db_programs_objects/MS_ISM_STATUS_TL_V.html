MS_ISM_STATUS_TL_V 

  CREATE OR REPLACE FORCE VIEW "MS_ISM_STATUS_TL_V" ("STATUS_NAME", "STATUS_ID", "LOCALE_ID", "LOCALE_STRING") AS 
  ((select 
nvl(tl.status_name,t.status_name) as status_name, t.status_id as status_id,t.locale_id as locale_id, t.locale_string as locale_string
from (select a.status_name,a.status_id, s.locale_id, s.locale_string  
    FROM ms_ism_status a, si_enabled_locales s) t, 
    ms_ism_status_tl tl
     WHERE t.status_id = tl.status_id(+) AND t.locale_id = tl.locale_id(+)))
