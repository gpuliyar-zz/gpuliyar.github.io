MS_ISM_001_ISSUE_LIST_REPORT_V 

  CREATE OR REPLACE FORCE VIEW "METRICSTREAM"."MS_ISM_001_ISSUE_LIST_REPORT_V" ("ISSUE_TITLE", "ISSUE_STATUS", "ISSUE_OWNER", "ISSUE_RATING", "PRIORITY", "SOURCE_TYPE", "ISSUE_DUE_DATE", "ISSUE_INITIATOR", "FIRST_IDENTIFIED_ON", "ISSUE_ID", "ISSUE_DETAILS", "NO_OF_ACTIONS", "ISSUE_OWNER_ORG", "ISSUE_APPROVERORG", "OWNERORG", "APPROVERORG", "PROGRESS_STATUS", "PID", "INSTANCEID", "ORG", "AREA_OF_COMPLIANCE", "CONTROLS", "RISKS", "CORE_OBJECTS", "EXCEPTION_TYPE") AS 
  SELECT /*@public*/ 
    (SELECT ''
      || a.issue_title
      || ''
    FROM DUAL
    )                                                                                                      AS issue_title,
    a.issue_status                                                                                         AS issue_status,
    ms_ism_utilities.get_user_full_name (a.issue_owner)                                                   AS issue_owner,
    ms_apps_utilities.get_display_value (100000, 'MS ISM Risk Rating', a.issue_risk_rating, 1, NULL, NULL) AS issue_rating,
    ms_apps_utilities.get_display_value (100000, 'MS ISM Priority', a.issue_risk_priority, 1, NULL, NULL)  AS priority,
    ms_apps_utilities.get_display_value (100000, 'MS ISM Source Type', a.issue_source_type, 1, NULL, NULL) AS source_type,
    a.issue_due_date                                                                                       AS issue_due_date,
    ms_ism_utilities.get_user_full_name (a.dd_created_by)                                                 AS issue_initiator,
    a.issue_identified_on                                                                                  AS first_identified_on,
    a.issue_id                                                                                             AS issue_id,
    REGEXP_REPLACE (a.issue_details, '<[^>]+>', NULL)                                                      AS issue_details,
    (SELECT COUNT (1) FROM ms_ism_action WHERE issue_id = a.issue_id
    ) AS no_of_actions,
    (SELECT org_entity_name
    FROM si_org_entities_t
    WHERE org_entity_id = a.issue_ownerorg
    ) AS issue_owner_org,
    (SELECT org_entity_name
    FROM si_org_entities_t
    WHERE org_entity_id = a.issue_approverorg
    ) AS issue_approverorg,
    a.issue_ownerorg,
    a.issue_approverorg,
    ms_apps_utilities.get_display_value (100000, 'MS ISM Update Status', a.progress_status, 1, NULL, NULL) AS progress_status,
    a.dd_process_instance_id                                                                               AS pid,
    a.dd_instance_id                                                                                       AS instanceid,
    /*ms_ism_helper.get_comma_seprated_value ('IAO_TITLE', 'MS_ISM_ISSUE_IAO', a.issue_id, 'IAO')            AS org,
    ms_ism_helper.get_comma_seprated_value ('ATC_TITLE', 'MS_ISM_ISSUE_ATC', a.issue_id, 'ATC')            AS area_of_compliance,
    ms_ism_helper.get_comma_seprated_value ('IAC_TITLE', 'MS_ISM_ISSUE_IAC', a.issue_id, 'IAC')            AS controls,
    ms_ism_helper.get_comma_seprated_value ('IAR_TITLE', 'MS_ISM_ISSUE_IAR', a.issue_id, 'IAR')            AS risks,
    ms_ism_helper.get_comma_seprated_value ('ICO_TITLE', 'MS_ISM_ISSUE_ICO', a.issue_id, 'ICO')            AS core_objects*/
    ms_ism_helper.get_coma_seprtd_orb_value ('Organization(s)',a.issue_id)  AS org,
    ms_ism_helper.get_coma_seprtd_orb_value ('Area(s) of Compliance', a.issue_id) AS area_of_compliance,
    ms_ism_helper.get_coma_seprtd_orb_value ('Control(s)', a.issue_id) AS controls,
    ms_ism_helper.get_coma_seprtd_orb_value ('Risk(s)', a.issue_id) AS risks,
    ms_ism_helper.get_coma_seprtd_orb_value ('Core Object(s)', a.issue_id) AS core_objects,
    ms_apps_utilities.get_display_value (100000, 'MS ISM Classification Type', a.TYPE_OF_EXECUTION, 1, NULL, NULL) AS EXCEPTION_TYPE
  FROM ms_ism_issue a
    --     , ms_ism_issue_v c
    --    WHERE     a.issue_id = c.issue_id
    --          AND c.instance_id = (SELECT MAX (instance_id)
    --                                 FROM ms_ism_issue_v
    --                                WHERE issue_id = a.issue_id)
    --          AND c.instance_rec_num = 1
    --          AND
  WHERE a.issue_status NOT IN ('New', 'Issue Reviewed', 'Pending Review', 'Pending Approval') -- start of fixing of bug - 40918
    --          AND EXISTS
    --                 (SELECT previous_stage
    --                    FROM ms_ism_issue_v
    --                   WHERE issue_id = a.issue_id AND dd_current_stage = 4)
  AND a.reached_manage = 'Y'
