    A.Action_Priority                                                                                                 AS Priority_Stored,
    Ms_ism_helper.Fn_Get_Action_Type(C.Action_Type)                                                                   AS Action_Type,
    c.Action_Type                                                                                                     AS Action_Type_stored,
    A.action_per_complte                                                                                              AS Percentage_Completed,
    Regexp_Replace (A.action_details, '<[^>]+>', NULL)                                                                AS Action_Details,
    Regexp_Replace (A.Action_Work_Done, '<[^>]+>', NULL)                                                              AS Work_Done,
    Regexp_Replace (A.Action_Results, '<[^>]+>', NULL)                                                                AS Results,
    ms_ism_helper.get_comma_sep_value_for_mlov ('OTHER_IMPACT','MS_ISM_ACTION_IOR',a.action_id,'ORG') as other_impact,
    'Pushinfolet?id='

    || CHR (38)

    || 'emd=1'
    || CHR (38)
    || 'amp;'
    || 'x_edit_mode=No'
