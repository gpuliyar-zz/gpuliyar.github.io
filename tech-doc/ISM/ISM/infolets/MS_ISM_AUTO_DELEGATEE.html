SELECT DISTINCT
       MS_ISM_HELPER.get_select_url_for_rpt (
          a.USER_FULL_NAME,
          a.USER_NAME,
          'RADIO')
          AS Radio,
       a.user_name AS USER_NAME,
       a.USER_FULL_NAME AS USER_FULL_NAME,
       MS_ISM_HELPER.get_roles_for_delegatee(user_name) AS Roles,
       A.User_Id,''
  FROM ms_ism_org_flow_up_v a,
  (Select org_entity_id From si_org_entities_t
	start With Org_Entity_Id in (Select Distinct Org_Entity_Id From Ms_Ism_Org_Flow_Up_V B Where User_Name = :logged_user)
	connect by org_entity_id = prior parent_org_entity_id)tmp
 WHERE activity_name in ('ISM - Issue Owner','ISM - Implement Action')
       And User_Name Not In (:logged_user)
       And User_Name Not In (Select Delegatee From Ms_Ism_Auto_Delegation Where Logged_User=:logged_user And Remove_Delegation='no')
       AND a.org_entity_id = tmp.org_entity_id
