MS_GRC_ASSET_CLASS_RV 

  CREATE OR REPLACE FORCE VIEW "MS_GRC_ASSET_CLASS_RV" ("OBJECT_ID", "OBJECT_NAME", "OWNERS", "STATUS", "DESCRIPTION", "VALID_FROM", "VALID_UNTIL", "OBJECT_LEVEL", "OBJECT_PARENT", "ASSET_CLASS_TYPE", "CPE_OR_EQUIVALENT_DEFINITION", "RELATED_RISKS", "RELATED_FRAMEWORK_REFERENCES", "RELATED_REFERENCES", "RELATED_AREAS_OF_COMPLIANCE", "RELATED_CONTROLS", "RESTRICT_ACCESS_TO", "DD_OBJECT_TYPE", "DD_PROCESS_INSTANCE_ID", "DD_INSTANCE_ID", "DD_EDIT_FLAG", "OWNER_ORG_IDS", "RELATED_ASSETS", "RELATED_QUEST_PROC", "MODEL_TYPE", "EXPIRED_FLAG", "EDIT", "DISPLAY_NAME", "OBJECT_TYPE", "OVER_RIDE_ACTION", "EDIT_ACTION_PARAMETERS") AS 
  SELECT b.object_id,
    b.object_name,
    (SELECT ms_concat(usr.first_name
      || ' '
      || usr.last_name) AS owners
    FROM ms_grc_core_object_own d,
      si_users_t usr,
      si_ent e
    WHERE usr.user_name   = d.owners
    AND usr.enterprise_id = e.ENTERPRISE_ID
    AND b.OBJECT_ID       = D.OBJECT_ID
    )            AS OWNERS,
    b.obj_status AS STATUS,
    b.description,
    b.valid_from,
    b.valid_until,
    b.object_level     AS OBJECT_LEVEL,
    b.object_parent    AS OBJECT_PARENT,
    a.asset_class_type AS asset_class_type,
    a.cpe_or_equivalent_definition,
    (SELECT MS_GRC_FN_CONCAT (orb.tgt_obj_id)
    FROM ms_grc_orb_relations_pub_v orb,
      ms_grc_risk base
    WHERE src_obj_type   = 'MS_GRC_ASSET_CLASS'
    AND orb.tgt_obj_type = 'MS_GRC_RISK'
    AND orb.src_obj_id   = b.object_id
    AND orb.tgt_obj_id   = base.object_id
    ) AS Related_risks,
    (SELECT MS_GRC_FN_CONCAT(orb.tgt_obj_id)
    FROM ms_grc_orb_relations_pub_v orb,
      ms_grc_model_reference base
    WHERE src_obj_type   = 'MS_GRC_ASSET_CLASS'
    AND orb.tgt_obj_type = 'MS_GRC_MODEL_REFERENCE'
    AND orb.src_obj_id   = b.object_id
    AND orb.tgt_obj_id   = base.object_id
    ) AS Related_framework_references,
    (SELECT MS_GRC_FN_CONCAT(orb.tgt_obj_id)
    FROM ms_grc_orb_relations_pub_v orb,
      ms_grc_reference base
    WHERE src_obj_type   = 'MS_GRC_ASSET_CLASS'
    AND orb.tgt_obj_type = 'MS_GRC_REFERENCE'
    AND orb.src_obj_id   = b.object_id
    AND orb.tgt_obj_id   = base.object_id
    ) AS related_references,
    (SELECT MS_GRC_FN_CONCAT(orb.tgt_obj_id)
    FROM ms_grc_orb_relations_pub_v orb,
      ms_grc_area_of_complia base
    WHERE src_obj_type   = 'MS_GRC_ASSET_CLASS'
    AND orb.tgt_obj_type = 'MS_GRC_AREA_OF_COMPLIANCE'
    AND orb.src_obj_id   = b.object_id
    AND orb.tgt_obj_id   = base.object_id
    ) AS Related_Areas_of_Compliance,
    (SELECT MS_GRC_FN_CONCAT(orb.tgt_obj_id)
    FROM ms_grc_orb_relations_pub_v orb,
      ms_grc_compliance_obje base
    WHERE src_obj_type   = 'MS_GRC_ASSET_CLASS'
    AND orb.tgt_obj_type = 'MS_GRC_CONTROL'
    AND orb.src_obj_id   = b.object_id
    AND orb.tgt_obj_id   = base.object_id
    ) AS Related_Controls,
    b.restrict_access_to,
    a.dd_object_type,
    a.dd_process_instance_id,
    a.dd_instance_id,
    NVL(a.dd_edit_flag, 'Y') AS DD_EDIT_FLAG,
    (SELECT MS_CONCAT(OWNER_ORGANIZATIONS)
    FROM MS_GRC_ASSET_CLASS_OGN E
    WHERE E.OBJECT_ID = B.OBJECT_ID
    ) AS owner_org_ids,
    (SELECT ms_concat(orb.tgt_obj_id)
    FROM ms_grc_orb_relations_pub_v orb,
      ms_grc_core_object base
    WHERE src_obj_type   = 'MS_GRC_ASSET_CLASS'
    AND orb.tgt_obj_type = 'MS_GRC_ASSET'
    AND orb.src_obj_id   = b.object_id
    AND orb.tgt_obj_id   = base.object_id
    ) AS RELATED_ASSETS,
    (SELECT ms_concat(orb.tgt_obj_id)
    FROM ms_grc_orb_relations_pub_v orb,
      ms_grc_question_proc base
    WHERE src_obj_type   = 'MS_GRC_ASSET_CLASS'
    AND orb.tgt_obj_type = 'MS_GRC_QUESTION_PROCEDURE'
    AND orb.src_obj_id   = a.object_id
    AND orb.tgt_obj_id   = base.object_id
    )                                                            AS RELATED_QUEST_PROC,
    NULL                                                         AS model_type,
    ms_Grc_utilities.INCLUDE_eXPIRED(B.VALID_FROM,B.VALID_UNTIL) AS EXPIRED_FLAG,
    (SELECT ms_Grc_utilities.Get_Edit_Url(b.DD_OBJECT_TYPE,a.DD_PROCESS_INSTANCE_ID, a.DD_INSTANCE_ID,NVL(a.DD_EDIT_FLAG,'Y'),b.object_name)
    FROM dual
    )                AS EDIT,
    b.object_name    AS Display_name,
    a.dd_object_type AS Object_Type,
    'Y'              AS OVER_RIDE_ACTION,
    'proc='
    || TO_CHAR(a.dd_process_instance_id)
    ||'|instid='
    ||TO_CHAR(a.dd_instance_id)
    ||'|edit_flag='
    ||NVL(a.dd_edit_flag,'Y') AS EDIT_ACTION_PARAMETERS
  FROM ms_grc_asset_class a,
    ms_grc_core_object b
  WHERE a.object_id    = b.object_id
  AND a.DD_OBJECT_TYPE = 'MS_GRC_ASSET_CLASS'
