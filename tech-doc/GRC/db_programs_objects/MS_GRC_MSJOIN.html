IS
  Static
FUNCTION ODCIAggregateInitialize(
    io_SrcContext IN OUT MS_GRC_MSJOIN)
  RETURN NUMBER
IS
BEGIN
  io_SrcContext           := MS_GRC_MSJOIN(NULL,NULL,NULL);
  io_SrcContext.Delimiter := ',';
  RETURN ODCIConst.Success;
END ODCIAggregateInitialize;
Member FUNCTION ODCIAggregateIterate(
    Self  IN OUT MS_GRC_MSJOIN,
    Value IN VARCHAR2)
  RETURN NUMBER
IS
  temp_val VARCHAR2(4000);
  gv_delimiter_space varchar2(10) := ', ';
  gv_delimiter varchar2(10) := ',';
  gv_double_quotes VARCHAR2(10) :='"';
  gv_esc_double_quotes VARCHAR2(10) :='""';
    lv_value VARCHAR2(4000) := Value;

BEGIN
 IF INSTR (Value, gv_delimiter) > 0 OR INSTR (Value, gv_double_quotes) > 0 THEN
      lv_value:= gv_double_quotes||REPLACE(lv_value,gv_double_quotes,gv_esc_double_quotes)||gv_double_quotes;
  END IF;
   If Self.ResultString Is Null Then
            Self.ResultString := Self.ResultString || lv_value;
         Else
            Self.ResultString := Self.ResultString || Self.Delimiter || lv_value;
         End If;

  RETURN ODCIConst.Success;
END ODCIAggregateIterate;
Member FUNCTION ODCIAggregateTerminate(
    Self IN MS_GRC_MSJOIN,
    o_ReturnValue OUT VARCHAR2,
    i_Flags IN NUMBER)

     o_Returnvalue := Self.ResultString;
  RETURN ODCIConst.Success;
END ODCIAggregateTerminate;
Member FUNCTION ODCIAggregateMerge(
    Self   IN OUT MS_GRC_MSJOIN,
    i_Ctx2 IN MS_GRC_MSJOIN)
  RETURN NUMBER
IS
BEGIN
  IF Self.ResultString IS NULL AND i_Ctx2.ResultString IS NOT NULL THEN
    Self.ResultString     := i_Ctx2.ResultString;
  Elsif Self.ResultString IS NOT NULL AND i_Ctx2.ResultString IS NOT NULL THEN
    Self.ResultString     := Self.ResultString || Self.Delimiter || i_Ctx2.ResultString;
  END IF;
  RETURN ODCIConst.Success;
END ODCIAggregateMerge;
END;
