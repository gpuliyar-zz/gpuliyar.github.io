MS_GRC_REQUIREMENT_RV 

  CREATE OR REPLACE FORCE VIEW "MS_GRC_REQUIREMENT_RV" ("OBJECT_NAME", "OBJECT_ID", "OWNERS", "VALID_UNTIL", "VALID_FROM", "DD_OBJECT_TYPE", "DESCRIPTION", "REQUIREMENT_TYPE", "CLASSIFICATION", "IMPACT", "PENALTIES", "KEY_REQUIREMENT", "SECTION_NO", "DD_PROCESS_INSTANCE_ID", "DD_INSTANCE_ID", "DD_EDIT_FLAG", "RESTRICT_ACCESS_TO", "STATUS", "OBJECT_PARENT", "OBJECT_LEVEL", "FREQUENCY", "OBJECT_NAME_URL", "OWNER_ORG_IDS", "APPLIES_TO_ORGANIZATIONS", "RELATED_STANDARDS", "RELATED_FUNCTIONS", "RELATED_PROCESSES", "RELATED_CONTROLS", "RELATED_FRAMEWORK_REFERENCES", "RELATED_RISKS", "RELATED_AREAS_OF_COMPLIANCE", "RELATED_REFERENCES", "REL_FRAME_REFERENCE_TYPE", "DISPLAY_NAME", "OVER_RIDE_ACTION", "EDIT_ACTION_PARAMETERS", "EXPIRED_FLAG") AS 
  SELECT a.object_name,
    a.object_id,
    (SELECT ms_concat(usr.first_name
      || ' '
      || usr.last_name) AS owners
    FROM ms_grc_requirement_own d,
      si_users_t usr ,
      si_ent e
    WHERE usr.user_name   = d.owners
    AND USR.ENTERPRISE_ID = E.ENTERPRISE_ID
    AND a.OBJECT_ID       = D.OBJECT_ID
    ) AS OWNERS,
    valid_until,
    valid_from,
    DD_OBJECT_TYPE,
    DESCRIPTION,
    requirement_type           AS REQUIREMENT_TYPE,
    requirement_classification AS CLASSIFICATION,
    requirement_impact         AS impact,
    requirement_penalties penalties,
     NVL(key_requirement, 'no') as key_requirement,
    section_no,
    dd_process_instance_id,
    dd_instance_id,
    NVL (dd_edit_flag, 'Y') AS DD_EDIT_FLAG,
    restrict_access_to,
    OBJ_STATUS      AS status,
    a.object_parent AS OBJECT_PARENT,
    OBJECT_LEVEL,
    FREQUENCY,
    ms_grc_utilities.get_edit_url (dd_object_type, dd_process_instance_id, dd_instance_id, NVL(DD_EDIT_FLAG,'Y'), a.OBJECT_NAME) AS object_name_url,
    (SELECT MS_CONCAT (OWNER_ORGANIZATIONS)
    FROM ms_grc_requirement_ORG E
    WHERE E.OBJECT_ID = A.OBJECT_ID
    ) AS owner_org_ids,
    (SELECT ms_concat(orb.tgt_obj_id)
    FROM ms_grc_orb_relations_pub_v orb,
      SI_ORG_ENTITIES_T base
    WHERE src_obj_type   = 'MS_GRC_REQUIREMENT'
    AND orb.tgt_obj_type = 'MS_GRC_ORGANIZATION'
    AND orb.src_obj_id   = a.object_id
    AND orb.tgt_obj_id   = base.org_entity_id
    ) AS Applies_to_Organizations,
    (SELECT ms_concat(orb.tgt_obj_id)
    FROM MS_GRC_ORB_RELATIONS_PUB_V ORB,
      ms_grc_area_of_complia base
    WHERE src_obj_type   = 'MS_GRC_REQUIREMENT'
    AND orb.tgt_obj_type = 'MS_GRC_STANDARD'
    AND orb.src_obj_id   = a.object_id
    AND orb.tgt_obj_id   = base.object_id
    ) AS RELATED_STANDARDS ,
    (SELECT ms_concat(orb.tgt_obj_id)
    FROM MS_GRC_ORB_RELATIONS_PUB_V ORB,
      ms_grc_function base
    WHERE src_obj_type   = 'MS_GRC_REQUIREMENT'
    AND orb.tgt_obj_type = 'MS_GRC_FUNCTION'
    AND orb.src_obj_id   = a.object_id
    AND ORB.TGT_OBJ_ID   = BASE.OBJECT_ID
    ) AS RELATED_FUNCTIONS ,
    (SELECT ms_concat(orb.tgt_obj_id)
    FROM MS_GRC_ORB_RELATIONS_PUB_V ORB,
      ms_grc_core_object base
    WHERE SRC_OBJ_TYPE   = 'MS_GRC_REQUIREMENT'
    AND orb.tgt_obj_type = 'MS_GRC_PROCESS'
    AND orb.src_obj_id   = a.object_id
    AND ORB.TGT_OBJ_ID   = BASE.OBJECT_ID
    ) AS Related_Processes ,
    (SELECT ms_concat(orb.tgt_obj_id)
    FROM MS_GRC_ORB_RELATIONS_PUB_V ORB,
      ms_grc_compliance_obje base
    WHERE SRC_OBJ_TYPE   = 'MS_GRC_REQUIREMENT'
    AND orb.tgt_obj_type = 'MS_GRC_CONTROL'
    AND orb.src_obj_id   = a.object_id
    AND ORB.TGT_OBJ_ID   = BASE.OBJECT_ID
    ) AS Related_Controls ,
    (SELECT ms_concat(orb.tgt_obj_id)
    FROM ms_grc_orb_relations_pub_v orb,
      ms_grc_model_reference base
    WHERE src_obj_type   = 'MS_GRC_REQUIREMENT'
    AND orb.tgt_obj_type = 'MS_GRC_MODEL_REFERENCE'
    AND orb.src_obj_id   = a.object_id
    AND ORB.TGT_OBJ_ID   = BASE.OBJECT_ID
    ) AS RELATED_FRAMEWORK_REFERENCES,
    (SELECT ms_concat(orb.tgt_obj_id)
    FROM MS_GRC_ORB_RELATIONS_PUB_V ORB,
      ms_grc_risk base
    WHERE src_obj_type   = 'MS_GRC_REQUIREMENT'
    AND orb.tgt_obj_type = 'MS_GRC_RISK'
    AND orb.src_obj_id   = a.object_id
    AND ORB.TGT_OBJ_ID   = BASE.OBJECT_ID
    ) AS RELATED_RISKS,
    (SELECT ms_concat(orb.tgt_obj_id)
    FROM MS_GRC_ORB_RELATIONS_PUB_V ORB,
      ms_grc_area_of_complia base
    WHERE src_obj_type   = 'MS_GRC_REQUIREMENT'
    AND orb.tgt_obj_type = 'MS_GRC_AREA_OF_COMPLIANCE'
    AND orb.src_obj_id   = a.object_id
    AND ORB.TGT_OBJ_ID   = BASE.OBJECT_ID
    ) AS RELATED_AREAS_OF_COMPLIANCE,
    (SELECT ms_concat(orb.tgt_obj_id)
    FROM MS_GRC_ORB_RELATIONS_PUB_V ORB,
      ms_grc_reference base
    WHERE SRC_OBJ_TYPE   = 'MS_GRC_REQUIREMENT'
    AND orb.tgt_obj_type = 'MS_GRC_REFERENCE'
    AND orb.src_obj_id   = a.object_id
    AND ORB.TGT_OBJ_ID   = BASE.OBJECT_ID
    )                                                                                             AS RELATED_REFERENCES,
    MS_GRC_UTILITIES.GET_MODEL_REFERENCE_TYPE(OBJECT_ID,DD_OBJECT_TYPE,'MS_GRC_MODEL_REFERENCE' ) as REL_FRAME_REFERENCE_TYPE,
    OBJECT_NAME as DISPLAY_NAME,
    'Y' AS OVER_RIDE_ACTION,
    'proc='
    || TO_CHAR(a.dd_process_instance_id)
    ||'|instid='
    ||TO_CHAR(a.dd_instance_id)
    ||'|edit_flag='
    ||NVL(A.DD_EDIT_FLAG,'Y')                                AS EDIT_ACTION_PARAMETERS,
    MS_GRC_UTILITIES.INCLUDE_EXPIRED(VALID_FROM,VALID_UNTIL) AS EXPIRED_FLAG
  FROM MS_GRC_REQUIREMENT A
  where DD_OBJECT_TYPE = 'MS_GRC_REQUIREMENT'
