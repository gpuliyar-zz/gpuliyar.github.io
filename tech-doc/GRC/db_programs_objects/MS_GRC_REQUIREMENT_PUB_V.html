  CREATE OR REPLACE FORCE VIEW "MS_GRC_REQUIREMENT_PUB_V" ("OBJECT_ID", "OBJECT_NAME", "OBJECT_NAME_URL", "DESCRIPTION", "DESCRIPTION_HTM", "OWNERS", "OWNER_ORGANIZATIONS", "RESTRICT_ACCESS_TO", "OBJ_STATUS", "VALID_FROM", "VALID_UNTIL", "OBJECT_LEVEL", "OBJECT_PARENT", "SECTION_NO", "REQUIREMENT_CLASSIFICATION", "REQUIREMENT_TYPE", "REQUIREMENT_IMPACT", "REQUIREMENT_PENALTIES", "REQUIREMENT_PENALTIES_HTM", "FREQUENCY", "NEXT_SCHEDULED_DATE", "START_AFTER", "DUE_BY", "RTF_ATTACH", "DD_OBJECT_TYPE", "DD_PROCESS_INSTANCE_ID", "DD_INSTANCE_ID", "INITIATED_BY", "CREATED_ON", "MODIFIED_ON", "LEVEL_1_APPROVER", "LEVEL_2_APPROVER", "KEY_REQUIREMENT", "TYPE_OF_COMPLIANCE") AS 
  SELECT /*@public*/ OBJECT_ID,
    OBJECT_NAME,
    ms_grc_utilities.get_edit_url (A.dd_object_type,
                                               A.dd_process_instance_id,
                                               A.dd_instance_id,
                                               NVL(a.DD_EDIT_FLAG,'Y'),
                                               A.OBJECT_NAME)            AS OBJECT_NAME_URL,
    regexp_replace(DESCRIPTION, '<[^>]+>', NULL) AS DESCRIPTION,
    DESCRIPTION                                  AS DESCRIPTION_HTM,
    (SELECT MS_CONCAT(OWNERS)
    FROM MS_GRC_REQUIREMENT_OWN B
    WHERE B.OBJECT_ID = A.OBJECT_ID
    ) AS OWNERS,
    (SELECT MS_CONCAT(OWNER_ORGANIZATIONS)
    FROM MS_GRC_REQUIREMENT_ORG B
    WHERE B.OBJECT_ID = A.OBJECT_ID
    ) AS OWNER_ORGANIZATIONS,
    RESTRICT_ACCESS_TO,
    obj_STATUS,
    VALID_FROM,
    VALID_UNTIL,
    OBJECT_LEVEL,
    OBJECT_PARENT,
    SECTION_NO,
    REQUIREMENT_CLASSIFICATION,
    REQUIREMENT_TYPE,
    REQUIREMENT_IMPACT,
    regexp_replace(REQUIREMENT_PENALTIES, '<[^>]+>', NULL) AS REQUIREMENT_PENALTIES,
    REQUIREMENT_PENALTIES                                  AS REQUIREMENT_PENALTIES_HTM,
    FREQUENCY,
    NEXT_sCHEDULED_DATE,
    START_aFTER,
    DUE_BY,
    RTF_aTTACH,
    DD_OBJECT_TYPE,
    DD_PROCESS_INSTANCE_ID,
    DD_INSTANCE_ID,
    NVL(OBJ_MODIFIED_BY,OBJ_CREATED_BY) AS INITIATED_BY,
    DD_CREATED_ON AS CREATED_ON,
    DD_MODIFIED_ON AS MODIFIED_ON,
    LEVEL_1_APPROVER AS LEVEL_1_APPROVER ,
	LEVEL_2_APPROVER AS LEVEL_2_APPROVER ,
	KEY_REQUIREMENT as KEY_REQUIREMENT,
	TYPE_OF_COMPLIANCE as TYPE_OF_COMPLIANCE

  from MS_GRC_REQUIREMENT a
