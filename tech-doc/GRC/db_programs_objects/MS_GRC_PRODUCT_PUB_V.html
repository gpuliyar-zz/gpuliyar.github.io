MS_GRC_PRODUCT_PUB_V 

  CREATE OR REPLACE FORCE VIEW "MS_GRC_PRODUCT_PUB_V" ("OBJECT_ID", "OBJECT_NAME", "OBJECT_NAME_URL", "DESCRIPTION", "DESCRIPTION_HTM", "OWNERS", "OWNER_ORGANIZATIONS", "PRODUCT_TYPE", "PRODUCT_LOCATION", "SALES_CHANNEL", "RESTRICT_ACCESS_TO", "OBJ_STATUS", "PRODUCT_CATEGORY", "OBJECT_LEVEL", "OBJECT_PARENT", "VALID_FROM", "VALID_UNTIL", "RTF_ATTACH", "DD_OBJECT_TYPE", "DD_PROCESS_INSTANCE_ID", "DD_INSTANCE_ID", "INITIATED_BY", "CREATED_ON", "MODIFIED_ON", "IS_ORPHAN", "LEVEL_1_APPROVER", "LEVEL_2_APPROVER") AS 
  SELECT /*@public*/ OBJECT_ID,
    OBJECT_NAME,
    ms_grc_utilities.get_edit_url (A.dd_object_type,
                                               A.dd_process_instance_id,
                                               A.dd_instance_id,
                                               NVL(a.DD_EDIT_FLAG,'Y'),
                                               A.OBJECT_NAME)            AS OBJECT_NAME_URL,
    regexp_replace(DESCRIPTION, '<[^>]+>', NULL) AS DESCRIPTION,
    DESCRIPTION                                  AS DESCRIPTION_HTM,
    (SELECT MS_CONCAT(OWNERS)
    FROM ms_grc_product_OWN B
    WHERE B.OBJECT_ID = A.OBJECT_ID
    ) AS OWNERS,
    (SELECT MS_CONCAT(OWNER_ORGANIZATIONS)
    FROM MS_GRC_PRODUCT_OGN B
    WHERE B.OBJECT_ID = A.OBJECT_ID
    ) AS OWNER_ORGANIZATIONS,
    (SELECT MS_CONCAT(PRODUCT_TYPE)
    FROM ms_grc_product_typ K
    WHERE K.OBJECT_ID = A.OBJECT_ID
    ) AS PRODUCT_TYPE,
    (SELECT MS_CONCAT(PRODUCT_LOCATION)
    FROM ms_grc_product_loc p
    WHERE p.OBJECT_ID = A.OBJECT_ID
    ) AS PRODUCT_LOCATION,
    (SELECT MS_CONCAT(SALES_CHANNEL)
    FROM ms_grc_product_sch q
    WHERE q.OBJECT_ID = A.OBJECT_ID
    ) AS SALES_CHANNEL,
    RESTRICT_ACCESS_TO,
    obj_STATUS,
    PRODUCT_CATEGORY,
    OBJECT_LEVEL,
    OBJECT_PARENT,
    VALID_FROM,
    VALID_UNTIL,
    RTF_aTTACH,
    DD_OBJECT_TYPE,
    DD_PROCESS_INSTANCE_ID,
    DD_INSTANCE_ID,
    NVL(OBJ_MODIFIED_BY,OBJ_CREATED_BY) AS INITIATED_BY,
    dd_CREATED_ON AS CREATED_ON,
    dd_MODIFIED_ON AS MODIFIED_ON, 
	(SELECT DECODE (count(1),0,2,1)
            FROM MS_GRC_ORB_RELATIONS_PUB_V
           WHERE SRC_OBJ_TYPE = 'MS_GRC_PRODUCT'
             AND SRC_OBJ_ID = A.object_id
             and SRC_OBJ_TYPE <> 'MS_GRC_ORGANIZATION'
             and TGT_OBJ_TYPE <> 'MS_GRC_ORGANIZATION'
             And Rownum =1) As IS_ORPHAN,
	 LEVEL_1_APPROVER AS LEVEL_1_APPROVER ,
	LEVEL_2_APPROVER AS LEVEL_2_APPROVER
  FROM MS_GRC_PRODUCT a
