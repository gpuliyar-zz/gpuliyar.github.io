MS_GRC_EXCEPTION_RV 

  CREATE OR REPLACE FORCE VIEW "MS_GRC_EXCEPTION_RV" ("EXCEPTION_ID", "EXCEPTION_FOR", "TYPE_EXCEPTION", "VALID_FROM", "EXPIRES_ON", "STATUS", "OBJ_CREATED_ON", "OBJ_CREATED_BY", "OBJ_MODIFIED_ON", "OBJ_MODIFIED_BY", "APPROVER", "ITEMS", "ITEMS_DISP", "EXPLANATION", "EXCEPTION_APPROVER", "APPLIES_TO_ORGANIZATIONS", "RELATED_AREAS_OF_COMPLIANCE", "RELATED_ASSET_CLASSES", "RELATED_ASSETS", "RELATED_CONTROLS", "RELATED_FRAMEWORK_REFERENCES", "RELATED_PROCESSES", "RELATED_REFERENCES", "RELATED_REG_BODIES", "RELATED_REQUIREMENTS", "RELATED_RISKS", "RELATED_STANDARDS", "DD_OBJECT_TYPE", "DD_PROCESS_INSTANCE_ID", "DD_INSTANCE_ID", "DD_EDIT_FLAG", "OBJECT_TYPE", "OVER_RIDE_ACTION", "EDIT_ACTION_PARAMETERS", "EDIT", "EXPIRED_FLAG") AS 
  SELECT a.object_id Exception_ID,--a.exception_for exception_for_stored_val,
    exception_for,
    type_exception,
    a.valid_from valid_from,
    a.Valid_Until Expires_On,
    a.obj_status Status,
    a.OBJ_CREATED_ON  AS OBJ_CREATED_ON,
    a.OBJ_CREATED_BY  AS OBJ_CREATED_BY,
    a.OBJ_MODIFIED_ON AS OBJ_MODIFIED_ON,
    a.OBJ_MODIFIED_BY AS OBJ_MODIFIED_BY,
    a.approver        AS Approver,
    (SELECT ms_concat(itm.select_items)
    FROM Ms_Grc_Exception_itm itm
    WHERE itm.object_id = a.object_id
    ) AS items,
    --            b.Select_Items Items,
    (SELECT (Ms_Grc_Utilities.Exception_Select_Items_Names(ms_concat(itm.select_items) || ',',
																						 exception_for))
     FROM Ms_Grc_Exception_itm itm
    WHERE itm.object_id = a.object_id
    ) AS items_disp,
     a.explanation Explanation,
    a.approver AS Exception_Approver,
    (SELECT MS_GRC_FN_CONCAT(orb.tgt_obj_id)
    FROM ms_grc_orb_relations_pub_v orb,
      si_org_entities_t base
    WHERE src_obj_type   = 'MS_GRC_EXCEPTION'
    AND orb.tgt_obj_type = 'MS_GRC_ORGANIZATION'
    AND orb.src_obj_id   = a.object_id
    AND orb.tgt_obj_id   = base.org_entity_id
    ) AS Applies_to_Organizations,
    (SELECT MS_GRC_FN_CONCAT(orb.tgt_obj_id)
    FROM ms_grc_orb_relations_pub_v orb,
      ms_grc_area_of_complia base
    WHERE src_obj_type   = 'MS_GRC_EXCEPTION'
    AND orb.tgt_obj_type = 'MS_GRC_AREA_OF_COMPLIANCE'
    AND orb.src_obj_id   = a.object_id
    AND orb.tgt_obj_id   = base.object_id
    ) AS RELATED_AREAS_OF_COMPLIANCE,
    (SELECT MS_GRC_FN_CONCAT(orb.tgt_obj_id)
    FROM ms_grc_orb_relations_pub_v orb,
      ms_grc_core_object base
    WHERE src_obj_type   = 'MS_GRC_EXCEPTION'
    AND orb.tgt_obj_type = 'MS_GRC_ASSET_CLASS'
    AND orb.src_obj_id   = a.object_id
    AND orb.tgt_obj_id   = base.object_id
    ) AS RELATED_ASSET_CLASSES,
    (SELECT MS_GRC_FN_CONCAT(orb.tgt_obj_id)
    FROM ms_grc_orb_relations_pub_v orb,
      ms_grc_core_object base
    WHERE src_obj_type   = 'MS_GRC_EXCEPTION'
    AND orb.tgt_obj_type = 'MS_GRC_ASSET'
    AND orb.src_obj_id   = a.object_id
    AND orb.tgt_obj_id   = base.object_id
    ) AS RELATED_ASSETS,
    (SELECT MS_GRC_FN_CONCAT(orb.tgt_obj_id)
    FROM ms_grc_orb_relations_pub_v orb,
      ms_grc_compliance_obje base
    WHERE src_obj_type   = 'MS_GRC_EXCEPTION'
    AND orb.tgt_obj_type = 'MS_GRC_CONTROL'
    AND orb.src_obj_id   = a.object_id
    AND orb.tgt_obj_id   = base.object_id
    ) AS RELATED_CONTROLS,
    (SELECT MS_GRC_FN_CONCAT(orb.tgt_obj_id)
    FROM ms_grc_orb_relations_pub_v orb,
      MS_GRC_MODEL_REFERENCE base
    WHERE src_obj_type   = 'MS_GRC_EXCEPTION'
    AND orb.tgt_obj_type = 'MS_GRC_MODEL_REFERENCE'
    AND orb.src_obj_id   = a.object_id
    AND orb.tgt_obj_id   = base.object_id
    ) AS RELATED_FRAMEWORK_REFERENCES,
    (SELECT MS_GRC_FN_CONCAT(orb.tgt_obj_id)
    FROM ms_grc_orb_relations_pub_v orb,
      ms_grc_core_object base
    WHERE src_obj_type   = 'MS_GRC_EXCEPTION'
    AND orb.tgt_obj_type = 'MS_GRC_PROCESS'
    AND orb.src_obj_id   = a.object_id
    AND orb.tgt_obj_id   = base.object_id
    ) AS RELATED_PROCESSES,
    (SELECT MS_GRC_FN_CONCAT(orb.tgt_obj_id)
    FROM ms_grc_orb_relations_pub_v orb,
      ms_grc_reference base
    WHERE src_obj_type   = 'MS_GRC_EXCEPTION'
    AND orb.tgt_obj_type = 'MS_GRC_REFERENCE'
    AND orb.src_obj_id   = a.object_id
    AND orb.tgt_obj_id   = base.object_id
    ) AS RELATED_REFERENCES,
    (SELECT MS_GRC_FN_CONCAT(orb.tgt_obj_id)
    FROM ms_grc_orb_relations_pub_v orb,
      ms_grc_regulatory_body base
    WHERE src_obj_type   = 'MS_GRC_EXCEPTION'
    AND orb.tgt_obj_type = 'MS_GRC_REGULATORY_BODY'
    AND orb.src_obj_id   = a.object_id
    AND orb.tgt_obj_id   = base.object_id
    ) AS RELATED_REG_BODIES,
    (SELECT MS_GRC_FN_CONCAT(orb.tgt_obj_id)
    FROM ms_grc_orb_relations_pub_v orb,
      ms_grc_requirement base
    WHERE src_obj_type   = 'MS_GRC_EXCEPTION'
    AND orb.tgt_obj_type = 'MS_GRC_REQUIREMENT'
    AND orb.src_obj_id   = a.object_id
    AND orb.tgt_obj_id   = base.object_id
    ) AS RELATED_REQUIREMENTS,
    (SELECT MS_GRC_FN_CONCAT(orb.tgt_obj_id)
    FROM ms_grc_orb_relations_pub_v orb,
      ms_grc_risk base
    WHERE src_obj_type   = 'MS_GRC_EXCEPTION'
    AND orb.tgt_obj_type = 'MS_GRC_RISK'
    AND orb.src_obj_id   = a.object_id
    AND orb.tgt_obj_id   = base.object_id
    ) AS RELATED_RISKS,
    (SELECT MS_GRC_FN_CONCAT(orb.tgt_obj_id)
    FROM ms_grc_orb_relations_pub_v orb,
      ms_grc_area_of_complia base
    WHERE src_obj_type   = 'MS_GRC_EXCEPTION'
    AND orb.tgt_obj_type = 'MS_GRC_STANDARD'
    AND orb.src_obj_id   = a.object_id
    AND orb.tgt_obj_id   = base.object_id
    ) AS RELATED_STANDARDS ,
    A.DD_OBJECT_TYPE,
    A.DD_PROCESS_INSTANCE_ID,
    A.DD_INSTANCE_ID,
    A.DD_EDIT_FLAG,
    a.dd_object_type AS Object_type,
    'Y'              AS OVER_RIDE_ACTION,
    'proc='
    || TO_CHAR(a.dd_process_instance_id)
    ||'|instid='
    ||TO_CHAR(a.dd_instance_id)
    ||'|edit_flag='
    ||NVL(a.dd_edit_flag,'Y')                                                                                                        AS EDIT_ACTION_PARAMETERS,
    ms_grc_utilities.get_edit_url(a.dd_object_type, a.dd_process_instance_id, a.dd_instance_id, NVL(a.dd_edit_flag,'Y'),a.object_id) AS EDIT,
    ms_Grc_utilities.INCLUDE_eXPIRED(a.VALID_FROM,a.VALID_UNTIL)                                                                     AS EXPIRED_FLAG
  FROM MS_GRC_EXCEPTION a
