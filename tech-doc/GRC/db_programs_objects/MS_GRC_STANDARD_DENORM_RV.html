  CREATE OR REPLACE FORCE VIEW "MS_GRC_STANDARD_DENORM_RV" ("OBJECT_ID", "OBJECT_NAME", "LOCALE_ID", "LOCALE", "DD_PROCESS_INSTANCE_ID", "DD_INSTANCE_ID", "OVER_RIDE_ACTION", "DD_OBJECT_TYPE", "OBJECT_NAME_URL", "EDIT_ACTION_PARAMETERS", "OBJECT_STATUS", "OBJECT_LEVEL", "STD_TYPE", "DESCRIPTION", "PARENT", "VALID_FROM", "VALID_UNTIL", "RESTRICT_ACCESS_TO", "OWNER_IDS", "OWNER_NAME", "OWNER_ORG_IDS", "OWNER_ORG_NAMES", "LEVEL_1_APPROVER", "LEVEL_2_APPROVER", "ATTACHMENTS", "UPDATED_BY", "UPDATED_ON", "CREATED_BY", "CREATED_ON", "REL_AOC_OBJ_NAME_AGG", "REL_ASSET_CLASS_OBJ_NAME_AGG", "REL_ASSET_OBJ_NAME_AGG", "REL_CTRL_OBJ_NAME_AGG", "REL_EVIDENCE_OBJ_NAME_AGG", "REL_EXCEPTION_OBJ_NAME_AGG", "REL_FIN_ACT_OBJ_NAME_AGG", "REL_FUNCTION_OBJ_NAME_AGG", "REL_KPIKRI_DEF_OBJ_NAME_AGG", "REL_MOD_REF_OBJ_NAME_AGG", "REL_OBJECTIVE_OBJ_NAME_AGG", "REL_ORGANIZATION_NAMES", "REL_ORG_PROFILE_OBJ_NAME_AGG", "REL_PROCESS_OBJ_NAME_AGG", "REL_PRODUCT_NAME_AGG", "REL_QUE_PROC_OBJ_NAME_AGG", "REL_REFRENCES_OBJ_NAME_AGG", "REL_REG_BODY_OBJ_NAME_AGG", "REL_REQUIREMENT_OBJ_NAME_AGG", "REL_RISK_OBJ_NAME_AGG", "REL_STANDARD_OBJ_NAME_AGG", "OBJ_PARENT_ID", "LEVEL_1_APPR", "LEVEL_2_APPR", "UPPER_OBJECT_NAME") AS 
  SELECT stdSumm.object_id,
	   stdSumm.object_name,
	   stdMlsSumm.locale_id,
	   stdMlsSumm.LOCALE,
     stdSumm.DD_PROCESS_INSTANCE_ID,
      stdSumm.DD_INSTANCE_ID,
      stdSumm.OVER_RIDE_ACTION,
      Stdsumm.Dd_Object_Type,
	   Stdsumm.Object_Name_Url As Object_Name_Url,
	   Stdsumm.Edit_Action_Parameters as EDIT_ACTION_PARAMETERS,
	   stdMlsSumm.OBJECT_STATUS,
	   stdMlsSumm.OBJECT_LEVEL,
	   stdMlsSumm.STD_TYPE,
	   stdSumm.DESCRIPTION,
	   stdSumm.PARENT,
	   stdSumm.VALID_FROM,
	   stdSumm.VALID_UNTIL,
	   stdSumm.RESTRICT_ACCESS_TO,
	   OWNER_IDS,
	   OWNER_NAME,
	   OWNER_ORG_IDS,
	   OWNER_ORG_NAMES,
	   (SELECT usr.first_name
			   || ' '
			   || usr.last_name
		  FROM si_users_t usr
		 WHERE stdSumm.LEVEL_1_APPROVER = usr.user_name
	   ) AS LEVEL_1_APPROVER,
	   (SELECT usr.first_name
			   || ' '
			   || usr.last_name
		  FROM si_users_t usr
		 WHERE stdSumm.LEVEL_2_APPROVER = usr.user_name
	   ) AS LEVEL_2_APPROVER,
	   Attachments,
	   (SELECT usr.first_name
			   || ' '
			   || usr.last_name
		  FROM si_users_t usr
		 WHERE stdSumm.Updated_By = usr.user_name
	   ) AS Updated_By,
	   UPDATED_ON,
	   (SELECT usr.first_name
			   || ' '
			   || usr.last_name
		  FROM si_users_t usr
		 WHERE stdSumm.CREATED_BY = usr.user_name
	   ) AS CREATED_BY,
	   CREATED_ON,
	   REL_AOC_OBJ_NAME_AGG,
	   REL_ASSET_CLASS_OBJ_NAME_AGG,
	   REL_ASSET_OBJ_NAME_AGG,
	   REL_CTRL_OBJ_NAME_AGG,
	   REL_EVIDENCE_OBJ_NAME_AGG,
	   REL_EXCEPTION_OBJ_NAME_AGG,
	   REL_FIN_ACT_OBJ_NAME_AGG,
	   REL_FUNCTION_OBJ_NAME_AGG,
	   REL_KPIKRI_DEF_OBJ_NAME_AGG,
	   REL_MOD_REF_OBJ_NAME_AGG,
	   REL_OBJECTIVE_OBJ_NAME_AGG,
	   REL_ORGANIZATION_NAMES,
	   REL_ORG_PROFILE_OBJ_NAME_AGG,
	   REL_PROCESS_OBJ_NAME_AGG,
	   REL_PRODUCT_NAME_AGG,
	   REL_QUE_PROC_OBJ_NAME_AGG,
	   REL_REFRENCES_OBJ_NAME_AGG,
	   REL_REG_BODY_OBJ_NAME_AGG,
	   REL_REQUIREMENT_OBJ_NAME_AGG,
	   REL_RISK_OBJ_NAME_AGG,
	   REL_STANDARD_OBJ_NAME_AGG,
     STDSUMM.OBJ_PARENT_ID,
     stdSumm.LEVEL_1_APPROVER as LEVEL_1_APPR,
     stdSumm.LEVEL_2_APPROVER as LEVEL_2_APPR,
     stdSumm.upper_object_name
  FROM MS_GRC_STANDARD_SUMMARY stdSumm,
	   MS_GRC_STANDARD_MLS_SUMMARY stdMlsSumm
 Where Stdsumm.Obj_Std_Summ_Wid = Stdmlssumm.Obj_Std_Summ_Wid
