MS_GRC_QUES_ENTP_RV 

  CREATE OR REPLACE FORCE VIEW "MS_GRC_QUES_ENTP_RV" ("OBJECT_ID", "OBJECT_NAME", "OBJECT_TYPE", "LOCALE_ID", "LOCALE", "DD_PROCESS_INSTANCE_ID", "DD_INSTANCE_ID", "OVER_RIDE_ACTION", "DD_OBJECT_TYPE", "OBJECT_NAME_URL", "EDIT_ACTION_PARAMETERS", "OBJECT_STATUS", "QUESTION", "OBJECTIVE", "VALID_FROM", "VALID_UNTIL", "RESTRICT_ACCESS_TO", "OWNER_IDS", "OWNER_NAME", "OWNER_ORG_IDS", "OWNER_ORG_NAMES", "LEVEL_1_APPROVER", "LEVEL_2_APPROVER", "ATTACHMENTS", "UPDATED_BY", "UPDATED_ON", "CREATED_BY", "CREATED_ON", "IS_ORPHAN", "REL_DATA_LINK", "RESPONSE_TYPE", "QUESTION_NATURE", "QUESTION_TYPE", "QUESTION_CAT", "UPPER_OBJECT_NAME") AS 
  SELECT QueSumm.object_id,
    QueSumm.object_name,
	'MS_GRC_QUESTION_PROCEDURE' as OBJECT_TYPE , 
    QueMlsSumm.locale_id,
    QueMlsSumm.LOCALE,
    QueSumm.DD_PROCESS_INSTANCE_ID,
    QueSumm.DD_INSTANCE_ID,
    QueSumm.OVER_RIDE_ACTION,
    Quesumm.Dd_Object_Type,
    Quesumm.Object_Name_Url        AS Object_Name_Url,
    Quesumm.Edit_Action_Parameters AS EDIT_ACTION_PARAMETERS,
    QueMlsSumm.OBJECT_STATUS,
    QueSumm.QUESTION,
    QueSumm.OBJECTIVE,
    QueSumm.VALID_FROM,
    QueSumm.VALID_UNTIL,
    QueSumm.RESTRICT_ACCESS_TO,
    OWNER_IDS,
    OWNER_NAME,
    OWNER_ORG_IDS,
    OWNER_ORG_NAMES,
    (SELECT usr.first_name
      || ' '
      || usr.last_name
    FROM si_users_t usr
    WHERE QueSumm.LEVEL_1_APPROVER = usr.user_name
    ) AS LEVEL_1_APPROVER,
    (SELECT usr.first_name
      || ' '
      || usr.last_name
    FROM si_users_t usr
    WHERE QueSumm.LEVEL_2_APPROVER = usr.user_name
    ) AS LEVEL_2_APPROVER,
    Attachments,
    (SELECT usr.first_name
      || ' '
      || usr.last_name
    FROM si_users_t usr
    WHERE QueSumm.Updated_By = usr.user_name
    ) AS Updated_By,
    UPDATED_ON,
    (SELECT usr.first_name
      || ' '
      || usr.last_name
    FROM si_users_t usr
    WHERE QueSumm.CREATED_BY = usr.user_name
    ) AS CREATED_BY,
    CREATED_ON,
    IS_ORPHAN,
    Lovvalue.displayed_value 			  AS REL_DATA_LINK,
    QueMlsSumm.RESPONSE_TYPE,
    QueMlsSumm.QUESTION_NATURE,
    QueMlsSumm.QUESTION_TYPE,
    QueMlsSumm.QUESTION_CAT,
    QueSumm.UPPER_OBJECT_NAME
  FROM MS_GRC_QUE_PROC_SUMMARY QueSumm,
    MS_GRC_QUE_PROC_MLS_SUMMARY QueMlsSumm,
	Ms_Qs_List_Of_Values Lovvalue,
    Ms_Qs_Lov_Names Lovname
  WHERE Quesumm.Obj_Que_Proc_Summ_Wid = Quemlssumm.Obj_Que_Proc_Summ_Wid
  AND Lovvalue.Org_Entity_Id           = 1
AND Lovname.Lov_Name                 = 'MS_GRC_LABLES'
and Lovvalue.stored_value = 'CLICK_HERE'
AND Lovvalue.Lov_Id                  = Lovname.Lov_Id
AND Lovvalue.Locale_id = QueMlsSumm.locale_id
