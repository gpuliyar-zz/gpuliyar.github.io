CREATE OR REPLACE TYPE BODY "MS_GRC_CONCAT_TY" IS
  STATIC FUNCTION ODCIAggregateInitialize(sctx  IN OUT  MS_GRC_CONCAT_TY)
    RETURN NUMBER IS
  BEGIN
    sctx := MS_GRC_CONCAT_TY(EMPTY_CLOB());
    RETURN ODCIConst.Success;
  END;

  MEMBER FUNCTION ODCIAggregateIterate(self   IN OUT  MS_GRC_CONCAT_TY,
                                       value  IN      VARCHAR2 )
    RETURN NUMBER IS
    myChar  VARCHAR2(4000);
  BEGIN
    IF SELF.g_string IS NULL AND value IS NOT NULL
    THEN
      SELF.g_string := value;
    ELSIF SELF.g_string IS NOT NULL AND value IS NOT NULL
    THEN
      SELF.g_string := self.g_string || ',' || value;
    END IF;
    
    RETURN ODCIConst.Success;
  END;

  MEMBER FUNCTION ODCIAggregateTerminate(self         IN   MS_GRC_CONCAT_TY,
                                         returnValue  OUT  CLOB,
                                         flags        IN   NUMBER)
    RETURN NUMBER IS
  BEGIN
    IF substr(SELF.g_string,1,1) = ','
    THEN
      returnValue := SUBSTR(SELF.g_string, 2, dbms_lob.getlength(SELF.g_string));
    ELSE
      returnValue := SELF.g_string;
    END IF;
    
    RETURN ODCIConst.Success;
  END;

  MEMBER FUNCTION ODCIAggregateMerge(self  IN OUT  MS_GRC_CONCAT_TY,
                                     ctx2  IN      MS_GRC_CONCAT_TY)
    RETURN NUMBER IS
  BEGIN
    IF SELF.g_string IS NULL AND ctx2.g_string IS NOT NULL
    THEN
      SELF.g_string := ctx2.g_string;
    ELSIF SELF.g_string IS NOT NULL AND ctx2.g_string IS NOT NULL
    THEN
      SELF.g_string := self.g_string || ',' || ctx2.g_string;
    END IF;
    RETURN ODCIConst.Success;
  END;
END;
