MS_GRC_RISK_RV 

  CREATE OR REPLACE FORCE VIEW "MS_GRC_RISK_RV" ("OBJECT_NAME", "OBJECT_ID", "OWNERS", "VALID_UNTIL", "VALID_FROM", "DESCRIPTION", "DD_OBJECT_TYPE", "RISK_CATEGORIES", "TYPE_RISK", "DD_PROCESS_INSTANCE_ID", "DD_INSTANCE_ID", "DD_EDIT_FLAG", "RESTRICT_ACCESS_TO", "OVER_RIDE_ACTION", "OBJECT_TYPE", "EDIT_ACTION_PARAMETERS", "OBJECT_LEVEL", "DISPLAY_NAME", "OBJECT_NAME_URL", "RELATED_FRAMEWORK_REFERENCES", "RELATED_CORE_OBJECTS", "RELATED_RISKS", "RELATED_PROCESS", "RELATED_ASSETS", "RELATED_ASSET_CLASSES", "RELATED_CONTROL", "RELATED_REQUIREMENT", "RELATED_REGULATORY_BODY", "RELATED_FUNCTION", "RELATED_REFERENCE", "APPLIES_TO_ORGANIZATIONS", "RELATED_AREAS_OF_COMPLIANCE", "RELATED_FINANCIAL_ACCOUNTS", "CORE_OBJECT_TYPE", "RELATED_OBJECTIVES", "OWNER_ORG_IDS", "OBJ_STATUS", "PRI_OBJECT_PARENT", "OBJECT_PARENT", "EXPIRED_FLAG", "KEY_RISK") AS 
  SELECT a.object_name,
    a.object_id,
    (SELECT ms_concat(usr.first_name
      || ' '
      || usr.last_name) AS owners
    FROM ms_grc_risk_own d,
      si_users_t usr ,
      si_ent e
    WHERE usr.user_name   = d.owners
    AND usr.enterprise_id = e.ENTERPRISE_ID
    AND a.OBJECT_ID       = D.OBJECT_ID
    ) AS OWNERS,
    valid_until,
    valid_from,
    description,
    dd_object_type,
    /* (  SELECT ms_concat(displayed_value)
    FROM z lovvalue,ms_grc_risk_cat c
    WHERE a.object_id = c.object_id
    and  lov_name   = 'MS GRC Risk Category'
    AND stored_value = category_risk
    ) AS*/
    (
    SELECT ms_concat(CATEGORY_RISK)
    FROM ms_grc_risk_cat c
    WHERE a.object_id = c.object_id
    ) AS RISK_CATEGORIES,
    /* (  SELECT displayed_value
    FROM z lovvalue
    WHERE lov_name   = 'MS GRC Risk Classification Type'
    AND stored_value = type_risk
    ) AS*/
    TYPE_RISK,
    dd_process_instance_id,
    dd_instance_id,
    NVL (dd_edit_flag, 'Y') AS dd_edit_flag,
    restrict_access_to,
    'Y'            AS OVER_RIDE_ACTION,
    DD_OBJECT_TYPE AS OBJECT_TYPE,
    'proc='
    || TO_CHAR (dd_process_instance_id)
    || '|instid='
    || TO_CHAR (dd_instance_id)
    || '|edit_flag='
    || NVL (dd_edit_flag, 'Y') AS EDIT_ACTION_PARAMETERS,
    /* (  SELECT displayed_value
    FROM z lovvalue
    WHERE lov_name   = 'MS_GRC_LEVEL'
    AND stored_value = object_level
    ) AS*/
    object_level,
    SUBSTR (object_name, 1, 30) display_name,
    ms_grc_utilities.get_edit_url (dd_object_type, dd_process_instance_id, dd_instance_id, NVL (dd_edit_flag, 'Y'), a.object_name ) AS object_name_url,
    (SELECT ms_concat(orb.tgt_obj_id)
    FROM ms_grc_orb_relations_pub_v orb,
      ms_grc_model_reference base
    WHERE src_obj_type   = 'MS_GRC_RISK'
    AND orb.tgt_obj_type = 'MS_GRC_MODEL_REFERENCE'
    AND orb.src_obj_id   = a.object_id
    AND orb.tgt_obj_id   = base.object_id
    ) AS Related_framework_references,
    (SELECT ms_concat(orb.tgt_obj_id)
    FROM ms_grc_orb_relations_pub_v orb,
      ms_grc_core_object base
    WHERE src_obj_type    = 'MS_GRC_RISK'
    AND orb.tgt_obj_type IN('MS_GRC_ASSET','MS_GRC_ASSET_CLASS','MS_GRC_PROCESS')
    AND orb.src_obj_id    = a.object_id
    AND orb.tgt_obj_id    = base.object_id
    ) AS Related_core_objects,
    (SELECT ms_concat(orb.tgt_obj_id)
    FROM ms_grc_orb_relations_pub_v orb,
      ms_grc_risk base
    WHERE src_obj_type   = 'MS_GRC_RISK'
    AND orb.tgt_obj_type = 'MS_GRC_RISK'
    AND orb.src_obj_id   = a.object_id
    AND orb.tgt_obj_id   = base.object_id
    ) AS Related_risks,
    (SELECT ms_concat(orb.tgt_obj_id)
    FROM ms_grc_orb_relations_pub_v orb,
      ms_grc_core_object base
    WHERE src_obj_type   = 'MS_GRC_RISK'
    AND orb.tgt_obj_type = 'MS_GRC_PROCESS'
    AND orb.src_obj_id   = a.object_id
    AND orb.tgt_obj_id   = base.object_id
    ) AS related_process,
    (SELECT ms_concat(orb.tgt_obj_id)
    FROM ms_grc_orb_relations_pub_v orb,
      ms_grc_core_object base
    WHERE src_obj_type   = 'MS_GRC_RISK'
    AND orb.tgt_obj_type = 'MS_GRC_ASSET'
    AND orb.src_obj_id   = a.object_id
    AND orb.tgt_obj_id   = base.object_id
    ) AS related_assets,
    (SELECT ms_concat(orb.tgt_obj_id)
    FROM ms_grc_orb_relations_pub_v orb,
      ms_grc_core_object base
    WHERE src_obj_type   = 'MS_GRC_RISK'
    AND orb.tgt_obj_type = 'MS_GRC_ASSET_CLASS'
    AND orb.src_obj_id   = a.object_id
    AND orb.tgt_obj_id   = base.object_id
    ) AS related_asset_classes,
    (SELECT ms_concat(orb.tgt_obj_id)
    FROM ms_grc_orb_relations_pub_v orb,
      ms_grc_compliance_obje base
    WHERE src_obj_type   = 'MS_GRC_RISK'
    AND orb.tgt_obj_type = 'MS_GRC_CONTROL'
    AND orb.src_obj_id   = a.object_id
    AND orb.tgt_obj_id   = base.object_id
    ) AS related_control,
    (SELECT ms_concat(orb.tgt_obj_id)
    FROM ms_grc_orb_relations_pub_v orb,
      MS_GRC_REQUIREMENT base
    WHERE src_obj_type   = 'MS_GRC_RISK'
    AND orb.tgt_obj_type = 'MS_GRC_REQUIREMENT'
    AND orb.src_obj_id   = a.object_id
    AND orb.tgt_obj_id   = base.object_id
    ) AS related_requirement,
    (SELECT ms_concat(orb.tgt_obj_id)
    FROM ms_grc_orb_relations_pub_v orb,
      MS_GRC_REGULATORY_BODY base
    WHERE src_obj_type   = 'MS_GRC_RISK'
    AND orb.tgt_obj_type = 'MS_GRC_REGULATORY_BODY'
    AND orb.src_obj_id   = a.object_id
    AND orb.tgt_obj_id   = base.object_id
    ) AS related_regulatory_body,
    (SELECT ms_concat(orb.tgt_obj_id)
    FROM ms_grc_orb_relations_pub_v orb,
      ms_grc_function base
    WHERE src_obj_type   = 'MS_GRC_RISK'
    AND orb.tgt_obj_type = 'MS_GRC_FUNCTION'
    AND orb.src_obj_id   = a.object_id
    AND orb.tgt_obj_id   = base.object_id
    ) AS related_function,
    (SELECT ms_concat(orb.tgt_obj_id)
    FROM ms_grc_orb_relations_pub_v orb,
      MS_GRC_REFERENCE base
    WHERE src_obj_type   = 'MS_GRC_RISK'
    AND orb.tgt_obj_type = 'MS_GRC_REFERENCE'
    AND orb.src_obj_id   = a.object_id
    AND orb.tgt_obj_id   = base.object_id
    ) AS related_reference,
    (SELECT ms_concat(base.org_entity_ID)
    FROM ms_grc_orb_relations_pub_v orb,
      si_org_entities_t base
    WHERE src_obj_type   = 'MS_GRC_RISK'
    AND orb.tgt_obj_type = 'MS_GRC_ORGANIZATION'
    AND orb.src_obj_id   = a.object_id
    AND orb.tgt_obj_id   = base.org_entity_id
    ) AS applies_to_organizations,
    (SELECT ms_concat(orb.tgt_obj_id)
    FROM ms_grc_orb_relations_pub_v orb,
      ms_grc_area_of_complia base
    WHERE src_obj_type   = 'MS_GRC_RISK'
    AND orb.tgt_obj_type = 'MS_GRC_AREA_OF_COMPLIANCE'
    AND orb.src_obj_id   = a.object_id
    AND orb.tgt_obj_id   = base.object_id
    ) AS related_areas_of_compliance,
    (SELECT ms_concat(orb.tgt_obj_id)
    FROM ms_grc_orb_relations_pub_v orb,
      ms_grc_financial_accou base
    WHERE src_obj_type   = 'MS_GRC_RISK'
    AND orb.tgt_obj_type = 'MS_GRC_FINANCIAL_ACCOUNTS'
    AND orb.src_obj_id   = a.object_id
    AND orb.tgt_obj_id   = base.object_id
    ) AS RELATED_FINANCIAL_ACCOUNTS,
    (SELECT ms_grc_utilities.Get_Relationships (object_id, dd_object_type, 'MS_GRC_PROCESS','OBJECT_TYPE')
    FROM DUAL
    ) AS core_object_type,
    (SELECT ms_concat(orb.tgt_obj_id)
    FROM ms_grc_orb_relations_pub_v orb,
      MS_GRC_OBJECTIVES base
    WHERE src_obj_type   = 'MS_GRC_RISK'
    AND orb.tgt_obj_type = 'MS_GRC_OBJECTIVES'
    AND orb.src_obj_id   = a.object_id
    AND orb.tgt_obj_id   = base.object_id
    ) AS related_objectives,
    (SELECT ms_concat (owner_organizations)
    FROM ms_grc_risk_ogn e
    WHERE e.object_id = a.object_id
    )            AS owner_org_ids,
    a.OBJ_STATUS AS OBJ_STATUS,
    (SELECT ms_concat(object_name)
    FROM MS_GRC_RISK
    WHERE object_id IN
      (SELECT b.pri_objecT_parent FROM MS_GRC_RISK_PRI B WHERE a.OBJECT_ID=B.OBJECT_ID
      )
    )                                                        AS PRI_OBJECT_PARENT,
    a.object_parent as object_parent,
    MS_GRC_UTILITIES.INCLUDE_EXPIRED(VALID_FROM,VALID_UNTIL) AS EXPIRED_FLAG,
    KEY_RISK
  FROM MS_GRC_RISK a
