MS_GRC_FRAMEWORKREF_EDS_V 

  CREATE OR REPLACE FORCE VIEW "MS_GRC_FRAMEWORKREF_EDS_V" ("ID", "NAME", "TYPE", "SOURCE", "STATUS", "DESCRIPTION", "OWNER_ORG_IDS", "OWNER_ORGS", "OWNER_IDS", "OWNER_NAME", "L1_APPROVER", "L2_APPROVER", "RESTRICT_ACCESS_TO", "VALID_FROM", "VALID_UNTIL", "CREATED_BY", "CREATED_ON", "MODIFIED_BY", "MODIFIED_ON", "ATTACHMENTS") AS 
  SELECT /*@public*/
A.OBJECT_ID AS ID,
A.OBJECT_NAME AS NAME,
(SELECT Z.DISPLAYED_VALUE FROM MS_QS_LOV_ORG_VALUES_V Z WHERE Z.LOV_NAME = 'MS GRC Model Reference Type' AND Z.STORED_VALUE=A.TYPE_SV
and z.locale_id=ms_grc_common_utils.fn_get_default_locale_id) AS TYPE,
(SELECT Z.DISPLAYED_VALUE FROM MS_QS_LOV_ORG_VALUES_V Z WHERE Z.LOV_NAME = 'MS GRC Source' AND Z.STORED_VALUE=A.SOURCE_SV
and z.locale_id=ms_grc_common_utils.fn_get_default_locale_id) AS SOURCE,
(SELECT Z.STATUS_NAME FROM MS_GRC_STATUS Z WHERE Z.STATUS_CODE=A.OBJ_STATUS_SV
and z.locale_id=ms_grc_common_utils.fn_get_default_locale_id) AS STATUS,
REGEXP_REPLACE(A.DESCRIPTION,'<[^>]+>', NULL) AS DESCRIPTION,
A.OWNER_ORG_IDS AS OWNER_ORG_IDS,
A.OWNER_ORG_NAMES AS OWNER_ORGS,
A.OWNER_IDS AS OWNER_IDS,
A.OWNER_NAME AS OWNER_NAME,
A.LEVEL_1_APPROVER AS l1_approver,
A.LEVEL_2_APPROVER AS l2_approver,
(SELECT Z.DISPLAYED_VALUE FROM MS_QS_LOV_ORG_VALUES_V Z WHERE Z.LOV_NAME = 'MS GRC Restrict Access To' AND Z.STORED_VALUE=A.RESTRICT_ACCESS_TO
and z.locale_id=ms_grc_common_utils.fn_get_default_locale_id) AS Restrict_Access_To,
A.VALID_FROM AS VALID_FROM,
A.VALID_UNTIL AS VALID_UNTIL,
A.CREATED_BY AS CREATED_BY,
A.CREATED_ON AS created_on,
A.UPDATED_BY AS modified_by,
A.UPDATED_ON AS modified_on,
A.ATTACHMENTS AS ATTACHMENTS
FROM MS_GRC_MOD_REF_SUMMARY A
