MS_GRC_ASSET_CLASS_PUB_V 

  CREATE OR REPLACE FORCE VIEW "MS_GRC_ASSET_CLASS_PUB_V" ("OBJECT_ID", "OBJECT_NAME", "OBJECT_NAME_URL", "DESCRIPTION", "DESCRIPTION_HTM", "OWNERS", "OWNER_ORGANIZATIONS", "RESTRICT_ACCESS_TO", "OBJ_STATUS", "VALID_FROM", "VALID_UNTIL", "RTF_ATTACH", "DD_OBJECT_TYPE", "DD_PROCESS_INSTANCE_ID", "DD_INSTANCE_ID", "OBJ_PART", "OBJ_VENDOR", "OBJ_PRODUCT", "OBJ_VERSION", "OBJ_UPDATE", "OBJ_EDITION", "OBJ_LANGUAGE", "INITIATED_BY", "CREATED_ON", "MODIFIED_ON") AS 
  SELECT /*@public*/ a.OBJECT_ID,
    A.OBJECT_NAME,
    ms_grc_utilities.get_edit_url (A.dd_object_type,
                                               A.dd_process_instance_id,
                                               A.dd_instance_id,
                                               NVL(C.DD_EDIT_FLAG,'Y'),
                                               A.OBJECT_NAME)            AS OBJECT_NAME_URL,
    regexp_replace(a.DESCRIPTION, '<[^>]+>', NULL) AS DESCRIPTION,
    a.DESCRIPTION                                  AS DESCRIPTION_HTM,
    (SELECT MS_CONCAT(OWNERS)
    FROM ms_Grc_core_object_OWN B
    WHERE B.OBJECT_ID = A.OBJECT_ID
    ) AS OWNERS,
    (SELECT MS_CONCAT(OWNER_ORGANIZATIONS)
    FROM ms_Grc_core_object_Ogn B
    WHERE B.OBJECT_ID = A.OBJECT_ID
    ) AS OWNER_ORGANIZATIONS,
    a.RESTRICT_ACCESS_TO,
    a.obj_STATUS,
    a.VALID_FROM,
    a.VALID_UNTIL,
    a.RTF_aTTACH,
    a.DD_OBJECT_TYPE,
    a.DD_PROCESS_INSTANCE_ID,
    a.DD_INSTANCE_ID,
    c.obj_part,
    c.obj_Vendor,
    c.obj_product,
    c.obj_version,
    c.obj_update,
    C.OBJ_EDITION,
    C.OBJ_LANGUAGE,
    NVL(A.OBJ_MODIFIED_BY,A.OBJ_CREATED_BY) AS INITIATED_BY,
    C.DD_CREATED_ON AS CREATED_ON,
    C.DD_MODIFIED_ON AS MODIFIED_ON
  FROM ms_Grc_asset_class c ,
    ms_grc_core_object a
  WHERE c.object_id    = a.object_id
  AND A.DD_OBJECT_TYPE = 'MS_GRC_ASSET_CLASS'
