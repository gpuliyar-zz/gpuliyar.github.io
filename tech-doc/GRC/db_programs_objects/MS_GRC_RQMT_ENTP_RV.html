  CREATE OR REPLACE FORCE VIEW "MS_GRC_RQMT_ENTP_RV" ("OBJECT_NAME", "OBJECT_ID", "OBJECT_TYPE", "LOCALE", "LOCALE_ID", "OWNER_IDS", "OWNER_NAME", "OWNER_ORG_IDS", "OWNER_ORG_NAMES", "KEY_REQUIREMENT", "COMPLIANCE_TYPE", "LEVEL_1_APPROVER", "LEVEL_2_APPROVER", "VALID_UNTIL", "VALID_FROM", "DESCRIPTION", "PENALTIES", "CATEGORY", "IMPACT", "SECTION_NO", "FREQUENCY", "DD_PROCESS_INSTANCE_ID", "DD_INSTANCE_ID", "DD_OBJECT_TYPE", "DD_EDIT_FLAG", "RESTRICT_ACCESS_TO", "OVER_RIDE_ACTION", "EDIT_ACTION_PARAMETERS", "OBJECT_LEVEL", "OBJECT_NAME_URL", "REL_DATA_LINK", "OBJECT_STATUS", "OBJECT_PARENT", "CREATED_ON", "CREATED_BY", "UPDATED_BY", "UPDATED_ON", "UPPER_OBJECT_NAME", "OBJ_PARENT_ID") AS 
  SELECT req_summ.object_name,
    req_summ.object_id,
    'MS_GRC_REQUIREMENT' as OBJECT_TYPE ,
    req_mls.LOCALE,
    req_mls.LOCALE_ID,
    req_summ.OWNER_IDS,
    req_summ.OWNER_NAME,
    req_summ.OWNER_ORG_IDS,
    req_summ.OWNER_ORG_NAMES,
    req_mls.KEY_REQUIREMENT,
    req_mls.COMPLIANCE_TYPE,
    (SELECT usr.first_name
      || ' '
      || usr.last_name
    FROM si_users_t usr
    WHERE req_summ.level_1_approver = usr.user_name
    ) AS level_1_approver,
    (SELECT usr.first_name
      || ' '
      || usr.last_name
    FROM si_users_t usr
    WHERE req_summ.level_2_approver = usr.user_name
    ) AS level_2_approver,
    req_summ.valid_until,
    req_summ.valid_from,
    req_summ.description,
    req_summ.PENALTIES,
    req_mls.CATEGORY,
    req_mls.IMPACT,
    req_summ.SECTION_NO,
    req_mls.FREQUENCY,
    req_summ.dd_process_instance_id,
    req_summ.dd_instance_id,
    req_summ.Dd_Object_Type,
    NVL (req_summ.dd_edit_flag, 'Y') AS dd_edit_flag,
    req_summ.restrict_access_to,
    'Y'                                   AS Over_Ride_Action,
    Req_Summ.Edit_Action_Parameters       AS EDIT_ACTION_PARAMETERS,
    Req_Mls.Object_Level                  AS Object_Level,
    Req_Summ.Object_Name_Url              AS object_name_url,
   Lovvalue.displayed_value 			  AS REL_DATA_LINK,
    req_mls.OBJECT_STATUS,
    req_summ.OBJECT_PARENT,
    req_summ.CREATED_ON,
    (SELECT usr.first_name
      || ' '
      || usr.last_name
    FROM si_users_t usr
    WHERE req_summ.CREATED_BY = usr.user_name
    ) AS CREATED_BY,
    (SELECT usr.first_name
      || ' '
      || usr.last_name
    FROM si_users_t usr
    WHERE req_summ.UPDATED_BY = usr.user_name
    ) AS UPDATED_BY,
    req_summ.UPDATED_ON,
    REQ_SUMM.UPPER_OBJECT_NAME
    ,REQ_SUMM.OBJ_PARENT_ID
  FROM MS_GRC_REQ_SUMMARY req_summ,
    MS_GRC_REQ_MLS_SUMMARY req_mls,
    Ms_Qs_List_Of_Values Lovvalue,
    Ms_Qs_Lov_Names Lovname
  WHERE Req_Mls.Object_Id = Req_Summ.Object_Id
  AND Lovvalue.Org_Entity_Id           = 1
AND Lovname.Lov_Name                 = 'MS_GRC_LABLES'
and Lovvalue.stored_value = 'CLICK_HERE'
AND Lovvalue.Lov_Id                  = Lovname.Lov_Id
AND Lovvalue.Locale_id = req_mls.locale_id
