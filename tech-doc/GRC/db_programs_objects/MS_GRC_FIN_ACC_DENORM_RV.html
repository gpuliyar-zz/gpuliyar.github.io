MS_GRC_FIN_ACC_DENORM_RV 

  CREATE OR REPLACE FORCE VIEW "MS_GRC_FIN_ACC_DENORM_RV" ("OBJECT_ID", "OBJECT_NAME", "LOCALE_ID", "LOCALE", "DD_PROCESS_INSTANCE_ID", "DD_INSTANCE_ID", "OVER_RIDE_ACTION", "DD_OBJECT_TYPE", "OBJECT_NAME_URL", "EDIT_ACTION_PARAMETERS", "OBJECT_STATUS", "DESCRIPTION", "VALID_FROM", "VALID_UNTIL", "RESTRICT_ACCESS_TO", "OWNER_IDS", "OWNER_NAME", "OWNER_ORG_IDS", "OWNER_ORG_NAMES", "LEVEL_1_APPROVER", "LEVEL_2_APPROVER", "ATTACHMENTS", "UPDATED_BY", "UPDATED_ON", "CREATED_BY", "CREATED_ON", "IS_ORPHAN", "REL_AOC_OBJ_NAME_AGG", "REL_ASSET_CLASS_OBJ_NAME_AGG", "REL_ASSET_OBJ_NAME_AGG", "REL_CTRL_OBJ_NAME_AGG", "REL_EVIDENCE_OBJ_NAME_AGG", "REL_EXCEPTION_OBJ_NAME_AGG", "REL_FIN_ACT_OBJ_NAME_AGG", "REL_FUNCTION_OBJ_NAME_AGG", "REL_KPIKRI_DEF_OBJ_NAME_AGG", "REL_MOD_REF_OBJ_NAME_AGG", "REL_OBJECTIVE_OBJ_NAME_AGG", "REL_ORGANIZATION_NAMES", "REL_ORG_PROFILE_OBJ_NAME_AGG", "REL_PROCESS_OBJ_NAME_AGG", "REL_PRODUCT_NAME_AGG", "REL_QUE_PROC_OBJ_NAME_AGG", "REL_REFRENCES_OBJ_NAME_AGG", "REL_REG_BODY_OBJ_NAME_AGG", "REL_REQUIREMENT_OBJ_NAME_AGG", "REL_RISK_OBJ_NAME_AGG", "REL_STANDARD_OBJ_NAME_AGG", "FIN_CATEGORY", "FIN_TYPE", "ASSERTIONS", "LEVEL_1_APPR", "LEVEL_2_APPR") AS 
  SELECT finSumm.object_id,
    finSumm.object_name,
    finMlsSumm.locale_id,
    finMlsSumm.LOCALE,
    finSumm.DD_PROCESS_INSTANCE_ID,
    finSumm.DD_INSTANCE_ID,
    finSumm.OVER_RIDE_ACTION,
    Finsumm.Dd_Object_Type,
    Finsumm.Object_Name_Url        AS object_name_url,
    finSumm.Edit_Action_Parameters AS EDIT_ACTION_PARAMETERS,
    finMlsSumm.OBJECT_STATUS,
    finSumm.DESCRIPTION,
    finSumm.VALID_FROM,
    finSumm.VALID_UNTIL,
    finSumm.RESTRICT_ACCESS_TO,
    OWNER_IDS,
    OWNER_NAME,
    OWNER_ORG_IDS,
    OWNER_ORG_NAMES,
    (SELECT usr.first_name
      || ' '
      || usr.last_name
    FROM si_users_t usr
    WHERE finSumm.LEVEL_1_APPROVER = usr.user_name
    ) AS LEVEL_1_APPROVER,
    (SELECT usr.first_name
      || ' '
      || usr.last_name
    FROM si_users_t usr
    WHERE finSumm.LEVEL_2_APPROVER = usr.user_name
    ) AS LEVEL_2_APPROVER,
    Attachments,
    (SELECT usr.first_name
      || ' '
      || usr.last_name
    FROM si_users_t usr
    WHERE finSumm.Updated_By = usr.user_name
    ) AS Updated_By,
    UPDATED_ON,
    (SELECT usr.first_name
      || ' '
      || usr.last_name
    FROM si_users_t usr
    WHERE finSumm.CREATED_BY = usr.user_name
    ) AS CREATED_BY,
    CREATED_ON,
    IS_ORPHAN,
    REL_AOC_OBJ_NAME_AGG,
    REL_ASSET_CLASS_OBJ_NAME_AGG,
    REL_ASSET_OBJ_NAME_AGG,
    REL_CTRL_OBJ_NAME_AGG,
    REL_EVIDENCE_OBJ_NAME_AGG,
    REL_EXCEPTION_OBJ_NAME_AGG,
    REL_FIN_ACT_OBJ_NAME_AGG,
    REL_FUNCTION_OBJ_NAME_AGG,
    REL_KPIKRI_DEF_OBJ_NAME_AGG,
    REL_MOD_REF_OBJ_NAME_AGG,
    REL_OBJECTIVE_OBJ_NAME_AGG,
    REL_ORGANIZATION_NAMES,
    REL_ORG_PROFILE_OBJ_NAME_AGG,
    REL_PROCESS_OBJ_NAME_AGG,
    REL_PRODUCT_NAME_AGG,
    REL_QUE_PROC_OBJ_NAME_AGG,
    REL_REFRENCES_OBJ_NAME_AGG,
    REL_REG_BODY_OBJ_NAME_AGG,
    REL_REQUIREMENT_OBJ_NAME_AGG,
    REL_RISK_OBJ_NAME_AGG,
    REL_STANDARD_OBJ_NAME_AGG,
    to_char(finMlsSumm.FIN_CATEGORY),
    to_char(finMlsSumm.FIN_TYPE),
    FINMLSSUMM.ASSERTIONS,
    LEVEL_1_APPROVER as LEVEL_1_APPR,
    LEVEL_1_APPROVER as LEVEL_2_APPR
  FROM MS_GRC_FIN_ACC_SUMMARY finSumm,
    MS_GRC_FIN_ACC_MLS_SUMMARY finMlsSumm
  WHERE FINSUMM.OBJ_FIN_SUMM_WID = FINMLSSUMM.OBJ_FIN_SUMM_WID
