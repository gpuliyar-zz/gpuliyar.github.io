MS_GRC_ORG_ORGANIZATION_V 

  CREATE OR REPLACE FORCE VIEW "MS_GRC_ORG_ORGANIZATION_V" ("ORG_ENTITY_ID", "PARENT_ORG_ENTITY_ID", "ORG_ENTITY_NAME", "PARENT_ORG_ENTITY_NAME", "ORG_DESCRIPTION", "ORG_LOCATION", "ORG_LEVEL", "ORG_HIERARCHY_NAME", "ORG_START_DATE", "ORG_END_DATE", "ORG_CREATED_BY", "ORG_CREATION_DATE", "ORG_LAST_UPDATE_DATE", "ORG_LAST_UPDATED_BY", "LOC_START_DATE", "LOC_END_DATE", "LOC_CREATED_BY", "LOC_CREATION_DATE", "LOC_LAST_UPDATE_DATE", "LOC_LAST_UPDATED_BY", "OHL_START_DATE", "OHL_END_DATE", "OHL_CREATED_BY", "OHL_CREATION_DATE", "OHL_LAST_UPDATE_DATE", "OHL_LAST_UPDATED_BY") AS 
  SELECT org.org_entity_id   AS org_entity_id
, org.parent_org_entity_id    AS parent_org_entity_id
, org.org_entity_name          AS org_entity_name
, porg.org_entity_name       AS parent_org_entity_name
, org.description                   AS org_description
, loc.location_name            AS org_location
, ohl.org_level                       AS org_level
, ohl.org_hierarchy_name          AS org_hierarchy_name
,org.start_date                   AS org_start_date
,org.end_date                    AS org_end_date
,org.created_by                  AS org_created_by
,org.creation_date              AS org_creation_date
,org.last_update_date          AS org_last_update_date
,org.last_updated_by          AS org_last_updated_by
,loc.start_date              AS loc_start_date
,loc.end_date               AS loc_end_date
,loc.created_by            AS loc_created_by
,loc.creation_date        AS loc_creation_date
,loc.last_update_date       AS loc_last_update_date
,loc.last_updated_by        AS loc_last_updated_by
,ohl.start_date              AS ohl_start_date
,ohl.end_date               AS ohl_end_date
,ohl.created_by            AS ohl_created_by
,ohl.creation_date        AS ohl_creation_date
,ohl.last_update_date       AS ohl_last_update_date
,ohl.last_updated_by        AS ohl_last_updated_by
FROM si_org_entities org
, si_org_locations loc
, si_org_hierarchy_levels ohl
, si_org_entities porg
WHERE org.location_id = loc.location_id (+)
AND SYSDATE BETWEEN org.start_date AND Nvl(org.end_date, SYSDATE)
AND org.org_hierarchy_id = ohl.org_hierarchy_id
AND org.parent_org_entity_id = porg.org_entity_id (+)
Order By org.org_entity_name
