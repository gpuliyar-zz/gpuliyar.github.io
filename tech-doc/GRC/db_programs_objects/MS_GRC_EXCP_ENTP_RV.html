MS_GRC_EXCP_ENTP_RV 

  CREATE OR REPLACE FORCE VIEW "MS_GRC_EXCP_ENTP_RV" ("OBJECT_ID", "LOCALE_ID", "LOCALE", "EXCEPTION_FOR", "TYPE_EXCEPTION", "VALID_FROM", "EXPIRES_ON", "STATUS", "OBJ_CREATED_ON", "OBJ_CREATED_BY", "OBJ_MODIFIED_ON", "OBJ_MODIFIED_BY", "APPROVER", "ITEMS", "ITEMS_DISP", "EXPLANATION", "EXCEPTION_APPROVER", "REL_DATA_LINK", "OBJECT_TYPE", "DD_PROCESS_INSTANCE_ID", "DD_INSTANCE_ID", "DD_EDIT_FLAG", "OVER_RIDE_ACTION", "EDIT_ACTION_PARAMETERS", "EDIT", "EXPIRED_FLAG") AS 
  SELECT a.object_id as object_id,
loc.Locale_id , 
loc.locale_string as Locale,
    (SELECT Displayed_Value
      FROM Ms_Qs_List_Of_Values Lovvalue,
        Ms_Qs_Lov_Names Lovname,
        Si_Ent Ent,
        MS_GRC_EXCEPTION expc
      WHERE Lovname.Lov_Name     = 'MS_GRC_Exception_For'
      AND Lovvalue.Lov_Id        = Lovname.Lov_Id
      AND Lovvalue.Org_Entity_Id = 1
      AND Lovvalue.Enterprise_Id = Ent.Enterprise_Id
      AND Lovname.Enterprise_Id  = Ent.Enterprise_Id
     AND Lovvalue.Locale_Id     = loc.LOCALE_ID
      AND expc.object_id        = a.object_id
      AND expc.Exception_For          = Lovvalue.Stored_Value) as exception_for,
    (SELECT Displayed_Value
      FROM Ms_Qs_List_Of_Values Lovvalue,
        Ms_Qs_Lov_Names Lovname,
        Si_Ent Ent,
        MS_GRC_EXCEPTION expc
      WHERE Lovname.Lov_Name     = 'MS_GRC_Type_Of_Exception'
      AND Lovvalue.Lov_Id        = Lovname.Lov_Id
      AND Lovvalue.Org_Entity_Id = 1
      AND Lovvalue.Enterprise_Id = Ent.Enterprise_Id
      AND Lovname.Enterprise_Id  = Ent.Enterprise_Id
      AND Lovvalue.Locale_Id     = loc.LOCALE_ID
      AND expc.object_id        = a.object_id
      AND expc.type_Exception          = Lovvalue.Stored_Value) as type_exception,
    a.valid_from valid_from,
    a.Valid_Until Expires_On,
    a.obj_status Status,
    a.OBJ_CREATED_ON  AS OBJ_CREATED_ON,
    a.OBJ_CREATED_BY  AS OBJ_CREATED_BY,
    a.OBJ_MODIFIED_ON AS OBJ_MODIFIED_ON,
    a.OBJ_MODIFIED_BY AS OBJ_MODIFIED_BY,
    a.approver        AS Approver,
    (SELECT ms_concat(itm.select_items)
    FROM Ms_Grc_Exception_itm itm
    WHERE itm.object_id = a.object_id
    ) AS items,

    (SELECT (Ms_Grc_Utilities.Exception_Select_Items_Names(ms_concat(itm.select_items) || ',',
																						 exception_for))
     FROM Ms_Grc_Exception_itm itm
    WHERE itm.object_id = a.object_id
    ) AS items_disp,
     a.explanation Explanation,
    a.approver AS Exception_Approver,
   Lovvalue.displayed_value as REL_DATA_LINK,
    A.DD_OBJECT_TYPE as object_type,
    A.DD_PROCESS_INSTANCE_ID ,
    A.DD_INSTANCE_ID ,
    A.DD_EDIT_FLAG,
    'Y'              AS OVER_RIDE_ACTION,
    'proc='
    || TO_CHAR(a.dd_process_instance_id)
    ||'|instid='
    ||TO_CHAR(a.dd_instance_id)
    ||'|edit_flag='
    ||NVL(a.dd_edit_flag,'Y')                                                                                                        AS EDIT_ACTION_PARAMETERS,
    ms_grc_utilities.get_edit_url(a.dd_object_type, a.dd_process_instance_id, a.dd_instance_id, NVL(a.dd_edit_flag,'Y'),a.object_id) AS EDIT,
    ms_Grc_utilities.INCLUDE_eXPIRED(a.VALID_FROM,a.VALID_UNTIL)                                                                     AS EXPIRED_FLAG
  FROM MS_GRC_EXCEPTION a,
  Ms_Qs_List_Of_Values Lovvalue,
  Ms_Qs_Lov_Names Lovname, 
  si_locales loc 
where Lovvalue.Org_Entity_Id   = 1
    AND Lovname.Lov_Name         = 'MS_GRC_LABLES'
    and Lovvalue.stored_value 	= 'CLICK_HERE'
    AND Lovvalue.Lov_Id          = Lovname.Lov_Id
     and Lovvalue.Locale_id  = loc.locale_id
     AND loc.enabled_flag = 'Y'
