  UPDATED_BY,

  VALID_UNTIL,
  QUESTION,
  OBJECTIVE,
  OWNER_ORG_NAMES,
  OWNER_NAME,
  LEVEL_1_APPROVER,
  LEVEL_2_APPROVER,
  REL_DATA_LINK as REL_ORGANIZATION_NAMES,
  REL_DATA_LINK as REL_AOC_OBJ_NAME_AGG,
  REL_DATA_LINK as REL_ASSET_OBJ_NAME_AGG,
  REL_DATA_LINK as REL_ASSET_CLASS_OBJ_NAME_AGG,
  REL_DATA_LINK as REL_CTRL_OBJ_NAME_AGG,
  REL_DATA_LINK as REL_PROCESS_OBJ_NAME_AGG,
  REL_DATA_LINK as REL_REFRENCES_OBJ_NAME_AGG,
  REL_DATA_LINK as REL_REQUIREMENT_OBJ_NAME_AGG,
  REL_DATA_LINK as REL_STANDARD_OBJ_NAME_AGG,
  (MS_GRC_UTILITIES.FN_GET_REPORT_JSON((SELECT METRIC_ID FROM SI_METRICS_T WHERE METRIC_NAME='MS_GRC_QUESTION_PROCEDURE'),rv.DD_PROCESS_INSTANCE_ID,(SELECT DD_INSTANCE_ID FROM MS_GRC_QUESTION_PROC WHERE OBJECT_ID=rv.OBJECT_ID),
'''flag=1'||CHR(38)||'emd=1'||CHR(38)||'x_edit_mode=Yes'||CHR (38)|| 'wrapper= No'|| CHR (38)|| 'mode=view''')) AS REPORT_JSON
FROM MS_GRC_QUES_ENTP_RV RV
WHERE LOCALE =
  (SELECT LOCALE FROM SI_USERS_T WHERE USER_NAME = :pv_object_status1
  )
AND NVL(is_orphan, 1)=NVL(:pv_orphan_object,NVL(is_orphan,1))
AND EXISTS
    (SELECT column_value
    FROM TABLE(CAST(ms_grc_rep_browser_fil_denorm.fn_QUES_PROC_report_filters(ms_grc_array_type(:pv_object_id,:pv_object_name,:pv_object_status,:pv_response_type,:pv_nature,:pv_question_type,:pv_category,:pv_created_last_no_days,to_char(:pv_object_id0,'YYYY-MM-DD'),TO_CHAR(:pv_object_id1,'YYYY-MM-DD'),:pv_object_id2,:pv_object_id3,TO_CHAR(:pv_object_id4,'YYYY-MM-DD'),TO_CHAR(:pv_object_id5,'YYYY-MM-DD'),TO_CHAR(:pv_object_id6,'YYYY-MM-DD'), TO_CHAR(:pv_object_id7,'YYYY-MM-DD'), :pv_object_id8,:pv_object_id9,:pv_object_name0,:pv_object_name1,:pv_object_name2,:pv_object_name3,:pv_object_name4,:pv_object_name5,NULL,:pv_object_name6,NULL,NULL,NULL,NULL,NULL,NULL,:pv_object_name7,NULL,NULL,:pv_object_name8,:pv_object_name9,NULL,:pv_object_status0,'MS_GRC_QUESTION_PROCEDURE',:pv_object_status1))as ms_varchar2_array))
  WHERE column_value = OBJECT_ID)order by UPPER_OBJECT_NAME
