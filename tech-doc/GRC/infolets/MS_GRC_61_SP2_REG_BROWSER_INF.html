SELECT
OBJECT_NAME as DISPLAY_NAME,
DD_OBJECT_TYPE as OBJECT_TYPE,
OVER_RIDE_ACTION,
EDIT_ACTION_PARAMETERS as VIEW_ACTION_PARAMETERS,
OBJECT_ID as OBJECT_ID
FROM ms_grc_reg_denorm_drv
WHERE NVL(is_orphan, 1)=NVL(:pv_orphan_object,NVL(is_orphan,1))
AND EXISTS
    (SELECT column_value
    FROM TABLE(cast(ms_grc_rep_browser_fil_denorm.fn_reg_report_filter(ms_grc_array_type(:pv_regulatory_id,:pv_regulatory_name,:pv_regulatory_status,:pv_created_last_no_days,to_char(:pv_created_from,'YYYY-MM-DD'),TO_CHAR(:pv_created_until,'YYYY-MM-DD'),:pv_expiring_in_no_days,:pv_expired_in_no_days,TO_CHAR(:pv_expired_from,'YYYY-MM-DD'),TO_CHAR(:pv_expired_until,'YYYY-MM-DD'),TO_CHAR(:pv_valid_from,'YYYY-MM-DD'),TO_CHAR(:pv_valid_until,'YYYY-MM-DD'),:pv_owner_organizations,:pv_owners,:pv_l1_approver,:pv_l2_approver,:pv_applies_to_organizations,:pv_related_areas_of_compliance,NULL,NULL,NULL,:pv_regulatory_name0,NULL,NULL,NULL,NULL,NULL,NULL,:pv_regulatory_name1,NULL,NULL,:pv_regulatory_name2,NULL,:pv_regulatory_name3,:pv_regulatory_name4,'MS_GRC_REGULATORY_BODY',:pv_regulatory_name5)) as ms_varchar2_array)) 
    WHERE column_value = OBJECT_ID
    ) ORDER BY  UPPER_OBJECT_NAME
