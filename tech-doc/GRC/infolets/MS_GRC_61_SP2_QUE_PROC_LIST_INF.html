ms_grc_que_proc_mls_summary quemlssumm

from table(cast(ms_grc_rep_browser_fil_denorm.fn_ques_proc_report_filters(ms_grc_array_type(:pv_object_id,:pv_object_name,:pv_object_status,:pv_response_type,:pv_nature,:pv_question_type,:pv_category,:pv_created_last_no_days,to_char(:pv_created_from,'YYYY-MM-DD'),to_char(:pv_created_until,'YYYY-MM-DD'),:pv_expiring_in_no_days,:pv_expired_in_no_days,to_char(:pv_expired_from,'YYYY-MM-DD'),to_char(:pv_expired_until,'YYYY-MM-DD'),to_char(:pv_valid_from,'YYYY-MM-DD'), to_char(:pv_valid_until,'YYYY-MM-DD'), :pv_owner_organizations,:pv_owners,:pv_l1_approver,:pv_l2_approver,:pv_applies_to_organizations,:pv_related_aoc,:pv_related_assets,
:pv_related_asset_classes,null, :pv_related_controls,null, null,null,null,null,null,:pv_related_process,null,null,:pv_related_reference,:pv_related_requirements, null,:pv_related_standards,'MS_GRC_QUESTION_PROCEDURE',:LOGIN_ID)) as MS_VARCHAR2_ARRAY))) 
and QUEMLSSUMM.LOCALE_ID =  (select LOCALE_ID from SI_USERS_T T1 ,SI_LOCALES T2 where USER_NAME = :LOGIN_ID  and T1.LOCALE = T2.LOCALE_STRING)
