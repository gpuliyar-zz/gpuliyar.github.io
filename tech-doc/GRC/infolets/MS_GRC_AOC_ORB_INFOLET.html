SELECT
V.object_id as Area_of_Compliance_ID,
V.OBJECT_NAME as Area_of_Compliance_Name,	
v.object_status as Status,					
v.aoc_type as aoc_type,	
V.created_on as created_on,
v.created_by as created_by,
v.updated_on as modified_on,
v.updated_by as modification_requested_by,
v.valid_from as valid_from,
v.valid_until as valid_until ,
v.description as description,
v.owner_org_names as Owner_Organizations,
v.owner_name as owners,
v.level_1_approver as Level_1_Approver,
v.level_2_approver as Level_2_Approver,
'MS_GRC_AREA_OF_COMPLIANCE'  as Additional_column_3,
ms_grc_utilities.get_form_url (dd_object_type, object_id) as form_url,
rel_id  as rel_config_id
FROM ms_grc_AOC_DENORM_RV V,
     MS_GRC_REL_DEFN A
WHERE v.LOCALE = (SELECT LOCALE FROM si_users_t where USER_NAME = :LOGIN_ID)
AND src_obj_type   = :SRC_OBJ_TYPE
AND a.dest_obj_type = 'MS_GRC_AREA_OF_COMPLIANCE'
AND a.enabled = 'Y'
AND EXISTS
(
	SELECT COLUMN_VALUE
	FROM 
	TABLE( CAST(ms_grc_orb_report_filter.fn_AOC_ORB_filters(:LOGIN_ID, :Area_of_Compliance_Name,:Area_of_Compliance_ID,:AOC_TYPE)AS ms_varchar2_array) ) 
	WHERE COLUMN_VALUE = v.OBJECT_ID
)
order by upper(V.OBJECT_NAME)
