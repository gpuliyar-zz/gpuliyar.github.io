SELECT ASCL_SUMM.OBJECT_ID AS OBJECT_ID, 
  ASCL_SUMM.OBJECT_NAME AS OBJECT_NAME,
  MLS_SUMM.ASSET_CLASS_TYPE AS ASSET_CLASS_TYPE,
  MLS_SUMM.OBJECT_LEVEL AS OBJECT_LEVEL,
  ASCL_SUMM.OBJECT_PARENT_NAME AS OBJECT_PARENT_NAME,
  'MS_GRC_ASSET_CLASS' as Additional_column3,
NULL as form_url,
'MS_GRC_ASSET_CLASS' as Additional_column4,
ASCL_SUMM.UPPER_OBJECT_NAME AS UPPER_OBJECT_NAME
FROM MS_GRC_ASSET_CLASS_SUMMARY ASCL_SUMM,MS_GRC_ASSET_CLASS_MLS_SUMMARY MLS_SUMM
where ASCL_SUMM.OBJECT_ID = MLS_SUMM.OBJECT_ID
and MLS_SUMM.LOCALE_ID = (select LOCALE_ID from SI_USERS_T T1 ,SI_LOCALES T2 where USER_NAME = :LOGIN_ID and T1.LOCALE = T2.LOCALE_STRING)
AND EXISTS
  (SELECT COLUMN_VALUE 
    FROM TABLE( CAST (ms_grc_orb_report_filter.fn_asset_class_ORB_filters(:LOGIN_ID, :Asset_Class_Name, :Asset_Class_ID,:Object_Level,:Object_Type) AS ms_varchar2_array)) 
WHERE COLUMN_VALUE = ASCL_SUMM.OBJECT_ID
  )
AND TRUNC(sysdate) BETWEEN NVL(TRUNC(ASCL_SUMM.valid_from),TRUNC(sysdate)) AND NVL(TRUNC(ASCL_SUMM.valid_until),TRUNC(sysdate))
ORDER BY UPPER(OBJECT_NAME)
