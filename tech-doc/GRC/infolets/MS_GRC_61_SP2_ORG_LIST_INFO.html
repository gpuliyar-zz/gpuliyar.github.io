select AOCSUMM.OBJECT_ID as OBJECT_ID,
AOCSUMM.object_name as object_name,
AOCMLSSUMM.OBJECT_LEVEL as OBJECT_LEVEL,
aocSumm.object_parent as OBJECT_PARENT,
AOCMLSSUMM.BASEL_II_ENTITY as BASEL_II_ENTITY,
AOCMLSSUMM.OBJECT_STATUS as OBJECT_STATUS,
aocsumm.created_on as created_on,
(SELECT USER_FULL_NAME FROM MS_GRC_USER_FULL_NAME_V where user_name = aocSumm.CREATED_BY) AS CREATED_BY,
updated_on,
(SELECT USER_FULL_NAME FROM MS_GRC_USER_FULL_NAME_V where user_name = aocSumm.Updated_By) AS Updated_By,
AOCSUMM.VALID_FROM as VALID_FROM,
AOCSUMM.VALID_UNTIL as VALID_UNTIL,
AOCSUMM.DESCRIPTION as DESCRIPTION,
AOCSUMM.OWNER_ORG_NAMES as OWNER_ORG_NAMES,
aocsumm.owner_name as owner_name,
(SELECT USER_FULL_NAME FROM MS_GRC_USER_FULL_NAME_V where user_name = aocSumm.level_1_approver) AS level_1_approver,
(SELECT USER_FULL_NAME FROM MS_GRC_USER_FULL_NAME_V where user_name = aocSumm.LEVEL_2_APPROVER) AS LEVEL_2_APPROVER,
REL_OBJECTIVE_OBJ_NAME_AGG,
REL_REFRENCES_OBJ_NAME_AGG
,aocsumm.REPORT_JSON as REPORT_JSON
,LEVEL_1_APPROVER AS LEVEL_1_APPR,
LEVEL_2_APPROVER AS LEVEL_2_APPR
,aocSumm.parent_object_id AS parent_object_id
from  MS_GRC_ORG_PRO_SUMMARY AOCSUMM,
ms_grc_org_pro_mls_summary aocmlssumm
where AOCMLSSUMM.OBJECT_ID = AOCSUMM.OBJECT_ID
and AOCMLSSUMM.OBJECT_ID in (select column_value
from table(cast(ms_grc_rep_browser_fil_denorm.fn_org_report_filter(ms_grc_array_type(:pv_object_id,:pv_object_name,:pv_object_status,:pv_basel_entity,:pv_created_last_no_days,to_char(:pv_created_from,'YYYY-MM-DD'), TO_CHAR(:pv_created_until,'YYYY-MM-DD'),:pv_expiring_in_no_days,:pv_object_id0,:pv_object_id1,:pv_object_id2,TO_CHAR(:pv_object_id3,'YYYY-MM-DD'),TO_CHAR(:pv_object_id4,'YYYY-MM-DD'),TO_CHAR(:pv_object_id5,'YYYY-MM-DD'),TO_CHAR(:pv_object_id6,'YYYY-MM-DD'), :pv_object_id7,:pv_object_id8,:pv_object_id9,:pv_object_name0, null,null,null,
null,null,null,null,null,null,null,:pv_object_name1,null,null,null,null,null,:pv_object_name2, null,null,'MS_GRC_ORG_PROFILE',:pv_object_name3)) as MS_VARCHAR2_ARRAY)))
and  AOCMLSSUMM.LOCALE_ID = (select LOCALE_ID from SI_USERS_T T1 ,SI_LOCALES T2 where USER_NAME = :pv_object_name3 and T1.LOCALE = T2.LOCALE_STRING)
and nvl(is_orphan, 1)=nvl(:pv_orphan_object,nvl(is_orphan,1))
order by upper(object_name)
