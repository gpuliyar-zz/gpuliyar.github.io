    FROM TABLE(CAST(ms_grc_rep_browser_fil_denorm.fn_ASSET_report_filters(ms_grc_array_type(:pv_object_id,:pv_object_name,:pv_asset_type,:pv_object_status,:pv_criticality,:pv_confidentiality, :pv_Integrity,:pv_availability,:pv_serial,:pv_object_id0, :pv_object_id1, to_char(:pv_object_id2,'YYYY-MM-DD'),TO_CHAR(:pv_object_id3,'YYYY-MM-DD'),:pv_object_id4,:pv_object_id5,:pv_object_id6,:pv_object_id7,TO_CHAR(:pv_object_id8,'YYYY-MM-DD'),TO_CHAR(:pv_object_id9,'YYYY-MM-DD'),TO_CHAR(:pv_object_name0,'YYYY-MM-DD'), TO_CHAR(:pv_object_name1,'YYYY-MM-DD'), :pv_object_name2,:pv_object_name3,:pv_object_name4,:pv_object_name5,:pv_object_name6,NULL,:pv_object_name7,:pv_object_name8,NULL, :pv_object_name9,NULL, NULL,NULL,NULL,:pv_asset_type0,NULL,NULL,:pv_asset_type1,NULL,:pv_asset_type2,:pv_asset_type3, :pv_asset_type4,:pv_asset_type5,'MS_GRC_ASSET',:pv_asset_type6)) AS ms_varchar2_array))  reqList  
  WHERE COLUMN_VALUE = OBJECT_ID
  )order by upper_object_name
