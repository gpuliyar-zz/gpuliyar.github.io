SELECT object_id AS Reference_ID,
  object_name    AS Reference_Name
FROM
  (SELECT OBJECT_ID,
    OBJECT_NAME,
    dd_object_type,
    (SELECT ms_concat (d.OWNER_ORGANIZATIONS)
    FROM ms_grc_reference_ORG d
    WHERE a.object_id = d.object_id
    ) AS OWNER_ORGANIZATIONS,
    restrict_access_to,
    valid_from,
    valid_until
  FROM MS_GRC_reference a
  ) t1
WHERE DD_OBJECT_TYPE                                                                                                      ='MS_GRC_REFERENCE'
AND ms_grc_utilities.ms_grc_SECURITY(t1.dd_object_type, t1.restrict_access_to, t1.OWNER_ORGANIZATIONS, :LOGIN_ID, 'VIEW') = 'ACCESS'
AND TRUNC(sysdate) BETWEEN NVL(TRUNC(t1.valid_from),TRUNC(sysdate)) AND NVL(TRUNC(t1.valid_until),TRUNC(sysdate))
ORDER BY OBJECT_NAME ASC
