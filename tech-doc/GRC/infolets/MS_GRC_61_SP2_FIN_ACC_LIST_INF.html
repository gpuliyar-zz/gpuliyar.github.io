SELECT finSumm.object_id   AS Financial_Account_ID,
finSumm.object_name  AS Financial_Account_Name,
finMlsSumm.FIN_TYPE         AS Financial_Account_Type,
finMlsSumm.FIN_CATEGORY     AS Financial_Account_Category,
finMlsSumm.object_status    AS Status,
finSumm.created_on       AS created_on,
(SELECT USER_FULL_NAME FROM MS_GRC_USER_FULL_NAME_V where user_name =finSumm.CREATED_BY) AS CREATED_BY,
finSumm.updated_on       AS modified_on,
(SELECT USER_FULL_NAME FROM MS_GRC_USER_FULL_NAME_V where user_name =finSumm.Updated_By) AS modification_requested_by,
finSumm.valid_from       AS valid_from,
finSumm.valid_until      AS valid_until ,
finSumm.description      AS description,
finSumm.owner_org_names  AS Owner_Organizations,
finSumm.owner_name       AS owners,
(SELECT USER_FULL_NAME FROM MS_GRC_USER_FULL_NAME_V where user_name =finSumm.LEVEL_1_APPROVER) AS LEVEL_1_APPROVER,
(SELECT USER_FULL_NAME FROM MS_GRC_USER_FULL_NAME_V where user_name =finSumm.LEVEL_2_APPROVER) AS LEVEL_2_APPROVER,
finMlsSumm.Assertions,
REL_OBJECTIVE_OBJ_NAME_AGG AS Objectives,
rel_refrences_obj_name_agg AS REFERENCES,
ATTACHMENTS                as ATTACHED_FILES,
finSumm.REPORT_JSON AS REPORT_JSON
,FINSUMM.LEVEL_1_APPROVER as LEVEL_1_APPR,
FINSUMM.LEVEL_2_APPROVER as LEVEL_2_APPR,
finSumm.UPPER_OBJECT_NAME as UPPER_OBJECT_NAME
FROM MS_GRC_FIN_ACC_SUMMARY finSumm,
MS_GRC_FIN_ACC_MLS_SUMMARY finMlsSumm
where FINMLSSUMM.OBJECT_ID = FINSUMM.OBJECT_ID
and finMlsSumm.OBJECT_ID IN (select column_value
from table( cast(MS_GRC_REP_BROWSER_FIL_DENORM.FN_FIN_ACC_REPORT_FILTERS(MS_GRC_ARRAY_TYPE(:pv_object_id,:pv_object_name,:pv_fin_type,:pv_fin_cat,:pv_object_status,:pv_created_last_no_days, TO_CHAR(:pv_created_from,'YYYY-MM-DD'),TO_CHAR(:pv_created_until,'YYYY-MM-DD'),:pv_expiring_in_no_days,:pv_expired_in_no_days,TO_CHAR(:pv_expired_from,'YYYY-MM-DD'),TO_CHAR(:pv_expired_until,'YYYY-MM-DD'),TO_CHAR(:pv_valid_from,'YYYY-MM-DD'),TO_CHAR(:pv_valid_until,'YYYY-MM-DD'), :pv_owner_organizations,:pv_owners,:pv_l1_approver,:pv_l2_approver,null,null,null,null,:pv_assertions,null,null, null,null,null,:pv_objectives,null,null,null,null,null,:pv_related_reference, null,null,'MS_GRC_FINANCIAL_ACCOUNTS',:LOGIN_ID))
as MS_VARCHAR2_ARRAY)))
and finMlsSumm.locale_id = (select LOCALE_ID from SI_USERS_T T1 ,SI_LOCALES T2 where USER_NAME = :LOGIN_ID and T1.LOCALE = T2.LOCALE_STRING)
AND NVL(is_orphan, 1)=NVL(:pv_orphan_object,NVL(is_orphan,1))
