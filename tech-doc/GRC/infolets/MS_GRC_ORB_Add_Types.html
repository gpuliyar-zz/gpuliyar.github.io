select rel_id,ve_title,src_obj_type,dest_obj_type,object_type_abb from
(
SELECT   rel_id, ORB_REL_TITLE AS ve_title, src_obj_type, dest_obj_type,display_order
,object_type_abb
    FROM ms_grc_rel_defn a, MS_GRC_ORB_CONFIG_FORM_TITLE b
   WHERE src_obj_type = :SRC_OBJ_TYPE
     AND a.dest_obj_type = b.object_Type
    and a.enabled= 'Y'
     and b.locale='en_US'
    )       
ORDER BY ve_title
