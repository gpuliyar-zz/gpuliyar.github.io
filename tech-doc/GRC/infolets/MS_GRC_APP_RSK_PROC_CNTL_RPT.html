SELECT V_PROCESS_ID,
  V_PROCESS_NAME,
  V_PROCESS_STATUS_DISP,
  V_PROCESS_LEVEL_DISP,
  V_PROCESS_PARENT_DISP,
  V_PROCESS_OWNER_ORG_DISP as V_PROCESS_OWNER_ORG,
  V_PROCESS_OWNERS_DISP as V_PROCESS_OWNERS,
  V_RISK_ID,
  V_RISK_NAME,
  V_RISK_STATUS_DISP,
  V_RISK_LEVEL_DISP,
  V_RISK_PARENT_DISP,
  V_RISK_OWNER_ORGS_DISP as V_RISK_OWNER_ORGS,
  V_RISK_OWNERS_DISP,
  V_RISK_CATEGORY_DISP,
  V_CONTROL_ID,
  V_CONTROL_NAME,
  V_CONTROL_STATUS_DISP,
  V_CONTROL_LEVEL_DISP,
   (SELECT object_name
FROM MS_GRC_COMPLIANCE_OBJE comp
WHERE comp.object_id    = V_CONTROL_PARENT_DISP
AND comp.dd_object_type = 'MS_GRC_CONTROL'
) as  V_CONTROL_PARENT_DISP,
  V_CONTROL_OWNER_ORGS_DISP as V_CONTROL_OWNER_ORGS,
  V_CONTROL_OWNERS_DISP,
  V_ORG_RELATED_TO_PROCESS_DISP as V_ORG_REL_TO_PROC,
  V_ORG_RELATED_TO_RISK_DISP as V_ORG_REL_TO_RISK,
  V_ORG_RELATED_TO_CONTROL_DISP  as  V_ORG_REL_TO_CON
