PROCESS Data Browser All 
SELECT
OBJECT_NAME as DISPLAY_NAME,
DD_OBJECT_TYPE as OBJECT_TYPE,
OVER_RIDE_ACTION,
EDIT_ACTION_PARAMETERS as VIEW_ACTION_PARAMETERS,
OBJECT_ID as OBJECT_ID
FROM ms_grc_process_denorm_drv
WHERE NVL(is_orphan, 1)=NVL(:PV_Orphans,NVL(is_orphan,1))
AND EXISTS
  (SELECT COLUMN_VALUE
  FROM TABLE(CAST(ms_grc_rep_browser_fil_denorm.fn_process_rpt_filter(ms_grc_array_type(:pv_id, :pv_object_name, :pv_obj_status, :pv_type, :pv_category, :pv_created_last_no_days, TO_CHAR(:pv_created_from,'YYYY-MM-DD'), TO_CHAR(:pv_created_until,'YYYY-MM-DD'), :pv_to_expire_in_no_days, :pv_expired_in_no_days, :pv_object_level, :pv_id3, TO_CHAR(:pv_id4,'YYYY-MM-DD'), TO_CHAR(:pv_expired_until,'YYYY-MM-DD'), TO_CHAR(:pv_valid_from,'YYYY-MM-DD'), TO_CHAR(:pv_valid_until,'YYYY-MM-DD'), :pv_owner_organizations, :pv_owners, :pv_l1_approver, :pv_l2_approver, :PV_APPLIES_TO_ORGANIZATIONS, :PV_RELATED_AREAS_OF_COMPLIANCE, :PV_RELATED_ASSETS, NULL, NULL, :PV_RELATED_CONTROL, NULL, NULL, :PV_RELATED_FUNCTIONS, NULL, :PV_RELATED_OBJECTIVES, NULL, :PV_RELATED_PROCESSES, :PV_RELATED_PRODUCTS, NULL, :PV_RELATED_REQUIREMENTS, :PV_RELATED_REFERENCES, :PV_RELATED_RISKS, NULL, 'MS_GRC_PROCESS', :LOGIN_ID))AS ms_varchar2_array))
  WHERE column_value = OBJECT_ID
  ) ORDER BY UPPER_OBJECT_NAME
