MS_GRC_CONTROL_OWNERSHIP_RPT 

  CREATE OR REPLACE FORCE VIEW "MS_GRC_CONTROL_OWNERSHIP_RPT" ("OBJECT_ID", "OBJECT_NAME", "LEGACY_ID", "OWNER_ORGS", "COMPLIANCE_CATEGORY", "KEY_CONTROL", "FREQUENCY", "OWNERS", "ATO_ORGANIZATIONS", "RESTRICT_ACCESS_TO", "DD_OBJECT_TYPE", "DUMMY1", "DUMMY2", "DUMMY3", "DUMMY4", "DUMMY5", "OWNER_ORG_IDS", "ATO_ORG_IDS", "OWNER_IDS", "OVER_RIDE_ACTION", "EDIT_ACTION_PARAMETERS", "DD_EDIT_FLAG", "DD_INSTANCE_ID", "DD_PROCESS_INSTANCE_ID") AS 
  SELECT  a.object_id,
            a.object_name,
            a.legacy_id,
            (SELECT   ms_concat(ms_apps_utilities.get_display_values (
                                   ms_grc_utilities.active_ent_id,
                                   'ORGENTITY',
                                   d.owner_organizations,
                                   ','
                                ))
               FROM   ms_grc_compliance_obje_ogn d
              WHERE   a.object_id = d.object_id)
               owner_orgs,
--            ms_apps_utilities.get_display_value (
--               ms_grc_utilities.active_ent_id,
--               'MS_GRC_COMPLIANCE_CATEGORY',
--               compliance_category
--            )  COMPLIANCE_CATEGORY,
--  ms_apps_utilities.get_display_values ( MS_GRC_UTILITIES.active_ent_id, 'MS_GRC_Nature_of_Relationship',
--(SELECT MS_CONCAT (E.PURPOSE)
--FROM MS_GRC_COMPLIANCE_OBJE_PUR e
--where a.OBJECT_ID = E.OBJECT_ID
--), ',' ) AS COMPLIANCE_CATEGORY,  
(SELECT MS_CONCAT (E.PURPOSE)
FROM MS_GRC_COMPLIANCE_OBJE_PUR e
where a.OBJECT_ID = E.OBJECT_ID
) AS COMPLIANCE_CATEGORY,  
--ms_apps_utilities.get_display_value (
--               ms_grc_utilities.active_ent_id,
--               'MS GRC Yes No',
--               key_compliance
--            )
              key_compliance as  key_control,
--            ms_apps_utilities.get_display_value (
--               ms_grc_utilities.active_ent_id,
--               'MS_GRC_Frequency',
--               a.frequency
--            )
             a.frequency  as  frequency,
            
            ms_apps_utilities.get_user_full_name (b.owners) as Owners,
--            (select ms_apps_utilities.get_user_full_name(b.owners)  from ms_grc_compliance_obje_own b where a.object_id=b.object_id)
--             as Owners,
            (SELECT   ms_grc_utilities.get_relationships (
                         a.object_id,
                         a.dd_object_type,
                         'MS_GRC_ORGANIZATION'
                      )
               FROM   DUAL)
               AS ato_organizations,
            restrict_access_to,
            a.dd_object_type,
            a.dd_process_instance_id AS dummy1,
            a.dd_object_type AS dummy2,
            a.dd_instance_id AS dummy3,
            NVL (a.dd_edit_flag, 'Y') AS dummy4,
            ms_grc_utilities.get_edit_url (a.dd_object_type,
                                           a.dd_process_instance_id,
                                           a.dd_instance_id,
                                           NVL (a.dd_edit_flag, 'Y'),
                                           a.object_name)
               AS dummy5,
            (SELECT   ms_concat (d.owner_organizations)
               FROM   ms_grc_compliance_obje_ogn d
              WHERE   a.object_id = d.object_id)
               AS owner_org_ids,
            (SELECT   ms_grc_utilities.get_relationships (
                         a.object_id,
                         'MS_GRC_CONTROL',
                         'MS_GRC_ORGANIZATION',
                         'OBJECT_ID'
                      )
               FROM   DUAL)
               ato_org_ids,
            (SELECT   ms_concat (owners)
               FROM   ms_grc_compliance_obje_own
              WHERE   object_id = a.object_id)
               owner_ids,
            'Y' AS over_ride_action,
               'proc='
            || TO_CHAR (a.dd_process_instance_id)
            || '|instid='
            || TO_CHAR (a.dd_instance_id)
            || '|edit_flag='
            || NVL (c.dd_edit_flag, 'Y')
               AS edit_action_parameters,
            NVL (c.dd_Edit_flag, 'Y'),
            a.dd_instance_id,
            a.dd_process_instance_id
            
     FROM   ms_grc_compliance_obje a,
            ms_grc_compliance_obje_own b,
            ms_Grc_control c
    where   B.OBJECT_ID = a.OBJECT_ID 
    and B.OBJECT_ID = C.OBJECT_ID
