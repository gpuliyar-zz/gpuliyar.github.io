MS_GRC_PROC_ENTP_RV 

  CREATE OR REPLACE FORCE VIEW "MS_GRC_PROC_ENTP_RV" ("OBJECT_NAME", "OBJECT_ID", "OBJECT_TYPE", "LOCALE_ID", "LOCALE", "DESCRIPTION", "OWNER_IDS", "OWNER_NAME", "OWNER_ORG_IDS", "OWNER_ORG_NAMES", "VALID_UNTIL", "VALID_FROM", "TYPE", "CATEGORY", "OBJECT_LEVEL", "DD_PROCESS_INSTANCE_ID", "DD_INSTANCE_ID", "DD_EDIT_FLAG", "RESTRICT_ACCESS_TO", "PARENT", "STATUS", "REL_DATA_LINK", "DD_OBJECT_TYPE", "OVER_RIDE_ACTION", "EDIT_ACTION_PARAMETERS", "ATTACHMENTS", "CREATED_ON", "CREATED_BY", "UPDATED_ON", "UPDATED_BY", "LEVEL_1_APPROVER", "LEVEL_2_APPROVER", "OBJECT_NAME_URL", "IS_ORPHAN", "UPPER_OBJECT_NAME", "OBJ_PARENT_ID") AS 
  SELECT ProcSumm.OBJECT_NAME AS OBJECT_NAME,
    ProcSumm.Object_id        AS OBJECT_ID,
	'MS_GRC_PROCESS' as OBJECT_TYPE , 
    ProcMlsSumm.LOCALE_ID,
    ProcMlsSumm.LOCALE,
    ProcSumm.description AS DESCRIPTION,
    ProcSumm.OWNER_IDS,
    ProcSumm.OWNER_NAME,
    ProcSumm.OWNER_ORG_IDS,
    ProcSumm.OWNER_ORG_NAMES,
    procsumm.valid_until                  AS VALID_UNTIL,
    procsumm.valid_from                   AS VALID_FROM,
    ProcMlsSumm.PROCESS_TYPE              AS TYPE,
    ProcMlsSumm.PROCESS_CATEGORY          AS CATEGORY,
    ProcMlsSumm.object_level              AS OBJECT_LEVEL,
    procsumm.dd_process_instance_id       AS DD_PROCESS_INSTANCE_ID,
    procsumm.dd_instance_id               AS DD_INSTANCE_ID,
    NVL (procsumm.dd_edit_flag, 'Y')      AS DD_EDIT_FLAG,
    procsumm.restrict_access_to           AS RESTRICT_ACCESS_TO,
    ProcSumm.OBJ_PARENT                   AS PARENT,
    ProcMlsSumm.OBJECT_STATUS             AS STATUS,
    Lovvalue.displayed_value 			  AS REL_DATA_LINK, 
    procsumm.dd_object_type               AS DD_OBJECT_TYPE,
    'Y'                                   AS Over_Ride_Action,
    Procsumm.Edit_Action_Parameters       AS EDIT_ACTION_PARAMETERS,
    procsumm.attachments                  AS ATTACHMENTS,
    procsumm.CREATED_on,
    (SELECT usr.first_name
      || ' '
      || usr.last_name
    FROM si_users_t usr
    WHERE procsumm.CREATED_BY = usr.user_name
    ) AS CREATED_BY,
    procsumm.UPDATED_on,
    (SELECT usr.first_name
      || ' '
      || usr.last_name
    FROM si_users_t usr
    WHERE procsumm.UPDATED_BY = usr.user_name
    ) AS UPDATED_BY,
    (SELECT usr.first_name
      || ' '
      || usr.last_name
    FROM si_users_t usr
    WHERE ProcSumm.level_1_approver = usr.user_name
    ) AS level_1_approver,
    (SELECT usr.first_name
      || ' '
      || usr.last_name
    FROM si_users_t usr
    WHERE ProcSumm.level_2_approver = usr.user_name
    )                        AS Level_2_Approver,
    Procsumm.Object_Name_Url AS object_name_url,
    ProcSumm.is_orphan,
    PROCSUMM.UPPER_OBJECT_NAME,
    PROCSUMM.OBJ_PARENT_ID
  FROM MS_GRC_PROCESS_SUMMARY ProcSumm,
    MS_GRC_PROCESS_MLS_SUMMARY ProcMlsSumm,
	Ms_Qs_List_Of_Values Lovvalue,
    Ms_Qs_Lov_Names Lovname
  WHERE Procsumm.Object_Id = Procmlssumm.Object_Id
  AND Lovvalue.Org_Entity_Id           = 1
AND Lovname.Lov_Name                 = 'MS_GRC_LABLES'
and Lovvalue.stored_value = 'CLICK_HERE'
AND Lovvalue.Lov_Id                  = Lovname.Lov_Id
AND Lovvalue.Locale_id = ProcMlsSumm.locale_id
