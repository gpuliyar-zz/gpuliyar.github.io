MS_GRC_PRODUCT_RV 

  CREATE OR REPLACE FORCE VIEW "MS_GRC_PRODUCT_RV" ("OBJECT_NAME", "OBJECT_ID", "OBJECT_LEVEL", "OBJ_STATUS", "OBJECT_PARENT", "PRODUCT_CATEGORY", "PRODUCT_LOCATION", "SALES_CHANNEL", "OWNERS", "VALID_UNTIL", "VALID_FROM", "RESTRICT_ACCESS_TO", "DD_EDIT_FLAG", "DD_OBJECT_TYPE", "DD_INSTANCE_ID", "DD_PROCESS_INSTANCE_ID", "DISPLAY_NAME", "OBJECT_TYPE", "OVER_RIDE_ACTION", "EDIT_ACTION_PARAMETERS", "EXPIRED_FLAG", "DESCRIPTION", "APPLIES_TO_ORGANIZATIONS", "RELATED_PROCESSES", "RELATED_RISKS", "RELATED_CONTROLS", "RELATED_AREAS_OF_COMPLIANCE", "RELATED_REQUIREMENTS", "RELATED_FINANCIAL_ACCOUNTS", "RELATED_FUNCTIONS", "RELATED_OBJECTIVES", "RELATED_FRAMEWORK_REFERENCES", "RELATED_REFERENCES", "RELATED_PRODUCTS", "EDIT", "OWNER_ORG_IDS", "PRODUCT_TYPE", "OBJ_CREATED_BY", "OBJ_CREATED_ON", "OBJ_MODIFIED_BY", "OBJ_MODIFIED_ON", "ATTACHMENTS", "LEVEL_1_APPROVER", "LEVEL_2_APPROVER") AS 
  SELECT object_name,
    object_id,
    object_level,
    obj_status,
    object_parent ,
    product_category,
    (SELECT ms_concat(product_location)
    FROM ms_grc_product_loc l
    WHERE l.object_id =a.object_id
    ) AS product_location,
    (SELECT ms_concat(sales_channel)
    FROM ms_grc_product_sch l
    WHERE l.object_id=a.object_id
    ) AS sales_channel,
    (SELECT ms_concat(usr.first_name
      || ' '
      || usr.last_name) AS owners
    FROM ms_grc_product_own d,
      si_users_t usr,
      si_ent e
    WHERE usr.user_name   = d.owners
    AND usr.enterprise_id = e.ENTERPRISE_ID
    AND d.object_id       =a.object_id
    ) AS OWNERS,
    valid_until,
    valid_from,
    restrict_access_to,
    NVL (DD_EDIT_FLAG, 'Y') AS DD_EDIT_FLAG,
    dd_object_type,
    dd_instance_id,
    DD_PROCESS_INSTANCE_ID,
   SUBSTR(a.object_name,1,30)AS display_name,
    dd_object_type          AS Object_Type,
    'Y'                     AS OVER_RIDE_ACTION,
    'proc='
    || TO_CHAR(dd_process_instance_id)
    ||'|instid='
    ||TO_CHAR(dd_instance_id)
    ||'|edit_flag='
    ||NVL(dd_edit_flag,'Y')                                AS EDIT_ACTION_PARAMETERS,
    ms_Grc_utilities.INCLUDE_items(VALID_FROM,VALID_UNTIL) AS EXPIRED_FLAG,
    DESCRIPTIOn,
    (SELECT ms_concat(orb.tgt_obj_id)
    FROM ms_grc_orb_relations_pub_v orb,
      SI_ORG_ENTITIES base
    WHERE src_obj_type   = 'MS_GRC_PRODUCT'
    AND orb.tgt_obj_type = 'MS_GRC_ORGANIZATION'
    AND orb.tgt_obj_id   = base.org_entity_id
    AND orb.src_obj_id   =a.object_id
    ) AS APPLIES_TO_ORGANIZATIONS,
    (SELECT ms_concat(orb.tgt_obj_id)
    FROM ms_grc_orb_relations_pub_v orb,
      ms_grc_core_object base
    WHERE src_obj_type   = 'MS_GRC_PRODUCT'
    AND orb.tgt_obj_type = 'MS_GRC_PROCESS'
    AND orb.tgt_obj_id   = base.object_id
    AND orb.src_obj_id   =a.object_id
    ) AS RELATED_Processes,
    (SELECT ms_concat(orb.tgt_obj_id)
    FROM ms_grc_orb_relations_pub_v orb,
      ms_grc_risk base
    WHERE src_obj_type   = 'MS_GRC_PRODUCT'
    AND orb.tgt_obj_type = 'MS_GRC_RISK'
    AND orb.tgt_obj_id   = base.object_id
    AND orb.src_obj_id   =a.object_id
    ) AS RELATED_Risks,
    (SELECT ms_concat(orb.tgt_obj_id)
    FROM ms_grc_orb_relations_pub_v orb,
      ms_grc_compliance_obje base
    WHERE src_obj_type   = 'MS_GRC_PRODUCT'
    AND orb.tgt_obj_type = 'MS_GRC_CONTROL'
    AND orb.tgt_obj_id   = base.object_id
    AND orb.src_obj_id   =a.object_id
    ) AS RELATED_Controls,
    (SELECT ms_concat(orb.tgt_obj_id)
    FROM ms_grc_orb_relations_pub_v orb,
      ms_grc_area_of_complia base
    WHERE src_obj_type   = 'MS_GRC_PRODUCT'
    AND orb.tgt_obj_type = 'MS_GRC_AREA_OF_COMPLIANCE'
    AND orb.tgt_obj_id   = base.object_id
    AND orb.src_obj_id   =a.object_id
    ) AS RELATED_areas_of_compliance,
    (SELECT ms_concat(orb.tgt_obj_id)
    FROM ms_grc_orb_relations_pub_v orb,
      ms_grc_requirement base
    WHERE src_obj_type   = 'MS_GRC_PRODUCT'
    AND orb.tgt_obj_type = 'MS_GRC_REQUIREMENT'
    AND orb.tgt_obj_id   = base.object_id
    AND orb.src_obj_id   =a.object_id
    ) AS RELATED_REQUIREMENTS,
    (SELECT ms_concat(orb.tgt_obj_id)
    FROM ms_grc_orb_relations_pub_v orb,
      ms_grc_financial_accou base
    WHERE src_obj_type   = 'MS_GRC_PRODUCT'
    AND orb.tgt_obj_type = 'MS_GRC_FINANCIAL_ACCOUNTS'
    AND orb.tgt_obj_id   = base.object_id
    AND orb.src_obj_id   =a.object_id
    ) AS RELATED_FINANCIAL_ACCOUNTS,
    (SELECT ms_concat(orb.tgt_obj_id)
    FROM ms_grc_orb_relations_pub_v orb,
      ms_grc_function base
    WHERE src_obj_type   = 'MS_GRC_PRODUCT'
    AND orb.tgt_obj_type = 'MS_GRC_FUNCTION'
    AND orb.tgt_obj_id   = base.object_id
    AND orb.src_obj_id   =a.object_id
    ) AS RELATED_Functions,
    (SELECT ms_concat(orb.tgt_obj_id)
    FROM ms_grc_orb_relations_pub_v orb,
      ms_grc_objectives base
    WHERE src_obj_type   = 'MS_GRC_PRODUCT'
    AND orb.tgt_obj_type = 'MS_GRC_OBJECTIVES'
    AND orb.tgt_obj_id   = base.object_id
    AND orb.src_obj_id   =a.object_id
    ) AS RELATED_Objectives,
    (SELECT ms_concat(orb.tgt_obj_id)
    FROM ms_grc_orb_relations_pub_v orb,
      ms_grc_model_reference base
    WHERE src_obj_type   = 'MS_GRC_PRODUCT'
    AND orb.tgt_obj_type = 'MS_GRC_MODEL_REFERENCE'
    AND orb.tgt_obj_id   = base.object_id
    AND orb.src_obj_id   =a.object_id
    ) AS RELATED_FRAMEWORK_REFERENCES,
    (SELECT ms_concat(orb.tgt_obj_id)
    FROM ms_grc_orb_relations_pub_v orb,
      ms_grc_reference base
    WHERE src_obj_type   = 'MS_GRC_PRODUCT'
    AND orb.tgt_obj_type = 'MS_GRC_REFERENCE'
    AND orb.tgt_obj_id   = base.object_id
    AND orb.src_obj_id   =a.object_id
    ) AS RELATED_REFERENCES,
    (SELECT ms_concat(orb.tgt_obj_id)
    FROM ms_grc_orb_relations_pub_v orb,
      ms_grc_product base
    WHERE src_obj_type   = 'MS_GRC_PRODUCT'
    AND orb.tgt_obj_type = 'MS_GRC_PRODUCT'
    AND orb.tgt_obj_id   = base.object_id
    AND orb.src_obj_id   =a.object_id
    )                                                                                                                            AS RELATED_PRODUCTS,
    MS_GRC_UTILITIES.GET_EDIT_URL (DD_OBJECT_TYPE, DD_PROCESS_INSTANCE_ID, DD_INSTANCE_ID, NVL (DD_EDIT_FLAG, 'Y'), OBJECT_NAME) AS EDIT,
    (SELECT ms_concat (owner_organizations)
    FROM ms_grc_product_ogn E
    WHERE E.OBJECT_ID = a.OBJECT_ID
    ) AS OWNER_ORG_IDS,
    (SELECT MS_CONCAT(PRODUCT_TYPE)
    FROM MS_GRC_PRODUCT_TYP T
    WHERE T.OBJECT_ID=a.OBJECT_ID
    )AS Product_type,
    (SELECT usr.first_name
      || ' '
      || usr.last_name
    FROM si_users_t usr
    WHERE OBJ_CREATED_BY = usr.user_name
    ) AS CREATED_BY,
    OBJ_CREATED_ON,
    (SELECT usr.first_name
      || ' '
      || usr.last_name
    FROM si_users_t usr
    WHERE obj_modified_by = usr.user_name
    ) AS UPDATED_BY,
    obj_modified_on,
    attachments,
    level_1_approver,
    level_2_approver
  from MS_GRC_PRODUCT a
