MS_GRC_MODEL_REFERENCE_RV 

  CREATE OR REPLACE FORCE VIEW "MS_GRC_MODEL_REFERENCE_RV" ("OBJECT_NAME", "OBJECT_ID", "OWNERS", "VALID_UNTIL", "VALID_FROM", "DESCRIPTION", "DD_OBJECT_TYPE", "DD_PROCESS_INSTANCE_ID", "DD_INSTANCE_ID", "DD_EDIT_FLAG", "RESTRICT_ACCESS_TO", "OBJ_STATUS", "DISPLAY_NAME", "OBJ_TYPE", "CONTROL", "PROCESS", "RISK", "RELATED_FRAMEWORK_REFERENCE", "RELATED_REFERENCE", "OBJECT_NAME_URL", "OBJ_SOURCE", "EXPIRED_FLAG", "OVER_RIDE_ACTION", "EDIT_ACTION_PARAMETERS", "OWNER_ORG_IDS") AS 
  SELECT a.object_name,
    a.object_id,
    (SELECT ms_concat(usr.first_name
      || ' '
      || usr.last_name) AS owners
    FROM ms_grc_model_reference_own d,
      si_users_t usr  , si_ent e
    WHERE usr.user_name = d.owners
    and  usr.enterprise_id = e.ENTERPRISE_ID
    AND a.OBJECT_ID     = D.OBJECT_ID) AS OWNERS,
    valid_until,
    valid_from,
   description,
    dd_object_type,
    dd_process_instance_id,
    dd_instance_id,
    NVL (dd_edit_flag, 'Y') AS DD_EDIT_FLAG,
    restrict_access_to,
    OBJ_STATUS,
    SUBSTR (object_name, 1, 30) display_name ,
    obj_type,
   (SELECT MS_GRC_FN_CONCAT(orb.tgt_obj_id)
    FROM ms_grc_orb_relations_pub_v orb,
    ms_grc_compliance_obje base
    WHERE src_obj_type   = 'MS_GRC_MODEL_REFERENCE'
    AND orb.tgt_obj_type = 'MS_GRC_CONTROL'
    AND orb.src_obj_id   = a.object_id
    AND orb.tgt_obj_id   = base.object_id
    ) AS control,
   (SELECT MS_GRC_FN_CONCAT(orb.tgt_obj_id)
    FROM ms_grc_orb_relations_pub_v orb,
    ms_grc_core_object base
    WHERE src_obj_type   = 'MS_GRC_MODEL_REFERENCE'
    AND orb.tgt_obj_type = 'MS_GRC_PROCESS'
    AND orb.src_obj_id   = a.object_id
     AND orb.tgt_obj_id   = base.object_id
     ) AS process,
    (SELECT MS_GRC_FN_CONCAT(orb.tgt_obj_id)
     FROM ms_grc_orb_relations_pub_v orb,
     ms_grc_risk base
     WHERE src_obj_type   = 'MS_GRC_MODEL_REFERENCE'
     AND orb.tgt_obj_type = 'MS_GRC_RISK'
     AND orb.src_obj_id   = a.object_id
     AND orb.tgt_obj_id   = base.object_id
     ) AS risk,
    (SELECT MS_GRC_FN_CONCAT(orb.tgt_obj_id)
     FROM ms_grc_orb_relations_pub_v orb,
     ms_grc_model_reference base
     WHERE src_obj_type   = 'MS_GRC_MODEL_REFERENCE'
     AND orb.tgt_obj_type = 'MS_GRC_MODEL_REFERENCE'
     AND orb.src_obj_id   = a.object_id
     AND orb.tgt_obj_id   = base.object_id
     ) AS related_framework_reference,
    (SELECT MS_GRC_FN_CONCAT(orb.tgt_obj_id)
     FROM ms_grc_orb_relations_pub_v orb,
     ms_grc_reference base
     WHERE src_obj_type   = 'MS_GRC_MODEL_REFERENCE'
     AND orb.tgt_obj_type = 'MS_GRC_REFERENCE'
     AND orb.src_obj_id   = a.object_id
     AND orb.tgt_obj_id   = base.object_id
     )AS Related_reference,
    ms_grc_utilities.get_edit_url (dd_object_type, dd_process_instance_id, dd_instance_id, NVL(DD_EDIT_FLAG,'Y'), a.OBJECT_NAME) AS object_name_url,
    obj_source,
     ms_Grc_utilities.INCLUDE_eXPIRED(a.VALID_FROM,a.VALID_UNTIL) AS EXPIRED_FLAG,
      'Y' AS OVER_RIDE_ACTION,
      'proc='|| TO_CHAR(dd_process_instance_id)||'|instid='||TO_CHAR(dd_instance_id)||'|edit_flag='||NVL(dd_edit_flag,'Y')  AS EDIT_ACTION_PARAMETERS,
    (SELECT MS_CONCAT (OWNER_ORGANIZATIONS)
    FROM ms_grc_model_reference_org E
    WHERE E.OBJECT_ID = A.OBJECT_ID
    ) AS owner_org_ids
  FROM ms_grc_model_reference a
