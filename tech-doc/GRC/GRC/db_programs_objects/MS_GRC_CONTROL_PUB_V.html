MS_GRC_CONTROL_PUB_V 

  CREATE OR REPLACE FORCE VIEW "MS_GRC_CONTROL_PUB_V" ("OBJECT_ID", "OBJECT_NAME", "DESCRIPTION", "DESCRIPTION_HTM", "OWNERS", "OWNER_ORGANIZATIONS", "RESTRICT_ACCESS_TO", "OBJ_STATUS", "VALID_FROM", "VALID_UNTIL", "OBJECT_LEVEL", "OBJECT_PARENT", "LEGACY_ID", "COMPLIANCE_CATEGORY", "COMPLIANCE_TYPE", "KEY_COMPLIANCE", "FREQUENCY", "FREQUENCY_DATE", "START_AFTER", "DUE_BY", "NEXT_SCHEDULED_DATE", "RTF_ATTACH", "DD_OBJECT_TYPE", "DD_PROCESS_INSTANCE_ID", "DD_INSTANCE_ID", "INITIATED_BY", "CREATED_ON", "MODIFIED_ON", "CREATED_BY", "UPDATED_BY", "LEVEL_1_APPROVER", "LEVEL_2_APPROVER", "PURPOSE", "CONTROL_TYPE", "NATURE_TYPE", "SOURCE_TYPE", "PRIORITY_TYPE", "OBJECT_NAME_URL") AS 
  SELECT /*@public*/ A.OBJECT_ID,
   A.OBJECT_NAME,
    regexp_replace(A.DESCRIPTION, '<[^>]+>', NULL) AS DESCRIPTION,
    A.DESCRIPTION                                  AS DESCRIPTION_HTM,
    (SELECT MS_CONCAT(OWNERS)
    FROM MS_GRC_COMPLIANCE_OBJE_OWN B
    WHERE B.OBJECT_ID = A.OBJECT_ID
    ) AS OWNERS,
    (SELECT MS_CONCAT(OWNER_ORGANIZATIONS)
    FROM MS_GRC_COMPLIANCE_OBJE_OGN B
    WHERE B.OBJECT_ID = A.OBJECT_ID
    ) AS OWNER_ORGANIZATIONS,
    A.RESTRICT_ACCESS_TO,
    A.obj_STATUS,
    A.VALID_FROM,
    A.VALID_UNTIL,
    A.OBJECT_LEVEL,
    A.OBJECT_PARENT,
    A.LEGACY_ID,
    A.compliance_category,
    A.COMPLIANCE_TYPE,
    DECODE(lower(KEY_COMPLIANCE),'yes',1,'no',2,2) AS KEY_COMPLIANCE,
    A.FREQUENCY,
    A.FREQUENCY_DATE,
    A.START_AFTER,
    A.DUE_BY,
    A.NEXT_SCHEDULED_DATE,
    A.RTF_aTTACH,
    A.DD_OBJECT_TYPE,
    A.DD_PROCESS_INSTANCE_ID,
    A.DD_INSTANCE_ID,
    NVL(A.OBJ_MODIFIED_BY,A.OBJ_CREATED_BY) AS INITIATED_BY,
    C.DD_CREATED_ON AS CREATED_ON,
    C.DD_MODIFIED_ON AS MODIFIED_ON,
	A.OBJ_CREATED_BY AS CREATED_BY ,
    A.OBJ_MODIFIED_BY AS UPDATED_BY,
	A.LEVEL_1_APPROVER AS LEVEL_1_APPROVER ,
	A.LEVEL_2_APPROVER AS LEVEL_2_APPROVER , 
	(SELECT MS_CONCAT(PURPOSE) FROM MS_GRC_CONTROL_PUR C WHERE C.OBJECT_ID = A.OBJECT_ID ) AS PURPOSE,
	(SELECT MS_CONCAT(CONTROL_TYPE) FROM MS_GRC_CONTROL_CTY D WHERE D.OBJECT_ID = A.OBJECT_ID ) AS CONTROL_TYPE,
	(SELECT MS_CONCAT(NATURE_TYPE) FROM MS_GRC_CONTROL_NRE E WHERE E.OBJECT_ID = A.OBJECT_ID) AS NATURE_TYPE,
	A.SOURCE_TYPE AS SOURCE_TYPE,
	A.PRIORITY_TYPE AS PRIORITY_TYPE,
	ms_grc_utilities.get_edit_url (A.dd_object_type,
                                               A.dd_process_instance_id,
                                               A.dd_instance_id,
                                               NVL(C.DD_EDIT_FLAG,'Y'),
                                               A.OBJECT_NAME)            AS OBJECT_NAME_URL
	
  FROM MS_GRC_COMPLIANCE_OBJE A,  MS_GRC_CONTROL C
  WHERE A.DD_OBJECT_TYPE = 'MS_GRC_CONTROL'
  AND C.OBJECT_ID = A.OBJECT_ID
