MS_GRC_ASSET_EDS_V 

  CREATE OR REPLACE FORCE VIEW "MS_GRC_ASSET_EDS_V" ("ID", "NAME", "STATUS", "LVL", "PARENT", "DESCRIPTION", "TYPE", "PHYSICAL_LOCATION", "BUSINESS_CRITICALITY", "CONFIDENTIALITY", "INTEGRITY", "AVAILABILITY", "RECOVER_RTO", "RECOVER_DURATION_RTO", "RECOVER_RPO", "RECOVER_DURATION_RPO", "RECOVER_WRT", "RECOVER_DURATION_WRT", "TD", "DURATION_MTD", "RTA", "DURATION_RTA", "OWNER_ORGANIZATION_IDS", "OWNER_ORGANIZATIONS", "OWNERS_IDS", "OWNERS", "VALID_FROM", "VALID_TO", "CREATED_BY", "CREATED_ON", "MODIFIED_BY", "MODIFIED_ON", "ATTACHMENTS") AS 
  SELECT 
A.OBJECT_ID AS ID,
A.OBJECT_NAME AS NAME,
(SELECT Z.STATUS_NAME FROM MS_GRC_STATUS Z WHERE Z.STATUS_CODE=A.OBJ_STATUS_SV
AND Z.LOCALE_ID=MS_GRC_COMMON_UTILS.fn_get_default_locale_id) AS Status,
(SELECT Z.DISPLAYED_VALUE FROM MS_QS_LOV_ORG_VALUES_V Z WHERE (TRIM(Z.LOV_NAME)='MS_GRC_LEVEL') AND Z.STORED_VALUE=A.OBJECT_LEVEL_SV
AND Z.LOCALE_ID=MS_GRC_COMMON_UTILS.fn_get_default_locale_id) 
AS lvl,
(SELECT Z.OBJECT_NAME FROM MS_GRC_ASSET_RV Z WHERE Z.OBJECT_ID=A.OBJECT_PARENT_ID) AS Parent,
REGEXP_REPLACE(A.DESCRIPTION,'<[^>]+>', NULL) AS Description,
(SELECT Z.DISPLAYED_VALUE FROM MS_QS_LOV_ORG_VALUES_V Z WHERE (TRIM(Z.LOV_NAME)='MS_GRC_Asset_Type')
 AND Z.STORED_VALUE=A.ASSET_TYPE_SV AND Z.LOCALE_ID=MS_GRC_COMMON_UTILS.fn_get_default_locale_id) AS type,
(SELECT Z.DISPLAYED_VALUE FROM MS_QS_LOV_ORG_VALUES_V Z WHERE (TRIM(Z.LOV_NAME)='MS GRC Physical Location') AND Z.STORED_VALUE=A.PHYSICAL_LOCATION_SV
AND Z.LOCALE_ID=MS_GRC_COMMON_UTILS.fn_get_default_locale_id) 
AS Physical_Location,
(SELECT Z.DISPLAYED_VALUE FROM MS_QS_LOV_ORG_VALUES_V Z WHERE (TRIM(Z.LOV_NAME)='MS_GRC_Business_Criticality') AND Z.STORED_VALUE=A.BUSINESS_CRITICALITY_SV
AND Z.LOCALE_ID=MS_GRC_COMMON_UTILS.fn_get_default_locale_id) 
AS Business_criticality,
(SELECT Z.DISPLAYED_VALUE FROM MS_QS_LOV_ORG_VALUES_V Z WHERE (TRIM(Z.LOV_NAME)='MS_GRC_Business_Criticality') AND Z.STORED_VALUE=A.CONFIDENTIALITY_SV
AND Z.LOCALE_ID=MS_GRC_COMMON_UTILS.fn_get_default_locale_id)
 AS Confidentiality,
(SELECT Z.DISPLAYED_VALUE FROM MS_QS_LOV_ORG_VALUES_V Z WHERE (TRIM(Z.LOV_NAME)='MS_GRC_Business_Criticality') AND Z.STORED_VALUE=A.INTEGRITY_SV
AND Z.LOCALE_ID=MS_GRC_COMMON_UTILS.fn_get_default_locale_id) 
AS Integrity,
(SELECT Z.DISPLAYED_VALUE FROM MS_QS_LOV_ORG_VALUES_V Z WHERE (TRIM(Z.LOV_NAME)='MS_GRC_Business_Criticality') AND Z.STORED_VALUE=A.AVAILABILITY_ASSET_SV
AND Z.LOCALE_ID=MS_GRC_COMMON_UTILS.fn_get_default_locale_id)
 AS Availability,
A.RECOVERY_TIME_OBJECTIVE AS Recover_RTO,
(SELECT Z.DISPLAYED_VALUE FROM MS_QS_LOV_ORG_VALUES_V Z WHERE (TRIM(Z.LOV_NAME)='MS_GRC_Duration_Values') AND Z.STORED_VALUE=A.DURATION_RTO_SV
AND Z.LOCALE_ID=MS_GRC_COMMON_UTILS.fn_get_default_locale_id) 
AS Recover_Duration_RTO,
A.RECOVERY_POINT_OBJECTIVE AS Recover_RPO,
(SELECT Z.DISPLAYED_VALUE FROM MS_QS_LOV_ORG_VALUES_V Z WHERE (TRIM(Z.LOV_NAME)='MS_GRC_Duration_Values') AND Z.STORED_VALUE=A.DURATION_RPO_SV
AND Z.LOCALE_ID=MS_GRC_COMMON_UTILS.fn_get_default_locale_id)
 AS Recover_Duration_RPO,
A.WORK_RECOVERY_TIME AS Recover_WRT,
(SELECT Z.DISPLAYED_VALUE FROM MS_QS_LOV_ORG_VALUES_V Z WHERE (TRIM(Z.LOV_NAME)='MS_GRC_Duration_Values') AND Z.STORED_VALUE=A.DURATION_WRT_SV
AND Z.LOCALE_ID=MS_GRC_COMMON_UTILS.fn_get_default_locale_id)
 AS Recover_Duration_WRT,
A.MAXIMUM_TOLERABLE_DOWNTIME AS TD,
(SELECT Z.DISPLAYED_VALUE FROM MS_QS_LOV_ORG_VALUES_V Z WHERE (TRIM(Z.LOV_NAME)='MS_GRC_Duration_Values') AND Z.STORED_VALUE=A.DURATION_MDT_SV
AND Z.LOCALE_ID=MS_GRC_COMMON_UTILS.fn_get_default_locale_id) 
AS Duration_MTD,
A.RECOVERY_TIME_ACTUAL AS RTA,
(SELECT Z.DISPLAYED_VALUE FROM MS_QS_LOV_ORG_VALUES_V Z WHERE (TRIM(Z.LOV_NAME)='MS_GRC_Duration_Values') AND Z.STORED_VALUE=A.DURATION_RTA_SV
AND Z.LOCALE_ID=MS_GRC_COMMON_UTILS.fn_get_default_locale_id) 
Duration_RTA,
A.OWNER_ORG_IDS AS Owner_Organization_Ids,
A.OWNER_ORG_NAMES AS Owner_Organizations,
A.OWNER_IDS AS Owners_Ids,
A.OWNER_NAME AS Owners,
A.VALID_FROM AS VALID_FROM,
A.VALID_UNTIL AS VALID_TO,
A.CREATED_BY AS CREATED_BY,
A.CREATED_ON AS created_on,
A.UPDATED_BY AS modified_by,
A.UPDATED_ON AS modified_on,
A.ATTACHMENTS AS ATTACHMENTS
FROM MS_GRC_ASSET_SUMMARY A
