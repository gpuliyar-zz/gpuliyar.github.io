  CREATE OR REPLACE FORCE VIEW "MS_GRC_REFERENCE_CV" ("ATTACHMENTS", "MODIF_REQ_BY_HIDDEN", "CREATED_BY_HIDDEN", "OWN_APP_CONFIG_TEMP", "CHECK_USER_AVAIL", "OBJ_STATUS_TEMP", "OWN_APP_CONFIG", "RTF_ATTACH", "OBJECT_ID", "ACTION_COMMENTS", "OBJ_MODIFIED_ON", "OBJ_CREATED_ON", "VALID_UNTIL", "VALID_FROM", "REFERENCE_DOCUMENT", "REFERENCE_URL", "DESCRIPTION", "OBJECT_NAME", "DD_CURRENT_STATUS", "DD_ACTION", "DD_PROCESS_CODE", "DD_ENTERPRISE_INFO", "INSTANCE_REC_NUM", "INSTANCE_ID", "METRIC_ID", "METRIC_RUN_DATE", "METRIC_COMPLETED_DATE", "LATEST_FLAG", "CREATED_BY", "CREATION_DATE", "PROCESS_FLOW_STATUS", "PROCESS_INSTANCE_ID", "MULTIROW_GROUP_NAME", "ELECTRONICALLY_SIGNED", "MULTIROW_REGION_ID", "UNIQUE_ROW_ID", "DESCRIPTION_HTM", "OBJ_MODIFIED_BY", "OBJ_STATUS", "OBJ_CREATED_BY", "RESTRICT_ACCESS_TO", "REFERENCE_URL_TYPE", "REFERENCE_TYPE", "REFERENCE_CATEGORY", "OWNER_ORGANIZATIONS", "OWNERS", "OBJECT_ACTION", "LEVEL_2_APPROVER", "LEVEL_1_APPROVER", "DD_CURRENT_STAGE", "DD_EVENT_USER_NAME", "DD_CURRENT_USER_NAME", "DD_OBJECT_TYPE", "LOCALE_ID", "LOCALE_STRING", "CONFIG_PARAM_APP", "WFI_DISPLAY", "WFI_STORED") AS 
  SELECT "ATTACHMENTS",
    "MODIF_REQ_BY_HIDDEN",
    "CREATED_BY_HIDDEN",
    "OWN_APP_CONFIG_TEMP",
    "CHECK_USER_AVAIL",
    "OBJ_STATUS_TEMP",
    "OWN_APP_CONFIG",
    "RTF_ATTACH",
    "OBJECT_ID",
    replace(DBMS_XMLGEN.CONVERT(ACTION_COMMENTS,1),chr(10),'
') as "ACTION_COMMENTS",
    "OBJ_MODIFIED_ON",
    "OBJ_CREATED_ON",
    "VALID_UNTIL",
    "VALID_FROM",
    "REFERENCE_DOCUMENT",
    "REFERENCE_URL",
    (CASE WHEN dbms_lob.getlength(DESCRIPTION_HTM)=0 THEN NULL ELSE  DESCRIPTION_HTM END)  AS DESCRIPTION,
    "OBJECT_NAME",
    "DD_CURRENT_STATUS",
    "DD_ACTION",
    "DD_PROCESS_CODE",
    "DD_ENTERPRISE_INFO",
    "INSTANCE_REC_NUM",
    "INSTANCE_ID",
    "METRIC_ID",
    "METRIC_RUN_DATE",
    "METRIC_COMPLETED_DATE",
    "LATEST_FLAG",
    "CREATED_BY",
    "CREATION_DATE",
    "PROCESS_FLOW_STATUS",
    "PROCESS_INSTANCE_ID",
    "MULTIROW_GROUP_NAME",
    "ELECTRONICALLY_SIGNED",
    "MULTIROW_REGION_ID",
    "UNIQUE_ROW_ID",
    "DESCRIPTION_HTM",
    ms_apps_utilities.get_user_full_name(OBJ_MODIFIED_BY) AS "OBJ_MODIFIED_BY",
    ms_grc_utilities.get_status_name (dd_current_user_name, obj_status ) "OBJ_STATUS",
    ms_apps_utilities.get_user_full_name(OBJ_CREATED_BY) AS "OBJ_CREATED_BY",
    ms_apps_utilities.get_display_value (MS_GRC_UTILITIES.active_ent_id, 'MS GRC Restrict Access To', restrict_access_to, 1, locale_id) restrict_access_to,
    ms_apps_utilities.get_display_value (MS_GRC_UTILITIES.active_ent_id, 'MS GRC Reference URL Type', reference_url_type, 1, locale_id) reference_url_type,
    ms_apps_utilities.get_display_value (MS_GRC_UTILITIES.active_ent_id, 'MS GRC Reference Type', reference_type, 1, locale_id) reference_type,
    ms_apps_utilities.get_display_value (MS_GRC_UTILITIES.active_ent_id, 'MS GRC Reference Category', reference_category, 1, locale_id) REFERENCE_CATEGORY,
    MS_APPS_UTILITIES.GET_DISPLAY_VALUES (MS_GRC_UTILITIES.ACTIVE_ENT_ID, 'ORGENTITY',OWNER_ORGANIZATIONS ) OWNER_ORGANIZATIONS,
    MS_GRC_UTILITIES.get_users (owners ) owners,
    ms_apps_utilities.get_display_value (MS_GRC_UTILITIES.active_ent_id, 'MS GRC Action', object_action, 1, locale_id) object_action,
    ms_apps_utilities.get_user_full_name (LEVEL_2_APPROVER) level_2_approver,
    ms_apps_utilities.get_user_full_name (level_1_approver) LEVEL_1_APPROVER,
    ms_grc_utilities.get_wf_stage_desc(loc.locale_id, DD_CURRENT_STAGE) "DD_CURRENT_STAGE",
    MS_APPS_UTILITIES.GET_USER_FULL_NAME (DD_EVENT_USER_NAME) DD_EVENT_USER_NAME,
    MS_APPS_UTILITIES.GET_USER_FULL_NAME (dd_current_user_name) dd_current_user_name,
    (SELECT FORM_TITLE
    FROM ms_grc_orb_config_form_title conf
    WHERE conf.object_type = DD_OBJECT_TYPE
    AND loc.LOCALE_ID      = conf.LOCALE_ID
    ) AS "DD_OBJECT_TYPE",
    loc.LOCALE_ID,
    loc.LOCALE_STRING,
        CONFIG_PARAM_APP,
    WFI_DISPLAY,
    WFI_STORED
      FROM MS_GRC_REFERENCE_V vwOut,
    (SELECT LOCALE_ID, LOCALE_STRING FROM si_locales WHERE enabled_flag = 'Y'
    ) loc
  WHERE (process_instance_id,instance_id) in 
 (SELECT process_instance_id,MAX(INSTANCE_ID) as INSTANCE_ID
                                            FROM MS_GRC_REFERENCE_V a
                                            WHERE process_instance_id=vwOut.process_instance_id
                                            and a.instance_rec_num = 1
                                            AND a.DD_CURRENT_STAGE != 'NONE'
                                            GROUP BY a.process_instance_id,
                                                     a.dd_current_stage,
                                                     a.CREATION_DATE)
