MS_GRC_CONTROL_RV 

  CREATE OR REPLACE FORCE VIEW "MS_GRC_CONTROL_RV" ("OBJECT_NAME", "OBJECT_ID", "OWNERS", "VALID_UNTIL", "VALID_FROM", "DESCRIPTION", "COMPLIANCE_TYPE", "COMPLIANCE_CATEGORY", "KEY_CONTROL", "APPLIES_TO_ORGANIZATIONS", "EDIT", "OWNER_ORG_IDS", "RESTRICT_ACCESS_TO", "OBJECT_LEVEL", "OBJ_STATUS", "OBJECT_PARENT", "FREQUENCY", "RELATED_PROCESSES", "RELATED_RISKS", "RELATED_AREAS_OF_COMPLIANCE", "RELATED_REQUIREMENTS", "RELATED_FRAMEWORK_REFERENCES", "RELATED_ASSERTIONS", "RELATED_OBJECTIVES", "RELATED_REFERENCES", "RELATED_FUNCTIONS", "CORE_OBJECT_TYPE", "RELATED_ASSETS", "RELATED_CORE_OBJECTS", "RELATED_ASSET_CLASSES", "DD_EDIT_FLAG", "DD_OBJECT_TYPE", "DD_INSTANCE_ID", "DD_PROCESS_INSTANCE_ID", "DISPLAY_NAME", "OBJECT_TYPE", "OVER_RIDE_ACTION", "EDIT_ACTION_PARAMETERS", "EXPIRED_FLAG", "PURPOSE", "CONTROL_TYPE", "NATURE", "SOURCE_TYPE", "PRIORITY_TYPE", "SAMPLE_SIZE") AS 
  SELECT a.object_name,
    a.object_id,
    (SELECT ms_concat(usr.first_name
      || ' '
      || usr.last_name) AS owners
    FROM ms_grc_compliance_obje_own d,
      si_users_t usr,
      si_ent e
    WHERE usr.user_name   = d.owners
    AND usr.enterprise_id = e.ENTERPRISE_ID
    AND b.OBJECT_ID       = D.OBJECT_ID
    AND a.object_id       = b.object_id
    ) AS OWNERS,
    valid_until,
    valid_from,
    DESCRIPTION,
    null as COMPLIANCE_TYPE,
    null as COMPLIANCE_CATEGORY,
    key_compliance as Key_control,
    (SELECT ms_concat(orb.tgt_obj_id)
    FROM ms_grc_orb_relations_pub_v orb,
      SI_ORG_ENTITIES base
    WHERE src_obj_type   = 'MS_GRC_CONTROL'
    AND orb.tgt_obj_type = 'MS_GRC_ORGANIZATION'
    AND orb.src_obj_id   = b.object_id
    AND orb.tgt_obj_id   = base.org_entity_id
    )                                                                                                                                      AS APPLIES_TO_ORGANIZATIONS,
    MS_GRC_UTILITIES.GET_EDIT_URL (A.DD_OBJECT_TYPE, A.DD_PROCESS_INSTANCE_ID, a.DD_INSTANCE_ID, NVL (b.DD_EDIT_FLAG, 'Y'), a.OBJECT_NAME) AS EDIT,
    (SELECT ms_concat (owner_organizations)
    FROM ms_grc_compliance_obje_ogn E
    WHERE E.OBJECT_ID = A.OBJECT_ID
    ) AS OWNER_ORG_IDS,
    a.restrict_access_to,
    a.object_level,
    a.obj_status,
    object_parent AS OBJECT_PARENT,
    a.frequency,
    (SELECT ms_concat(orb.tgt_obj_id)
    FROM ms_grc_orb_relations_pub_v orb,
      ms_grc_core_object base
    WHERE src_obj_type   = 'MS_GRC_CONTROL'
    AND orb.tgt_obj_type = 'MS_GRC_PROCESS'
    AND orb.src_obj_id   = b.object_id
    AND orb.tgt_obj_id   = base.object_id
    ) AS RELATED_Processes,
    (SELECT ms_concat(orb.tgt_obj_id)
    FROM ms_grc_orb_relations_pub_v orb,
      ms_grc_risk base
    WHERE src_obj_type   = 'MS_GRC_CONTROL'
    AND orb.tgt_obj_type = 'MS_GRC_RISK'
    AND orb.src_obj_id   = b.object_id
    AND orb.tgt_obj_id   = base.object_id
    ) AS RELATED_RISKS,
    (SELECT ms_concat(orb.tgt_obj_id)
    FROM ms_grc_orb_relations_pub_v orb,
      ms_grc_area_of_complia base
    WHERE src_obj_type   = 'MS_GRC_CONTROL'
    AND orb.tgt_obj_type = 'MS_GRC_AREA_OF_COMPLIANCE'
    AND orb.src_obj_id   = b.object_id
    AND orb.tgt_obj_id   = base.object_id
    ) AS RELATED_AREAS_OF_COMPLIANCE,
    (SELECT ms_concat(orb.tgt_obj_id)
    FROM ms_grc_orb_relations_pub_v orb,
      ms_grc_requirement base
    WHERE src_obj_type   = 'MS_GRC_CONTROL'
    AND orb.tgt_obj_type = 'MS_GRC_REQUIREMENT'
    AND orb.src_obj_id   = b.object_id
    AND orb.tgt_obj_id   = base.object_id
    ) AS RELATED_REQUIREMENTS,
    (SELECT ms_concat(orb.tgt_obj_id)
    FROM ms_grc_orb_relations_pub_v orb,
      ms_grc_model_reference base
    WHERE src_obj_type   = 'MS_GRC_CONTROL'
    AND orb.tgt_obj_type = 'MS_GRC_MODEL_REFERENCE'
    AND orb.src_obj_id   = b.object_id
    AND orb.tgt_obj_id   = base.object_id
    ) AS Related_Framework_References,
    (SELECT MS_GRC_UTILITIES.Get_Relationships (A.OBJECT_ID, a.dd_object_type, 'ASSERTION')
    FROM DUAL
    ) AS RELATED_ASSERTIONS,
    (SELECT ms_concat(orb.tgt_obj_id)
    FROM ms_grc_orb_relations_pub_v orb,
      ms_grc_objectives base
    WHERE src_obj_type   = 'MS_GRC_CONTROL'
    AND orb.tgt_obj_type = 'MS_GRC_OBJECTIVES'
    AND orb.src_obj_id   = b.object_id
    AND orb.tgt_obj_id   = base.object_id
    ) AS RELATED_OBJECTIVES,
    (SELECT ms_concat(orb.tgt_obj_id)
    FROM ms_grc_orb_relations_pub_v orb,
      ms_grc_reference base
    WHERE src_obj_type   = 'MS_GRC_CONTROL'
    AND orb.tgt_obj_type = 'MS_GRC_REFERENCE'
    AND orb.src_obj_id   = b.object_id
    AND orb.tgt_obj_id   = base.object_id
    ) AS RELATED_REFERENCES,
    (SELECT ms_concat(orb.tgt_obj_id)
    FROM ms_grc_orb_relations_pub_v orb,
      ms_grc_function base
    WHERE src_obj_type   = 'MS_GRC_CONTROL'
    AND orb.tgt_obj_type = 'MS_GRC_FUNCTION'
    AND orb.src_obj_id   = b.object_id
    AND orb.tgt_obj_id   = base.object_id
    ) AS RELATED_FUNCTIONS,
    (SELECT MS_GRC_UTILITIES.Get_Relationships (A.OBJECT_ID, a.dd_object_type, 'MS_GRC_PROCESS','OBJECT_TYPE')
    FROM DUAL
    ) AS CORE_OBJECT_TYPE,
    (SELECT ms_concat(orb.tgt_obj_id)
    FROM ms_grc_orb_relations_pub_v orb,
      ms_grc_core_object base
    WHERE src_obj_type   = 'MS_GRC_CONTROL'
    AND orb.tgt_obj_type = 'MS_GRC_ASSET'
    AND orb.src_obj_id   = b.object_id
    AND orb.tgt_obj_id   = base.object_id
    ) AS RELATED_ASSETS,
    (SELECT ms_concat(orb.tgt_obj_id)
    FROM ms_grc_orb_relations_pub_v orb,
      ms_grc_core_object base
    WHERE src_obj_type    = 'MS_GRC_CONTROL'
    AND orb.tgt_obj_type IN('MS_GRC_ASSET','MS_GRC_ASSET_CLASS','MS_GRC_PROCESS')
    AND orb.src_obj_id    = b.object_id
    AND orb.tgt_obj_id    = base.object_id
    ) AS Related_core_objects,
    (SELECT ms_concat(orb.tgt_obj_id)
    FROM ms_grc_orb_relations_pub_v orb,
      ms_grc_core_object base
    WHERE src_obj_type   = 'MS_GRC_CONTROL'
    AND orb.tgt_obj_type = 'MS_GRC_ASSET_CLASS'
    AND orb.src_obj_id   = b.object_id
    AND orb.tgt_obj_id   = base.object_id
    )                         AS RELATED_ASSET_CLASSES,
    NVL (b.DD_EDIT_FLAG, 'Y') AS DD_EDIT_FLAG,
    a.dd_object_type,
    a.dd_instance_id,
    a.DD_PROCESS_INSTANCE_ID,
    a.object_name AS display_name,
    a.dd_object_type          AS Object_Type,
    'Y'                       AS OVER_RIDE_ACTION,
    'proc='
    || TO_CHAR(a.dd_process_instance_id)
    ||'|instid='
    ||TO_CHAR(a.dd_instance_id)
    ||'|edit_flag='
    ||NVL(a.dd_edit_flag,'Y')                                  AS EDIT_ACTION_PARAMETERS,
    ms_Grc_utilities.INCLUDE_items(a.VALID_FROM,a.VALID_UNTIL) AS EXPIRED_FLAG,
     (select MS_CONCAT(Purpose)
    from MS_GRC_CONTROL_PUR P
    where P.OBJECT_ID = A.OBJECT_ID
    ) as Purpose, -- compliance category is changed to purpose.
    (select MS_CONCAT(CONTROL_TYPE)
        FROM MS_GRC_COMPLIANCE_OBJE_cty d
        where b.OBJECT_ID       = D.OBJECT_ID
        and a.OBJECT_ID       = B.OBJECT_ID
        ) as CONTROL_TYPE,
   (select MS_CONCAT(NATURE_TYPE)
  from MS_GRC_CONTROL_NRE N
    where N.OBJECT_ID = a.OBJECT_ID
    ) as Nature,
    --Compliance type is changed to Nature.
       a.SOURCE_TYPE,
       a.PRIORITY_TYPE,
       a.SAMPLE_SIZE
  FROM MS_GRC_COMPLIANCE_OBJE A,
    MS_GRC_CONTROL B
  WHERE a.DD_OBJECT_TYPE = 'MS_GRC_CONTROL'
  and a.OBJECT_ID        = B.OBJECT_ID
