MS_GRC_FINANCIAL_ACCOUNTS_RV 

  CREATE OR REPLACE FORCE VIEW "MS_GRC_FINANCIAL_ACCOUNTS_RV" ("OBJECT_NAME", "OBJECT_ID", "STATUS", "DESCRIPTION", "OWNERS", "ACCOUNT_BALANCE", "ACCOUNT_BALANCE_AS_OF", "SOX_MATERIAL", "VALID_UNTIL", "VALID_FROM", "DD_PROCESS_INSTANCE_ID", "DD_INSTANCE_ID", "DD_EDIT_FLAG", "RESTRICT_ACCESS_TO", "RELATED_PROCESSES", "RELATED_RISKS", "RELATED_REFERENCES", "RELATED_ORG_PROFILE", "ACCOUNT_TYPE", "ACCOUNT_CATEGORY", "EXPIRED_FLAG", "OWNER_ORG_IDS", "EDIT", "DISPLAY_NAME", "DD_OBJECT_TYPE", "OVER_RIDE_ACTION", "EDIT_ACTION_PARAMETERS") AS 
  SELECT object_name,
    object_id,
    Obj_Status AS Status,
    DESCRIPTION ,
    (SELECT ms_concat(usr.first_name
      || ' '
      || usr.last_name) AS owners
    FROM ms_grc_financial_accou_own d,
      si_users_t usr,
      si_ent e
    WHERE usr.user_name   = d.owners
    AND usr.enterprise_id = e.ENTERPRISE_ID
    AND a.OBJECT_ID       = D.OBJECT_ID
    ) AS OWNERS,
    account_balance,
    acc_balance_as_of AS account_balance_as_of,
    INITCAP (sox_material) AS sox_material,
    valid_until,
    valid_from,
    dd_process_instance_id,
    dd_instance_id,
    NVL (dd_edit_flag, 'Y') AS DD_EDIT_FLAG,
    --owner_organizations AS owner_org_ids,
    restrict_access_to,
    (SELECT MS_GRC_FN_CONCAT(orb.tgt_obj_id)
    FROM ms_grc_orb_relations_pub_v orb,
      ms_grc_core_object base
    WHERE src_obj_type   = 'MS_GRC_FINANCIAL_ACCOUNTS'
    AND orb.tgt_obj_type = 'MS_GRC_PROCESS'
    AND orb.src_obj_id   = a.object_id
    AND orb.tgt_obj_id   = base.object_id
    ) AS related_Processes,
    (SELECT MS_GRC_FN_CONCAT(orb.tgt_obj_id)
    FROM ms_grc_orb_relations_pub_v orb,
      ms_grc_risk base
    WHERE src_obj_type   = 'MS_GRC_FINANCIAL_ACCOUNTS'
    AND orb.tgt_obj_type = 'MS_GRC_RISK'
    AND orb.src_obj_id   = a.object_id
    AND orb.tgt_obj_id   = base.object_id
    ) AS related_risks,
    (SELECT MS_GRC_FN_CONCAT(orb.tgt_obj_id)
    FROM ms_grc_orb_relations_pub_v orb,
      ms_grc_reference base
    WHERE src_obj_type   = 'MS_GRC_FINANCIAL_ACCOUNTS'
    AND orb.tgt_obj_type = 'MS_GRC_REFERENCE'
    AND orb.src_obj_id   = a.object_id
    AND orb.tgt_obj_id   = base.object_id
    ) AS related_references,
    (SELECT MS_GRC_FN_CONCAT(orb.tgt_obj_id)
    FROM ms_grc_orb_relations_pub_v orb,
      ms_grc_org_profile base
    WHERE src_obj_type   = 'MS_GRC_FINANCIAL_ACCOUNTS'
    AND orb.tgt_obj_type = 'MS_GRC_ORG_PROFILE'
    AND orb.src_obj_id   = a.object_id
    AND orb.tgt_obj_id   = base.object_id
    )                                                            AS related_org_profile,
    obj_type                                                     AS Account_type,
    obj_category                                                 AS Account_category,
    ms_Grc_utilities.INCLUDE_eXPIRED(a.VALID_FROM,a.VALID_UNTIL) AS EXPIRED_FLAG,
    (SELECT MS_CONCAT (OWNER_ORGANIZATIONS)
    FROM MS_GRC_FINANCIAL_ACCOU_ORG E
    WHERE E.OBJECT_ID = A.OBJECT_ID
    ) AS owner_org_ids,
    (SELECT ms_Grc_utilities.Get_Edit_Url(a.DD_OBJECT_TYPE,a.DD_PROCESS_INSTANCE_ID, a.DD_INSTANCE_ID,NVL (a.DD_EDIT_FLAG, 'Y'),a.object_name)
    FROM dual
    )                AS EDIT,
    a.object_name    AS Display_name,
    a.dd_object_type AS dd_object_type,
    'Y'              AS OVER_RIDE_ACTION,
    'proc='
    || TO_CHAR(a.dd_process_instance_id)
    ||'|instid='
    ||TO_CHAR(a.dd_instance_id)
    ||'|edit_flag='
    ||NVL(a.dd_edit_flag,'Y') AS EDIT_ACTION_PARAMETERS
  FROM MS_GRC_FINANCIAL_ACCOU a
