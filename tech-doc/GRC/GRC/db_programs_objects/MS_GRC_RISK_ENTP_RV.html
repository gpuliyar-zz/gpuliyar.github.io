MS_GRC_RISK_ENTP_RV 

  CREATE OR REPLACE FORCE VIEW "MS_GRC_RISK_ENTP_RV" ("OBJECT_NAME", "OBJECT_ID", "LOCALE_ID", "LOCALE", "OWNER_ORG_IDS", "OWNER_ORG_NAMES", "OWNER_IDS", "OWNER_NAMES", "KEY_RISK", "LEVEL_1_APPROVER", "LEVEL_2_APPROVER", "VALID_UNTIL", "VALID_FROM", "DESCRIPTION", "RISK_CATEGORIES", "TYPE_RISK", "DD_PROCESS_INSTANCE_ID", "DD_INSTANCE_ID", "DD_EDIT_FLAG", "RESTRICT_ACCESS_TO", "OVER_RIDE_ACTION", "OBJECT_TYPE", "EDIT_ACTION_PARAMETERS", "OBJECT_LEVEL", "DISPLAY_NAME", "OBJECT_NAME_URL", "REL_DATA_LINK", "OBJECT_STATUS", "OBJECT_PARENT", "DD_OBJECT_TYPE", "CREATED_DATE", "CREATED_BY", "UPDATED_DATE", "UPDATED_BY", "IS_ORPHAN", "UPPER_OBJECT_NAME", "OBJ_PARENT_ID") AS 
  SELECT rsk_summ.OBJECT_NAME AS OBJECT_NAME,
    rsk_summ.object_id as object_id,
    mls_Summ.locale_id         AS locale_id,
    mls_summ.locale            AS locale,
    rsk_summ.OWNER_ORG_IDS     AS OWNER_ORG_IDS,
    rsk_summ.OWNER_ORG_NAMES   AS OWNER_ORG_NAMES,
    rsk_summ.OWNER_IDS         AS OWNER_IDS,
    rsk_summ.OWNER_NAME        AS Owner_names,
    INITCAP(mls_Summ.key_risk) AS key_risk,
    rsk_summ.level_1_approver  AS level_1_approver,
    rsk_summ.level_2_approver  AS level_2_approver,
    rsk_summ.valid_until       AS valid_until,
    rsk_summ.valid_from        AS valid_from,
    rsk_summ.description       AS description,
    mls_Summ.CAT_NAME_AGG      AS RISK_CATEGORIES,
    mls_Summ.TYPE_RISK         AS TYPE_RISK,
    rsk_summ.dd_process_instance_id,
    rsk_summ.dd_instance_id,
    NVL (rsk_summ.dd_edit_flag, 'Y') AS dd_edit_flag,
    rsk_summ.restrict_access_to,
    rsk_summ.OVER_RIDE_ACTION       AS OVER_RIDE_ACTION,
    Rsk_Summ.Dd_Object_Type         AS Object_Type,
    Rsk_Summ.Edit_Action_Parameters AS EDIT_ACTION_PARAMETERS,
    mls_Summ.object_level,
    (SELECT SUBSTR (rsk.object_name,1, 30)
    FROM MS_GRC_RISK rsk
    WHERE rsk.object_id = rsk_summ.object_id
    )                                     AS Display_Name,
    Rsk_Summ.Object_Name_Url              AS object_name_url,
    Lovvalue.displayed_value 			  AS REL_DATA_LINK,
    mls_Summ.obj_status               AS OBJECT_STATUS,
    rsk_summ.OBJECT_PARENT,
    DD_OBJECT_TYPE,
    rsk_summ.CREATED_DATE,
    (SELECT usr.first_name
      || ' '
      || usr.last_name
    FROM si_users_t usr
    WHERE rsk_summ.CREATED_BY = usr.user_name
    ) AS CREATED_BY,
    rsk_summ.UPDATED_DATE,
    (SELECT usr.first_name
      || ' '
      || usr.last_name
    FROM si_users_t usr
    WHERE rsk_summ.UPDATED_BY = usr.user_name
    ) AS UPDATED_BY,
    RSK_SUMM.IS_ORPHAN,
    RSK_SUMM.UPPER_OBJECT_NAME,
    RSK_SUMM.OBJ_PARENT_ID
  FROM MS_GRC_RISK_OBJ_SUMMARY rsk_summ,
    MS_GRC_RISK_OBJ_MLS_SUMMARY mls_Summ,
    Ms_Qs_List_Of_Values Lovvalue,
    Ms_Qs_Lov_Names Lovname
  WHERE Mls_Summ.Object_Id = Rsk_Summ.Object_Id
   AND Lovvalue.Org_Entity_Id           = 1
AND Lovname.Lov_Name                 = 'MS_GRC_LABLES'
and Lovvalue.stored_value = 'CLICK_HERE'
AND Lovvalue.Lov_Id                  = Lovname.Lov_Id
AND Lovvalue.Locale_id = mls_Summ.locale_id
