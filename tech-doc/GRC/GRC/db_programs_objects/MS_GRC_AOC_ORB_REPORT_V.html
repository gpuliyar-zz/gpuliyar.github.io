MS_GRC_AOC_ORB_REPORT_V 

  CREATE OR REPLACE FORCE VIEW "MS_GRC_AOC_ORB_REPORT_V" ("OBJECT_NAME", "OBJECT_ID", "OWNER_ORGANIZATIONS", "OWNERS", "VALID_UNTIL", "VALID_FROM", "DD_OBJECT_TYPE", "ATO_ORGANIZATIONS", "OWNER_ORG_IDS", "RESTRICT_ACCESS_TO", "AOC_TYPE") AS 
  SELECT a.object_name,
    a.object_id,
    (SELECT ms_concat(ms_apps_utilities.get_display_value ( (ms_grc_utilities.active_ent_id), 'ORGENTITY', owner_organizations ))
    FROM ms_grc_Area_of_complia_org d
    WHERE a.object_id = d.object_id
    ) AS owner_organizations,
    (SELECT ms_concat ( ms_apps_utilities.get_user_full_name (owners) )
    FROM ms_grc_Area_of_COmplia_own e
    WHERE a.object_id = e.object_id
    ) AS OWNERS,
    valid_until,
    valid_from,
    dd_object_type,
    ms_grc_utilities.get_relationships (object_id, dd_object_type, 'MS_GRC_ORGANIZATION') AS ato_organizations,
    (SELECT MS_CONCAT (OWNER_ORGANIZATIONS)
    FROM MS_GRC_Area_of_complia_ORG E
    WHERE E.OBJECT_ID = A.OBJECT_ID
    ) AS owner_org_ids,
    restrict_access_to,
     (SELECT ms_concat(ms_apps_utilities.get_display_value ( (ms_grc_utilities.active_ent_id), 'MS_GRC_AOC_TYPES', aoc_type ))
    FROM ms_grc_Area_of_complia_typ d
    WHERE a.object_id = d.object_id
    ) AS aoc_type
  FROM ms_grc_AREA_OF_COMPLIA a
