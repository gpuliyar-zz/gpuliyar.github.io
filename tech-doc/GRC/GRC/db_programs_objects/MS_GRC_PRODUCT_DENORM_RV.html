MS_GRC_PRODUCT_DENORM_RV 

  CREATE OR REPLACE FORCE VIEW "MS_GRC_PRODUCT_DENORM_RV" ("OBJECT_ID", "OBJECT_NAME", "EDIT", "OBJECT_LEVEL", "OBJECT_PARENT", "OBJ_LOCATION", "PRODUCT_CATEGORY_AGG", "PRODUCT_TYPE_AGG", "SALES_CHANNEL_AGG", "OBJ_STATUS", "CREATED_DATE", "CREATED_BY", "UPDATED_DATE", "UPDATED_BY", "VALID_FROM", "VALID_UNTIL", "DESCRIPTION", "OBJ_OWN_ORG_ID_AGG", "OBJ_OWN_ORG_NAME_AGG", "OBJ_OWNER_ID_AGG", "OBJ_OWNER_NAME_AGG", "LEVEL_1_APPROVER", "LEVEL_2_APPROVER", "REL_ORGANIZATION_NAMES", "REL_AOC_OBJ_NAME_AGG", "REL_ASSET_OBJ_NAME_AGG", "REL_CTRL_OBJ_NAME_AGG", "REL_OBJECTIVE_OBJ_NAME_AGG", "REL_PRODUCT_NAME_AGG", "REL_PROCESS_OBJ_NAME_AGG", "REL_REFRENCES_OBJ_NAME_AGG", "REL_RISK_OBJ_NAME_AGG", "LOCALE_ID", "LOCALE", "RESTRICT_ACCESS_TO", "DD_EDIT_FLAG", "DD_OBJECT_TYPE", "DD_INSTANCE_ID", "DD_PROCESS_INSTANCE_ID", "OVER_RIDE_ACTION", "DISPLAY_NAME", "EDIT_ACTION_PARAMETERS", "EXPIRED_FLAG", "ATTACHMENTS", "IS_ORPHAN", "OBJECT_PARENT_ID", "LEVEL_1_APPR", "LEVEL_2_APPR", "UPPER_OBJECT_NAME") AS 
  SELECT prdSumm.object_id,
    Prdsumm.Object_Name,
    Prdsumm.Object_Name_Url AS EDIT,
    mlsSumm.object_level,
    prdSumm.OBJECT_PARENT_NAME as object_parent,
	mlsSumm.LOCATIONS_AGG as OBJ_LOCATION,
    mlsSumm.Product_Category_Agg,
    mlsSumm.Product_Type_Agg,
    mlsSumm.Sales_Channel_Agg,
    mlsSumm.Obj_status,
    prdSumm.created_date,
     (SELECT usr.first_name
			   || ' '
			   || usr.last_name
		  FROM si_users_t usr
		 WHERE prdSumm.CREATED_BY = usr.user_name
	   ) AS CREATED_BY,
    prdSumm.updated_date,
    (SELECT usr.first_name
			   || ' '
			   || usr.last_name
		  FROM si_users_t usr
		 WHERE prdSumm.Updated_By = usr.user_name
	   ) as updated_by,
    prdSumm.valid_from,
    prdSumm.valid_until,
    PRDSUMM.DESCRIPTION,
    PRDSUMM.OWNER_ORG_IDS AS OBJ_OWN_ORG_ID_AGG,
    PRDSUMM.OWNER_ORG_NAMES AS OBJ_OWN_ORG_NAME_AGG,
    PRDSUMM.OWNER_IDS AS OBJ_OWNER_ID_AGG,
    prdSumm.OWNER_NAME as obj_owner_name_agg,
    (SELECT usr.first_name
			   || ' '
			   || usr.last_name
		  FROM si_users_t usr
		 Where  prdSumm.level_1_approver = Usr.User_Name
	   ) AS LEVEL_1_APPROVER,
    (SELECT usr.first_name
			   || ' '
			   || usr.last_name
		  FROM si_users_t usr
		 Where Prdsumm.Level_2_Approver = Usr.User_Name
	   ) AS LEVEL_2_APPROVER,
    prdSumm.REL_ORGANIZATION_NAMES,
    prdSumm.REL_AOC_OBJ_NAME_AGG,
    prdSumm.REL_ASSET_OBJ_NAME_AGG,
    prdSumm.REL_CTRL_OBJ_NAME_AGG,
    prdSumm.REL_OBJECTIVE_OBJ_NAME_AGG,
    prdSumm.REL_PRODUCT_NAME_AGG,
    prdSumm.REL_PROCESS_OBJ_NAME_AGG,
    prdSumm.REL_REFRENCES_OBJ_NAME_AGG,
    prdSumm.REL_RISK_OBJ_NAME_AGG,
    mlsSumm.LOCALE_ID,
    mlsSumm.LOCALE,
    prdSumm.restrict_access_to,
    prdSumm.dd_edit_flag,
    prdSumm.dd_object_type,
    prdSumm.dd_instance_id,
    prdSumm.dd_process_instance_id,
    prdSumm.over_ride_action,
    Substr(Prdsumm.Object_Name,1,30) As Display_Name,
    Prdsumm.Edit_Action_Parameters   AS EDIT_ACTION_PARAMETERS,
    MS_GRC_UTILITIES.include_items(prdSumm.valid_from, prdSumm.valid_until) AS EXPIRED_FLAG,
    PRDSUMM.ATTACHMENTS,
	PRDSUMM.IS_ORPHAN,
  prdSumm.OBJECT_PARENT_ID,
  LEVEL_1_APPROVER as LEVEL_1_APPR,
  LEVEL_2_APPROVER as LEVEL_2_APPR,
  prdSumm.upper_object_name
  FROM MS_GRC_PRODUCT_SUMMARY prdSumm,
    MS_GRC_PRODUCT_MLS_SUMMARY mlsSumm    
  Where  Prdsumm.Object_Id = Mlssumm.Object_Id
