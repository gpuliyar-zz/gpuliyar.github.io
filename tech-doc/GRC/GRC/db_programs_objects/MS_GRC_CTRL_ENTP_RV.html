MS_GRC_CTRL_ENTP_RV 

  CREATE OR REPLACE FORCE VIEW "MS_GRC_CTRL_ENTP_RV" ("OBJECT_ID", "OBJECT_NAME", "OBJECT_TYPE", "UPPER_OBJECT_NAME", "OBJECT_NAME_URL", "LOCALE_ID", "LOCALE", "KEY_CONTROL", "CONTROL_PURPOSE", "CONTROL_TYPE", "CONTROL_NATURE", "CONTROL_SOURCE", "CONTROL_PRIORITY", "OBJECT_LEVEL", "OBJECT_PARENT", "OBJECT_STATUS", "CONTROL_FREQUENCY", "CREATED_DATE", "CREATED_BY", "UPDATED_DATE", "UPDATED_BY", "VALID_FROM", "VALID_UNTIL", "DESCRIPTION", "OWNER_ORG_IDS", "OWNER_ORG_NAMES", "OWNER_IDS", "OWNER_NAMES", "LEVEL_1_APPROVER", "LEVEL_2_APPROVER", "RESTRICT_ACCESS_TO", "DD_EDIT_FLAG", "DD_OBJECT_TYPE", "DD_INSTANCE_ID", "DD_PROCESS_INSTANCE_ID", "OVER_RIDE_ACTION", "EDIT_ACTION_PARAMETERS", "EXPIRED_FLAG", "SAMPLE_SIZE", "IS_ORPHAN", "REL_DATA_LINK", "OBJECT_PARENT_ID") AS 
  SELECT contSumm.object_id
        ,contSumm.object_Name
		  ,'MS_GRC_CONTROL' as OBJECT_TYPE
        ,contSumm.upper_object_Name
        ,contSumm.OBJECT_NAME_URL
        ,mlsSumm.LOCALE_ID
        ,mlsSumm.LOCALE
        ,mlsSumm.Key_Control
        ,mlsSumm.control_purpose
        ,mlsSumm.control_type
        ,mlsSumm.control_nature
        ,mlsSumm.control_source
        ,mlsSumm.control_priority
        ,mlsSumm.object_level
       ,( SELECT object_name
            FROM MS_GRC_COMPLIANCE_OBJE comp
           WHERE comp.object_id    = contSumm.parent
             AND comp.dd_object_type = 'MS_GRC_CONTROL'
         ) as object_parent
       ,mlsSumm.Object_Status
       ,mlsSumm.control_frequency
       ,contSumm.Created_Date
       ,( SELECT usr.first_name
                 || ' '
                 || usr.last_name
            FROM si_users_t usr
           WHERE contSumm.Created_By = usr.user_name
         ) AS created_by
       ,contSumm.Updated_Date
       ,( SELECT usr.first_name
                 || ' '
                 || usr.last_name
            FROM si_users_t usr
           WHERE contSumm.Updated_By = usr.user_name
         ) AS Updated_By
       ,contSumm.valid_from
       ,contSumm.valid_until
       ,contSumm.description
       ,contSumm.OWNER_ORG_IDS   AS OWNER_ORG_IDS
       ,contSumm.OWNER_ORG_NAMES AS OWNER_ORG_NAMES
       ,contSumm.OWNER_IDS       AS OWNER_IDS
       ,contSumm.OWNER_NAME      AS OWNER_NAMES
       ,contSumm.level_1_approver
       ,contSumm.level_2_approver
       ,contSumm.restrict_access_to
       ,contSumm.dd_edit_flag
       ,contSumm.dd_object_type
       ,contSumm.dd_instance_id
       ,contSumm.dd_process_instance_id
       ,Contsumm.Over_Ride_Action
       ,Contsumm.Edit_Action_Parameters AS EDIT_ACTION_PARAMETERS
       ,contSumm.expired_flag
       ,contSumm.sample_size
       ,contSumm.is_orphan
	   ,lovvalue.displayed_value as rel_data_link
     ,contSumm.parent as object_parent_id   
   FROM MS_GRC_CONTROL_SUMMARY contSumm
       ,MS_GRC_CONTROL_MLS_SUMMARY mlsSumm
	   ,Ms_Qs_List_Of_Values Lovvalue
       ,Ms_Qs_Lov_Names Lovname
  WHERE CONTSUMM.OBJECT_ID = MLSSUMM.OBJECT_ID
    AND Lovvalue.Org_Entity_Id   = 1
    AND Lovname.Lov_Name         = 'MS_GRC_LABLES'
    and Lovvalue.stored_value 	= 'CLICK_HERE'
    AND Lovvalue.Lov_Id          = Lovname.Lov_Id
    AND Lovvalue.Locale_id 		= mlsSumm.locale_id
