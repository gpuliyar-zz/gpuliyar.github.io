MS_GRC_RISK_PUB_V 

  CREATE OR REPLACE FORCE VIEW "MS_GRC_RISK_PUB_V" ("OBJECT_ID", "OBJECT_NAME", "DESCRIPTION", "DESCRIPTION_HTM", "OWNERS", "OWNER_ORGANIZATIONS", "RESTRICT_ACCESS_TO", "OBJ_STATUS", "KEY_RISK", "LEVEL_1_APPROVER", "LEVEL_2_APPROVER", "VALID_FROM", "VALID_UNTIL", "OBJECT_LEVEL", "OBJECT_PARENT", "CATEGORY_RISK", "TYPE_RISK", "RTF_ATTACH", "DD_OBJECT_TYPE", "DD_PROCESS_INSTANCE_ID", "DD_INSTANCE_ID", "INITIATED_BY", "CREATED_ON", "MODIFIED_ON", "CREATED_BY", "UPDATED_BY", "IS_ORPHAN", "OBJECT_NAME_URL") AS 
  SELECT /*@public*/ OBJECT_ID,
    OBJECT_NAME,
    regexp_replace(DESCRIPTION, 'null;[^null;]+null;', NULL) AS DESCRIPTION,
    DESCRIPTION                                  AS DESCRIPTION_HTM,
    (SELECT MS_CONCAT(OWNERS)
    FROM MS_GRC_RISK_OWN B
    WHERE B.OBJECT_ID = A.OBJECT_ID
    ) AS OWNERS,
    (SELECT MS_CONCAT(OWNER_ORGANIZATIONS)
    FROM MS_GRC_RISK_OGN B
    WHERE B.OBJECT_ID = A.OBJECT_ID
    ) AS OWNER_ORGANIZATIONS,
    RESTRICT_ACCESS_TO,
    obj_STATUS,
    KEY_RISK ,
    LEVEL_1_APPROVER,
    LEVEL_2_APPROVER,
    VALID_FROM,
    VALID_UNTIL,
    OBJECT_LEVEL,
    (SELECT MS_CONCAT(PRI_OBJECT_PARENT)
    FROM MS_GRC_RISK_PRI B
    WHERE B.OBJECT_ID = A.OBJECT_ID
    ) AS OBJECT_PARENT,
    (SELECT MS_CONCAT(CATEGORY_RISK)
    FROM MS_GRC_RISK_CAT B
    WHERE B.OBJECT_ID = A.OBJECT_ID
    ) AS CATEGORY_RISK,
    TYPE_RISK,
    RTF_ATTACH,
    DD_OBJECT_TYPE,
    DD_PROCESS_INSTANCE_ID,
    DD_INSTANCE_ID,
    NVL(OBJ_MODIFIED_BY,OBJ_CREATED_BY) AS INITIATED_BY,
    DD_CREATED_ON AS CREATED_ON,
    DD_MODIFIED_ON AS MODIFIED_ON,
    OBJ_CREATED_BY AS CREATED_BY ,
    OBJ_MODIFIED_BY AS UPDATED_BY,
    (SELECT CASE WHEN count(1) > 0
                      THEN 
                        1
                      ELSE
                        2
                      END
            FROM MS_GRC_ORB_RELATIONS_PUB_V
           WHERE SRC_OBJ_TYPE = 'MS_GRC_RISK'
             AND SRC_OBJ_ID = A.object_id
             and SRC_OBJ_TYPE <>'MS_GRC_ORGANIZATION'
             and TGT_OBJ_TYPE <> 'MS_GRC_ORGANIZATION'
             And Rownum =1) As IS_ORPHAN,

                                        A.dd_process_instance_id,   
                                        A.dd_instance_id,   
                                        NVL(A.DD_EDIT_FLAG,'Y'),    
                                        A.OBJECT_NAME)                          AS OBJECT_NAME_URL
  FROM ms_grc_risk a
