MS_GRC_OBJECTIVES_RV 

  CREATE OR REPLACE FORCE VIEW "MS_GRC_OBJECTIVES_RV" ("OBJECT_NAME", "OBJECT_ID", "STATUS", "OWNERS", "VALID_UNTIL", "VALID_FROM", "DD_OBJECT_TYPE", "OBJECTIVE_TYPE", "DD_PROCESS_INSTANCE_ID", "DD_INSTANCE_ID", "DD_EDIT_FLAG", "RESTRICT_ACCESS_TO", "RELATED_PROCESSES", "RELATED_RISKS", "RELATED_CONTROLS", "RELATED_ORG_PROFILES", "RELATED_REFERENCES", "OWNER_ORG_IDS", "EXPIRED_FLAG", "DISPLAY_NAME", "OVER_RIDE_ACTION", "EDIT_ACTION_PARAMETERS", "DESCRIPTION", "EDIT", "OBJECTIVE_TYPE_NEW", "PARENT_OBJECTIVE_ID", "COSO_TYPE", "OBJECT_LEVEL") AS 
  SELECT A.OBJECT_NAME,
    a.object_id,
    obj_status AS status,
    (SELECT ms_concat(usr.first_name
      || ' '
      || usr.last_name) AS owners
    FROM ms_grc_objectives_own d,
      si_users_t usr,
      si_ent e
    WHERE usr.user_name  = d.owners
    AND a.object_id      = D.OBJECT_ID
    AND usr.enterprise_id=e.enterprise_id
    ) AS OWNERS,
    valid_until,
    VALID_FROM,
    DD_OBJECT_TYPE,
    TYPE_OBJ AS objective_type,
    dd_process_instance_id,
    dd_instance_id,
    NVL (dd_edit_flag, 'Y') AS DD_EDIT_FLAG,
    RESTRICT_ACCESS_TO,
    (SELECT ms_concat(orb.tgt_obj_id)
    FROM ms_grc_orb_relations_pub_v orb,
      ms_grc_core_object base
    WHERE src_obj_type   = 'MS_GRC_OBJECTIVES'
    AND orb.tgt_obj_type = 'MS_GRC_PROCESS'
    AND orb.src_obj_id   = a.object_id
    AND orb.tgt_obj_id   = base.object_id
    ) AS Related_Processes,
    (SELECT ms_concat(orb.tgt_obj_id)
    FROM ms_grc_orb_relations_pub_v orb,
      ms_grc_risk base
    WHERE src_obj_type   = 'MS_GRC_OBJECTIVES'
    AND orb.tgt_obj_type = 'MS_GRC_RISK'
    AND orb.src_obj_id   = a.object_id
    AND orb.tgt_obj_id   = base.object_id
    ) AS Related_Risks,
    (SELECT ms_concat(orb.tgt_obj_id)
    FROM ms_grc_orb_relations_pub_v orb,
      ms_grc_compliance_obje base
    WHERE src_obj_type   = 'MS_GRC_OBJECTIVES'
    AND orb.tgt_obj_type = 'MS_GRC_CONTROL'
    AND orb.src_obj_id   = a.object_id
    AND orb.tgt_obj_id   = base.object_id
    ) AS Related_Controls,
    (SELECT ms_concat(orb.tgt_obj_id)
    FROM ms_grc_orb_relations_pub_v orb,
      ms_grc_org_profile base
    WHERE src_obj_type   = 'MS_GRC_OBJECTIVES'
    AND orb.tgt_obj_type = 'MS_GRC_ORG_PROFILE'
    AND orb.src_obj_id   = a.object_id
    AND orb.tgt_obj_id   = base.object_id
    ) AS Related_Org_Profiles,
    (SELECT ms_concat(orb.tgt_obj_id)
    FROM ms_grc_orb_relations_pub_v orb,
      ms_grc_reference base
    WHERE src_obj_type   = 'MS_GRC_OBJECTIVES'
    AND orb.tgt_obj_type = 'MS_GRC_REFERENCE'
    AND orb.src_obj_id   = a.object_id
    AND orb.tgt_obj_id   = base.object_id
    ) AS Related_References,
    (SELECT MS_CONCAT (OWNER_ORGANIZATIONS)
    FROM MS_GRC_OBJECTIVES_ORG E
    WHERE E.OBJECT_ID = A.OBJECT_ID
    )                                                        AS OWNER_ORG_IDS,
    MS_GRC_UTILITIES.INCLUDE_EXPIRED(VALID_FROM,VALID_UNTIL) AS EXPIRED_FLAG,
    SUBSTR(a.object_name,1,30)                               AS Display_name,
    'Y'                                                      AS OVER_RIDE_ACTION,
    'proc='
    || TO_CHAR(a.dd_process_instance_id)
    ||'|instid='
    ||TO_CHAR(a.dd_instance_id)
    ||'|edit_flag='
    ||NVL(a.dd_edit_flag,'Y') AS EDIT_ACTION_PARAMETERS,
    DESCRIPTION,
    (SELECT ms_Grc_utilities.Get_Edit_Url(a.DD_OBJECT_TYPE, a.DD_PROCESS_INSTANCE_ID, a.DD_INSTANCE_ID, NVL(a.DD_EDIT_FLAG,'Y'), a.object_name)
    from DUAL
    ) as EDIT,OBJECTIVE_TYPE as OBJECTIVE_TYPE_NEW,
     PARENT_OBJECTIVE_NAME as PARENT_OBJECTIVE_ID,
        (select MS_CONCAT (Ms_Apps_Utilities.Get_Display_Value((Ms_Grc_Utilities.Active_Ent_Id), 'MS_GRC_COSO_Type', COSO_TYPE))
    from MS_GRC_OBJECTIVES_COS F
    where F.OBJECT_ID = a.OBJECT_ID
    ) as COSO_TYPE,
     'Level '||objective_level as Object_level
  FROM MS_GRC_OBJECTIVES A
  where DD_OBJECT_TYPE = 'MS_GRC_OBJECTIVES'
