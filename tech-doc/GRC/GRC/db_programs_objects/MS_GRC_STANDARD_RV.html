MS_GRC_STANDARD_RV 

  CREATE OR REPLACE FORCE VIEW "MS_GRC_STANDARD_RV" ("OBJECT_ID", "OBJECT_NAME", "OBJECT_LEVEL", "OBJECT_PARENT", "STANDARD_TYPE", "OWNERS", "OBJ_STATUS", "VALID_FROM", "VALID_UNTIL", "RELATED_AOC", "RELATED_REQUIRMENTS", "RELATED_REFERENCES", "RELATED_REGULATORY_BODIES", "RELATED_FUNCTION", "RELATED_QUESTION_PROCEDURE", "RELATED_ORGANIZATIONS", "APPLIES_TO_ORGANIZATIONS", "RESTRICT_ACCESS_TO", "DD_OBJECT_TYPE", "DD_PROCESS_INSTANCE_ID", "DD_INSTANCE_ID", "DD_EDIT_FLAG", "OVER_RIDE_ACTION", "DISPLAY_NAME", "EDIT_ACTION_PARAMETERS", "OWNER_ORG_IDS", "OBJECT_NAME_URL", "EXPIRED_FLAG", "DESCRIPTION") AS 
  SELECT a.object_id,
       a.object_name,
       /* (SELECT displayed_value
         FROM z lovvalue
         WHERE lov_name   = 'MS_GRC_LEVEL'
         AND stored_value = object_level
         ) AS object_level,*/
       object_level,
       (SELECT a.object_name
          FROM ms_grc_area_of_complia a
         WHERE a.object_id = b.object_parent
           AND a.dd_object_type = 'MS_GRC_STANDARD') AS OBJECT_PARENT,
       b.standard_type,
       (SELECT ms_concat(usr.first_name || ' ' || usr.last_name) AS owners
          FROM ms_grc_standard_own d, si_users_t usr, si_ent e
         WHERE usr.user_name = d.owners
           AND d.object_id = a.object_id
           and e.enterprise_id = usr.enterprise_id) AS OWNERS,
       a.OBJ_STATUS,
       a.valid_from,
       a.valid_until,
       (SELECT MS_GRC_FN_CONCAT(orb.tgt_obj_id)
          FROM ms_grc_orb_relations_pub_v orb, ms_grc_area_of_complia base
         WHERE src_obj_type = 'MS_GRC_STANDARD'
           AND orb.tgt_obj_type = 'MS_GRC_AREA_OF_COMPLIANCE'
           AND orb.src_obj_id = a.object_id
           AND orb.tgt_obj_id = base.object_id) AS Related_AOC,
       (SELECT MS_GRC_FN_CONCAT(orb.tgt_obj_id)
          FROM ms_grc_orb_relations_pub_v orb, MS_GRC_REQUIREMENT base
         WHERE src_obj_type = 'MS_GRC_STANDARD'
           AND orb.tgt_obj_type = 'MS_GRC_REQUIREMENT'
           AND orb.src_obj_id = a.object_id
           AND orb.tgt_obj_id = base.object_id) AS Related_requirments,
       (SELECT MS_GRC_FN_CONCAT(orb.tgt_obj_id)
          FROM ms_grc_orb_relations_pub_v orb, MS_GRC_REFERENCE base
         WHERE src_obj_type = 'MS_GRC_STANDARD'
           AND orb.tgt_obj_type = 'MS_GRC_REFERENCE'
           AND orb.src_obj_id = a.object_id
           AND orb.tgt_obj_id = base.object_id) AS related_references,
       (SELECT MS_GRC_FN_CONCAT(orb.tgt_obj_id)
          FROM ms_grc_orb_relations_pub_v orb, MS_GRC_REGULATORY_BODY base
         WHERE src_obj_type = 'MS_GRC_STANDARD'
           AND orb.tgt_obj_type = 'MS_GRC_REGULATORY_BODY'
           AND orb.src_obj_id = a.object_id
           AND orb.tgt_obj_id = base.object_id) AS Related_Regulatory_Bodies,
       (SELECT MS_GRC_FN_CONCAT(orb.tgt_obj_id)
          FROM ms_grc_orb_relations_pub_v orb, ms_grc_function base
         WHERE src_obj_type = 'MS_GRC_STANDARD'
           AND orb.tgt_obj_type = 'MS_GRC_FUNCTION'
           AND orb.src_obj_id = a.object_id
           AND orb.tgt_obj_id = base.object_id) AS related_function,
       (SELECT MS_GRC_FN_CONCAT(orb.tgt_obj_id)
          FROM ms_grc_orb_relations_pub_v orb, ms_grc_question_proc base
         WHERE src_obj_type = 'MS_GRC_STANDARD'
           AND orb.tgt_obj_type = 'MS_GRC_QUESTION_PROCEDURE'
           AND orb.src_obj_id = a.object_id
           AND orb.tgt_obj_id = base.object_id) AS related_question_procedure,
       null as   related_organizations,


                         (
 SELECT ms_concat(orb.tgt_obj_id )
  FROM ms_grc_orb_relations_pub_v orb,
    si_org_entities_t base
  WHERE src_obj_type   = 'MS_GRC_STANDARD'
  AND orb.tgt_obj_type = 'MS_GRC_ORGANIZATION'
  AND orb.src_obj_id   = a.object_id
  AND orb.tgt_obj_id   = base.org_entity_id
  ) AS APPLIES_TO_ORGANIZATIONS,
        a.restrict_access_to,
       a.dd_object_type,
       a.dd_process_instance_id,
       a.dd_instance_id,
       NVL(b.dd_edit_flag, 'Y') AS DD_EDIT_FLAG,
       'Y' AS OVER_RIDE_ACTION,
       SUBSTR (object_name, 1, 30) AS display_name ,
      ('proc='|| TO_CHAR(a.dd_process_instance_id)||'|instid='||TO_CHAR(a.dd_instance_id)||'|edit_flag='||NVL(b.DD_EDIT_FLAG,'Y'))  AS EDIT_ACTION_PARAMETERS,
       (SELECT MS_CONCAT(OWNER_ORGANIZATIONS)
          FROM MS_GRC_STANDARD_org E
         WHERE E.OBJECT_ID = a.OBJECT_ID
           AND e.object_id = b.object_id) AS owner_org_ids,
       ms_grc_utilities.get_edit_url(a.dd_object_type,
                                     a.dd_process_instance_id,
                                     a.dd_instance_id,
                                     NVL(b.DD_EDIT_FLAG, 'Y'),
                                     a.OBJECT_NAME) AS object_name_url,

       CASE
      WHEN VALID_UNTIL IS NULL
      THEN '1'
      WHEN TRUNC (VALID_UNTIL) >= TRUNC (SYSDATE)
      THEN '1'
      ELSE '2'
    END
        AS EXPIRED_FLAG,
       DESCRIPTION

  FROM ms_grc_area_of_complia a, ms_grc_standard b
 WHERE a.object_id = b.object_id
