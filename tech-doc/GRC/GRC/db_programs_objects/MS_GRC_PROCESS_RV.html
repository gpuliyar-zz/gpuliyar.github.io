MS_GRC_PROCESS_RV 

  CREATE OR REPLACE FORCE VIEW "MS_GRC_PROCESS_RV" ("OBJECT_NAME", "OBJECT_ID", "DESCRIPTION", "OWNERS", "VALID_UNTIL", "VALID_FROM", "TYPE", "CATEGORY", "OBJECT_LEVEL", "CONFIDENTIALITY", "INTEGRITY", "PROCESS_AVAILABILITY", "BUSINESS_CRITICALITY", "DD_PROCESS_INSTANCE_ID", "DD_INSTANCE_ID", "DD_EDIT_FLAG", "OWNER_ORG_IDS", "RESTRICT_ACCESS_TO", "PARENT", "STATUS", "RELATED_RISKS", "RELATED_CONTROLS", "RELATED_AREAS_OF_COMPLIANCE", "RELATED_REQUIREMENTS", "RELATED_FRAMEWORK_REFERENCES", "RELATED_OBJECTIVES", "RELATED_FINANCIAL_ACCOUNTS", "RELATED_REFERENCES", "RELATED_FUNCTIONS", "APPLIES_TO_ORGANIZATIONS", "CORE_OBJECT_TYPE", "DISPLAY_NAME", "DD_OBJECT_TYPE", "OVER_RIDE_ACTION", "EDIT_ACTION_PARAMETERS", "EXPIRED_FLAG", "EDIT") AS 
  SELECT A.OBJECT_NAME,
    a.object_id,
    DESCRIPTION,
    (SELECT ms_concat(usr.first_name
      || ' '
      || usr.last_name) AS owners
    FROM ms_grc_core_object_own d,
      si_users_t usr ,
      si_ent e
    WHERE usr.user_name   = d.owners
    AND USR.ENTERPRISE_ID = E.ENTERPRISE_ID
    AND A.OBJECT_ID       = D.OBJECT_ID
    ) AS OWNERS,
    VALID_UNTIL,
    VALID_FROM,
    core_TYPE     AS TYPE,
    core_category AS CATEGORY,
    OBJECT_LEVEL,
    confidentiality,
    INTEGRITY,
    AVAILABILITY_PROCESS AS PROCESS_AVAILABILITY,
    business_criticality AS Business_Criticality,
    b.dd_process_instance_id,
    b.dd_instance_id,
    NVL (b.dd_edit_flag, 'Y') AS DD_EDIT_FLAG,
    (SELECT MS_CONCAT (OWNER_ORGANIZATIONS)
    FROM MS_GRC_cORE_OBJECT_OGN E
    WHERE E.OBJECT_ID = A.OBJECT_ID
    ) AS owner_org_ids,
    restrict_access_to,
    (select OBJECT_ID
    FROM ms_grc_core_object
    WHERE OBJECT_ID = a.OBJECT_PARENT
    )          AS parent,
    obj_status AS STATUS,
    (SELECT ms_concat(orb.tgt_obj_id)
    FROM ms_grc_orb_relations_pub_v orb,
      ms_grc_risk base
    WHERE src_obj_type   = 'MS_GRC_PROCESS'
    AND orb.tgt_obj_type = 'MS_GRC_RISK'
    AND orb.src_obj_id   = a.object_id
    AND orb.tgt_obj_id   = base.object_id
    ) AS RELATED_RISKS,
    (SELECT ms_concat(orb.tgt_obj_id)
    FROM ms_grc_orb_relations_pub_v orb,
      ms_grc_compliance_obje base
    WHERE src_obj_type   = 'MS_GRC_PROCESS'
    AND orb.tgt_obj_type = 'MS_GRC_CONTROL'
    AND orb.src_obj_id   = a.object_id
    AND orb.tgt_obj_id   = base.object_id
    ) AS RELATED_CONTROLS,
    (SELECT ms_concat(orb.tgt_obj_id)
    FROM ms_grc_orb_relations_pub_v orb,
      ms_grc_area_of_complia base
    WHERE src_obj_type   = 'MS_GRC_PROCESS'
    AND orb.tgt_obj_type = 'MS_GRC_AREA_OF_COMPLIANCE'
    AND orb.src_obj_id   = a.object_id
    AND orb.tgt_obj_id   = base.object_id
    ) AS RELATED_AREAS_OF_COMPLIANCE,
    (SELECT ms_concat(orb.tgt_obj_id)
    FROM ms_grc_orb_relations_pub_v orb,
      ms_grc_requirement base
    WHERE src_obj_type   = 'MS_GRC_PROCESS'
    AND orb.tgt_obj_type = 'MS_GRC_REQUIREMENT'
    AND orb.src_obj_id   = a.object_id
    AND orb.tgt_obj_id   = base.object_id
    ) AS RELATED_REQUIREMENTS,
    (SELECT ms_concat(orb.tgt_obj_id)
    FROM ms_grc_orb_relations_pub_v orb,
      ms_grc_model_reference base
    WHERE src_obj_type   = 'MS_GRC_PROCESS'
    AND orb.tgt_obj_type = 'MS_GRC_MODEL_REFERENCE'
    AND orb.src_obj_id   = a.object_id
    AND orb.tgt_obj_id   = base.object_id
    ) AS RELATED_FRAMEWORK_REFERENCES,
    (SELECT ms_concat(orb.tgt_obj_id)
    FROM ms_grc_orb_relations_pub_v orb,
      ms_grc_objectives base
    WHERE src_obj_type   = 'MS_GRC_PROCESS'
    AND orb.tgt_obj_type = 'MS_GRC_OBJECTIVES'
    AND orb.src_obj_id   = a.object_id
    AND orb.tgt_obj_id   = base.object_id
    ) AS RELATED_OBJECTIVES,
    (SELECT ms_concat(orb.tgt_obj_id)
    FROM ms_grc_orb_relations_pub_v orb,
      ms_grc_financial_accou base
    WHERE src_obj_type   = 'MS_GRC_PROCESS'
    AND orb.tgt_obj_type = 'MS_GRC_FINANCIAL_ACCOUNTS'
    AND orb.src_obj_id   = a.object_id
    AND orb.tgt_obj_id   = base.object_id
    ) AS RELATED_FINANCIAL_ACCOUNTS,
    (SELECT ms_concat(orb.tgt_obj_id)
    FROM ms_grc_orb_relations_pub_v orb,
      ms_grc_reference base
    WHERE src_obj_type   = 'MS_GRC_PROCESS'
    AND orb.tgt_obj_type = 'MS_GRC_REFERENCE'
    AND orb.src_obj_id   = a.object_id
    AND orb.tgt_obj_id   = base.object_id
    ) AS RELATED_REFERENCES,
    (SELECT ms_concat(orb.tgt_obj_id)
    FROM ms_grc_orb_relations_pub_v orb,
      ms_grc_function base
    WHERE src_obj_type   = 'MS_GRC_PROCESS'
    AND orb.tgt_obj_type = 'MS_GRC_FUNCTION'
    AND orb.src_obj_id   = a.object_id
    AND orb.tgt_obj_id   = base.object_id
    ) AS RELATED_FUNCTIONS,
    (SELECT ms_concat(orb.tgt_obj_id)
    FROM ms_grc_orb_relations_pub_v orb,
      si_org_entities_t base
    WHERE src_obj_type   = 'MS_GRC_PROCESS'
    AND orb.tgt_obj_type = 'MS_GRC_ORGANIZATION'
    AND orb.src_obj_id   = a.object_id
    AND orb.tgt_obj_id   = base.org_entity_id
    ) AS APPLIES_TO_ORGANIZATIONS,
    (SELECT display_name
    FROM ms_apps_entity_v B
    WHERE B.DB_TABLE_NAME = A.DD_OBJECT_TYPE
    ) AS CORE_OBJECT_TYPE,
    SUBSTR(OBJECT_NAME,1,30) DISPLAY_NAME,
    a.dd_object_type AS dd_object_type,
    'Y'              AS OVER_RIDE_ACTION,
    'proc='
    || TO_CHAR(a.dd_process_instance_id)
    ||'|instid='
    ||TO_CHAR(a.dd_instance_id)
    ||'|edit_flag='
    ||NVL(A.DD_EDIT_FLAG,'Y')                                    AS EDIT_ACTION_PARAMETERS,
    MS_GRC_UTILITIES.INCLUDE_EXPIRED(A.VALID_FROM,A.VALID_UNTIL) AS EXPIRED_FLAG,
    (SELECT ms_Grc_utilities.Get_Edit_Url(a.DD_OBJECT_TYPE, a.DD_PROCESS_INSTANCE_ID, a.DD_INSTANCE_ID, NVL(a.DD_EDIT_FLAG,'Y'), a.object_name)
    FROM DUAL
    ) AS EDIT
  FROM ms_grc_core_object a,
    MS_GRC_PROCESS b
  WHERE A.OBJECT_ID    = B.OBJECT_ID
  and B.DD_OBJECT_TYPE = 'MS_GRC_PROCESS'
