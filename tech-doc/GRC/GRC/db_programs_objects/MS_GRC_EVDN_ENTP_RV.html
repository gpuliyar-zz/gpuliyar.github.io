MS_GRC_EVDN_ENTP_RV 

  CREATE OR REPLACE FORCE VIEW "MS_GRC_EVDN_ENTP_RV" ("OBJECT_ID", "OBJECT_NAME", "OBJECT_TYPE", "LOCALE_ID", "LOCALE", "DD_PROCESS_INSTANCE_ID", "DD_INSTANCE_ID", "OVER_RIDE_ACTION", "DD_OBJECT_TYPE", "OBJECT_NAME_URL", "EDIT_ACTION_PARAMETERS", "OBJECT_STATUS", "DESCRIPTION", "VALID_FROM", "VALID_UNTIL", "RESTRICT_ACCESS_TO", "OWNER_IDS", "OWNER_NAME", "OWNER_ORG_IDS", "OWNER_ORG_NAMES", "LEVEL_1_APPROVER", "LEVEL_2_APPROVER", "ATTACHMENTS", "UPDATED_BY", "UPDATED_ON", "CREATED_BY", "CREATED_ON", "UPPER_OBJECT_NAME", "REL_DATA_LINK") AS 
  SELECT eviSumm.object_id
        ,eviSumm.object_name
		,'MS_GRC_EVIDENCE' as OBJECT_TYPE
        ,eviMlsSumm.locale_id
        ,eviMlsSumm.LOCALE
        ,eviSumm.DD_PROCESS_INSTANCE_ID
        ,eviSumm.DD_INSTANCE_ID
        ,eviSumm.OVER_RIDE_ACTION
        ,Evisumm.Dd_Object_Type
        ,Evisumm.Object_Name_Url        AS Object_Name_Url
        ,Evisumm.Edit_Action_Parameters AS EDIT_ACTION_PARAMETERS
        ,eviMlsSumm.OBJECT_STATUS
        ,eviSumm.DESCRIPTION
        ,eviSumm.VALID_FROM
        ,eviSumm.VALID_UNTIL
        ,eviSumm.RESTRICT_ACCESS_TO
        ,OWNER_IDS
        ,OWNER_NAME
        ,OWNER_ORG_IDS
        ,OWNER_ORG_NAMES
        ,( SELECT usr.first_name
                  || ' '
                  || usr.last_name
             FROM si_users_t usr
            WHERE eviSumm.LEVEL_1_APPROVER = usr.user_name
          ) AS LEVEL_1_APPROVER
        ,( SELECT usr.first_name
                  || ' '
                  || usr.last_name
             FROM si_users_t usr
            WHERE eviSumm.LEVEL_2_APPROVER = usr.user_name
         ) AS LEVEL_2_APPROVER
        ,Attachments
        ,( SELECT usr.first_name
                  || ' '
                  || usr.last_name
             FROM si_users_t usr
            WHERE eviSumm.Updated_By = usr.user_name
         ) AS Updated_By
        ,UPDATED_ON
        ,(SELECT usr.first_name
                 || ' '
                 || usr.last_name
            FROM si_users_t usr
           WHERE eviSumm.CREATED_BY = usr.user_name
          ) AS CREATED_BY
        ,CREATED_ON
        ,evisumm.UPPER_OBJECT_NAME
		,Lovvalue.displayed_value as REL_DATA_LINK
    FROM MS_GRC_EVIDENCE_SUMMARY eviSumm
        ,MS_GRC_EVIDENCE_MLS_SUMMARY eviMlsSumm
	    ,Ms_Qs_List_Of_Values Lovvalue
        ,Ms_Qs_Lov_Names Lovname
   WHERE Evisumm.Obj_Evi_Summ_Wid = Evimlssumm.Obj_Evi_Summ_Wid
     AND Lovvalue.Org_Entity_Id   = 1
     AND Lovname.Lov_Name         = 'MS_GRC_LABLES'
     and Lovvalue.stored_value 	= 'CLICK_HERE'
     AND Lovvalue.Lov_Id          = Lovname.Lov_Id
     AND Lovvalue.Locale_id 		= eviMlsSumm.locale_id
