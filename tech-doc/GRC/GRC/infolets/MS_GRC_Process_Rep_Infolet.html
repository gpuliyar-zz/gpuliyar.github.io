SELECT ProcSumm.Object_Id  AS OBJECT_ID,
ProcSumm.OBJECT_NAME AS Object_Name,
ProcMlsSumm.object_level              AS OBJECT_LEVEL,
ProcSumm.OBJ_PARENT                   AS PARENT,
ProcMlsSumm.PROCESS_TYPE              AS TYPE,
ProcMlsSumm.PROCESS_CATEGORY          AS CATEGORY,
PROCMLSSUMM.OBJECT_STATUS   as OBJ_STATUS,
procsumm.created_on as created_on,
(SELECT USER_FULL_NAME FROM MS_GRC_USER_FULL_NAME_V where user_name = ProcSumm.created_by) as CREATED_BY,
procsumm.updated_on as updated_on,
(SELECT USER_FULL_NAME FROM MS_GRC_USER_FULL_NAME_V where user_name = procsumm.UPDATED_BY) as UPDATED_BY,
PROCSUMM.VALID_FROM as VALID_FROM,
PROCSUMM.VALID_UNTIL as VALID_UNTIL,
PROCSUMM.DESCRIPTION as DESCRIPTION,
PROCSUMM.OWNER_ORG_NAMES as OWNER_ORG_NAMES,
procsumm.owner_name as owner_name,
(SELECT USER_FULL_NAME FROM MS_GRC_USER_FULL_NAME_V where user_name = ProcSumm.level_1_approver) AS level_1_approver,
(SELECT USER_FULL_NAME FROM MS_GRC_USER_FULL_NAME_V where user_name = ProcSumm.level_2_approver )AS Level_2_Approver,
ProcSumm.REL_ORGANIZATION_NAMES AS Applies_To_Organizations,
ProcSumm.REL_AOC_OBJ_NAME_AGG            AS Related_Areas_Of_Compliance,
ProcSumm.REL_ASSET_OBJ_NAME_AGG          AS Related_Assets,
ProcSumm.REL_CTRL_OBJ_NAME_AGG        AS Related_Control,
ProcSumm.REL_FUNCTION_OBJ_NAME_AGG       AS Related_Function,
ProcSumm.REL_OBJECTIVE_OBJ_NAME_AGG      AS Related_Objectives,
ProcSumm.REL_PROCESS_OBJ_NAME_AGG        AS Related_Process,
ProcSumm.REL_PRODUCT_NAME_AGG        AS Related_Products,
ProcSumm.REL_REFRENCES_OBJ_NAME_AGG      AS related_reference,
ProcSumm.REL_REQUIREMENT_OBJ_NAME_AGG    AS related_requirement,
ProcSumm.REL_RISK_OBJ_NAME_AGG           AS Related_risks
, ProcSumm.Report_json AS REPORT_JSON
,ProcSumm.OBJ_PARENT_ID as OBJ_PARENT_ID,
ProcSumm.Level_1_Approver as Level_1_Appr,
ProcSumm.Level_2_Approver as Level_2_Appr
,ProcSumm.UPPER_OBJECT_NAME AS UPPER_OBJECT_NAME
FROM MS_GRC_PROCESS_SUMMARY ProcSumm,    MS_GRC_PROCESS_MLS_SUMMARY ProcMlsSumm
where ProcMlsSumm.object_id=ProcSumm.object_id and  ProcMlsSumm.object_id in (select column_value
FROM TABLE(CAST(ms_grc_rep_browser_fil_denorm.fn_process_rpt_filter(ms_grc_array_type(:pv_id, :pv_object_name, :pv_obj_status, :pv_type, :pv_category, :pv_created_last_no_days, TO_CHAR(:pv_created_from,'YYYY-MM-DD'), TO_CHAR(:pv_created_until,'YYYY-MM-DD'), :pv_id0, :pv_id1, :pv_id2, :pv_id3, TO_CHAR(:pv_id4,'YYYY-MM-DD'), TO_CHAR(:pv_id5,'YYYY-MM-DD'), TO_CHAR(:pv_id6,'YYYY-MM-DD'), TO_CHAR(:pv_id7,'YYYY-MM-DD'), :pv_id8, :pv_id9, :pv_object_name0, :pv_object_name1, :pv_object_name2, :pv_object_name3, :PV_RELATED_ASSETS, NULL, NULL, :PV_RELATED_CONTROL, NULL, NULL, :PV_RELATED_FUNCTIONS, NULL, :PV_RELATED_OBJECTIVES, NULL, :PV_RELATED_PROCESSES, :PV_RELATED_PRODUCTS, NULL, :PV_RELATED_REQUIREMENTS, :PV_RELATED_REFERENCES, :PV_RELATED_RISKS, NULL, 'MS_GRC_PROCESS', :LOGIN_ID))AS ms_varchar2_array))
)and ProcMlsSumm.locale_id = (select LOCALE_ID from SI_USERS_T T1 ,SI_LOCALES T2 where USER_NAME = :LOGIN_ID  and T1.LOCALE = T2.LOCALE_STRING) 
and NVL(IS_ORPHAN, 1)=NVL(:PV_Orphans,NVL(IS_ORPHAN,1)) order by upper_object_name
