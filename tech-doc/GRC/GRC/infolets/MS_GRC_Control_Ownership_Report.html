SELECT object_id,
  object_name,
  legacy_id,
  OWNER_ORGS,
  MS_APPS_UTILITIES.GET_DISPLAY_VALUES (MS_GRC_UTILITIES.ACTIVE_ENT_ID, 'MS_GRC_COMPLIANCE_CATEGORY', COMPLIANCE_CATEGORY, ',', 1,(SELECT LOCALE_ID FROM SI_LOCALES WHERE LOCALE_STRING = (SELECT LOCALE FROM SI_USERS_T WHERE USER_NAME=:LOGIN_ID)))  AS  COMPLIANCE_CATEGORY,
  MS_APPS_UTILITIES.GET_DISPLAY_VALUE (MS_GRC_UTILITIES.ACTIVE_ENT_ID,'MS_GRC_YES_NO_MLS',KEY_CONTROL,1,(SELECT LOCALE_ID FROM SI_LOCALES WHERE LOCALE_STRING = (SELECT LOCALE FROM SI_USERS_T WHERE USER_NAME=:LOGIN_ID)))  AS  KEY_CONTROL,
  ms_apps_utilities.get_display_value (ms_grc_utilities.active_ent_id,'MS_GRC_Frequency',frequency,1,(select LOCALE_ID from SI_LOCALES where LOCALE_STRING = (SELECT LOCALE FROM SI_USERS_T WHERE USER_NAME=:LOGIN_ID)))  as  frequency,
  owners,
  ato_organizations,
  restrict_access_to,
  dd_object_type,
  dummy1,
  dummy2,
  dummy3,
  dummy4,
  dummy5,
  owner_org_ids,
  ato_org_ids,
  OWNER_IDS,
  MS_GRC_UTILITIES.GET_EDIT_URL(DD_OBJECT_TYPE, DD_PROCESS_INSTANCE_ID, DD_INSTANCE_ID, DD_EDIT_FLAG,OBJECT_NAME) AS EDIT_FLAG,
  (MS_GRC_UTILITIES.FN_GET_REPORT_JSON((SELECT METRIC_ID FROM SI_METRICS_T WHERE METRIC_NAME = 'MS_GRC_CONTROL'), DD_PROCESS_INSTANCE_ID, DD_INSTANCE_ID,
   '''flag=1'||CHR(38)||'emd=1'||CHR(38)||'x_edit_mode=Yes'||CHR (38)|| 'wrapper= No'|| CHR (38)|| 'mode=view''')) as report_json  
FROM MS_GRC_CONTROL_OWNERSHIP_RPT
where object_id in  (SELECT column_value
      FROM TABLE(MS_GRC_SECURITY.GET_OBJECT_LIST_BY_USER(:LOGIN_ID, DD_OBJECT_TYPE))
      WHERE COLUMN_VALUE IS NOT NULL
      )
