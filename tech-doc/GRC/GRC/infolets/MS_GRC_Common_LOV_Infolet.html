Select c.Stored_Value,
       c.Displayed_Value,
       c.Lov_Name,
       c.Locale_Id,
       c.Enterprise_Id,
       c.Lov_Id,
       c.Lov_Value_Id,
       c.Org_Entity_Id,
       c.Org_Entity_Name,
       c.Start_Date,
       c.End_Date
From   Ms_Qs_Lov_Org_Values_v c,
si_locales a,si_users_t b 
where  c.locale_id= a.locale_id
and a.locale_string = b.locale and b.user_name = :LOGIN_ID
order by c.display_order
