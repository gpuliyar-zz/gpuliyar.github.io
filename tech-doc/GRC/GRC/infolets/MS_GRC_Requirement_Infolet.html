,req_summ.UPPER_OBJECT_NAME as UPPER_OBJECT_NAME
FROM MS_GRC_REQ_SUMMARY req_summ,
    MS_GRC_REQ_MLS_SUMMARY req_mls
	where req_mls.object_id=req_summ.object_id
	and req_mls.object_id in (SELECT column_value
FROM TABLE(CAST(MS_GRC_REP_BROWSER_FIL_DENORM.FN_REQ_RPT_FILTER(ms_grc_array_type(:Req_ID,:Req_Name,:Status,:Sec_No,:Obj_Level,:Obj_Parent,:Key_Obj,:Req_Freq,:Obj_Category,:Req_ID0,:Req_ID1,:Req_ID2,TO_CHAR(:Req_ID3,'YYYY-MM-DD'),TO_CHAR(:Req_ID4,'YYYY-MM-DD'),:Req_ID5,:Req_ID6,TO_CHAR(:Req_ID7,'YYYY-MM-DD'),TO_CHAR(:Req_ID8,'YYYY-MM-DD'),TO_CHAR(:Req_ID9,'YYYY-MM-DD'),to_char(:Req_Name0,'YYYY-MM-DD'),:Req_Name1,:Req_Name2,:Req_Name3,:Req_Name4,:Req_Name5,:Req_Name6,:Req_Name7,NULL,NULL,:Req_Name8,:Req_Name9,:Status1,:Status0,NULL,:Status2,NULL,:Status3,:Status4,NULL,NULL,:Status5,:Status6,:Status7,'MS_GRC_REQUIREMENT',:Status8))AS MS_VARCHAR2_ARRAY))
) and req_mls.locale_id = (select LOCALE_ID from SI_USERS_T T1 ,SI_LOCALES T2 where USER_NAME = :Status8  and T1.LOCALE = T2.LOCALE_STRING)
order by upper_object_name
