Select c.Stored_Value,
       c.Displayed_Value,
       c.Lov_Name,
       c.Locale_Id,
        c.Start_Date,
       c.End_Date
From   MS_GRC_LOV_ORG_VALUES_V c,
si_locales a,si_users_t b 
where  c.locale_id= a.locale_id
and a.locale_string = b.locale and b.user_name = :LOGIN_ID AND (TRIM (LOV_NAME) = 'MS GRC Question Type')
order by c.Displayed_Value
