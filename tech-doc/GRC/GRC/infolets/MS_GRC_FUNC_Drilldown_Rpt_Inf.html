SELECT ctrl.object_id                                                        AS object_id ,
  ctrl.object_name                                                           AS object_name ,
  ctrl.object_status                                                       AS object_status ,
  inn.rel_dir_type                                                           AS relationship_type,
(MS_GRC_UTILITIES.FN_GET_REPORT_JSON((select METRIC_ID from SI_METRICS_T where METRIC_NAME='MS_GRC_FUNCTION'),ctrl.DD_PROCESS_INSTANCE_ID,(select DD_INSTANCE_ID from MS_GRC_FUNCTION where OBJECT_ID=ctrl.OBJECT_ID),
'''flag=1'||CHR(38)||'emd=1'||CHR(38)||'x_edit_mode=Yes'||CHR (38)|| 'wrapper= No'|| CHR (38)|| 'mode=view''')) AS REPORT_JSON  
FROM
  (SELECT DISTINCT tgt_obj_id,
    Src_Obj_Id ,
    rel_dir_type
  FROM ms_grc_orb_relations_pub_v vw
  WHERE vw.tgt_obj_type ='MS_GRC_FUNCTION'
  AND vw.src_obj_type   = :src_obj_type
  ) inn,
  MS_GRC_FUNC_ENTP_RV ctrl
WHERE inn.Src_Obj_Id = :object_id
AND inn.tgt_obj_id   = ctrl.object_id
and 
ctrl.LOCALE = (SELECT LOCALE FROM si_users_t where USER_NAME = :LOGIN_ID)
order by ctrl.upper_object_name
