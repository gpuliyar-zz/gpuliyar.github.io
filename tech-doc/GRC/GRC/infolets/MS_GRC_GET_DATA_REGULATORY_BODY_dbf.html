SELECT   object_id,
         OBJECT_NAME,
         VALID_FROM,
         VALID_UNTIL,
         RESTRICT_ACCESS_TO,
         DD_PROCESS_INSTANCE_ID,
         DD_INSTANCE_ID,
         OWNER_ORGANIZATIONS,
         display_name,
         DD_OBJECT_TYPE object_type,
         OVER_RIDE_ACTION,
         EXPIRED_FLAG,EDIT_ACTION_PARAMETERS,OBJECT_ID
  FROM   (SELECT   OBJECT_ID,
                   OBJECT_NAME,
                   SUBSTR (object_name, 1, 30) display_name,
                   dd_object_type object_type,
                   'Y' AS OVER_RIDE_ACTION,
                   ms_Grc_utilities.INCLUDE_eXPIRED (VALID_FROM, VALID_UNTIL)
                      AS EXPIRED_FLAG,
                   VALID_FROM,
                   VALID_UNTIL,
                   RESTRICT_ACCESS_TO,
                   DD_OBJECT_TYPE,
                   DD_PROCESS_INSTANCE_ID,'proc='|| TO_CHAR(dd_process_instance_id)||'|instid='||TO_CHAR(dd_instance_id)||'|edit_flag='||NVL(dd_edit_flag,'Y')  AS EDIT_ACTION_PARAMETERS,
                   DD_INSTANCE_ID,
                   (SELECT   ms_concat (d.OWNER_ORGANIZATIONS)
                      FROM   ms_grc_regulatory_body_ORG d
                     WHERE   a.object_id = d.object_id)
                      AS OWNER_ORGANIZATIONS
            FROM   ms_grc_regulatory_body a) t1
 WHERE   ms_grc_utilities.ms_grc_security (dd_object_type,
                                                   restrict_access_to,
                                                   t1.OWNER_ORGANIZATIONS,
                                                   :LOGIN_ID,
                                                   'VIEW') = 'ACCESS'
 AND TRUNC(sysdate) BETWEEN NVL(TRUNC(valid_from),TRUNC(sysdate)) AND NVL(TRUNC(valid_until),TRUNC(sysdate)) 
                                                   order by OBJECT_NAME asc
