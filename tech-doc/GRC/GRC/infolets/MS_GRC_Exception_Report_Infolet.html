select Exception_Id,Exception_for,
      type_of_exception,Applies_To_Organizations,
      OBJ_CREATED_ON,
      OBJ_CREATED_BY,
      OBJ_MODIFIED_ON,
      OBJ_MODIFIED_BY,
      Related_Areas_Of_Compliance,Related_Asset_Classes,Related_Assets,
      Related_Controls,RELATED_FRAMEWORK_REFERENCES,Related_Processes,
      Related_References,Related_Requirements,Related_Standards,
      valid_from,
      expires_on,
      status,
      items,
      (SELECT usr.first_name
	|| ' '
	|| usr.last_name
   FROM si_users_t usr
  WHERE exception_approver = usr.user_name
  ) as  exception_approver,
      explanation,
      edit
,
 exception_approver AS APPROVER_1,
  (MS_GRC_UTILITIES.FN_GET_REPORT_JSON((SELECT METRIC_ID FROM SI_METRICS_T 
WHERE METRIC_NAME='MS_GRC_EXCEPTION'),DD_PROCESS_INSTANCE_ID,(SELECT MAX(INSTANCE_ID) FROM MS_GRC_EXCEPTION_V WHERE OBJECT_ID=OBJECT_ID),
'''flag=1'||CHR(38)||'emd=1'||CHR(38)||'x_edit_mode=Yes'||CHR (38)|| 'wrapper= No'|| CHR (38)|| 'mode=view''')) as report_json,Related_Risks
   from table(ms_grc_reportbrowser_filters.GET_Filters_exception(ms_grc_array_type(:Exception_Id,:Type_Of_Exception,:Exception_For,:Item_User_Name,:Status,:Created_in_the_last_days,to_char(:Created_From,'YYYY-MM-DD'),to_char(:Created_Until,'YYYY-MM-DD'),:Expiring_in_the_last_days,:Exception_Id0,TO_CHAR(:Exception_Id1,'YYYY-MM-DD'),TO_CHAR(:Exception_Id2,'YYYY-MM-DD'),
              TO_CHAR(:Exception_Id3,'YYYY-MM-DD'),TO_CHAR(:Exception_Id4,'YYYY-MM-DD'),:Exception_Id5,:Exception_Id6,:Exception_Id7,:Exception_Id8,:Exception_Id9,:Type_Of_Exception0,:Type_Of_Exception1,:Type_Of_Exception2,:Type_Of_Exception3,:Type_Of_Exception4,:Type_Of_Exception5,'',:Type_Of_Exception6)))
  where exists(SELECT 1
        FROM SI_USER_ORG_ACT_VIEW
        WHERE activity_name IN ('GRC - View GRC Object','GRC - View All GRC Objects')
        AND user_name = :Type_Of_Exception6)
