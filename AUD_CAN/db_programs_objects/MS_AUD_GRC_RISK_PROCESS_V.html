MS_AUD_GRC_RISK_PROCESS_V 

                                                
  CREATE OR REPLACE FORCE VIEW "AUDDEVSP3"."MS_AUD_GRC_RISK_PROCESS_V" ("OBJ_REL_ID", "SRC_OBJ_TYPE", "SRC_OBJ_ID", "TGT_OBJ_TYPE", "TGT_OBJ_ID", "REL_COMMENTS", "APPLICABLE_TO", "REL_DIR_TYPE", "ADDN_DETLS") AS 
  SELECT "OBJ_REL_ID",
    "SRC_OBJ_TYPE",
    "SRC_OBJ_ID",
    "TGT_OBJ_TYPE",
    "TGT_OBJ_ID",
    "REL_COMMENTS",
    "APPLICABLE_TO",
    "REL_DIR_TYPE",
    "ADDN_DETLS"
  FROM ms_grc_orb_relations_pub_v
  WHERE UPPER (src_obj_type) = 'MS_GRC_PROCESS'
  AND UPPER (tgt_obj_type)   = 'MS_GRC_RISK'
  UNION ALL
  SELECT  "OBJ_REL_ID",
    "SRC_OBJ_TYPE",
    "SRC_OBJ_ID",
    "TGT_OBJ_TYPE",
    "TGT_OBJ_ID",
    "REL_COMMENTS",
    "APPLICABLE_TO",
    "REL_DIR_TYPE",
    "ADDN_DETLS"
  FROM  
  (SELECT /*+ FIRST_ROWS(10) */rel_orb.REL_INST_ID AS OBJ_REL_ID,
    defn.SRC_OBJ_TYPE        AS src_obj_type,
    rel.REL_SOURCE_OBJECT_ID AS src_obj_id,
    defn.dest_obj_type       AS tgt_obj_type,
    rel_orb.tgt_obj_id       AS tgt_obj_id,
    rel_orb.rel_comments     AS rel_comments,
    rel_orb.applicable_to    AS applicable_to,
    'Related'                AS REL_DIR_TYPE,
    Additional_details       AS ADDN_DETLS
  FROM ms_aud_ae_rel_defn defn,
    ms_Grc_orb_relations rel,
    ms_Grc_orb_relations_orb rel_orb
  WHERE rel.rel_source_id = rel_orb.rel_source_id
  AND defn.rel_id         = rel_orb.rel_config_id
  UNION ALL
  SELECT /*+ FIRST_ROWS(10) */ REL_ORB.REL_INST_ID AS OBJ_REL_ID,
    defn.dest_obj_type       AS src_obj_type,
    rel_orb.tgt_obj_id       AS src_obj_id,
    defn.SRC_OBJ_TYPE        AS tgt_obj_type,
    rel.REL_SOURCE_OBJECT_ID AS tgt_obj_id,
    rel_orb.rel_comments     AS rel_comments,
    rel_orb.applicable_to    AS applicable_to,
    'Induced'                AS REL_DIR_TYPE,
    Additional_details       AS ADDN_DETLS
  FROM ms_aud_ae_rel_defn defn,
    ms_Grc_orb_relations rel,
    ms_Grc_orb_relations_orb rel_orb
  WHERE REL.REL_SOURCE_ID = REL_ORB.REL_SOURCE_ID
  AND defn.rel_id         = rel_orb.rel_config_id)
  WHERE UPPER (src_obj_type) = 'MS_AUD_AUDITABLE_ENTITY'
  AND UPPER (tgt_obj_type)   = 'MS_GRC_RISK'
