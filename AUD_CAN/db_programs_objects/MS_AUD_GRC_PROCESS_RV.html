MS_AUD_GRC_PROCESS_RV 

                                                
  CREATE OR REPLACE FORCE VIEW "AUDDEVSP3"."MS_AUD_GRC_PROCESS_RV" ("DD_OBJECT_TYPE", "OBJECT_NAME", "OBJECT_ID", "DESCRIPTION", "OWNERS", "OWNER_ORGANIZATIONS", "VALID_UNTIL", "VALID_FROM", "TYPE", "CATEGORY", "OBJECT_LEVEL", "DD_PROCESS_INSTANCE_ID", "DD_INSTANCE_ID", "DD_EDIT_FLAG", "OWNER_ORG_IDS", "RESTRICT_ACCESS_TO", "PARENT", "STATUS", "RELATED_RISKS", "RELATED_CONTROLS", "RELATED_AREAS_OF_COMPLIANCE", "RELATED_REQUIREMENTS", "RELATED_FRAMEWORK_REFERENCES", "RELATED_OBJECTIVES", "RELATED_FINANCIAL_ACCOUNTS", "RELATED_REFERENCES", "RELATED_FUNCTIONS", "APPLIES_TO_ORGANIZATIONS", "CORE_OBJECT_TYPE") AS 
  SELECT a.dd_object_type, a.object_name, a.object_id, REGEXP_REPLACE(description, '<[^>]+>', NULL) description,
       (SELECT ms_concat (ms_grc_utilities.get_users (d.owners))
          FROM ms_grc_core_object_own d
         WHERE a.object_id = d.object_id) owners,
       (SELECT ms_concat
                  (ms_apps_utilities.get_display_values
                                              (ms_grc_utilities.active_ent_id,
                                               'ORGENTITY',
                                               d.owner_organizations,
                                               ','
                                              )
                  )
          FROM ms_grc_core_object_ogn d
         WHERE a.object_id = d.object_id) owner_organizations,
       valid_until, valid_from,
       (SELECT displayed_value
          FROM ms_qs_list_of_values lovvalue,ms_qs_lov_names lovname
          WHERE lovvalue.lov_id = lovname.lov_id
              AND lovvalue.org_entity_id = 1
              AND lovname.lov_name in('MS_AUD_AE_Process_Type','MS GRC Process Type')
              AND lovvalue.locale_id = nvl(cast(si_db_locale_sv.getvalue as number),1009) 
              and stored_value=a.core_type) TYPE,
       (SELECT displayed_value
          FROM ms_qs_list_of_values lovvalue,ms_qs_lov_names lovname 
          WHERE lovvalue.lov_id = lovname.lov_id
              AND lovvalue.org_entity_id = 1
              AND lovname.lov_name in('MS GRC Core Category','MS_AUD_AE_Category_Type')
              AND lovvalue.locale_id = nvl(cast(si_db_locale_sv.getvalue as number),1009) 
              and stored_value=a.core_category
              ) CATEGORY,
       ms_apps_utilities.get_display_value
                                (ms_grc_utilities.active_ent_id,
                                 'MS_GRC_LEVEL',
                                 object_level
                                ) object_level,
       b.dd_process_instance_id, b.dd_instance_id,
       NVL (a.dd_edit_flag, 'Y') AS dd_edit_flag,
       (SELECT ms_concat (owner_organizations)
          FROM ms_grc_core_object_ogn e
         WHERE e.object_id = a.object_id) AS owner_org_ids,
       restrict_access_to, (SELECT object_name
                              FROM ms_grc_core_object
                             WHERE object_id = a.object_parent) AS PARENT,
       obj_status AS status,
       (SELECT ms_aud_utilities.fn_get_ae_rel
                                          (a.object_id,
                                           'MS_GRC_RISK'
                                          )
          FROM DUAL) AS related_risks,
       (SELECT ms_aud_utilities.fn_get_ae_rel
                                       (a.object_id,
                                        'MS_GRC_CONTROL'
                                       )
          FROM DUAL) AS related_controls,
       (SELECT ms_aud_utilities.fn_get_ae_rel
                  (a.object_id,
                   'MS_GRC_AREA_OF_COMPLIANCE'
                  )
          FROM DUAL) AS related_areas_of_compliance,
       (SELECT ms_aud_utilities.fn_get_ae_rel
                                (a.object_id,
                                 'MS_GRC_REQUIREMENT'
                                )
          FROM DUAL) AS related_requirements,
       (SELECT ms_aud_utilities.fn_get_ae_rel
                    (a.object_id,
                     'MS_GRC_MODEL_REFERENCE'
                    )
          FROM DUAL) AS related_framework_references,
       (SELECT ms_aud_utilities.fn_get_ae_rel
                                   (a.object_id,
                                    'MS_GRC_OBJECTIVES'
                                   )
          FROM DUAL) AS related_objectives,
       (SELECT ms_aud_utilities.fn_get_ae_rel
                   (a.object_id,
                    'MS_GRC_FINANCIAL_ACCOUNTS'
                   )
          FROM DUAL) AS related_financial_accounts,
       (SELECT ms_aud_utilities.fn_get_ae_rel
                                    (a.object_id,
                                     'MS_GRC_REFERENCE'
                                    )
          FROM DUAL) AS related_references,
       (SELECT ms_aud_utilities.fn_get_ae_rel
                                      (a.object_id,
                                       'MS_GRC_FUNCTION'
                                      )
          FROM DUAL) AS related_functions,
       (SELECT ms_aud_utilities.fn_get_ae_rel
                           (a.object_id,
                            'MS_GRC_ORGANIZATION'
                           )
          FROM DUAL) AS applies_to_organizations,
       NVL ((SELECT display_name
               FROM ms_apps_entity_v b
              WHERE b.db_table_name = a.dd_object_type),
            'Auditable Entity'
           ) AS core_object_type
  FROM ms_grc_core_object a, ms_aud_auditable_entit b
 WHERE b.dd_object_type = 'MS_AUD_AUDITABLE_ENTITY'
 AND a.object_id = b.object_id
