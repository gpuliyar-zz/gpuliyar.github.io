DECODE(a.instance_id,MS_AUD_UTILITIES.min_instance(a.audit_id),b.audit_title, NULL) END AS Audit_Title, CASE WHEN a.audit_status =34 AND a.audit_action =3 THEN NVL(ms_aud_utilities.fn_get_plan_title(a.plan_title),'Adhoc') ELSE DECODE(a.instance_id, MS_AUD_UTILITIES.min_instance(a.audit_id),NVL(ms_aud_utilities.fn_get_plan_title(a.plan_title),'Adhoc'), NULL) END AS Plan_Title, CASE WHEN a.audit_status =34 AND a.audit_action =3 THEN ms_aud_utilities.fn_get_org_display_value(a.conducted_by_org) ELSE DECODE(a.instance_id, MS_AUD_UTILITIES.min_instance(a.audit_id),ms_aud_utilities.fn_get_org_display_value(a.conducted_by_org), NULL) END AS ORGANIZATION, CASE WHEN a.audit_status =34 AND a.audit_action =3 THEN CAST (NULL AS VARCHAR2 (1000)) ELSE DECODE(a.instance_id, MS_AUD_UTILITIES.min_instance(a.audit_id),CAST (NULL AS VARCHAR2 (1000)), NULL) END AS Purpose, CASE WHEN a.audit_status =34 AND a.audit_action =3 THEN NVL(ms_apps_utilities.GET_USER_FULL_NAME(b.lead_auditor),ms_aud_utilities.fn_get_mlov_lead_aud_disp(ms_aud_utilities.mlov_values (a.audit_id,'AUDIT_ID','lead_auditor_mlov','ms_aud_audit_alm'))) ELSE DECODE(a.instance_id, MS_AUD_UTILITIES.min_instance(a.audit_id),NVL(ms_apps_utilities.GET_USER_FULL_NAME(b.lead_auditor),ms_aud_utilities.fn_get_mlov_lead_aud_disp(ms_aud_utilities.mlov_values (a.audit_id,'AUDIT_ID','lead_auditor_mlov','ms_aud_audit_alm'))),NULL) END AS Lead_Auditor, CASE WHEN a.audit_status =34 AND a.audit_action =3 THEN NVL(ms_apps_utilities.GET_USER_FULL_NAME(b.audit_manager), ms_aud_utilities.fn_get_mlov_aud_man_disp(ms_aud_utilities.mlov_values (a.audit_id,'AUDIT_ID','audit_manager_mlov','ms_aud_audit_amm'))) ELSE DECODE(a.instance_id, MS_AUD_UTILITIES.min_instance(a.audit_id),NVL(ms_apps_utilities.GET_USER_FULL_NAME(b.audit_manager),ms_aud_utilities.fn_get_mlov_aud_man_disp(ms_aud_utilities.mlov_values (a.audit_id,'AUDIT_ID','audit_manager_mlov','ms_aud_audit_amm'))),NULL) END AS Audit_Manager, CASE WHEN a.audit_status =34 AND a.audit_action =3 THEN ms_aud_utilities.fn_get_url('MS_AUD_MANAGE_AUDIT',a.process_instance_id,a.audit_id,a.instance_id,ms_aud_utilities.FN_GET_STATUS_DISPLAY_NAME(a.audit_status),'2','3') ELSE ms_aud_utilities.fn_get_url('MS_AUD_MANAGE_AUDIT',a.process_instance_id,a.audit_id,a.instance_id,ms_aud_utilities.FN_GET_STATUS_DISPLAY_NAME(a.audit_status),'2','3') END       AS Status, TO_CHAR(a.creation_date,'MM/DD/YYYY HH24:MI:SS') AS "Date/Time", CASE WHEN a.audit_status =34 AND a.audit_action =3 THEN ms_aud_utilities.trail_user(a.audit_id,a.instance_id,a.audit_status) ELSE ms_aud_utilities.trail_user(a.audit_id,a.instance_id,a.audit_status) END AS User_Name, CASE WHEN a.audit_status =34 AND a.audit_action =3 THEN DECODE (ms_aud_utilities.FN_GET_STATUS_DISPLAY_NAME(a.audit_status),'Final Report',NULL,'Closed',NULL,'Closed (Re-opened)',NULL,a.audit_comments) ELSE DECODE (ms_aud_utilities.FN_GET_STATUS_DISPLAY_NAME(a.audit_status),'Final Report',NULL,'Closed',NULL,'Closed (Re-opened)',NULL,a.audit_comments) END AS Comments, NULL AS Electronic_Signatures, a.audit_id, PROCESS_INSTANCE_ID, instance_id FROM (SELECT * FROM MS_AUD_MANAGE_AUDIT_V  WHERE AUDIT_ID = :1 AND INSTANCE_REC_NUM = 1 AND MS_AUD_UTILITIES.TRAIL_STATUS(AUDIT_ID,INSTANCE_ID,AUDIT_STATUS) >0 AND MS_AUD_AUDIT_SECURITY.CHECK_VIEW_ACCESS(AUDIT_ID,DD_OBJECT_TYPE,RESTRICT_ACCESS_TO,CONDUCTED_BY_ORG,:2,'VIEW') = 'ACCESS' AND AUDIT_ID IS NOT NULL) A, (SELECT * FROM MS_AUD_AUDIT WHERE STATUS NOT IN ('2','31','44','51')) B WHERE A.AUDIT_ID = B.AUDIT_ID AND ( (a.audit_status =34 AND a.audit_action =3) or (a.audit_status =34 AND a.audit_action NOT IN ('2','4') ) OR (a.audit_status in (33,35,36,37,38,4)) or (a.audit_status = 57 AND a.audit_action in (36,3))
or (a.audit_status in (8,58) AND a.audit_action is not null) ) 
ORDER BY a.creation_date
