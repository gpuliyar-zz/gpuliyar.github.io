                                                        select CmpCount,NonCount,Qname,QuestionId from
    (select CmpCount,NonCount,Qname,QuestionId,(CmpCount+NonCount) as Total from
    (select NVL(p.Comp,0) as CmpCount,NVL(q.NonComp,0) as NonCount,p.QnameNon as Qname,p.QstId as QuestionId from
    (select count(a.question_id) as Comp,a.question_procedure_no as QnameNon,a.question_id as QstId
    from 
    ms_aud_chk_question_qst a,ms_aud_checklist_task b,ms_aud_audit c,ms_chk_chklist_master d
    where b.checklist_id=a.qst_checklist_id and
    d.checklist_id=a.qst_checklist_id and
    b.checklist_id=d.checklist_id and
    b.task_id=a.qst_task_id and
    a.qst_audit_id=b.audit_id and
    a.qst_audit_id=c.audit_id and
    b.audit_id=c.audit_id and 
    b.checklist_type='6' and
    a.qs_response is not null and
    a.qs_response in ('4','2','3') and
    c.audit_status not in ('4','8') and
    c.audit_start_date between :1+1 and :2
    and
    (ms_apps_security_pkg.check_object_access (d.dd_object_type, d.restrict_access_to,
    ms_chk_utilities.fn_get_mlov_values (b.checklist_id,'owner_organizations','ms_chk_chklist_master_org'),:3,'VIEW')='ACCESS')
    group by a.question_id,a.question_procedure_no ) p,
   ( select count(a.question_id) as NonComp,a.question_procedure_no as QnameCmp,a.question_id as QstId
    from 
    ms_aud_chk_question_qst a,ms_aud_checklist_task b,ms_aud_audit c,ms_chk_chklist_master d
    where b.checklist_id=a.qst_checklist_id and
    d.checklist_id=a.qst_checklist_id and
    b.checklist_id=d.checklist_id and
    b.task_id=a.qst_task_id and
    a.qst_audit_id=b.audit_id and
    a.qst_audit_id=c.audit_id and
    b.audit_id=c.audit_id and 
    b.checklist_type='6' and
    a.qs_response is not null and
    a.qs_response='4' and
    c.audit_status not in ('4','8') and
    c.audit_start_date between :1+1 and :2
    and
    (ms_apps_security_pkg.check_object_access (d.dd_object_type, d.restrict_access_to,
    ms_chk_utilities.fn_get_mlov_values (b.checklist_id,'owner_organizations','ms_chk_chklist_master_org'),:3,'VIEW')='ACCESS')
    group by a.question_id,a.question_procedure_no) q
    where p.QstId=q.QstId(+))
    order by Total desc)
    where rownum<=5
    and Total<>0
