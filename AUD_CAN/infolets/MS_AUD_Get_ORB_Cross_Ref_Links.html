                                                        SELECT locale_id, ms_concat(stored_value
  ||'^*^'
  || displayed_value
) AS report_detail
FROM
  (
Select Stored_Value,
       Displayed_Value,
       Lov_Name,
       Locale_Id,
       Enterprise_Id,
       Lov_Id,
       Lov_Value_Id,
       Org_Entity_Id,
       Org_Entity_Name,
       Start_Date,
       End_Date
From   Ms_Qs_Lov_Org_Values_v
where  (End_Date is null or End_Date > sysdate) 
        and start_date <=sysdate
        and (trim(LOV_Name) = 'MS AUD Reference Type')
        and :1 = :1
) group by locale_id
