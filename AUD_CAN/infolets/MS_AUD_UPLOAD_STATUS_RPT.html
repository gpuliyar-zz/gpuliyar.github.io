MS_AUD_RPT_Upload_Status Report SYSTEMI 
                                                        SELECT MODULE,
  FORM_NAME,
  UPLOAD_TEMPLATE_NAME,
  UPLOAD_PROCESS_INSTANCE_ID,
  UPLOAD_INSTANCE_ID,
  STATUS,
  STATUS_DESCRIPTION,
  CREATED_ON,
  '"'||SUCCESS_TEMPLATE||'",8' AS SUCCESS_TEMPLATE ,
  '"'||ERROR_TEMPLATE||'",8' AS ERROR_TEMPLATE,
  USER_OPERATION
FROM MS_APPS_DU_UPLOAD_STATUS_LOG
WHERE UPLOAD_PROCESS_INSTANCE_ID IN
  (SELECT UNPVT.VALUE
  FROM MS_AUD_DATA_UPLOAD A UNPIVOT ( VALUE FOR ATTRIBUTE IN (AUDIT_PID,
  PLAN_SCOPE_PID, DRAFT_ISSUES_PID, 
  AUDIT_TASK_PID,CHECKLIST_TASK_PID, CONTROL_TEST_TASK_PID,AUDIT_REPORT_PID, PLAN_PID, AUDITABLE_ENTITY_PID, CHECKLIST_PID ) ) UNPVT
  WHERE PARENT_PID = NVL(:2,PARENT_PID)
  AND UPLOAD_TYPE  =:1 
  ) order by  UPLOAD_PROCESS_INSTANCE_ID desc
