                                                        select  ms_aud_utilities.get_select_url(scenario_name,scenario_id,'RADIO') as "Check Box", 
        scenario_id as "Scenario ID",
        scenario_name as "Scenario Name"
from
(
  select scenario_id, scenario_name from (select scenario_id, scenario_name
  from table (ms_aud_f001_helper.func_get_scenarios_from_risk (:1)) 
  where ms_aud_utilities.fn_get_count_scenarios (:1) > 0
  union all
select case when :1 is null then null  else 'NA' end , case when :1 is null then null  else 'NA' end
  from dual)    
 )
 where (upper(scenario_name) like nvl(upper('%'||:2||'%'),upper(scenario_name)) 
      or upper(scenario_id) like nvl(upper('%'||:2||'%'),upper(scenario_id))
      )
order by scenario_name
