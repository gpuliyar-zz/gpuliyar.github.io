                                                        SELECT  distinct aa.audit_title,
        aa.plan_title,
        ms_aud_utilities.fn_get_org_display_value(aa.conducted_by_org) as Organization,
        CAST (null AS VARCHAR2 (1000)) as Purpose,
        ms_apps_utilities.GET_USER_FULL_NAME(aa.lead_auditor) as Lead_Auditor,
        ms_apps_utilities.GET_USER_FULL_NAME(aa.audit_manager) as Audit_Manager,
                    (SELECT DISTINCT    '<a target="_self" '
                          || 'href="javascript'
                          || CHR (58)
                          || 'callreportdetails(''Pushinfolet?id='
                          || ms_apps_utilities.get_infolet_id
                                                        ('MS_AUD_MANAGE_AUDIT')
                          || CHR (38)
                          || 'proc='
                          || process_instance_id
                          || CHR (38)
                          || 'flag=1'
                          || CHR (38)
                          || 'emd=2'
                          || CHR (38)
                          || 'x_edit_mode=No'
                          || CHR (38)
                          || 'audit_id='
                          || audit_id
                          || CHR (38)
                          || 'wrapper=no'')">'
                          || ms_aud_utilities.fn_get_status_display_name
                                                                 (audit_status)
                          || '</a>'
                     FROM ms_aud_manage_audit_v
                    WHERE instance_id = (SELECT MAX (instance_id)
                                           FROM ms_aud_manage_audit_v
                                          WHERE audit_id = :1)) as Status,(SELECT DISTINCT    
                          ms_aud_utilities.fn_get_status_display_name
                                                                 (audit_status)                          
                     FROM ms_aud_manage_audit_v
                    WHERE instance_id = (SELECT MAX (instance_id)
                                           FROM ms_aud_manage_audit_v
                                          WHERE audit_id = :1)) as Audit_Status,
                 to_char(aa.creation_date,'Mon/DD/YYYY') as "Date/Time",
                ms_apps_utilities.GET_USER_FULL_NAME(aa.dd_current_user_name) as User_Name, aa.audit_comments as Comments,
                aa.electronically_signed as Electronic_Signatures, aa.audit_id,aa.PROCESS_INSTANCE_ID, aa.instance_id
           FROM ms_aud_manage_audit_v aa, ms_aud_audit_trail bb
           where aa.audit_id = bb.audit_id
           and aa.process_instance_id = bb.pid
           and aa.instance_id = bb.instance_id
           AND bb.instance_id = (SELECT MAX (instance_id)
                             FROM ms_aud_audit_trail
                            WHERE audit_id = :1)  
           and aa.audit_id = :1 
           and bb.audit_id = :1 
           and aa.audit_status is not null
             and aa.instance_rec_num = 1
             and ms_aud_audit_security.CHECK_VIEW_ACCESS(aa.audit_id,aa.dd_object_type,aa.restrict_access_to,aa.conducted_by_org,:2,'VIEW') = 'ACCESS'
             order by Audit_Status
