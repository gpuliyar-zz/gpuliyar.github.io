FUNCTION get_pmo_audit_id (i_team_id NUMBER) RETURN VARCHAR2 AS x_unique_id VARCHAR2 (4000); BEGIN SELECT unique_id INTO x_unique_id FROM MS_AUD_AUDIT_SCHEDULES WHERE NVL (audit_id, audit_schedule_id) = (SELECT PROJECT_ID FROM MS_RSM_PROJECT_TEAM WHERE team_id = i_team_id); RETURN x_unique_id; EXCEPTION WHEN OTHERS THEN RETURN NULL; END;
