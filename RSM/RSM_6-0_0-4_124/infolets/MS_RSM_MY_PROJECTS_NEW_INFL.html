                                                        SELECT RSM_TEAM.TEAM_NAME As TEAM_NAME,    
       RSM_TEAM.START_DATE As START_DATE,    
       RSM_TEAM.END_DATE As END_DATE,    
       MS_RSM_UTILITIES.GET_TEAM_MANAGER(RSM_TEAM.TEAM_ID) AS TEAM_MANAGER,  
       ms_rsm_utilities.GET_TEAM_MANAGER_L2(RSM_TEAM.TEAM_ID) As TEAM_L2_MANAGER,  
       RSM_TEAM.TEAM_TYPE As TEAM_TYPE,    
       RSM_TEAM.TEAM_ID As TEAM_ID,    
       (SELECT ms_concat (FIRST_NAME || ' ' || LAST_NAME) user_full_name    
          FROM si_users_t    
         WHERE user_NAME IN    
                   (SELECT TEAM_MANAGER from ms_rsm_project_team_TMR where team_id=rsm_team.team_id)) AS TEAM_MANAGER_FULL_NAME,    
       (SELECT ms_concat (FIRST_NAME || ' ' || LAST_NAME) user_full_name    
          FROM si_users_t    
         WHERE user_NAME IN    
                  (SELECT TEAM_L2_MANAGER from ms_rsm_project_team_TM2 where team_id=rsm_team.team_id)) AS TEAM_L2_MANAGER_FULL_NAME,    
       PROJECT_ID As PROJECT_ID,    
       ORG_LEVEL As ORG_LEVEL,    
         (select ms_rsm_process.get_rsm_budget_url((select ms_apps_utilities.get_infolet_id('MS_RSM_MODIFY_TEAM_BUDGETS')    
                                                   from dual),    
                                                 (select budget_pid    
                                                    from ms_rsm_budget_header    
                                                   where team_id = RSM_TEAM.team_id),    
                                                 null,    
                                                 19,    
                                                 RSM_TEAM.team_id,    
                                                 :1)    
          from dual) AS budget_url,    
       NULL AS NULL2,    
       NULL AS NULL3    
  From Ms_Rsm_Project_Team Rsm_Team    
 WHERE NVL(team_status,0)<>59 and NVL(END_DATE,sysdate)>=(SYSDATE-MS_RSM_UTILS.GET_PARAMETER_VALUE('PAST_TIME_PERIOD_FOR_PROJECTS_ENDED')) and (INSTR(','||MS_RSM_UTILITIES.GET_TEAM_MANAGER(TEAM_ID) ||',',','|| :1||',')>0    
 OR INSTR (','||ms_rsm_utilities.GET_TEAM_MANAGER_L2(TEAM_ID)||',',','|| :1||',') > 0    
 OR Team_Id IN (select distinct team_id from ms_rsm_res_request where team_id=Team_Id and active='Y' and resources=ms_apps_utilities.get_user_id(:1)))
