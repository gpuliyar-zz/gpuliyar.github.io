                                                        select
(select ms_rsm_process.get_rsm_rpm_edit_url_for_team(
(SELECT MS_APPS_UTILITIES.GET_INFOLET_ID('MS_AUD_RSM_MANAGE_TEAM_RESOURCE') FROM DUAL),
NVL(MANAGE_RESOURCE_PID,-1),A.TEAM_NAME,34,A.TEAM_ID,45,-1,:1) FROM DUAL) AS EDIT_URL,
A.TEAM_NAME AS TEAM_NAME,
a.team_id As team_id,
(select MS_RSM_UTILITIES.get_user_full_name(ms_rsm_utilities.GET_TEAM_MANAGER(a.team_id)) from dual) as team_manager_fullname,
(SELECT MS_RSM_UTILITIES.GET_USER_FULL_NAME(MS_RSM_UTILITIES.GET_TEAM_MANAGER_L2(A.TEAM_ID)) FROM DUAL) AS TEAM_L2_MANAGER_FULLNAME,
A.START_DATE AS START_DATE,
a.end_date As end_date,
NULL AS TOTAL_BUDGETED_EFFORT ,
MS_RSM_UTILITIES.GET_TEAM_MANAGER(A.TEAM_ID) AS TEAM_MANAGER,
ms_rsm_utilities.GET_TEAM_MANAGER_L2(a.team_id) As team_l2_manager,
null as open_resource_count,
null as total_allocated,
(select ms_rsm_process.get_concat_resources(a.team_id) from dual) as resource_name,
NULL AS ACTUAL_EFFORT,
( SELECT  MS_RSM_PROCESS.GET_RSM_BUDGET_URL((SELECT MS_APPS_UTILITIES.GET_INFOLET_ID('MS_RSM_MODIFY_TEAM_BUDGETS') FROM DUAL),
(SELECT BUDGET_PID FROM MS_RSM_BUDGET_HEADER WHERE TEAM_ID=A.TEAM_ID),NULL,19,A.TEAM_ID, :1)  FROM DUAL) AS MODIFY_URL,
NULL AS NULL1,
null as null2,
null as null3
from ms_rsm_project_team a
WHERE (TRUNC(NVL(A.END_DATE,SYSDATE)) >= TRUNC(SYSDATE))
AND ((SUBSTR(MS_RSM_UTILITIES.GET_TEAM_MANAGER(A.TEAM_ID),INSTR(MS_RSM_UTILITIES.GET_TEAM_MANAGER(A.TEAM_ID), :1),LENGTH(:1)) = (:1 ))OR(SUBSTR(MS_RSM_UTILITIES.GET_TEAM_MANAGER_L2(A.TEAM_ID),INSTR(MS_RSM_UTILITIES.GET_TEAM_MANAGER_L2(A.TEAM_ID), :1),LENGTH(:1)) = (:1 )) OR (TEAM_ID IN (SELECT DISTINCT TEAM_ID FROM MS_RSM_TEAM_RESOURCES WHERE TEAM_ID=TEAM_ID AND RESOURCE_NAME=:1 )))
order by a.start_date desc,a.team_name
