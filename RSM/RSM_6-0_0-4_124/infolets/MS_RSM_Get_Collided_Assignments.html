SELECT REQUEST_ID AS REQUEST_ID,  
  RESOURCE_POOL AS RESOURCE_POOL,  
 RESOURCE_NAME AS RESOURCE_NAME,  
  Project_Id As Project_Id,  
  PROJECT_NAME AS PROJECT_NAME,  
  START_DATE AS START_DATE,  
  END_DATE AS END_DATE,  
  NO_OF_HOURS AS NO_OF_HOURS,  
  Modified_On As Modified_On,  
  MODIFIED_BY AS MODIFIED_BY,  
  ACTION_TAKEN AS ACTION_TAKEN,  
  USER_NAME As USER_NAME,  
  RESOURCE_POOL_ID AS RESOURCE_POOL_ID,  
  RESOURCE_ID As RESOURCE_ID  
from  
(SELECT B.PRIMARY_KEY AS REQUEST_ID,  
  (Select Resource_Pool_Name From Ms_Rsm_Resource_Pool Where Resource_Pool_Id=A.Resource_Pool) As Resource_Pool,  
  nvl(ms_rsm_utils.GET_USER_FULL_NAME(TO_NUMBER(A.RESOURCES)),'Unassigned') AS RESOURCE_NAME,  
  'PROJ-'||A.TEAM_ID As Project_Id,  
  (SELECT TEAM_NAME FROM MS_RSM_PROJECT_TEAM WHERE TEAM_ID=A.TEAM_ID ) AS PROJECT_NAME,  
  nvl(B.ALLOCATED_START_DATE,b.requested_start_date) AS START_DATE,  
  nvl(B.ALLOCATED_END_DATE,b.requested_end_date) AS END_DATE,  
  NVL(B.ALLOCATED_HOURS,B.REQUESTED_HOURS) AS NO_OF_HOURS,  
  NVL(A.DD_MODIFIED_ON,A.DD_CREATED_ON) AS MODIFIED_ON,  
  NVL(ms_rsm_utils.GET_USER_FULL_NAME( A.DD_MODIFIED_BY),ms_rsm_utils.GET_USER_FULL_NAME( A.DD_CREATED_BY)) AS MODIFIED_BY,  
  'Updated' AS ACTION_TAKEN,  
  ms_rsm_utils.GET_USER_FULL_NAME(:1) AS USER_NAME,  
  A.RESOURCE_POOL AS RESOURCE_POOL_ID,  
  A.RESOURCES As RESOURCE_ID  
  From Ms_Rsm_Res_Request A,Ms_Rsm_Res_Request_Hrs B  
  Where A.PRIMARY_KEY In (Select Id From Ms_Rsm_Sch_Data_Collision Where Type='REQUEST' And User_Name=:1)  
  And A.PRIMARY_KEY=B.PRIMARY_KEY AND A.ACTIVE='Y'  
UNION  
SELECT B.PRIMARY_KEY AS REQUEST_ID,  
  NULL AS RESOURCE_POOL,  
  null AS RESOURCE_NAME,  
  null As Project_Id,  
  null AS PROJECT_NAME,  
  null AS START_DATE,  
  null AS END_DATE,  
  null AS NO_OF_HOURS,  
 NVL(A.DD_MODIFIED_ON,A.DD_CREATED_ON) As Modified_On,  
  NVL(ms_rsm_utils.GET_USER_FULL_NAME( A.DD_MODIFIED_BY),ms_rsm_utils.GET_USER_FULL_NAME( A.DD_CREATED_BY)) AS MODIFIED_BY,  
  'Deleted' AS ACTION_TAKEN,  
  ms_rsm_utils.GET_USER_FULL_NAME(:1) AS USER_NAME,  
  NULL AS RESOURCE_POOL_ID,  
  NULL As RESOURCE_ID  
  From Ms_Rsm_Res_Request A,Ms_Rsm_Res_Request_Hrs B  
  Where A.Primary_Key In (Select Id From Ms_Rsm_Sch_Data_Collision Where Type='REQUEST' And User_Name=:1) 
  AND A.PRIMARY_KEY=B.PRIMARY_KEY AND A.ACTIVE='N')
