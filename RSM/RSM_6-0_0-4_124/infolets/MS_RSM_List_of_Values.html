                                                        SELECT   lov_id,
         lov_name,
         stored_value,
         displayed_value,
         start_date,
         end_date,
         enterprise_id,
         org_entity_id
    FROM MS_RSM_LIST_OF_VALUES_V
   WHERE NVL (TRUNC (start_date), TRUNC (SYSDATE - 1)) <= TRUNC (SYSDATE)
     AND NVL (TRUNC (end_date), TRUNC (SYSDATE + 1)) >= TRUNC (SYSDATE)
     and lov_name like'%RSM%'
ORDER BY lov_name,display_order
