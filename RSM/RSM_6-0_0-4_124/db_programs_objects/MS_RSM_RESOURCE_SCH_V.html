MS_RSM_RESOURCE_SCH_V 

                                                
  CREATE OR REPLACE FORCE VIEW "RSMDEV"."MS_RSM_RESOURCE_SCH_V" ("ORG_ROLE_ID", "ENTERPRISE_ID", "ORG_ENTITY_ID", "ROLE_ID", "ROLE_NAME", "ORG_ENTITY_NAME", "USER_ID", "USER_NAME", "USER_FULL_NAME", "ACTIVITY_ID", "ACTIVITY_NAME", "ACTIVITY_SHORT_NAME") AS 
  select  A.ORG_ROLE_ID,
        b.ENTERPRISE_ID,
        a.ORG_ENTITY_ID,
        a.ROLE_ID,
        c.ROLE_NAME,
        b.ORG_ENTITY_NAME,
        e.USER_ID,
        e.USER_NAME,
        first_name||' '||decode(middle_initial,'','',middle_initial||' ')||last_name user_full_name,
		f.activity_id,
		f.activity_name,
		f.activity_short_name
from
        SI_USERS e,
        SI_USER_ORG_ROLES d,
        SI_ORG_ROLES a,
        SI_ORG_ENTITIES b,
        SI_ROLES c,
		si_activities_t f ,
        si_role_activities g 
where     d.ORG_ROLE_ID = a.ORG_ROLE_ID
and        b.ORG_ENTITY_ID = a.ORG_ENTITY_ID
and        c.role_id = a.role_id
and        E.USER_ID = d.USER_ID
AND     c.enterprise_id = b.enterprise_id
and f.activity_id    = g.activity_id
  AND g.role_id          = a.role_id
  AND g.role_id          = c.role_id
AND SYSDATE BETWEEN (NVL (e.start_date, SYSDATE - 1))
                              AND NVL (e.end_date, SYSDATE + 1)
