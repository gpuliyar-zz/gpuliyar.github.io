FUNCTION CHECK_FORM_ACCESS ( ip_pushform_id IN NUMBER, ip_process_instance_id IN NUMBER, ip_user_name IN VARCHAR2 ) RETURN NUMBER AS LV_count NUMBER; LV_sql_stmt VARCHAR(32767); LV_object_type VARCHAR2(100); LV_restrict_access_to VARCHAR2(100); LV_owner_organizations VARCHAR2(4000); LV_action VARCHAR2(100); LV_security_all VARCHAR2(100); LV_org_list VARCHAR_TBL_TYPE; LV_cntr NUMBER; LV_org_found BOOLEAN; LV_restriction VARCHAR2(30); LV_ownership VARCHAR2(30); BEGIN LV_action := 'EDIT'; CODE_POS := 1; SELECT COUNT(1) INTO LV_count FROM SI_EVENT_ASSIGNMENTS WHERE INFOLET_ID = ip_pushform_id AND PROCESS_INSTANCE_ID = ip_process_instance_id AND USER_ID = ( SELECT USER_ID FROM SI_USERS WHERE UPPER(USER_NAME) = UPPER(ip_user_name) ); CODE_POS := 2; IF LV_count > 0 THEN CODE_POS := 3; RETURN 1; ELSE CODE_POS := 3; SELECT SECURITY_RESTRICTION, SECURITY_OWNERSHIP INTO LV_restriction, LV_ownership FROM MS_APPS_ENTITY_V WHERE DB_TABLE_NAME IN ( SELECT ENTITY_NAME FROM MS_APPS_VISUAL_ENTITY_COMP_V WHERE VE_NAME = ( SELECT METRIC_NAME FROM SI_METRICS_T WHERE METRIC_ID = ip_pushform_id ) ) AND SECURITY_RESTRICTION IS NOT NULL AND SECURITY_OWNERSHIP IS NOT NULL; CODE_POS := 4; LV_sql_stmt := 'SELECT DD_OBJECT_TYPE, ' || LV_restriction || ',' || LV_ownership || ' FROM SI_'|| ip_pushform_id || '_T S WHERE PROCESS_INSTANCE_ID = :1 AND INSTANCE_ID = ( SELECT MAX(INSTANCE_ID) FROM SI_' || ip_pushform_id || '_T S1 WHERE S1.PROCESS_INSTANCE_ID = S.PROCESS_INSTANCE_ID ) AND ROWNUM = 1'; CODE_POS := 5; EXECUTE IMMEDIATE LV_sql_stmt INTO LV_object_type, LV_restrict_access_to, LV_owner_organizations USING ip_process_instance_id; CODE_POS := 6; IF ip_user_name IS NULL OR LV_restrict_access_to IS NULL OR LV_owner_organizations IS NULL THEN RETURN 0; END IF; CODE_POS := 7; IF MS_APPS_SECURITY_PKG.EDIT_SECURITY.EXISTS(LV_object_type) THEN IF MS_APPS_SECURITY_PKG.EDIT_SECURITY(LV_object_type).INFO_TIMESTAMP < SYSDATE - INTERVAL '0 01:00:00' DAY TO SECOND THEN FETCH_SECURITY_INFO(LV_object_type); END IF; ELSE FETCH_SECURITY_INFO(LV_object_type); END IF; CODE_POS := 8; IF NOT MS_APPS_SECURITY_PKG.EDIT_SECURITY.EXISTS(LV_object_type) THEN RETURN 0; END IF; CODE_POS := 9; LV_security_all := MS_APPS_SECURITY_PKG.EDIT_SECURITY(LV_object_type).SECURITY_EDIT_ALL; CODE_POS := 10; IF MS_APPS_SECURITY_PKG.EDIT_ALL_ACCESS.EXISTS(ip_user_name) THEN IF MS_APPS_SECURITY_PKG.EDIT_ALL_ACCESS(ip_user_name).INFO_TIMESTAMP < SYSDATE - INTERVAL '0 01:00:00' DAY TO SECOND THEN FETCH_USER_ACCESS(LV_action, LV_security_all, ip_user_name); END IF; ELSE FETCH_USER_ACCESS(LV_action, LV_security_all, ip_user_name); END IF; CODE_POS := 11; IF MS_APPS_SECURITY_PKG.EDIT_ALL_ACCESS.EXISTS(ip_user_name) THEN IF MS_APPS_SECURITY_PKG.EDIT_ALL_ACCESS(ip_user_name).ACCESS_FLAG = 'ACCESS' THEN RETURN 1; END IF; END IF; CODE_POS := 12; LV_security_all := MS_APPS_SECURITY_PKG.EDIT_SECURITY(LV_object_type).SECURITY_EDIT; CODE_POS := 13; IF MS_APPS_SECURITY_PKG.USER_ACT_ORG.EXISTS(ip_user_name) THEN IF MS_APPS_SECURITY_PKG.USER_ACT_ORG(ip_user_name).ACT_LIST.EXISTS(LV_security_all) THEN IF MS_APPS_SECURITY_PKG.USER_ACT_ORG(ip_user_name).ACT_LIST(LV_security_all).INFO_TIMESTAMP < SYSDATE - INTERVAL '0 01:00:00' DAY TO SECOND THEN FETCH_ORGS(LV_security_all, ip_user_name); END IF; ELSE FETCH_ORGS(LV_security_all, ip_user_name); END IF; ELSE FETCH_ORGS(LV_security_all, ip_user_name); END IF; CODE_POS := 14; IF MS_APPS_SECURITY_PKG.USER_ACT_ORG.EXISTS(ip_user_name) THEN IF MS_APPS_SECURITY_PKG.USER_ACT_ORG(ip_user_name).ACT_LIST.EXISTS(LV_security_all) THEN LV_org_found := FALSE; LV_org_list := MS_APPS_APP_BUILDER_PKG.SPLIT_BY_DELIMITER(LV_owner_organizations); FOR LV_cntr IN LV_org_list.FIRST..LV_org_list.LAST LOOP LV_org_found := MS_APPS_SECURITY_PKG.USER_ACT_ORG(ip_user_name).ACT_LIST(LV_security_all).ORG_LIST.EXISTS(LV_org_list(LV_cntr)); IF LV_org_found THEN EXIT; END IF; END LOOP; IF LV_org_found THEN RETURN 1; ELSE IF LV_restrict_access_to = 'N' THEN RETURN 1; ELSE RETURN 0; END IF; END IF; ELSE IF CHECK_OBJECT_ACCESS ( ip_object_name => LV_object_type, ip_restrict_access_to => LV_restrict_access_to, ip_owners_organization => LV_owner_organizations, ip_user_name => ip_user_name, ip_action => 'VIEW' ) = 'ACCESS' THEN RETURN 2; ELSE RETURN 0; END IF; END IF; END IF; CODE_POS := 15; RETURN 1; EXCEPTION WHEN OTHERS THEN ERROR_CODE := SQLCODE; ERROR_MESSAGE := SUBSTR(SQLERRM, 1, 4000); RETURN 0; END CHECK_FORM_ACCESS;
