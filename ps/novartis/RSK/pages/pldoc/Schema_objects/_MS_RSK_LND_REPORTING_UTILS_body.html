FUNCTION get_distinct_mlov( IP_MAIN_STRING IN VARCHAR2, LV_QUERY_LIST IN VARCHAR2) RETURN VARCHAR2 AS LV_MAIN_LIST1 VARCHAR2_ARRAY := VARCHAR2_ARRAY (); LV_MAIN_LIST2 VARCHAR2_ARRAY := VARCHAR2_ARRAY (); GLOBAL_STRING1 VARCHAR2(31000) :=IP_MAIN_STRING; GLOBAL_STRING2 VARCHAR2(31000); BEGIN IF TRIM(LV_QUERY_LIST) IS NOT NULL THEN GLOBAL_STRING1 := IP_MAIN_STRING||','||LV_QUERY_LIST; ELSE GLOBAL_STRING1 := IP_MAIN_STRING; END IF; SELECT DISTINCT column_value bulk collect into LV_MAIN_LIST1 from table (MS_APPS_UTILITIES.PARSE_STRING (GLOBAL_STRING1, ',')) where column_value is not null; IF lv_main_list1.COUNT > 0 THEN FOR j IN 1 .. lv_main_list1.COUNT LOOP IF TRIM(GLOBAL_STRING2) IS NOT NULL THEN GLOBAL_STRING2 :=GLOBAL_STRING2||','||LV_MAIN_LIST1 (J); ELSE GLOBAL_STRING2 :=LV_MAIN_LIST1 (J); END IF; END LOOP; END IF; RETURN GLOBAL_STRING2; EXCEPTION WHEN OTHERS THEN RETURN NULL; END get_distinct_mlov;
