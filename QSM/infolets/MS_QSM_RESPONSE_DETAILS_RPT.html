                                                        SELECT que_request_name,
  substr(que_request_id,1,instr(que_request_id,'#')-1) que_request_id,
  (SELECT que_name
  FROM ms_Qsm_questionnaire
  WHERE internal_questnr_id=questionnaire_id
  )                AS questionnaire_name,
  questionnaire_id AS questionnaire_id ,
  section_name,
  question_name,
  question_pk,
  SORT_ID,
  (select first_name||' '||last_name from SI_USERS where user_name = case when (length(RESPONDENT)-length(replace(RESPONDENT,',','')))>0 then RESPONDED_BY else RESPONDENT end )Respondent ,
  response,
  RESP_COMMENTS resp_comments
FROM
  (SELECT que_request_name,
    que_request_id||'#'||qst_response_id que_request_id,
    questionnaire_id,
    section_name,
    question_name,
    question_pk,
    TO_NUMBER(REGEXP_REPLACE(section_id, '[^[:digit:'
    || CHR(93)
    || CHR(93)
    || '')
    ||regexp_replace(DECODE(qst_parent_type,'QST',question_fk,question_pk), '[^[:digit:'
    || chr(93)
    || chr(93))
    ||REGEXP_REPLACE(QUESTION_PK, '[^[:digit:'
    || CHR(93)
    || CHR(93)
    || ''))              AS SORT_ID,
    QUESTION_DESCRIPTION AS QUESTION_DESCRIPTION,
    DECODE(c.response_type,'1',TO_CHAR(c.resp_number),'2',TO_CHAR(c.resp_date),'3',c.resp_number,'4',
    (SELECT ms_concat(lov.display_value)
    FROM ms_qsm_resp_question_rml rml,
      ms_qsm_qst_lov_data lov
    WHERE rml.qst_response_id=a.que_response_id
    AND rml.question_pk      =c.QUESTION_PK
    AND rml.question_pk      =lov.question_id
    AND rml.RESP_MLOV        =lov.stored_value
    ),'5',c.resp_clob,'6',
    (SELECT lov.display_value
    FROM ms_qsm_qst_lov_data lov
    WHERE LOV.QUESTION_ID = C.QUESTION_PK
    AND lov.stored_value  = c.resp_radio
    ),'7',
    (SELECT lov.display_value
    FROM ms_qsm_qst_lov_data lov
    WHERE lov.question_id = c.question_pk
    AND lov.stored_value  = c.resp_dd
    ),'8',substr(resp_attachment,1,instr(resp_attachment,'#')-1),'9',resp_sprd,'10',
    (SELECT lov.display_value
    FROM ms_qsm_qst_lov_data lov
    WHERE LOV.QUESTION_ID = C.QUESTION_PK
    AND lov.stored_value  = c.resp_radio_b
    ),'11',
    (SELECT ms_concat(lov.display_value)
    FROM ms_qsm_resp_question_rml rml,
      ms_qsm_qst_lov_data lov
    WHERE rml.qst_response_id=a.que_response_id
    AND rml.question_pk      =c.QUESTION_PK
    AND rml.question_pk      =lov.question_id
    AND rml.RESP_MLOV        =lov.stored_value
    ))response,
   a.RESPONDENT,
   a.RESPONDED_BY,
    RESP_COMMENTS,
    section_id
  FROM ms_qsm_questnr_resp a,
    ms_qsm_resp_section b,
    ms_qsm_resp_question c,
    ms_qsm_scheduler_details d
  WHERE a.que_response_id     =b.section_fk
  AND a.que_response_id       =c.qst_response_id
  AND b.SECTION_ID            = c.QUESTION_SEC_ID
  AND a.DD_PROCESS_INSTANCE_ID=d.PROCESS_INSTANCE_ID
  AND ((:1          ='1'
  AND a.questionnaire_id      =:2)
  OR (:1            ='2'
  AND a.que_request_id        =:2))
  AND (instr(','
    ||NVL(:3,b.SECTION_ID)
    ||',',','
    ||b.SECTION_ID
    ||',',1,1)>0)
  AND (instr(','
    ||NVL(:4,c.question_pk)
    ||',',','
    ||c.question_pk
    ||',',1,1)     >0)
  AND (instr(','
    ||NVL(:5,a.RESPONDENT)
    ||',',','
    ||a.RESPONDENT
    ||',',1,1)>0)
  )
WHERE NVL(response,'$_$')=NVL(response,NVL(response,'$_$'))
GROUP BY que_request_name,
  substr(que_request_id,1,instr(que_request_id,'#')-1),
  questionnaire_id,
  section_name,
  question_name,
  question_pk,
  SORT_ID,
  response,
  section_id,
  case when (length(RESPONDENT)-length(replace(RESPONDENT,',','')))>0 then RESPONDED_BY else RESPONDENT end,
  RESP_COMMENTS,
  substr(que_request_id,instr(que_request_id,'#')+1)
ORDER BY SORT_ID
