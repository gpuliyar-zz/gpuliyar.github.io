                                                        SELECT ONGOING_DESC, QUE_REQUEST_TYPE, QTR_DESC, RESP_COUNT_BD,RESP_COUNT_L7,RESP_COUNT_G7 
FROM (
SELECT DUE_FLAG, NULL ONGOING_DESC, 
       QUE_REQUEST_TYPE, QTR_DESC, COUNT(QUE_RESPONSE_ID) RESP_COUNT
FROM ( 
    SELECT RESPONDED_ON,DUE_DATE,trunc(sysdate),
    CASE 
       WHEN  TRUNC(DUE_DATE) IS null THEN 'ERR'
       WHEN TRUNC(NVL(RESPONDED_ON,SYSDATE)) <= TRUNC((DUE_DATE))  THEN 'BD'
       WHEN TRUNC(nvl(RESPONDED_ON,SYSDATE)) <= TRUNC((DUE_DATE+7))  THEN 'L7'
       ELSE 'G7'
    END DUE_FLAG, QUE_REQUEST_TYPE, QUE_RESPONSE_ID, QTR_DESC
    FROM MS_QSM_QUESTNR_RESP_RPT_V RV, MS_QSM_QUARTER_RPT_V Q
    WHERE TRUNC(CREATION_DATE) between q.qtr_from and q.qtr_to
    --TRUNC(CREATION_DATE)>ADD_MONTHS(TRUNC(SYSDATE),-1)
AND QUE_REQUEST_TYPE='2'
    AND QUE_REQUEST_ID IN (
            SELECT IQ.QUE_REQUEST_ID
            FROM MS_QSM_INIT_QUESTNR IQ, MS_QSM_QUESTNR_SCH QS
            WHERE MS_APPS_SECURITY_PKG.CHECK_OBJECT_ACCESS(iq.DD_OBJECT_TYPE,iq.RESTRICT_ACCESS_TO,
                      (Select ms_concat(Owner_Organizations) from Ms_Qsm_Init_Questnr_Owo iqo Where Iqo.Que_Request_Id=Iq.Que_Request_Id )
                      ,:1,'VIEW') = 'ACCESS'  
              AND IQ.QUE_REQUEST_ID=QS.SCH_REQ_ID        
              AND QS.sch_frequency not in ('3','6') --exclude Live and RunNow                     
                      )
      )                      
GROUP BY DUE_FLAG, QUE_REQUEST_TYPE, QTR_DESC 
)
PIVOT ( SUM(RESP_COUNT) FOR DUE_FLAG in ('BD' RESP_COUNT_BD,'L7' RESP_COUNT_L7, 'G7' RESP_COUNT_G7)
      )
