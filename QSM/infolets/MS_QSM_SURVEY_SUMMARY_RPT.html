                                                        select que_request_name,que_request_id,(select que_name from ms_Qsm_questionnaire where internal_questnr_id=questionnaire_id) AS questionnaire_name,questionnaire_id AS questionnaire_id
,section_name,question_name,question_pk,SORT_ID,
response
,count(1) AS RESP_COUNT,ROUND((COUNT(1)/SUM(count(question_pk)) OVER (PARTITION BY question_pk,que_request_id))*100,2)  AS PERCENTAGE
from
(select que_request_name,que_request_id,questionnaire_id,section_name,question_name,question_pk,
TO_NUMBER(REGEXP_REPLACE(section_id, '[^[:digit:' || CHR(93) || CHR(93) || '')||regexp_replace(decode(qst_parent_type,'QST',question_fk,question_pk), '[^[:digit:' || chr(93) || chr(93))||REGEXP_REPLACE(QUESTION_PK, '[^[:digit:' || CHR(93) || CHR(93) || '')) AS SORT_ID,QUESTION_DESCRIPTION AS QUESTION_DESCRIPTION,
decode(c.response_type,'1',to_char(c.resp_number),'2',to_char(c.resp_date),'3',c.resp_number,'4',(select ms_concat(lov.display_value) 
from ms_qsm_resp_question_rml rml,ms_qsm_qst_lov_data lov 
where rml.qst_response_id=a.que_response_id
and rml.question_pk =c.QUESTION_PK
and rml.question_pk=lov.question_id
and rml.RESP_MLOV=lov.stored_value),'5',c.resp_clob,'6',(select lov.display_value 
from ms_qsm_qst_lov_data lov 
where LOV.QUESTION_ID = C.QUESTION_PK
and lov.stored_value = c.resp_radio),'7',(select lov.display_value 
from ms_qsm_qst_lov_data lov 
where lov.question_id = c.question_pk
and lov.stored_value = c.resp_dd),'8',substr(resp_attachment,1,instr(resp_attachment,'#',1)-1),'9',substr(substr(qst_json_value,instr(qst_json_value,'"SPRD_LINK"',1)+13),1,instr(substr(qst_json_value,instr(qst_json_value,'"SPRD_LINK"',1)+13),'"',1)-1),'10',(select lov.display_value 
from ms_qsm_qst_lov_data lov 
where LOV.QUESTION_ID = C.QUESTION_PK
and lov.stored_value = c.resp_radio_b),'11',(select ms_concat(lov.display_value) 
from ms_qsm_resp_question_rml rml,ms_qsm_qst_lov_data lov 
where rml.qst_response_id=a.que_response_id
and rml.question_pk =c.QUESTION_PK
and rml.question_pk=lov.question_id
and rml.RESP_MLOV=lov.stored_value))response,resp_comments,section_id
from ms_qsm_questnr_resp a,ms_qsm_resp_section b,ms_qsm_resp_question  c,ms_qsm_scheduler_details d
where  a.que_response_id=b.section_fk
and a.que_response_id=c.qst_response_id
and b.SECTION_ID = c.QUESTION_SEC_ID
and a.DD_PROCESS_INSTANCE_ID=d.PROCESS_INSTANCE_ID
and ((:1='1' AND a.questionnaire_id=:2) OR (:1='2' AND a.que_request_id=:2))
and (instr(','||nvl(:3,b.SECTION_ID)||',',','||b.SECTION_ID||',',1,1)>0)
and (instr(','||nvl(:4,c.question_pk)||',',','||c.question_pk||',',1,1)>0)
and (:6 is null or :6 = to_char(trunc(d.sent_out_on),'DD/MM/YYYY'))
)
where (instr(','||NVL(:5,response)||',',','||response||',',1,1)>0)
group by que_request_name,que_request_id,questionnaire_id,section_name,question_name,question_pk,SORT_ID,
response,section_id
order by SORT_ID
