                                                        SELECT DISTINCT 
          a.QUE_REQUEST_NAME,
          a.QUE_REQUEST_ID,
          a.DD_CREATED_BY,
          a.QUE_APPLIES_TO,
          (SELECT displayed_value
           FROM ms_qs_list_of_values_t
           WHERE lov_id =
                  (SELECT lov_id
                     FROM ms_qs_lov_names
                    WHERE lov_name = 'MS_QSM_Questionnaire_Type')
           AND stored_value = a.QUE_REQUEST_TYPE) AS QUE_REQUEST_TYPE,
          (SELECT COUNT(1) FROM MS_QSM_QUESTNR_RESP B WHERE B.QUE_REQUEST_ID= A.QUE_REQUEST_ID) RESPONDENT_COUNT,
          (SELECT SUM(DECODE(B.RESP_STATUS,4,0,1)) FROM MS_QSM_QUESTNR_RESP B WHERE B.QUE_REQUEST_ID= A.QUE_REQUEST_ID) RESPONSE_COUNT,
          (select ms_concat(c.owner_organizations) from MS_QSM_INIT_QUESTNR_owo c
           where a.QUE_REQUEST_ID = c.QUE_REQUEST_ID ) as owner_organization,
          ( select ms_concat(ms_apps_utilities.get_user_full_name(d.owners )) from MS_QSM_INIT_QUESTNR_ows d
          where a.QUE_REQUEST_ID = d.QUE_REQUEST_ID ) as owners
          FROM MS_QSM_INIT_QUESTNR a
