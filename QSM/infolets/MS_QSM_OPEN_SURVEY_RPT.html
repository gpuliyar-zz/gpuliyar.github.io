                                                        SELECT QUE_REQUEST_NAME,QUE_REQUEST_TYPE,MS_QSM_RPT_UTIL.GET_DISPLAY_VALUE( :1, 'MS_QSM_Init_Questionnaire_Type', a.QUE_REQUEST_TYPE,a.DD_OBJECT_TYPE ) AS QUE_REQUEST_TYPE_DISP,
MANAGE_SURVEY,MS_QSM_RPT_UTIL.GET_DISPLAY_VALUE( :1, 'MS_QSM_MANAGE_SURVEY', a.MANAGE_SURVEY,a.DD_OBJECT_TYPE ) AS MANAGE_SURVEY_DISP,
QUE_APPLIES_TO, (SELECT c.OBJECT_NAME
                     FROM MS_QSM_APPLIES_TO c
                    WHERE c.OBJECT_ID = a.QUE_APPLIES_TO
                          AND LOCALE_ID = (SELECT Y.LOCALE_ID
                                    FROM SI_USERS_T x, SI_LOCALES y
                                   WHERE X.LOCALE = Y.LOCALE_STRING
                                         AND X.USER_NAME = :1)) AS QUE_APPLIES_TO_DISP,SCH_END_BY,SCH_CLOSE_DATE,
MS_QSM_RPT_UTIL.GET_DISPLAY_VALUE( :1, 'MS_QSM_Frequency', a.SCH_FREQUENCY, a.DD_OBJECT_TYPE ) AS SCH_FREQUENCY,
DD_OBJECT_TYPE,MS_QSM_RPT_UTIL.GET_URL(a.DD_OBJECT_TYPE,a.DD_PROCESS_INSTANCE_ID, a.QUE_REQUEST_NAME)  as NAME_LINK,sch_frequency as  SCH_FREQUENCY_STORED
FROM MS_QSM_OPEN_SURVEYS_RPT_V a
WHERE  
(a.OWNER_ORGANIZATIONS IS NULL or 
MS_APPS_SECURITY_PKG.CHECK_OBJECT_ACCESS(a.DD_OBJECT_TYPE,NVL(a.RESTRICT_ACCESS_TO,'N'),a.OWNER_ORGANIZATIONS,:1,'VIEW') = 'ACCESS'
)
AND NVL(a.SCH_FREQUENCY,'X') <>'3'
