                                                        SELECT QUE_ID,
QUE_NAME,
QUE_TYPE_DISP,
APPLIES_TO_DISP,
OWNER_ORGANIZATIONS_DISP,
OWNERS_DISP,
NAME_LINK,
NO_OF_PLANS,
EXPIRED_FLAG,
VALID_FROM,
VALID_UNTILL,
LOGIN_ID,
VERSION_NUMBER
FROM
(SELECT q.que_id,
Q.QUE_NAME,
MS_QSM_RPT_UTIL.GET_DISPLAY_VALUE(:1,'MS_QSM_Questionnaire_Type' ,Q.QUE_TYPE,Q.DD_OBJECT_TYPE) as QUE_TYPE_DISP,
MS_QSM_RPT_UTIL.GET_APPLIES_TO_DISPLAY_VALUE(:1, Q.APPLIES_TO) AS APPLIES_TO_DISP,
Q.OWNER_ORGANIZATIONS_DISP,
Q.OWNERS_DISP,
MS_QSM_RPT_UTIL.GET_URL(Q.DD_OBJECT_TYPE,Q.DD_PROCESS_INSTANCE_ID, Q.QUE_NAME) AS NAME_LINK,
EXPIRED_FLAG,
(SELECT COUNT(P.QUE_REQUEST_ID)
FROM MS_QSM_INIT_QUESTNR_RPT_V P
WHERE Q.QUE_ID =P.QUESTIONNAIRE_ID
AND MS_APPS_SECURITY_PKG.CHECK_OBJECT_ACCESS(P.DD_OBJECT_TYPE, P.RESTRICT_ACCESS_TO, P.OWNER_ORGANIZATIONS,:1,'VIEW') = 'ACCESS'
) NO_OF_PLANS,
VALID_FROM,
VALID_UNTILL,
:1 as LOGIN_ID, 
VERSION_NUMBER
from MS_QSM_QUESTIONNAIRE_RPT_V Q
where MS_APPS_SECURITY_PKG.CHECK_OBJECT_ACCESS(Q.DD_OBJECT_TYPE, Q.RESTRICT_ACCESS_TO, Q.OWNER_ORGANIZATIONS,:1,'VIEW') = 'ACCESS'
and (:2 IS NULL and expired_flag =2) OR ( (:2 IS NOT NULL and :2='1' and expired_flag IN (1,2)) OR ( :2 IS NOT NULL and :2='2' and expired_flag IN (2)) )
)
