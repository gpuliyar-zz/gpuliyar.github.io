                                                        SELECT QUE_REQUEST_NAME,case when length(SECTION_NAME)>20 then cast(SECTION_NAME as varchar2(20))||'...' else SECTION_NAME end||': '|| QUESTION_NAME as QUESTION_NAME,RESPONDENT,response,RESP_COMMENTS FROM 
(select QUE_REQUEST_NAME,que_request_id,QUESTION_NAME,RESPONDENT,QUESTION_FK,QUESTION_PK,INSTANCE_ID,MAX(INSTANCE_ID) OVER (PARTITION BY PROCESS_INSTANCE_ID) MAX_IID,
       DECODE(response_type,
                  '1',TO_CHAR(RESP_NUMBER),
                  '2',TO_CHAR(resp_date),
                  '3',resp_number,
                  '4',   (SELECT ms_concat(lov.display_value)    FROM ms_qsm_qst_lov_data lov    WHERE LOV.QUESTION_ID = a.QUESTION_PK    AND instr(','||resp_mlov||',',','||lov.stored_value||',') >0     ),
                  '5',resp_clob,
                  '6',    (SELECT lov.display_value    FROM ms_qsm_qst_lov_data lov    WHERE LOV.QUESTION_ID = a.QUESTION_PK    AND lov.stored_value  = resp_radio    ),
                  '7',    (SELECT lov.display_value    FROM ms_qsm_qst_lov_data lov    WHERE lov.question_id = a.QUESTION_PK    AND lov.stored_value  = resp_dd    ),
                  '8',substr(resp_attachment,1,instr(resp_attachment,'#',1)-1),
                  '9',substr(substr(qst_json_value,instr(qst_json_value,'"SPRD_LINK"',1)+13),1,instr(substr(qst_json_value,instr(qst_json_value,'"SPRD_LINK"',1)+13),'"',1)-1),
                  '10',    (SELECT lov.display_value    FROM ms_qsm_qst_lov_data lov    WHERE LOV.QUESTION_ID = a.QUESTION_PK    AND lov.stored_value  = resp_radio_b    ),
                  '11',    (SELECT ms_concat(lov.display_value)     FROM ms_qsm_qst_lov_data lov    WHERE LOV.QUESTION_ID = a.QUESTION_PK    AND instr(','||CHECKBOX_HIDDEN_VALUE||',',','||lov.stored_value||',') >0    ))response,
       RESP_COMMENTS,
       (select SECTION_NAME from MS_QSM_SECTION where SECTION_ID_PK=a.QUESTION_SEC_ID ) SECTION_NAME,RESP_STATUS
       from MS_QSM_QUESTNR_RESP_DUP_V a
       where QUE_REQUEST_ID = :1
      )
      WHERE INSTANCE_ID=MAX_IID
      and QUESTION_NAME is not null
--      and SECTION_NAME is not null
      and RESP_STATUS IN ('1','5','6','7','9')--1	Approval Pending 2	Cancelled 3	Active 4	In Process 5	Responded 6	Completed 7	Certified 8	Not Certified 9	Approved     --and INSTANCE_ID in (select max(INSTANCE_ID) from MS_QSM_QUESTNR_RESP_DUP_V b where PROCESS_INSTANCE_ID=b.PROCESS_INSTANCE_ID and DD_CURRENT_STAGE in ('APPROVE','PUBLISH') group by PROCESS_INSTANCE_ID,DD_CURRENT_STAGE)
      START WITH SUBSTR(QUESTION_FK,1,3)='SEC'
CONNECT BY PRIOR QUESTION_PK = QUESTION_FK
ORDER SIBLINGS BY TO_NUMBER(REGEXP_REPLACE(QUESTION_PK, '[^[:digit:' || CHR(93) || CHR(93) || ''))
