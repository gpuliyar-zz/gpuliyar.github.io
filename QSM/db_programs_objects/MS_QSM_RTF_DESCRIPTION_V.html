MS_QSM_RTF_DESCRIPTION_V 

                                                
  CREATE OR REPLACE FORCE VIEW "NOVT_DEV"."MS_QSM_RTF_DESCRIPTION_V" ("DESCRIPTION", "DESC_TYPE", "Q_ID", "LOCALE_ID") AS 
  SELECT NVL(TRIM (y.qst_description), x.qst_description) as description,
        'D' as DESC_TYPE,
        x.qst_id_pk as q_id,
       NVL (y.locale_id, x.locale_id) as locale_id
  FROM (SELECT qst_description, locale_id, qst_id_pk
          FROM ms_qsm_question a , si_enabled_locales b
         --WHERE a.qst_id_pk = :QST_ID_PK 
         ) x,
            ms_qsm_question_tl y
        WHERE  x.qst_id_pk = y.qst_id_pk(+) 
      AND x.locale_id = y.locale_id(+)
      --AND :TYPE_VAL='D' 
UNION ALL
SELECT NVL(TRIM (y.CERTIFICATION_TEXT), x.CERTIFICATION_TEXT) as description,
        'R' as DESC_TYPE,
        x.INTERNAL_QUESTNR_ID as q_id,
       NVL (y.locale_id, x.locale_id)as locale_id FROM (
SELECT CERTIFICATION_TEXT, locale_id, INTERNAL_QUESTNR_ID
          FROM ms_qsm_questionnaire a , si_enabled_locales b
         --WHERE a.INTERNAL_QUESTNR_ID = :QST_ID_PK 
        ) x,
       ms_qsm_questionnaire_tl y
WHERE x.INTERNAL_QUESTNR_ID = y.INTERNAL_QUESTNR_ID(+) 
 --AND :TYPE_VAL='R'
AND x.locale_id = y.locale_id(+)
