MS_QSM_REPORTING_PERIOD_TL_V 

                                                
  CREATE OR REPLACE FORCE VIEW "NOVT_DEV"."MS_QSM_REPORTING_PERIOD_TL_V" ("ASSIGNMENT_START_DATE", "ASSIGNMENT_END_DATE", "CREATED_BY", "CREATION_DATE", "LAST_UPDATED_BY", "LAST_UPDATE_DATE", "PERIOD_ID", "PERIOD_TYPE", "PERIOD_NAME", "PERIOD_START_DATE", "PERIOD_END_DATE", "LOCALE_ID") AS 
  (SELECT a.ASSIGNMENT_START_DATE,
           a.ASSIGNMENT_END_DATE,
           a.CREATED_BY,
           a.CREATION_DATE,
           a.LAST_UPDATED_BY,
           a.LAST_UPDATE_DATE,
           a.PERIOD_ID,
           a.PERIOD_TYPE,
           NVL(b.PERIOD_NAME,a.PERIOD_NAME),
           a.PERIOD_START_DATE,
           a.PERIOD_END_DATE,
           NVL (a.locale_id,
                (SELECT LOCALE_ID
                   FROM SI_LOCALES
                  WHERE LOCALE_STRING = 'en_US'))
      FROM  (select c.ASSIGNMENT_START_DATE,
           c.ASSIGNMENT_END_DATE,
           c.CREATED_BY,
           c.CREATION_DATE,
           c.LAST_UPDATED_BY,
           c.LAST_UPDATE_DATE,
           c.PERIOD_ID,
           c.PERIOD_TYPE,
           c.PERIOD_NAME,
           c.PERIOD_START_DATE,
           c.PERIOD_END_DATE,
           d.locale_id from MS_QSM_REPORTING_PERIOD c,SI_ENABLED_LOCALES d) a, MS_QSM_REPORTING_PERIOD_TL b
             WHERE a.PERIOD_ID = b.PERIOD_ID(+)
             AND  a.locale_id=b.locale_id(+))
