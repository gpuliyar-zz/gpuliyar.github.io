  CREATE OR REPLACE FORCE VIEW "NOVT_DEV"."MS_QSM_QUESTION_TL_V" ("QST_ID_PK", "QST_SECTION_ID", "QST_SECTION_FK", "QST_ID", "QST_NAME", "QST_NAME_TL", "QST_DESCRIPTION", "QST_DESCRIPTION_TL", "QST_LIST_OF_VALUES", "QST_LIST_OF_VALUES_TL", "LOCALE_ID", "QST_DESCRIPTION_HTM_TL", "QST_ACTION_PLAN") AS 
  SELECT a.qst_id_pk,
          a.qst_section_id,
          a.qst_section_fk,
          a.qst_id,
          a.qst_name,
          b.qst_name AS qst_name_tl,
          a.qst_description,
          b.qst_description AS qst_description_tl,
          a.qst_list_of_values,
          b.qst_list_of_values AS qst_list_of_values_tl,
          b.locale_id,
          b.qst_description_htm,
          b.QST_ACTION_PLAN
     --          c.RQ_LOCATION,
     --          C.RQ_DESCRIPTION
     FROM ms_qsm_question a, ms_qsm_question_tl b
    --          ms_qsm_rephrase_qst_rpq_tl c
    WHERE a.qst_id_pk = b.qst_id_pk
