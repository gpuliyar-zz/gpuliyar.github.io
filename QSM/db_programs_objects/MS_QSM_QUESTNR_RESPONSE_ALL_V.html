MS_QSM_QUESTNR_RESPONSE_ALL_V 

                                                
  CREATE OR REPLACE FORCE VIEW "NOVT_DEV"."MS_QSM_QUESTNR_RESPONSE_ALL_V" ("QUE_RESPONSE_ID", "QUE_REQUEST_ID", "QUE_REQUEST_NAME", "DD_OBJECT_TYPE", "PROCESS_INSTANCE_ID", "INSTANCE_ID", "INSTANCE_REC_NUM", "MULTIROW_REGION_ID", "RESPONDENT", "CREATION_DATE", "RESPONDED_ON", "DUE_DATE", "RESP_STATUS", "RESP_GRADE", "RESP_SCORE", "APPLIES_TO", "APPLIES_TO_OBJECT", "RESPONSE_APPROVER", "SECTION_FK", "QUESTION_ID", "QUESTION_NAME", "SECTION_NAME", "RESPONSE_TYPE", "SCR_SCORE", "SCR_GRADE", "SECTION_ID", "OVERALL_GRADE", "OVERALL_SCORE", "QUESTION_FK", "RESPONDED_BY", "ASSIGNEE") AS 
  SELECT a.que_response_id,
  a.que_request_id,
  a.QUE_REQUEST_NAME,
  a.dd_object_type,
  a.PROCESS_INSTANCE_ID,
  a.instance_id,
  a.instance_rec_num,
  a.multirow_region_id,
  --a.finding_id,
  a.respondent,
  a.creation_date,
  a.responded_on,
  a.DUE_DATE,
  a.resp_status,
  a.resp_grade,
  a.RESP_SCORE,
  a.applies_to,
  a.applies_to_object,
  a.response_approver,
  a.section_fk,
  a.question_id,
  a.question_name,
  a.SECTION_NAME,
  a.response_type,
  a.scr_score,
  a.scr_grade,
  a.section_id,
  a.overall_grade,
  a.overall_score,
  a.question_fk,
  a.responded_by,
  a.assignee
FROM MS_QSM_QUESTNR_RESP_DUP_V A,
  (SELECT v.process_instance_id,
    MAX(v.instance_id) instance_id
  FROM ms_qsm_questnr_resp_dup_v v
  WHERE v.instance_rec_num = 1
  GROUP BY process_instance_id
  ) b
WHERE a.process_instance_id=b.process_instance_id
AND a.instance_id          =b.instance_id
  /*A.INSTANCE_ID =
  (SELECT MAX (INSTANCE_ID)
  from ms_qsm_questnr_resp_dup_v b
  WHERE B.PROCESS_INSTANCE_ID = A.PROCESS_INSTANCE_ID)*/
AND instance_rec_num = 1
