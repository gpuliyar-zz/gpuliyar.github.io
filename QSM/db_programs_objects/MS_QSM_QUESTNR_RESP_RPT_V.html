MS_QSM_QUESTNR_RESP_RPT_V 

                                                
  CREATE OR REPLACE FORCE VIEW "NOVT_DEV"."MS_QSM_QUESTNR_RESP_RPT_V" ("QUE_RESPONSE_ID", "QUE_REQUEST_ID", "PROCESS_INSTANCE_ID", "INSTANCE_ID", "DD_OBJECT_TYPE", "ENT_ID", "QUE_REQUEST_NAME", "RESPONDENT", "RESPONDENT_DISP", "CREATION_DATE", "RESPONDED_ON", "DUE_DATE", "RESP_STATUS", "QUE_REQUEST_TYPE", "RESP_GRADE", "RESP_SCORE", "APPLIES_TO", "APPLIES_TO_OBJECT", "RESPONSE_APPROVER", "RESPONSE_APPROVER_DISP", "NO_OF_FINDINGS", "DUE_FLAG", "ISM_YN", "ASSIGNEE") AS 
  SELECT A.QUE_RESPONSE_ID,
          A.QUE_REQUEST_ID,
          A.PROCESS_INSTANCE_ID,
          A.INSTANCE_ID,
          DD_OBJECT_TYPE,
          A.ENT_ID,
          a.QUE_REQUEST_NAME,
          a.RESPONDENT,
          case when length(A.RESPONDENT)-length(replace(A.RESPONDENT,',')) >0 then 
          nvl(MS_APPS_UTILITIES.GET_USER_FULL_NAME (A.RESPONDED_BY),'Assigned to Role') else
          NVL (
             MS_APPS_UTILITIES.GET_USER_FULL_NAME (A.RESPONDENT),
             (SELECT MS_CONCAT (
                        MS_APPS_UTILITIES.GET_USER_FULL_NAME (COLUMN_VALUE))
                FROM TABLE (
                        ms_apps_utilities.parse_string (A.RESPONDENT, ',')))) end
             AS RESPONDENT_DISP,
          a.CREATION_DATE,
          a.RESPONDED_ON,
          a.DUE_DATE,
          a.RESP_STATUS,
          --RESP_STATUS_DISP,
          A.QUE_REQUEST_TYPE,
          --MS_APPS_UTILITIES.GET_DISPLAY_VALUES(a.ent_id,'MS_QSM_Init_Questionnaire_Type',A.QUE_REQUEST_TYPE,',') AS QUE_REQUEST_TYPE_DISP,
          A.RESP_GRADE,
          A.RESP_SCORE,
          A.APPLIES_TO,
          A.APPLIES_TO_OBJECT,
          a.RESPONSE_APPROVER,
          (SELECT MS_CONCAT (
                        MS_APPS_UTILITIES.GET_USER_FULL_NAME (COLUMN_VALUE))
                FROM TABLE (
                        ms_apps_utilities.parse_string (a.RESPONSE_APPROVER, ','))) RESPONSE_APPROVER_DISP,
          --MS_APPS_UTILITIES.GET_USER_FULL_NAME (a.RESPONSE_APPROVER)             AS RESPONSE_APPROVER_DISP,
          A.NO_OF_FINDINGS,
          a.DUE_FLAG,
          a.ISM_YN,
          a.ASSIGNEE
     FROM (  SELECT rv.que_response_id,
                    rv.que_request_id,
                    rv.process_instance_id,
                    rv.instance_id,
                    rv.dd_object_type,
                    ms_apps_utilities.
                    get_enterprise_id (
                       ms_apps_utilities.get_infolet_id (rv.dd_object_type))
                       AS ent_id,
                    RV.QUE_REQUEST_NAME,
                    RV.RESPONDENT,
                    RV.CREATION_DATE,
                    RESPONDED_ON,
                    RV.DUE_DATE,
                    RV.RESP_STATUS,
                    --(SELECT status_name FROM MS_QSM_STATUS WHERE status_id=rv.resp_status) AS resp_status_disp,
                    IQ.QUE_REQUEST_TYPE,
                    (SELECT GRADE_NAME
                       FROM MS_QSM_GRADE_CONFIG
                      WHERE GRADE_ID = RV.OVERALL_GRADE)
                       AS RESP_GRADE,
                    RV.OVERALL_SCORE AS RESP_SCORE,
                    rv.applies_to,
                    rv.applies_to_object,
                    RV.RESPONSE_APPROVER,
                    --                    (SELECT DECODE (I.ISM_YN,
                    --                                    'Y', COUNT (DISTINCT Finding_Id),
                    --                                    NULL)
                    --                       --From Ms_Qsm_Questnr_Response_V
                    --                       FROM Ms_Qsm_Questnr_Response_ALL_V
                    --                      WHERE     Que_Request_Id = Rv.Que_Request_Id
                    --                            AND Process_Instance_Id = Rv.Process_Instance_Id
                    --                            AND INSTANCE_ID = RV.INSTANCE_ID
                    --                            AND i.ISM_YN = 'Y')
                    --                       AS NO_OF_FINDINGS,
                    (SELECT DECODE (I.ISM_YN, 'Y', COUNT (FINDING_ID), NULL)
                       FROM MS_QSM_RESP_ISSUE
                      WHERE FINDING_FK = RV.QUE_RESPONSE_ID)
                       AS NO_OF_FINDINGS,
                    CASE
                       WHEN TRUNC (rv.DUE_DATE) IS NULL
                       THEN
                          'ERR'
                       WHEN TRUNC (NVL (rv.RESPONDED_ON, SYSDATE)) <=
                               TRUNC ( (rv.DUE_DATE))
                       THEN
                          'BD'
                       WHEN TRUNC (NVL (rv.RESPONDED_ON, SYSDATE)) <=
                               TRUNC ( (rv.DUE_DATE + 7))
                       THEN
                          'L7'
                       ELSE
                          'G7'
                    END
                       DUE_FLAG,
                    I.ISM_YN,
                    --From Ms_Qsm_Questnr_Response_V Rv, Ms_Qsm_Init_Questnr Iq
                    --From Ms_Qsm_Questnr_Resp_Dup_V Rv, Ms_Qsm_Init_Questnr Iq
                    RV.RESPONDED_BY,
                    ASSIGNEE
               FROM Ms_Qsm_Questnr_Response_ALL_V Rv,
                    Ms_Qsm_Init_Questnr Iq,
                    (SELECT DECODE (COUNT (*), 0, 'N', 'Y') ISM_YN
                       FROM ms_apps_events
                      WHERE EVENT_NAME = 'TRIGGER_ISSUE') I
              WHERE rv.que_request_id = iq.que_request_id
           ORDER BY RV.QUE_REQUEST_NAME) a
