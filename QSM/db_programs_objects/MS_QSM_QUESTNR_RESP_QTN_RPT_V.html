MS_QSM_QUESTNR_RESP_QTN_RPT_V 

                                                
  CREATE OR REPLACE FORCE VIEW "NOVT_DEV"."MS_QSM_QUESTNR_RESP_QTN_RPT_V" ("QUE_RESPONSE_ID", "QUE_REQUEST_NAME", "QST_SECTION_FK", "QST_ID_PK", "QUESTION_TEXT", "PROCESS_INSTANCE_ID", "MULTIROW_REGION_ID", "SECTION_NAME", "RESPONSE_TYPE", "RESPONSE_OPTIONS", "SCORE", "GRADE") AS 
  SELECT QUE_RESPONSE_ID,
          QUE_REQUEST_NAME,
          QST_SECTION_FK,
          QST_ID_PK,
          QUESTION_TEXT,
          PROCESS_INSTANCE_ID,
          MULTIROW_REGION_ID,
          SECTION_NAME,
          RESPONSE_TYPE,
          RESPONSE_OPTIONS,
          SCORE,
          (SELECT GRADE_NAME
             FROM MS_QSM_GRADE_CONFIG
            WHERE GRADE_ID = a.GRADE)
             AS GRADE
     FROM (SELECT                              --SECTION_FK AS QST_SECTION_FK,
                 QUESTION_FK AS QST_SECTION_FK,
                  QUESTION_ID AS QST_ID_PK,
                  QUESTION_NAME AS Question_Text,
                  PROCESS_INSTANCE_ID,
                  MULTIROW_REGION_ID,
                  SECTION_NAME,
                  MS_QSM_EMAIL_PARAMETER.GET_LOV_DISPLAY_NAME (RESPONSE_TYPE)
                     AS RESPONSE_TYPE,
                  NULL RESPONSE_OPTIONS,
                  QUE_REQUEST_NAME,
                  RESP_SCORE AS SCORE,
                  RESP_GRADE AS GRADE,
                  QUE_RESPONSE_ID
             --FROM MS_QSM_QUESTNR_RESPONSE_V V
             --FROM Ms_Qsm_Questnr_Resp_Dup_V V
             FROM MS_QSM_QUESTNR_RESPONSE_DATA_V V
            WHERE PROCESS_INSTANCE_ID =
                     (SELECT MAX (PROCESS_INSTANCE_ID) AS INSTID
                        --FROM MS_QSM_QUESTNR_RESPONSE_V
                        --FROM Ms_Qsm_Questnr_Resp_Dup_V
                        FROM MS_QSM_QUESTNR_RESPONSE_DATA_V
                       WHERE QUE_RESPONSE_ID = V.QUE_RESPONSE_ID)
                  AND (INSTANCE_ID) =
                         (SELECT MAX (INSTANCE_ID) AS INSTID
                            --FROM MS_QSM_QUESTNR_RESPONSE_V
                            --FROM Ms_Qsm_Questnr_Resp_Dup_V
                            FROM MS_QSM_QUESTNR_RESPONSE_DATA_V
                           WHERE QUE_RESPONSE_ID = V.QUE_RESPONSE_ID
                                 AND PROCESS_INSTANCE_ID =
                                        V.PROCESS_INSTANCE_ID)
                  AND (QUESTION_ID IS NOT NULL OR SECTION_FK IS NOT NULL)
           UNION
           SELECT                              --SECTION_FK AS QST_SECTION_FK,
                 NULL AS QST_SECTION_FK,
                  SECTION_ID,
                  QUESTION_NAME AS QUESTION_TEXT,
                  PROCESS_INSTANCE_ID,
                  MULTIROW_REGION_ID,
                  SECTION_NAME,
                  MS_QSM_EMAIL_PARAMETER.GET_LOV_DISPLAY_NAME (RESPONSE_TYPE)
                     AS RESPONSE_TYPE,
                  NULL RESPONSE_OPTIONS,
                  QUE_REQUEST_NAME,
                  RESP_SCORE,
                  RESP_GRADE,
                  QUE_RESPONSE_ID
             --FROM MS_QSM_QUESTNR_RESPONSE_V v
             --FROM Ms_Qsm_Questnr_Resp_Dup_V v
             FROM MS_QSM_QUESTNR_RESPONSE_DATA_V v
            WHERE PROCESS_INSTANCE_ID =
                     (SELECT MAX (PROCESS_INSTANCE_ID) AS INSTID
                        --FROM MS_QSM_QUESTNR_RESPONSE_V
                        --FROM Ms_Qsm_Questnr_Resp_Dup_V
                        FROM MS_QSM_QUESTNR_RESPONSE_DATA_V
                       WHERE QUE_RESPONSE_ID = V.QUE_RESPONSE_ID)
                  AND (INSTANCE_ID) =
                         (SELECT MAX (INSTANCE_ID) AS INSTID
                            --FROM MS_QSM_QUESTNR_RESPONSE_V
                            --FROM Ms_Qsm_Questnr_Resp_Dup_V
                            FROM MS_QSM_QUESTNR_RESPONSE_DATA_V
                           WHERE QUE_RESPONSE_ID = V.QUE_RESPONSE_ID
                                 AND PROCESS_INSTANCE_ID =
                                        V.PROCESS_INSTANCE_ID)
                  AND (SECTION_ID IS NOT NULL)) A
