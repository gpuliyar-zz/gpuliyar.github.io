MS_QSM_QUESTION_BA_V 

                                                
  CREATE OR REPLACE FORCE VIEW "NOVT_DEV"."MS_QSM_QUESTION_BA_V" ("CHECKBOX", "RADIO", "OBJECT_ID", "OBJECT_NAME", "RESPONSE_TYPE", "RESPONSE_TYPE_STORED", "LIST_OF_RESPONSE", "CATEGORIES", "BU_NAME", "OWNER_ORGANIZATIONS", "RESTRICT_ACCESS_TO", "DD_OBJECT_TYPE") AS 
  SELECT  '' AS CHECKBOX,
	   '' AS RADIO,
	   T1.OBJECT_ID,
	   OBJECT_NAME,RESPONSE_TYPE,RESPONSE_TYPE_STORED,LIST_OF_RESPONSE ,categories,bu_name,Owner_Organizations,RESTRICT_ACCESS_TO,DD_OBJECT_TYPE
FROM   (SELECT DISTINCT OBJECT_ID   AS OBJECT_ID,
						OBJECT_NAME AS OBJECT_NAME,Restrict_Access_To,DD_OBJECT_TYPE,Owner_Organizations,MS_APPS_UTILITIES.GET_DISPLAY_VALUE ( MS_GRC_UTILITIES.ACTIVE_ENT_ID, 'MS GRC Response Type', RESPONSE_TYPE )  AS RESPONSE_TYPE, RESPONSE_TYPE AS RESPONSE_TYPE_STORED,LIST_OF_RESPONSE
		FROM   MS_GRC_QUESTION_PROC_PUB_V)T1,
    MS_QSM_QUESTION_PRO_V T2
     WHERE T1.OBJECT_ID=T2.OBJECT_ID
      ORDER BY OBJECT_NAME
