MS_QSM_QUESTIONNAIRE_RPT_V 

                                                
  CREATE OR REPLACE FORCE VIEW "NOVT_DEV"."MS_QSM_QUESTIONNAIRE_RPT_V" ("QUE_ID", "ENT_ID", "DD_OBJECT_TYPE", "QUE_TYPE", "OWNER_ORGANIZATIONS", "OWNERS", "APPLIES_TO", "QUE_NAME", "OWNER_ORGANIZATIONS_DISP", "OWNERS_DISP", "VALID_FROM", "VALID_UNTILL", "EXPIRED_FLAG", "DD_PROCESS_INSTANCE_ID", "RESTRICT_ACCESS_TO", "VERSION_NUMBER") AS 
  SELECT q.que_id,
          q.ent_id,
          q.dd_object_type,
          q.que_type,
          OWNER_ORGANIZATIONS,
          q.Owners,
          q.applies_to,
          q.que_name,
          --ms_apps_utilities.get_display_values(q.ent_id,'MS_QSM_Questionnaire_Type',q.que_type,',') que_type_disp,
          --q.Applies_To_disp,
          Owner_organizations_Disp,
          Owners_Disp,
          VALID_FROM,
          VALID_UNTILL,
          CASE
             WHEN TRUNC (SYSDATE) <= NVL (valid_untill, SYSDATE) THEN '2'
             ELSE '1'
          END
             EXPIRED_FLAG,
          q.dd_process_instance_id,
          q.restrict_access_to,
          Version_Number
     FROM (  SELECT a.INTERNAL_QUESTNR_ID que_id,
                    MS_APPS_UTILITIES.
                    GET_ENTERPRISE_ID (
                       MS_APPS_UTILITIES.GET_INFOLET_ID (a.DD_OBJECT_TYPE))
                       ENT_ID,
                    a.DD_OBJECT_TYPE,
                    a.VALID_FROM,
                    a.VALID_UNTILL,
                    a.que_name,
                    a.que_type,
                    (SELECT ms_concat (b.applies_to)
                       FROM ms_qsm_questionnaire_apt b
                      WHERE b.internal_questnr_id = a.internal_questnr_id)
                       applies_to,
                    --(select ms_concat(c.OBJECT_NAME) from MS_QSM_QUESTIONNAIRE_APT b,MS_QSM_APPLIES_TO c  where  b.applies_to = c.object_id and b.internal_questnr_id=a.que_id) Applies_To_Disp,
                    /*(SELECT ms_concat (b.owner_organizations)
                       FROM MS_QSM_QUESTIONNAIRE_OWO b
                      WHERE b.internal_questnr_id = a.internal_questnr_id)*/
                      MS_QSM_UTILITIES.FN_GET_MLOV_VALUES('internal_questnr_id='''||internal_questnr_id||'''','OWNER_ORGANIZATIONS','MS_QSM_QUESTIONNAIRE_OWO')
                       OWNER_ORGANIZATIONS,
                   /* (SELECT ms_concat (DISTINCT c.org_entity_name)
                       FROM MS_QSM_QUESTIONNAIRE_OWO b,
                            ms_qsm_flow_down_org_v c
                      WHERE b.owner_organizations = c.org_entity_id
                            AND B.INTERNAL_QUESTNR_ID = A.INTERNAL_QUESTNR_ID
                            AND ROWNUM <=10)*/
                            MS_QSM_UTILITIES.FN_GET_MLOV_VALUES('B.OWNER_ORGANIZATIONS = C.ORG_ENTITY_ID AND B.internal_questnr_id='''||internal_questnr_id||'''','DISTINCT C.ORG_ENTITY_NAME','MS_QSM_QUESTIONNAIRE_OWO B, SI_ORG_ENTITIES_T C')
                       OWNER_ORGANIZATIONS_DISP,
                    /*(SELECT ms_concat (b.owners)
                       FROM MS_QSM_QUESTIONNAIRE_OWS B
                      WHERE b.internal_questnr_id = a.internal_questnr_id)*/
                      MS_QSM_UTILITIES.FN_GET_MLOV_VALUES('internal_questnr_id='''||internal_questnr_id||'''','OWNERS','MS_QSM_QUESTIONNAIRE_OWS')
                       OWNERS,
                   /* (SELECT MS_CONCAT (
                               MS_APPS_UTILITIES.GET_USER_FULL_NAME (B.OWNERS))
                       FROM MS_QSM_QUESTIONNAIRE_OWS B
                      WHERE B.INTERNAL_QUESTNR_ID = A.INTERNAL_QUESTNR_ID
                      AND ROWNUM <=10)*/
                      MS_QSM_UTILITIES.FN_GET_MLOV_VALUES('internal_questnr_id='''||internal_questnr_id||'''','MS_APPS_UTILITIES.GET_USER_FULL_NAME (OWNERS)','MS_QSM_QUESTIONNAIRE_OWS')
                       OWNERS_DISP,
                    A.DD_PROCESS_INSTANCE_ID,
                    A.RESTRICT_ACCESS_TO,
                    A.Version_Number
               FROM MS_QSM_QUESTIONNAIRE a
           ORDER BY a.que_name) q
