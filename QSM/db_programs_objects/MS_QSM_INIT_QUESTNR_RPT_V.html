MS_QSM_INIT_QUESTNR_RPT_V 

                                                
  CREATE OR REPLACE FORCE VIEW "NOVT_DEV"."MS_QSM_INIT_QUESTNR_RPT_V" ("QUE_REQUEST_ID", "QUESTIONNAIRE_ID", "QUE_REQUEST_NAME", "QUE_REQUEST_TYPE", "MANAGE_SURVEY", "ENT_ID", "DD_OBJECT_TYPE", "DD_PROCESS_INSTANCE_ID", "SCH_FREQUENCY", "SCH_NEXT_DATE", "LAST_SENT_DATE", "SENT", "PROGRESS", "COMPLETED", "SCH_END_BY", "OPEN_ISSUES", "ISSUES_PAST_DUE", "QUE_APPLIES_TO", "QUE_APPLIES_TO_DISP", "OWNER_ORGANIZATIONS", "OWNER_ORGANIZATIONS_DISP", "RESTRICT_ACCESS_TO", "SCH_START_DATE", "SCH_DUE_AFTER", "QUE_NAME", "OWNERS", "OWNERS_DISP", "DD_CREATED_BY") AS 
  SELECT a.QUE_REQUEST_ID,
          a.QUESTIONNAIRE_ID,
          a.QUE_REQUEST_NAME,
          a.QUE_REQUEST_TYPE,
          a.MANAGE_SURVEY,
          a.ENT_ID,
          a.DD_OBJECT_TYPE,
          a.DD_PROCESS_INSTANCE_ID,
          a.sch_frequency,
          a.sch_next_date,
          a.LAST_SENT_DATE,
          a.SENT,
          (a.SENT - a.COMPLETED) PROGRESS,
          a.COMPLETED,
          a.SCH_END_BY,
          a.OPEN_ISSUES,
          a.ISSUES_PAST_DUE,
          A.QUE_APPLIES_TO,
          (SELECT B.OBJECT_NAME
             FROM MS_QSM_APPLIES_TO B
            WHERE B.OBJECT_ID = a.QUE_APPLIES_TO
                  AND LOCALE_ID =
                         (SELECT Y.LOCALE_ID
                            FROM SI_USERS_T x, SI_LOCALES y
                           WHERE X.LOCALE = Y.LOCALE_STRING
                                 AND X.USER_NAME = A.DD_CREATED_BY))
             AS QUE_APPLIES_TO_DISP,
          OWNER_ORGANIZATIONS,
          OWNER_ORGANIZATIONS_DISP,
          RESTRICT_ACCESS_TO,
          SCH_START_DATE,
          SCH_DUE_AFTER,
          QUE_NAME,
          OWNERS,
          OWNERS_DISP,
          DD_CREATED_BY
     FROM (  SELECT A.QUE_REQUEST_ID,
                    A.QUESTIONNAIRE_ID,
                    A.QUE_REQUEST_NAME,
                    A.QUE_REQUEST_TYPE,
                    A.MANAGE_SURVEY,
                    MS_APPS_UTILITIES.
                    GET_ENTERPRISE_ID (
                       MS_APPS_UTILITIES.GET_INFOLET_ID (a.DD_OBJECT_TYPE))
                       ENT_ID,
                    a.DD_OBJECT_TYPE,
                    B.SCH_FREQUENCY,
                    b.SCH_NEXT_DATE,
                    (SELECT MAX (TRUNC (SENT_OUT_ON))
                       FROM MS_QSM_SCHEDULER_DETAILS SA
                      WHERE SA.SCH_REQ_ID = a.QUE_REQUEST_ID)
                       last_sent_date,
                   /* (SELECT COUNT (QUE_RESPONSE_ID)
                       FROM MS_QSM_QUESTNR_RESP_RPT_V
                      WHERE QUE_REQUEST_ID = a.QUE_REQUEST_ID)
                       SENT,*/
                       (SELECT COUNT(1) FROM MS_QSM_SCHEDULER_DETAILS X WHERE X.SCH_REQ_ID=A.QUE_REQUEST_ID)SENT,
                    (SELECT COUNT (QUE_RESPONSE_ID)
                       FROM MS_QSM_QUESTNR_RESP SA
                      WHERE SA.QUE_REQUEST_ID = A.QUE_REQUEST_ID)
                       COMPLETED,
                    B.SCH_END_BY,
                    MS_QSM_RPT_UTIL.
                    GET_ISSUE_COUNT ('OPEN', 'REQUEST_ID', QUE_REQUEST_ID)
                       OPEN_ISSUES,
                    MS_QSM_RPT_UTIL.
                    GET_ISSUE_COUNT ('OVERDUE', 'REQUEST_ID', QUE_REQUEST_ID)
                       ISSUES_PAST_DUE,
                    a.DD_PROCESS_INSTANCE_ID,
                    A.QUE_APPLIES_TO,
                   /* (SELECT MS_CONCAT (B.OWNER_ORGANIZATIONS)
                       FROM MS_QSM_INIT_QUESTNR_OWO B
                      WHERE B.QUE_REQUEST_ID = a.QUE_REQUEST_ID)*/
                      MS_QSM_UTILITIES.FN_GET_MLOV_VALUES('QUE_REQUEST_ID='''||QUE_REQUEST_ID||'''','OWNER_ORGANIZATIONS','MS_QSM_INIT_QUESTNR_OWO')
                       OWNER_ORGANIZATIONS,
                    /*(SELECT MS_CONCAT (DISTINCT C.ORG_ENTITY_NAME)
                       FROM MS_QSM_INIT_QUESTNR_OWO B, MS_QSM_FLOW_DOWN_ORG_V C
                      WHERE B.OWNER_ORGANIZATIONS = C.ORG_ENTITY_ID
                            AND B.QUE_REQUEST_ID = A.QUE_REQUEST_ID
                            AND ROWNUM<=10)*/
                            MS_QSM_UTILITIES.FN_GET_MLOV_VALUES('B.OWNER_ORGANIZATIONS = C.ORG_ENTITY_ID AND B.QUE_REQUEST_ID='''||QUE_REQUEST_ID||'''','DISTINCT C.ORG_ENTITY_NAME','MS_QSM_INIT_QUESTNR_OWO B, SI_ORG_ENTITIES_T C')
                       OWNER_ORGANIZATIONS_DISP,
                    /*(SELECT MS_CONCAT (B.OWNERS)
                       FROM MS_QSM_INIT_QUESTNR_OWS B
                      WHERE B.QUE_REQUEST_ID = a.QUE_REQUEST_ID)*/
                      MS_QSM_UTILITIES.FN_GET_MLOV_VALUES('QUE_REQUEST_ID='''||QUE_REQUEST_ID||'''','OWNERS','MS_QSM_INIT_QUESTNR_OWS')
                       OWNERS,
                    /*(SELECT MS_CONCAT (
                               MS_APPS_UTILITIES.GET_USER_FULL_NAME (B.OWNERS))
                       FROM MS_QSM_INIT_QUESTNR_OWS B
                      WHERE B.QUE_REQUEST_ID = a.QUE_REQUEST_ID
                      AND ROWNUM <=10)*/
                      MS_QSM_UTILITIES.FN_GET_MLOV_VALUES('QUE_REQUEST_ID='''||QUE_REQUEST_ID||'''','MS_APPS_UTILITIES.GET_USER_FULL_NAME (OWNERS)','MS_QSM_INIT_QUESTNR_OWS')
                       OWNERS_DISP,
                    A.RESTRICT_ACCESS_TO,
                    B.SCH_START_DATE,
                    B.SCH_DUE_AFTER,
                    C.QUE_NAME,
                    A.DD_CREATED_BY
               FROM MS_QSM_INIT_QUESTNR a,
                    MS_QSM_QUESTNR_SCH B,
                    MS_QSM_QUESTIONNAIRE C
              WHERE A.QUE_REQUEST_ID = B.SCH_REQ_ID(+)
                    AND a.questionnaire_id = c.INTERNAL_QUESTNR_ID
           ORDER BY a.QUE_REQUEST_NAME) a
